150 def _do_analysis(self): 

151 """ 

152 Do the phase RMS calculation, outlier analysis, 

153 and fill the dictionary with everything required for later scoring and  

154 plotting 

155 

156 inputs used are: 

157 self.cycletime, self.totaltime, self.PMinACA, 

158 self.outlierlimit, self.outlier_antennas, self.maxpoorant  

159 

160 uses functions: 

161 self._phase_rms_caltab, self.mad 

162 

163 fills: 

164 self.allResults 

165 

166 dict keys are: 

167 blphaserms, blphasermscycle, bllen, blname, blout, 

168 blphasermsbad, blphasermscyclebad, bllenbad, 

169 antphaserms, antphasermscycle, antname, 

170 phasermsP80, phasermscycleP80, blP80, blP80orig 

171 """ 

172 

173 # Call to phase_rms_caltab 

174 # gets baseline based phase RMS and antenna based phase RMS for 

175 # the total time (i.e. length of BP scan) and  

176 # the cycle time (time it takes to cycle the start of a phase cal scan to the next - ties with a 'decohernce time' over the target) 

177 if self.cycletime < self.totaltime: 

178 self.allResult = self._phase_rms_caltab(timeScale=self.cycletime) # if cycle time is shorter we pass the option so it gets assessed 

179 else: 

180 self.allResult = self._phase_rms_caltab() # otherwise no options added and cycletime value will equal total time value 

181 

182 # Check the cycle time calculation is not all nans - might happen if there are considerable flags 

183 # i.e. if a large fraction of data > self.flag_tolerance are flagged 

184 cycleResultFinite = len(np.array(self.allResult['blphasermscycle'])[np.isfinite(self.allResult['blphasermscycle'])]) 

185 

186 # if it is all flagged, theory to scale down total time with 'lower' scaling constant power (Maud et al. 2022) 

187 if cycleResultFinite == 0: 

188 self.allResult['blphasermscycle'] = np.array(self.allResult['blphaserms'])*(self.cycletime/self.totaltime)**0.3 

189 

190 if self.PMinACA: 

191 # we need to exclude PM antennas from the calculation as they 'can' be too 'long' baselines 

192 # and not really useful for the understanding of the target likely phase RMS  

193 # where only CM used in the images. Thus cut the PM from the upper 80 cut 

194 blACAid = np.array([blid for blid in range(len(self.allResult['blname'])) if 'PM' not in self.allResult['blname'][blid]]) 

195 antACAid = np.array([antid for antid in range(len(self.allResult['antname'])) if 'PM' not in self.allResult['antname'][antid]]) 

196 

197 # Reset allResult to exclude the PM baselines, and PM ants only 

198 bl_keys = ['blphaserms', 'blphasermscycle', 'bllen', 'blname'] # PIPE-1633 

199 ant_keys = ['antphaserms', 'antphasermscycle', 'antname'] # PIPE-1633 

200 for bl_key in bl_keys: 

201 self.allResult[bl_key] = np.array(self.allResult[bl_key])[blACAid] 

202 for ant_key in ant_keys: 

203 self.allResult[ant_key] = np.array(self.allResult[ant_key])[antACAid] 

204 

205 # Get 80th percentile baseline length and ID of all baselines above it 

206 self.allResult['blP80orig'] = np.percentile(self.allResult['bllen'], 80) # was xy 

207 

208 # PIPE-1662 P80 to acount for flagged antennas, i.e work from P80 of the 'good' data 

209 self.allResult['blP80'] = np.percentile(np.array(self.allResult['bllen'])[np.isfinite(np.array(self.allResult['blphaserms']))], 80) 

210 ##### 

211 

212 ID_all80 = np.where(np.array(self.allResult['bllen'])> self.allResult['blP80'])[0] # was xy 

213 

214 # First calculation of the representative phase RMS values 

215 self.allResult['phasermsP80'] = np.median(np.array(self.allResult['blphaserms'])[ID_all80[np.isfinite(np.array(self.allResult['blphaserms'])[ID_all80])]]) 

216 self.allResult['phasermscycleP80'] = np.median(np.array(self.allResult['blphasermscycle'])[ID_all80[np.isfinite(np.array(self.allResult['blphasermscycle'])[ID_all80])]]) 

217 phaseRMScycleP80mad = self.mad(np.array(self.allResult['blphasermscycle'])[ID_all80[np.isfinite(np.array(self.allResult['blphasermscycle'])[ID_all80])]]) 

218 

219 # Begin the outlier checks 

220 # first check if any antennas are just above 100 deg phase RMS (this means ~pure phase noise for phases between -180 to +180 deg) 

221 # so sensible to identify these antennas - over total time 

222 ID_poorant = np.where(np.array(self.allResult['antphaserms'])[np.isfinite(self.allResult['antphaserms'])]>self.outlierlimit)[0] 

223 

224 if len(ID_poorant) > 0: 

225 for antout in ID_poorant: 

226 self.outlier_antennas.append(np.array(self.allResult['antname'])[np.isfinite(self.allResult['antphaserms'])][antout]) 

227 

228 # for what are yellow or red scores where phase RMS >50deg 

229 # we check for statistical outliers, and then clip these out and re-calculate the phaseRMS 

230 # in the case PL missed bad antennas were causing the high phase RMS 

231 # self.outlier_antennas is used/passed in the score function as it adjusts the score and message too 

232 if self.allResult['phasermscycleP80'] > 50.: 

233 statsoutlierlimit = self.allResult['phasermscycleP80'] + 4.*phaseRMScycleP80mad # tested limit works well 

234 # ant phase on cycle time ! 

235 ID_poorant = np.where(np.array(self.allResult['antphasermscycle'])[np.isfinite(self.allResult['antphasermscycle'])] > statsoutlierlimit)[0] 

236 if len(ID_poorant)>0: 

237 for antout in ID_poorant: 

238 self.outlier_antennas.append(np.array(self.allResult['antname'])[np.isfinite(self.allResult['antphaserms'])][antout]) 

239 

240 # max limit on number of 'bad' antennas to be allowed to exclude from calculations (set at 11 good as tested in PIPE692) 

241 # we clip them out and recalculate - score function also tracks and gives a warning 

242 # if >11, basically the data is rubbish, so don't clip and let scores be low 

243 if len(self.outlier_antennas) > 0 and len(self.outlier_antennas) < self.maxpoorant: 

244 

245 # crude way to get the index values is a loop over the baselines 

246 # is there a way to better search allResult['blname'] to check if any self.outlier_antennas are (or are not) in them 

247 ID_goodbl=[] 

248 ID_badbl=[] 

249 for idg, bln in enumerate(self.allResult['blname']): 

250 if (bln.split('-')[0] not in self.outlier_antennas) and (bln.split('-')[1] not in self.outlier_antennas): 

251 ID_goodbl.append(idg) 

252 else: # assume bad 

253 ID_badbl.append(idg) 

254 

255 # now simply make a common list of IDs from good ones and the ID_all80 list - which are want we want to assess 

256 ID_all80 = np.array(np.sort(np.array(list(set(ID_goodbl).intersection(ID_all80))))) 

257 

258 # recalculate the phase RMS that is bl >P80 and in the 'good' list 

259 self.allResult['phasermsP80'] = np.median(np.array(self.allResult['blphaserms'])[ID_all80[np.isfinite(np.array(self.allResult['blphaserms'])[ID_all80])]]) 

260 self.allResult['phasermscycleP80'] = np.median(np.array(self.allResult['blphasermscycle'])[ID_all80[np.isfinite(np.array(self.allResult['blphasermscycle'])[ID_all80])]]) 

261 

262 # store outliers for passing to plot function - will be plotted in 'shade/alpha' 

263 self.allResult['blphasermsbad'] = np.array(self.allResult['blphaserms'])[np.array(ID_badbl)] 

264 self.allResult['blphasermscyclebad'] = np.array(self.allResult['blphasermscycle'])[np.array(ID_badbl)] 

265 self.allResult['bllenbad'] = np.array(self.allResult['bllen'])[np.array(ID_badbl)] 

266 else: 

267 # none the 'bad' entires in dict  

268 self.allResult['blphasermsbad'] = None 

269 self.allResult['blphasermscyclebad'] = None 

270 self.allResult['bllenbad'] = None 

271 else: 

272 # this else is for <50deg phase RMS where we do not recalcualte the phase RMS as its low already 

273 # but we still want to identify any outliers to notify in the messages 

274 statsoutlierlimit = np.max([self.allResult['phasermscycleP80'] + 6.0 * phaseRMScycleP80mad, 2.0 * self.allResult['phasermscycleP80']]) 

275 # outlier on cycle time  

276 ID_poorant = np.where(np.array(self.allResult['antphasermscycle'])[np.isfinite(self.allResult['antphasermscycle'])]>statsoutlierlimit)[0] 

277 # add them to the list so score code picks them up if required and changes the messages 

278 if len(ID_poorant)>0: 

279 for antout in ID_poorant: 

280 self.outlier_antennas.append(np.array(self.allResult['antname'])[np.isfinite(self.allResult['antphaserms'])][antout]) 

281 

282 # none the 'bad' entires in dict  

283 self.allResult['blphasermsbad'] = None 

284 self.allResult['blphasermscyclebad'] = None 

285 self.allResult['bllenbad'] = None 

286 

287 # now loop over and log write the phase RMS parameters 

288 # sort by baseline len 

289 LOG.info(' Phase RMS calculated over the cycle time as function of baseline length') 

290 for bnblph in list(zip(np.array(self.allResult['blname'])[np.array(self.allResult['bllen']).argsort()],np.array(self.allResult['bllen'])[np.array(self.allResult['bllen']).argsort()], 

291 np.array(self.allResult['blphasermscycle'])[np.array(self.allResult['bllen']).argsort()])): 

292 

293 # add simple text to flagged (i.e. we made nan for the calculation), or outlier antennas (outlier comes first) 

294 if str(bnblph[0].split('-')[0]) in self.outlier_antennas or str(bnblph[0].split('-')[1]) in self.outlier_antennas: 

295 LOG.info(str(bnblph)+' - outlier') 

296 elif not np.isfinite(bnblph[2]): 

297 LOG.info(str(bnblph)+' - flagged') 

298 else: 

299 LOG.info(str(bnblph)) 

300 

301 if len(self.outlier_antennas) > 0: 

302 antoutstr = ",".join(self.outlier_antennas) 

303 LOG.info(" Possible high phase RMS on antenna(s): {}".format(antoutstr)) 

304 

305 # Save off spw, scan, field 

306 self.allResult['spw'] = self.spw 

307 self.allResult['scan'] = self.scan 

308 self.allResult['field'] = self.field

331 def _getbaselinesproj(self, field_id: Optional[int] = None) -> Dict[str, float]: 

332 """ 

333 Code to get the projected baseline from the openend  

334 visibilitiy file already - these are ordered in  

335 terms of antennas. This is a modified stand alone version 

336 similar to the getProjectedBaselines from Todd Hunter's AUs. 

337 

338 returns a dict of lengths which are BL name keyed 

339 e.g. bllens[DA41-DA42] - key is name ant 1 - dash - name ant 2 

340 """ 

341 with casa_tools.MSMDReader(self.vis) as msmd: 

342 spwchan = msmd.nchan(self.spw) 

343 datadescid = msmd.datadescids(spw=self.spw)[0] 

344 

345 with casa_tools.MSReader(self.vis) as ms: 

346 ms.selectinit(datadescid=datadescid) 

347 ms.select({'uvdist': [1e-9, 1e12]}) # avoid auto corr 

348 ms.selectchannel(1, 0, spwchan, 1) # data structure related 

349 ms.select({'scan_number': int(self.scan)}) 

350 if field_id: 

351 ms.select({'field_id': int(field_id)}) 

352 alldata = ms.getdata(['uvdist', 'antenna1', 'antenna2']) 

353 

354 # The length of e.g. alldata['uvdist'] is > total no. of Bls - it loops over all time stamps of the BP  

355 # we need a mean of the unique values (as Todd's aU) otherwise we just get the first time entry in the below 

356 bldict = {} 

357 uniBl = [] 

358 baselineLen = {} 

359 for allID in range(len(alldata['uvdist'])): 

360 myBl = '%s-%s' % (self.antlist[alldata['antenna1'][allID]], self.antlist[alldata['antenna2'][allID]]) 

361 thelen = alldata['uvdist'][allID] 

362 if myBl not in bldict: 

363 bldict[myBl] = [] 

364 bldict[myBl].append(thelen) 

365 uniBl.append(myBl) 

366 

367 uniBl = np.unique(uniBl) 

368 

369 for myBl in uniBl: 

370 baselineLen[myBl] = np.mean(bldict[myBl]) # this has a list for each 

371 

372 # order irrelavant as keyed here with BL Name 

373 return baselineLen

388 def _get_bandpass_scan_time(self) -> Tuple[float, float]: 

389 """ 

390 Read a caltable file and return time 

391 shifted to start at zero seconds starting time. 

392 Note these are the recored times in the caltable 

393 

394 uses inputs: 

395 self.caltable, self.spw, self.scan, self.antlist 

396  

397 :returns: total time of baseline scan, average integration time  

398 :rtype: float, float 

399 """ 

400 with casa_tools.TableReader(self.caltable) as tb: 

401 nant = len(self.antlist) 

402 timeX = [] 

403 antid = 0 

404 while len(timeX) < 2: 

405 tb1 = tb.query("ANTENNA1 == %s && SPECTRAL_WINDOW_ID == %s && SCAN_NUMBER == %s" % (antid, self.spw, self.scan)) 

406 timeX = tb1.getcol('TIME') 

407 antid += 1 

408 if antid == nant - 1: 

409 break 

410 tb1.close() 

411 

412 # If there is no time found (should never happen), throw. 

413 if len(timeX) < 2: 

414 raise Exception("No times found in bandpass caltable: {}".format(self.caltable)) 

415 

416 timeX = np.array(timeX) - timeX[0] 

417 total_bp_scan_time = timeX[-1] 

418 avg_integration_time = np.median(np.diff(timeX)) 

419 

420 return total_bp_scan_time, avg_integration_time

422 def _getcycletime(self) -> float: 

423 """ 

424 Computes the median time (in seconds) between visits to the specified intent. 

425 Note that other parts of the ALMA project consider the "cycleTime" to be the  

426 scan duration on the science target before going back to the phase calibrator, 

427 i.e. ignoring the duration of the phasecal scan, the ATM cal scans, the  

428 checksource, and all the slewing and overhead. 

429 

430 If the cycle time is not found, diverts to a lookup table (See: PIPE-1848) 

431  

432 This method is adapted from: 

433 - Todd Hunter ORIG in analysis Utils (cycleTime) 

434 - LM edited for this code 

435 

436 input used: 

437 self.vis 

438 

439 return: the cycletime (float) 

440 """ 

441 with casa_tools.MSMDReader(self.vis) as msmd: 

442 scans = msmd.scansforintent('*PHASE*') 

443 

444 # PIPE-1848: if there is no phase calibrator, we cannot get cycle time either 

445 if len(scans) == 0: 

446 # No phase calibrator in the data 

447 LOG.warning("Using lookup Cycle Time as there is no PHASE intent in {}".format(self.vis)) 

448 usecycle = self._lookupcycle() 

449 return usecycle 

450 

451 # Use the lookup if there is only one phase calibrator scan. See: PIPE-1848 

452 if len(scans) == 1: 

453 LOG.warning("Using lookup Cycle Time as there is only 1 PHASE calibrator scan for {}".format(self.vis)) 

454 usecycle = self._lookupcycle() 

455 return usecycle 

456 

457 # all correctly formed data should go here 

458 # now get the times for the scans and work out the cycle time  

459 times = [] 

460 

461 for scan in scans: 

462 times.append(np.min(msmd.timesforscan(scan))) 

463 

464 if len(times) == 1: 

465 LOG.warning("There was only 1 scan with this intent.") 

466 

467 diffs = np.diff(times) 

468 return np.median(diffs)

470 def _lookupcycle(self) -> float: 

471 """ 

472 Look up the nearest default cycle time 

473 using best practices for Baseline length and 

474 the frequency band. This is only needed when the 

475 returned cycle time is None, which only happens 

476 for malformed data with only one PHASE cal scan, 

477 that ultimately should not be coming to pipeline at all. 

478 

479 See: PIPE-1848 

480 """ 

481 if self.PMinACA: 

482 config = 0 

483 else: 

484 config = self._getconfig() # Configs run 1 to 10, 0 is ACA

621 def _get_cal_phase(self, ant: int) -> np.ndarray: 

622 """ 

623 Read a caltable file and select the 

624 phases from one pol (tested in PIPE692 as sufficient). 

625 If those phase data have a flag, the phase 

626 value is set to nan, which is dealt with in the correct 

627 way during the rest of the calulation. The phases  

628 are unwrapped, i.e. solved for the 2PI ambiguitiy  

629 

630 input required: 

631 ant (int) - the antenna to get the phases for  

632 

633 uses inputs: 

634 self.caltable, self.refantid, self.spw, self.scan 

635  

636 

637 returns: float array of the phases of an antenna 

638 """ 

639 with casa_tools.TableReader(self.caltable) as tb: 

640 tb1 = tb.query("ANTENNA1 == %s && ANTENNA2 == %s && SPECTRAL_WINDOW_ID == %s && SCAN_NUMBER == %s "%(ant, self.refantid, self.spw, self.scan)) 

641 cal_phases = tb1.getcol('CPARAM') 

642 cal_phases = np.angle(cal_phases[0][0]) ## in radians one pol only 

643 

644 # code works fine as tested in PIPE 692 with only XX pol extractions 

645 # not 'really' required to do both pols, but could do so, and also get an average 

646 # single and full-pol to check, this single pull will work, but with more then the order differs (c.f. renorm pol code) 

647 

648 # Exclude flagged data as it is extracted from the gaintable 

649 # this is antenna based only  

650 flags = tb1.getcol('FLAG') # [0][0] 

651 tb1.close() 

652 

653 cal_phases = [val if not flags[0][0][id_u] else np.nan for id_u, val in enumerate(cal_phases)] # so only one pol used 'X' 

654 cal_phases = np.array(cal_phases) 

655 

656 # Correct wraps in phase stream 

657 # Note: everything is in radians 

658 cal_phases = self.phase_unwrap(cal_phases) 

659 return cal_phases

661 def _phase_rms_caltab(self, antout: list=[], timeScale: float=None) -> Dict[str, str]: 

662 """ 

663 Run the loop over the caltable, work out the baseline based phases, 

664 and calculate the phase RMS. Also get the Phase RMS per antenna (with 

665 respect to the refant - i.e. ant based phase RMS). 

666  

667 inputs used: 

668 self.antlist, self.flag_tolerance, self.difftime, self.baselines 

669 

670 calls functions: 

671 self._get_cal_phase, self.ave_phase, self.std_overlapping_avg 

672  

673 returns: rms_results{} 

674 dict keys: blphaserms, bphasermscycle, bllen, blname, 

675 antphaserms, antphasermscycle, antname 

676 """ 

677 

678 # setup the result list 

679 rms_results = {} 

680 rms_results['blphaserms'] = [] 

681 rms_results['blphasermscycle'] = [] 

682 rms_results['bllen'] = [] 

683 rms_results['blname'] = [] 

684 rms_results['antphaserms'] = [] 

685 rms_results['antphasermscycle'] = [] 

686 rms_results['antname'] = [] 

687 

688 # PIPE-1661 baseline based flag awareness - these store flags  

689 # that we read in - just in case we 'need' this information 

690 rms_results['blflags'] = [] 

691 

692 # Get phase for the reference antenna. 

693 pHrefant = self._get_cal_phase(self.refantid) 

694 

695 # Number of antennas. 

696 nant = len(self.antlist) 

697 

698 # Loop over every antenna ID. 

699 iloop = np.arange(nant-1) 

700 for i in iloop: 

701 # Ant based parameters 

702 pHant1 = self._get_cal_phase(i) 

703 rms_results['antname'].append(self.antlist[i]) 

704 

705 # Subtract the phase for reference antenna from the phase for the 

706 # current antenna. Because phases can be non-zero, use 

707 # pHant1- pHrefant (or) pHrefant - pHant1, depending on which is 

708 # smaller antenna index. 

709 if self.refantid <= i: 

710 pHantalone = pHrefant - pHant1 

711 else: 

712 pHantalone = pHant1 - pHrefant 

713 

714 # Make an assessment of flagged data for that antenna 

715 if len(pHantalone[np.isnan(pHantalone)]) > self.flag_tolerance*len(pHantalone): 

716 rms_results['antphaserms'].append(np.nan) 

717 rms_results['antphasermscycle'].append(np.nan) 

718 

719 else: 

720 # Perform averaging over 10s, but work on the corrected w.r.t. 

721 # refant. This should be zero but in case there is a jump 

722 # (i.e. refant change), then this should now correctly have 

723 # taken this out. 

724 pHant_ave = self.ave_phase(pHantalone, self.difftime, over=10.0) 

725 rmspHant_ave = np.std(np.array(pHant_ave)[np.isfinite(pHant_ave)]) 

726 rms_results['antphaserms'].append(rmspHant_ave) 

727 if timeScale: 

728 rmspHant_ave_cycle = self.std_overlapping_avg(pHant_ave, self.difftime, over=timeScale) 

729 rms_results['antphasermscycle'].append(rmspHant_ave_cycle) 

730 else: 

731 rms_results['antphasermscycle'].append(rmspHant_ave) 

732 

733 jloop = np.arange(i+1, nant) # baseline loop 

734 for j in jloop: 

735 # Get the phases - needs to be the single read in table  

736 

737 # pHant1 is read in above already 

738 pHant2 = self._get_cal_phase(j) 

739 # phases from cal table come in an order, baseline then is simply the subtraction 

740 pH = pHant1 - pHant2 

741 

742 # fill baseline information now 

743 rms_results['blname'].append(self.antlist[i]+'-'+self.antlist[j]) 

744 # OLD (new) WAY from context - average all as overview 

745 # not direcrtly the BP only 

746 # rms_results['bllen'].append(float(self.baselines.get_baseline(i,j).length.value)) # from context input 

747 

748 rms_results['bllen'].append(float(self.baselines[self.antlist[i]+'-'+self.antlist[j]])) # from function 

749 

750 # PIPE-1661 - get baseline based flags 

751 blisflagged = self._isblflagged(i, j) 

752 

753 rms_results['blflags'].append(blisflagged) 

754 ###### 

755 

756 # make assessment if this is a bad antenna 

757 if self.antlist[i] in antout or self.antlist[j] in antout: 

758 rms_results['blphaserms'].append(np.nan) 

759 rms_results['blphasermscycle'].append(np.nan) 

760 

761 # PIPE-1661 assessment (could tie with if above but separated for clarity 

762 elif blisflagged: # can only be T/F bool 

763 rms_results['blphaserms'].append(np.nan) 

764 rms_results['blphasermscycle'].append(np.nan) 

765 ############### 

766 

767 # make an assessment of flagged data 

768 elif len(pH[np.isnan(pH)]) > self.flag_tolerance*len(pH): 

769 rms_results['blphaserms'].append(np.nan) 

770 rms_results['blphasermscycle'].append(np.nan) 

771 

772 else: 

773 # do averaing -> 10s 

774 # need to get time and then rebin 

775 pH_ave = self.ave_phase(pH, self.difftime, over=10.0) # for thermal/short term noise 

776 rmspH_ave = np.std(np.array(pH_ave)[np.isfinite(pH_ave)]) 

777 rms_results['blphaserms'].append(rmspH_ave) 

778 if timeScale: 

779 rmspH_ave_cycle = self.std_overlapping_avg(pH_ave, self.difftime, over=timeScale) 

780 rms_results['blphasermscycle'].append(rmspH_ave_cycle) 

781 else: 

782 rms_results['blphasermscycle'].append(rmspH_ave) 

783 

784 # set RMS output in degrees as we want 

785 for key_res in ['blphaserms', 'blphasermscycle', 'antphaserms', 'antphasermscycle']: 

786 rms_results[key_res] = np.degrees(rms_results[key_res]) 

787 

788 return rms_results

790 def _isblflagged(self, ant1: int, ant2: int): 

791 """ 

792 function to make the assessment of  

793 the flags that are saved and return if the 

794 full baseline is flagged or not 

795  

796 requires the self.blflags 

797 """ 

798 

799 # in checking the order of data read associated with each antenna 

800 # the ordering system is not 100% preordained it seems, 

801 # i.e. we have to select each time the index of bls in agreement with  

802 # the antenna pair and antenna list - then check the flags 

803 

804 # simply need to select the correct cut for the baseline in question 

805 # start with the bandpass, if totally flagged assume all data flagged  

806 # for that baseline 

807 

808 # speed assumption is we will hit a false before a true 

809 # flag for good data, so assume bad, set to good 

810 flaggedbl = True 

811 

812 idbl = np.where((self.blflags[1]==ant1) & (self.blflags[2]==ant2) & (self.blflags[3]==self.fieldId))[0] 

813 # loop over correct ant, integration time index  

814 for iduse in idbl: 

815 

816 if self.blflags[0].shape[0]*self.blflags[0].shape[1]!= np.sum(self.blflags[0][:,:,iduse]): 

817 flaggedbl = False 

818 break 

819 

820 # else here for Phase cal check if not flagged in BP 

821 if not flaggedbl: 

822 flaggedbl = True 

823 for phid in self.ph_ids: # usually one phase cal anyway but loop incase multiple 

824 idbl = np.where((self.blflagsref[1] == ant1) & (self.blflagsref[2] == ant2) & (self.blflagsref[3] == phid))[0] 

825 for iduse in idbl: 

826 if self.blflagsref[0].shape[0] * self.blflagsref[0].shape[1] != np.sum(self.blflagsref[0][:, :, iduse]): 

827 # if there is unflagged data in this antenna pair and time then 

828 # that baseline is, in fact, not fully flagged 

829 flaggedbl = False 

830 break 

831 

832 return flaggedbl

834 def _get_final_spw_and_blflags(self, inputsin, qa_spw_candidates) \ 

835 -> Tuple[int, Tuple[np.ndarray, np.ndarray, np.ndarray, np.ndarray]]: 

836 """ 

837 Select the best candidate SpW for the phase decoherence analysis based 

838 on ranked list and baseline flagging information. 

839 

840 Args: 

841 inputsin: inputs from caller task (expected to be hifa_spwphaseup) 

842 

843 Returns: 

844 2-Tuple containing: 

845 - selected SpW ID (integer) 

846 - 4-tuple representing baseline flagging info for selected SpW 

847 """ 

848 # PIPE-1871: from the ranked list of SpWs, pick the first SpW for which 

849 # the baselines are not fully flagged.

851 for qa_spw in qa_spw_candidates:

959 @staticmethod 

960 def std_overlapping_avg(phase: np.ndarray, diffTime: float, over: float=120.0) -> float: 

961 """ 

962 Calculate STD over a set time and return the average of all overlapping 

963 values of the standard deviation - overlapping estimator. This acts 

964 upon a phase-time stream of data. Will consider only finite values in  

965 the phase-time stream (i.e. phase array). The phase array must be  

966 unwrapped, i.e. continous, no breaks or 2PI ambiguities 

967  

968 This is the standard deviation, so it takes out the 

969 mean value. RMS with mean or fit removed provides 

970 the same value for zero-centered phases. 

971 

972 :param phase: any unwrapped input phase 

973 :type phase: array 

974 :param diffTime: the time between each data integration 

975 :type diffTime: float 

976 :param over: time in seconds to calculate the SD over 

977 :type over: float 

978 :returns: average standard deviation for the dataset calcualted over the input timescale  

979 :rtype: float 

980 """ 

981 

982 # Overlap in elements 

983 over = int(np.round(over / diffTime)) 

984 

985 # Use nan versions of numpy functions as some elements of 'phase' might be np.nan to indicate that it was flagged 

986 std_hold = [] 

987 

988 # Loop over the data in range rounded to size of time step 

989 for i in range(len(phase) - over): 

990 std_hold.append( 

991 np.nanstd(np.array(phase[i:i+over])) 

992 ) 

993 

994 std_mean = np.nanmean(std_hold) 

995 

996 return std_mean

999 def ave_phase(phase: np.ndarray, diffTime: np.ndarray, over: float=1.0) -> np.ndarray: 

1000 """ 

1001 Do an averaging/smoopthing on the phase data 

1002 The default for ALMA for phase statistics should be 10s. 

1003 For default 6s integration times this will average 2 values. 

1004 If input phases from the gain table have integration(diffTime) > 10s 

1005 then no averaging is made. 

1006 

1007 :param phase: phase series of the data to average 

1008 :type phase: array 

1009 :param diffTime: the average difference in time between each data value, i.e. each phase 

1010 :type diffTime: float
