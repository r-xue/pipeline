76 parang=None, applymode=None, flagbackup=None, flagsum=None, flagdetailedsum=None, 

77 parallel=None): 

78 """Initialize Inputs. 

79 

80 Args: 

81 context: Pipeline context object containing state information. 

82 

83 output_dir: Output directory. 

84 Defaults to None, which corresponds to the current working directory. 

85 

86 vis: The list of input MeasurementSets. Defaults to the list of MeasurementSets in the pipeline context. 

87 example: ``['X227.ms']`` 

88 

89 field: A string containing the list of field names or field ids to which the calibration will be applied. Defaults to all fields in the pipeline 

90 context. 

91 example: ``'3C279'``, ``'3C279,M82'`` 

92 

93 spw: The list of spectral windows and channels to which the calibration will be applied. Defaults to all science windows in the pipeline 

94 context. 

95 example: ``'17'``, ``'11,15'`` 

96 

97 antenna: The selection of antennas to which the calibration will be applied. Defaults to all antennas. Not currently supported. 

98 

99 intent: A string containing the list of intents against which the selected fields will be matched. Defaults to all supported intents 

100 in the pipeline context. 

101 example: ``'*TARGET*'`` 

102 

103 parang: Apply parallactic angle correction 

104 

105 applymode: Calibration apply mode 

106 

107 - 'calflag': calibrate data and apply flags from solutions 

108 - 'calflagstrict': (default) same as above except flag spws for which calibration is 

109 unavailable in one or more tables (instead of allowing them to pass 

110 uncalibrated and unflagged) 

111 - 'trial': report on flags from solutions, dataset entirely unchanged 

112 - 'flagonly': apply flags from solutions only, data not calibrated 

113 - 'flagonlystrict': same as above except flag spws for which calibration is 

114 unavailable in one or more tables 

115 - 'calonly': calibrate data only, flags from solutions NOT applied 

116 

117 flagbackup: Backup the flags before the apply 

118 

119 flagsum: Compute before and after flagging summary statistics 

120 

121 flagdetailedsum: Compute detailed before and after flagging statistics summaries. Parameter available only when if flagsum is ``True``. 

122 

123 parallel: Process multiple MeasurementSets in parallel using the casampi parallelization framework. 

124 

125 Options: ``'automatic'``, ``'true'``, ``'false'``, ``True``, ``False`` 

126 

127 Default: ``None`` (equivalent to ``False``) 

128 

129 """

133 self.context = context 

134 # vis must be set first, as other properties may depend on it 

135 self.vis = vis 

136 self.output_dir = output_dir

188 def merge_with_context(self, context): 

189 """ 

190 Merges these results with the given context by examining the context 

191 and marking any applied caltables, so removing them from subsequent 

192 on-the-fly calibration calculations. 

193 

194 See :method:`~pipeline.Results.merge_with_context` 

195 """ 

196 if not self.applied: 

197 LOG.error('No results to merge') 

198 

199 for calapp in self.applied: 

200 LOG.trace('Marking %s as applied' % calapp.as_applycal()) 

201 context.callibrary.mark_as_applied(calapp.calto, calapp.calfrom) 

202 

203 # Update data_column 

204 if self.data_type is not None: 

205 msobj = context.observing_run.get_ms(self.inputs['vis']) 

206 colname = 'CORRECTED_DATA' 

207 # Temporal workaround: restoredata merges context twice 

208 if msobj.get_data_column(self.data_type) != colname: 

209 msobj.set_data_column(self.data_type, colname)

223 return s

253 def prepare(self): 

254 inputs = self.inputs 

255 

256 # Get the target data selection for this task as a CalTo object 

257 calto = callibrary.get_calto_from_inputs(inputs) 

258 

259 # Now get the calibration state for that CalTo data selection. The 

260 # returned dictionary of CalTo:CalFroms specifies the calibrations to 

261 # be applied and the data selection to apply them to. While the CASA 

262 # callibrary could use the full, untrimmed calibration state for the 

263 # whole MS, we need the trimmed version for 1) the web log, where it 

264 # is used to state what specific calibrations were applied, and 2), 

265 # the pipeline callibrary, where we mark the target data selection as 

266 # having calibration applied. 

267 # 

268 # Note that no 'ignore' argument is given to get_calstate 

269 # (specifically, we don't say ignore=['calwt'] like many other tasks) 

270 # as applycal is a task that can handle calwt and so different values 

271 # of calwt should in this case result in different tasks. 

272 calstate = inputs.context.callibrary.get_calstate(calto) 

273 calstate = callibrary.fix_cycle0_data_selection(self.inputs.context, calstate) 

274 

275 # The 'hide_empty=True' is important here. The working calstate 

276 # contains empty state for MSes outside the scope of this task, i.e., 

277 # MSes in the context that are not specified in inputs.vis for this 

278 # task. These extra MSes cause problems in weblog code downstream, 

279 # which expects results.applied to refer to just the MSes specified in 

280 # the inputs. Using hide_empty gives the data the expected shape. 

281 merged = calstate.merged(hide_empty=True) 

282 

283 if os.getenv('DISABLE_CASA_CALLIBRARY', False): 

284 LOG.info('CASA callibrary disabled: reverting to non-callibrary applycal call') 

285 jobs = jobs_without_calapply(merged, inputs, self.modify_task_args) 

286 elif contains_uvcont_table(merged): 

287 LOG.info('Calibration state contains uvcont tables: reverting to non-callibrary applycal call') 

288 jobs = jobs_without_calapply(merged, inputs, self.modify_task_args) 

289 else: 

290 LOG.info('No uvcont tables in calibration state: using CASA callibrary applycal.') 

291 jobs = jobs_with_calapply(calstate, inputs, self.modify_task_args) 

292 

293 # if requested, schedule additional flagging tasks to determine 

294 # statistics 

295 if inputs.flagsum: 

296 summary_args = dict(vis=inputs.vis, mode='summary') 

297 # give subclasses a chance to tweak flag summary arguments 

298 summary_args = self._get_flagsum_arg(summary_args) 

299 # schedule a flagdata summary jobs either side of the applycal jobs 

300 jobs.insert(0, casa_tasks.flagdata(name='before', **summary_args)) 

301 jobs.append(casa_tasks.flagdata(name='applycal', **summary_args)) 

302 

303 if inputs.flagdetailedsum: 

304 ms = inputs.context.observing_run.get_ms(inputs.vis) 

305 # Schedule a flagdata job to determine flagging stats per spw 

306 # and per field 

307 flagkwargs = ["spw='{!s}' fieldcnt=True mode='summary' name='AntSpw{:0>3}'".format(spw.id, spw.id) 

308 for spw in ms.get_spectral_windows()] 

309 

310 # give subclasses a change to tweak flag arguments 

311 flagkwargs = self._tweak_flagkwargs(flagkwargs) 

312 

313 jobs.append(casa_tasks.flagdata(vis=inputs.vis, mode='list', inpfile=flagkwargs, flagbackup=False)) 

314 

315 # execute the jobs and capture the output 

316 job_results = [self._executor.execute(job) for job in jobs] 

317 flagdata_results = [job_result for job, job_result in zip(jobs, job_results) if job.fn_name == 'flagdata'] 

318 

319 applied_calapps = [callibrary.CalApplication(calto, calfroms) for calto, calfroms in merged.items()] 

320 

321 # give a dict like {'abc123.ms': 'path/to/callibrary'}. The use of 

322 # dict assumes that there is only one jobrequest per MS, which is true 

323 # when the CASA callibrary is used. 

324 vis_to_callib = {job.kw['vis']: job.kw['callib'] for job in jobs 

325 if job.fn_name == 'applycal' and 'callib' in job.kw} 

326 

327 result = ApplycalResults(applied_calapps, callib_map=vis_to_callib, 

328 data_type=self.applied_data_type) 

329 

330 # add and reshape the flagdata results if required 

331 if inputs.flagsum: 

332 result.summaries = [flagdata_results[0], flagdata_results[1]] 

333 if inputs.flagdetailedsum: 

334 reshaped_flagsummary = reshape_flagdata_summary(flagdata_results[2]) 

335 processed_flagsummary = self.process_flagsummary(reshaped_flagsummary) 

336 result.flagsummary = processed_flagsummary 

337 

338 return result

355def reshape_flagdata_summary(flagdata_result): 

356 """ 

357 Reshape a flagdata result so that results are grouped by field. 

358 

359 :param flagdata_result: 

360 :return: 

361 """ 

362 # Set into single dictionary report (single spw) if only one dict returned 

363 if not all([key.startswith('report') for key in flagdata_result]): 

364 flagdata_result = {'report0': flagdata_result} 

365 

366 flagsummary = collections.defaultdict(dict) 

367 for report_level, report in flagdata_result.items(): 

368 report_name = report['name'] 

369 report_type = report['type'] 

370 # report keys are all fieldnames with the exception of 'name' and 

371 # 'type', which are in there too. 

372 for field_name in [key for key in report if key not in ('name', 'type')]: 

373 # deepcopy to avoid modifying the results dict 

374 flagsummary[field_name][report_level] = copy.deepcopy(report[field_name]) 

375 flagsummary[field_name][report_level]['name'] = '{!s}Field_{!s}'.format(report_name, field_name) 

376 flagsummary[field_name][report_level]['type'] = report_type 

377 

378 return flagsummary

400def jobs_without_calapply(merged, inputs, mod_fn): 

401 jobs = [] 

402 # sort for a stable applycal order, to make diffs easier to parse 

403 for calto, calfroms in sorted(merged.items()): 

404 # if there's nothing to apply for this data selection, continue. This 

405 # should never be seen as merged is called with hide_empty=True 

406 if not calfroms: 

407 LOG.info('There is no calibration information for field %s intent %s spw %s in %s' % 

408 (str(calto.field), str(calto.intent), str(calto.spw), inputs.ms.basename)) 

409 continue 

410 

411 # arrange a calibration job for the unique data selection 

412 inputs.spw = calto.spw 

413 inputs.field = calto.field 

414 inputs.intent = calto.intent 

415 

416 task_args = inputs.to_casa_args() 

417 

418 # set the on-the-fly calibration state for the data selection. 

419 calapp = callibrary.CalApplication(calto, calfroms) 

420 task_args['gaintable'] = calapp.gaintable 

421 task_args['gainfield'] = calapp.gainfield 

422 task_args['spwmap'] = calapp.spwmap 

423 task_args['interp'] = calapp.interp 

424 task_args['calwt'] = calapp.calwt 

425 task_args['applymode'] = inputs.applymode 

426 

427 # give subclasses a chance to modify the task arguments 

428 task_args = mod_fn(task_args) 

429 

430 jobs.append(casa_tasks.applycal(**task_args)) 

431 

432 return jobs

435def jobs_with_calapply(calstate: IntervalCalState, inputs: ApplycalInputs, mod_fn: Callable): 

436 callibrary_file = '{}.s{}.{}.callibrary'.format(inputs.vis, 

437 inputs.context.task_counter, 

438 inputs.context.subtask_counter) 

439 ms = inputs.context.observing_run.get_ms(inputs.vis) 

440 calstate.export_to_casa_callibrary(ms, callibrary_file) 

441 

442 # No callibrary file will be created when the merged calstate does not 

443 # require the application of calibrations. 

444 if not os.path.exists(callibrary_file): 

445 LOG.info('No applycal job required for CASA callibrary: {}'.format(callibrary_file)) 

446 return [] 

447 

448 calstate_file = '{}.s{}.{}.calstate'.format(inputs.vis, 

449 inputs.context.task_counter, 

450 inputs.context.subtask_counter) 

451 with open(calstate_file, "w") as applyfile: 

452 applyfile.write('# Apply file for %s\n' % (os.path.basename(inputs.vis))) 

453 applyfile.write(calstate.as_applycal()) 

454 

455 task_args = inputs.to_casa_args() 

456 

457 # Don't delete spw, field, or intents as the inputs may request 

458 # calibration of a subset of the total MS. The CASA callibrary can 

459 # still define calibration for the whole MS, that's not a problem. 

460 for a in ['gaintable', 'gainfield', 'spwmap', 'interp', 'calwt']: 

461 if a in task_args: 

462 del task_args[a] 

463 

464 task_args['applymode'] = inputs.applymode 

465 task_args['docallib'] = True 

466 task_args['callib'] = callibrary_file 

467 

468 mod_fn(task_args) 

469 

470 return [casa_tasks.applycal(**task_args)]
