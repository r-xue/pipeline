54 def _get_tsys_map(self, context, ms): 

55 # Get the Tsys spw map by retrieving it from the first tsys CalFrom 

56 # that is present in the callibrary 

57 try: 

58 spwmap = utils.get_calfroms(context, ms.name, "tsys")[0].spwmap 

59 except IndexError: 

60 # proceed with 1:1 mapping 

61 spwmap = list(range(len(ms.spectral_windows))) 

62 

63 # we're only concerned about science spws 

64 science_spws = ms.get_spectral_windows(science_windows_only=True) 

65 science_spw_ids = [spw.id for spw in science_spws] 

66 

67 # contruct a dict mapping Tsys spw to science spws 

68 tsys_map = {} 

69 for tsys_spw in spwmap: 

70 science_spws = { 

71 science_id 

72 for science_id, mapped_tsys in enumerate(spwmap) 

73 if tsys_spw == mapped_tsys and science_id in science_spw_ids 

74 } 

75 tsys_map[tsys_spw] = sorted(science_spws) 

76 

77 return tsys_map

79 def post_process_plots(self, pcontext: Context, spw_map: Dict[int, List[int]], plots: List[Plot]) -> List[Plot]: 

80 """ 

81 Transform the 'foreign' plot wrappers coming from extern code into 

82 native wrappers more aligned with the pipeline and standard pipeline 

83 presentation. 

84 

85 We want to make as few modifications as possible to the extern 

86 contribution, keeping it free of pipeline context, domain objects, 

87 etc. until it can be properly adopted and refactored. This function 

88 modifies the one concession we did make for the pipeline, having the 

89 extern code return plot wrappers, to return plot wrappers as we'd 

90 expect from pipeline-native code. 

91 """ 

92 updated = copy.deepcopy(plots) 

93 

94 for plot in updated: 

95 ms = pcontext.observing_run.get_ms(plot.parameters["vis"]) 

96 # TsysDataClassFile recognises the intents 'bandpass', 'phasecal' and 'science' 

97 

98 # extern code sets: 

99 # - field ID to an int, can be lookup up in MS 

100 orig_field = plot.parameters["field"] 

101 field_list = ms.get_fields(field_id=orig_field) 

102 plot.parameters["field"] = ( 

103 field_list[0].name if field_list else str(orig_field) 

104 ) 

105 

106 # extern code sets: 

107 # - spw, from which we can identify Tsys spw and science spw 

108 tsys_spw = plot.parameters["tsys_spw"] 

109 plot.parameters["spw"] = spw_map[tsys_spw] 

110 

111 # extern code sets: 

112 # - intent to a string, can be mapped to a pipeline intent 

113 extern_to_pipeline_intent_map = { 

114 "bandpass": "BANDPASS", 

115 "phasecal": "PHASE", 

116 "science": "TARGET", 

117 } 

118 orig_intent = plot.parameters["intent"] 

119 plot.parameters["intent"] = extern_to_pipeline_intent_map.get( 

120 orig_intent, orig_intent 

121 ) 

122 

123 return updated
