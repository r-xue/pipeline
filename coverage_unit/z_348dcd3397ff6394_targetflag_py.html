62class SerialTargetflag(basetask.StandardTaskTemplate): 

63 Inputs = TargetflagInputs 

64 

65 def prepare(self): 

66 

67 inputs = self.inputs 

68 

69 # Initialize results. 

70 result = TargetflagResults() 

71 

72 cafflags = [] 

73 

74 # create a shortcut to the plotting function that pre-supplies the inputs and context 

75 plot_fn = functools.partial(create_plots, inputs, inputs.context) 

76 

77 # Check for any target intents 

78 eb_intents = inputs.context.observing_run.get_ms(inputs.vis).intents 

79 if 'TARGET' not in eb_intents: 

80 LOG.info('No target intents found.') 

81 return result 

82 

83 # Create back-up of flags. 

84 LOG.info('Creating back-up of "pre-targetflag" flagging state') 

85 flag_backup_name_pretgtf = 'before_tgtflag' 

86 task = casa_tasks.flagmanager(vis=inputs.vis, mode='save', versionname=flag_backup_name_pretgtf) 

87 self._executor.execute(task) 

88 

89 # Ensure that any flagging applied to the MS by this applycal are 

90 # reverted at the end, even in the case of exceptions. 

91 try: 

92 # Run applycal to apply pre-existing caltables and propagate their 

93 # corresponding flags to the MS. Should typically include Tsys, 

94 # bandpass, and spwphaseup tables, as well as WVR if 12-m antennas 

95 # are present, and antpos if any position corrections were made. 

96 LOG.info('Applying pre-existing cal tables.') 

97 acinputs = applycal.IFApplycalInputs( 

98 context=inputs.context, vis=inputs.vis, intent='TARGET', flagsum=False, flagbackup=False) 

99 actask = applycal.SerialIFApplycal(acinputs) 

100 acresult = self._executor.execute(actask, merge=True) 

101 # copy across the vis:callibrary dict to our result. This dict 

102 # will be inspected by the renderer to know if/which callibrary 

103 # files should be copied across to the weblog stage directory 

104 result.callib_map.update(acresult.callib_map) 

105 

106 # Create back-up of flags after applycal but before correctedampflag. 

107 LOG.info('Creating back-up of "after_tgtflag_applycal" flagging state') 

108 flag_backup_name_after_tgtflag_applycal = 'after_tgtflag_applycal' 

109 task = casa_tasks.flagmanager(vis=inputs.vis, mode='save', versionname=flag_backup_name_after_tgtflag_applycal) 

110 self._executor.execute(task) 

111 

112 # Find amplitude outliers and flag data. This needs to be done 

113 # per source / per field ID / per spw basis. 

114 LOG.info('Running correctedampflag to identify target source outliers to flag.') 

115 

116 # This task is called by the framework for each EB in the vis list. 

117 

118 # Call correctedampflag for the target intent. For that intent it 

119 # will loop over spw and field IDs to inspect the flags individually 

120 # per mosaic pointing. 

121 cafinputs = correctedampflag.Correctedampflag.Inputs( 

122 context=inputs.context, vis=inputs.vis, intent='TARGET', niter=1) 

123 caftask = correctedampflag.Correctedampflag(cafinputs) 

124 cafresult = self._executor.execute(caftask) 

125 

126 # Store correctedampflag result 

127 result.cafresult = cafresult 

128 

129 # Get new flag commands 

130 cafflags = cafresult.flagcmds() 

131 

132 # If new outliers were identified...make "after flagging" plots that 

133 # include both applycal flags and correctedampflag flags 

134 if cafflags: 

135 # Make "after calibration, after flagging" plots for the weblog 

136 LOG.info('Creating "after calibration, after flagging" plots') 

137 result.plots['after'] = plot_fn(flagcmds=cafflags, suffix='after') 

138 

139 # Restore the "after_tgtflag_applycal" backup of the flagging 

140 # state, so that the "before plots" only show things needing 

141 # to be flagged by correctedampflag 

142 LOG.info('Restoring back-up of "after_tgtflag_applycal" flagging state.') 

143 task = casa_tasks.flagmanager(vis=inputs.vis, mode='restore', versionname=flag_backup_name_after_tgtflag_applycal) 

144 self._executor.execute(task) 

145 # Make "after calibration, before flagging" plots for the weblog 

146 LOG.info('Creating "after calibration, before flagging" plots') 

147 result.plots['before'] = plot_fn(flagcmds=cafflags, suffix='before') 

148 

149 finally: 

150 # Restore the "pre-targetflag" backup of the flagging state, to 

151 # undo any flags that were propagated from caltables to the MS by 

152 # the applycal call. 

153 LOG.info('Restoring back-up of "pre-targetflag" flagging state.') 

154 task = casa_tasks.flagmanager(vis=inputs.vis, mode='restore', versionname=flag_backup_name_pretgtf) 

155 self._executor.execute(task) 

156 

157 if cafflags: 

158 # Re-apply the newly found flags from correctedampflag. 

159 LOG.info('Re-applying flags from correctedampflag.') 

160 fsinputs = FlagdataSetter.Inputs(context=inputs.context, vis=inputs.vis, table=inputs.vis, inpfile=[]) 

161 fstask = FlagdataSetter(fsinputs) 

162 fstask.flags_to_set(cafflags) 

163 _ = self._executor.execute(fstask) 

164 

165 return result

168 return results

180def create_plots(inputs, context, flagcmds, suffix=''): 

181 """ 

182 Return amplitude vs time plots for the given data column. 

183 

184 :param inputs: pipeline inputs 

185 :param context: pipeline context 

186 :param flagcmds: list of FlagCmd objects 

187 :param suffix: optional component to add to the plot filenames 

188 :return: dict of (x axis type => str, [plots,...]) 

189 """ 

190 # Exit early if weblog generation has been disabled 

191 if basetask.DISABLE_WEBLOG: 

192 return [], [] 

193 

194 calto = callibrary.CalTo(vis=inputs.vis) 

195 output_dir = context.output_dir 

196 

197 # Amplitude vs time plots 

198 amp_time_plots = AmpVsXChart('time', context, output_dir, calto, suffix=suffix).plot() 

199 

200 # Amplitude vs UV distance plots shall contain only the fields that were flagged 

201 flagged_spws = {flagcmd.spw for flagcmd in flagcmds} 

202 spw_field_dict = {int(spw): ','.join(sorted({flagcmd.field for flagcmd in flagcmds if flagcmd.spw==spw})) for spw in flagged_spws} 

203 amp_uvdist_plots = AmpVsXChart('uvdist', context, output_dir, calto, suffix=suffix, field=spw_field_dict).plot() 

204 

205 for spw, field in spw_field_dict.items(): 

206 LOG.info(f'Fields flagged for {inputs.vis} spw {spw}: {field}') 

207 

208 return {'time': amp_time_plots, 'uvdist': amp_uvdist_plots}
