556 inputs = self.inputs 

557 if inputs.overwrite or not os.path.exists(outfile): 

558 template_text = FLAGGING_TEMPLATE_HEADER.replace('___TITLESTR___', titlestr) 

559 with open(outfile, 'w') as f: 

560 f.writelines(template_text)

563 def _rename_J2000_to_ICRS(ms_path: str) -> None: 

564 # PIPE-2006: rename J2000 FIELD and SOURCE table directions to ICRS (only for ALMA cycles 0-2); 

565 # adapted from a script by D.Petry (ESO), 2016-03-04 

566 with casa_tools.MSMDReader(ms_path) as msmd: 

567 if 'ALMA' not in msmd.observatorynames(): 

568 return 

569 

570 basename = os.path.basename(ms_path) 

571 changed_columns = set() 

572 with casa_tools.TableReader(ms_path + '/FIELD', nomodify=False) as tb: 

573 for colname in ['PhaseDir_Ref', 'DelayDir_Ref', 'RefDir_Ref']: 

574 a = tb.getcol(colname) 

575 for i in range(len(a)): 

576 if a[i] == 0: # J2000 

577 a[i] = 21 # ICRS 

578 changed_columns.add(colname) 

579 tb.putcol(colname, a) 

580 if changed_columns: 

581 LOG.info(basename + ': changing coords from J2000 to ICRS in the FIELD table, columns ' + 

582 ', '.join(changed_columns)) 

583 

584 with casa_tools.TableReader(ms_path + '/SOURCE', nomodify=False) as tb: 

585 x = tb.getcolkeywords('DIRECTION') 

586 if x['MEASINFO']['Ref'] == 'J2000': 

587 x['MEASINFO']['Ref'] = 'ICRS' 

588 tb.putcolkeywords('DIRECTION', x) 

589 LOG.info(basename + ': changing coords from J2000 to ICRS in the SOURCE table')
