63 """Initialise SDImportDataInputs class. 

64 

65 Args: 

66 context: pipeline context 

67 

68 vis: List of visibility data files. These may be ASDMs, tar files of ASDMs, 

69 MSes, or tar files of MSes, If ASDM files are specified, they will be 

70 converted to MS format. 

71 

72 Example: ``vis=['X227.ms', 'asdms.tar.gz']`` 

73 

74 output_dir: path of output directory 

75 

76 asis: Creates verbatim copies of the ASDM tables in the output MS. 

77 The value given to this option must be a list of table names 

78 separated by space characters. Default value, None, is 

79 equivalent to the following list. 

80 

81 ``'SBSummary ExecBlock Annotation Antenna Station Receiver Source CalAtmosphere CalWVR SpectralWindow'`` 

82 

83 Example: ``'Receiver'``, ``''`` 

84 

85 process_caldevice: Import the ASDM caldevice table. 

86 

87 Example: ``True`` 

88 

89 Default: ``None`` (equivalent to ``True``) 

90 

91 session: List of sessions to which the visibility files belong. 

92 Defaults to a single session containing all the visibility files, 

93 otherwise a session must be assigned to each vis file. 

94 

95 Example: ``session=['Session_1', 'Sessions_2']`` 

96 

97 overwrite: Overwrite existing files on import. When converting 

98 ASDM to MS, if overwrite=False and the MS already exists in 

99 output directory, then this existing MS dataset will be used 

100 instead. 

101 

102 Default: ``None`` (equivalent to ``False``) 

103 

104 nocopy: Disable copying of MS to working directory 

105 

106 bdfflags: Apply BDF flags on import. 

107 

108 datacolumns: Dictionary defining the data types of existing columns. 

109 The format is: 

110 

111 ``{'data': 'data type 1'}`` 

112 

113 or 

114 

115 ``{'data': 'data type 1', 'corrected': 'data type 2'}`` 

116 

117 For ASDMs the data type can only be RAW and one can only specify 

118 it for the data column. 

119 For MSes one can define two different data types for the DATA and 

120 CORRECTED_DATA columns and they can be any of the known data types 

121 (RAW, REGCAL_CONTLINE_ALL, REGCAL_CONTLINE_SCIENCE, 

122 SELFCAL_CONTLINE_SCIENCE, REGCAL_LINE_SCIENCE, 

123 SELFCAL_LINE_SCIENCE, BASELINED, ATMCORR). 

124 The intent selection strings _ALL or _SCIENCE can be skipped. 

125 In that case the task determines this automatically by inspecting 

126 the existing intents in the dataset. 

127 Usually, a single datacolumns dictionary is used for all datasets. 

128 If necessary, one can define a list of dictionaries, one for each EB, 

129 with different setups per EB. If no type is specified, {'data':'raw'} 

130 will be assumed. 

131 

132 save_flagonline: Save flag commands, flagging template, imaging targets, to text files。 

133 

134 Default: ``None`` (equivalent to ``True``) 

135 

136 lazy: Use the lazy filter import. 

137 

138 Default: ``None`` (equivalent to ``False``) 

139 

140 with_pointing_correction: Apply pointing correction to DIRECTION. 

141 Add (ASDM::Pointing::encoder - ASDM::Pointing::pointingDirection) 

142 to the value to be written in MS::Pointing::direction. 

143 

144 Default: ``None`` (equivalent to ``True``) 

145 

146 createmms: Create an MMS. 

147 

148 Default: ``None`` (equivalent to ``False``) 

149 

150 ocorr_mode: Selection of baseline correlation to import. 

151 Valid only if input visibility is ASDM. See a document of CASA, 

152 casatasks::importasdm, for available options. 

153 

154 Default: ``None`` (equivalent to ``'ao'``) 

155 

156 hm_rasterscan: Heuristics method for raster scan analysis. 

157 Two analysis modes, time-domain analysis ('time') and 

158 direction analysis ('direction'), are available. 

159 

160 Default: ``None`` (equivalent to ``'time'``) 

161 

162 parallel: Execute using CASA HPC functionality, if available. 

163 

164 Options: ``'automatic'``, ``'true'``, ``'false'``, ``True``, ``False`` 

165 

166 Default: ``None`` (equivalent to ``'automatic'``) 

167 """ 

168 super().__init__(context, vis=vis, output_dir=output_dir, asis=asis, 

169 process_caldevice=process_caldevice, session=session, 

170 overwrite=overwrite, nocopy=nocopy, bdfflags=bdfflags, lazy=lazy, 

171 save_flagonline=save_flagonline, createmms=createmms, 

172 ocorr_mode=ocorr_mode, datacolumns=datacolumns) 

173 self.with_pointing_correction = with_pointing_correction 

174 self.hm_rasterscan = hm_rasterscan 

175 self.parallel = parallel

179 """SDImportDataResults is an equivalent class with ImportDataResults.

186 the resulting MeasurementSet domain objects and optionally the additional

246class SerialSDImportData(importdata.ImportData): 

247 """Data import execution task of SingleDish. 

248 

249 This class extends importdata.ImportData class, and methods execute main logics depends on it. 

250 """ 

251 

252 Inputs = SDImportDataInputs 

253 

254 def prepare(self, **parameters: Dict[str, Any]) -> SDImportDataResults: 

255 """Prepare job requests for execution. 

256 

257 Args: 

258 parameters: the parameters to pass through from the superclass. 

259 Returns: 

260 SDImportDataResults : result object 

261 """ 

262 # get results object by running super.prepare() 

263 results = super(SerialSDImportData, self).prepare() 

264 

265 # per MS inspection 

266 table_prefix = relative_path(os.path.join(self.inputs.context.name, 'MSDataTable.tbl'), 

267 self.inputs.output_dir) 

268 reduction_group_list = [] 

269 direction_rsh_result_list = [] 

270 time_rsh_result_list = [] 

271 org_directions_dict = {} 

272 for ms in results.mses: 

273 LOG.debug('Start inspection for %s' % ms.basename) 

274 table_name = os.path.join(table_prefix, ms.basename) 

275 inspector = inspection.SDInspection(self.inputs.context, table_name, ms=ms, hm_rasterscan=self.inputs.hm_rasterscan) 

276 reduction_group, org_directions, msglist, direction_rsh_result, time_rsh_result = self._executor.execute(inspector, merge=False) 

277 reduction_group_list.append(reduction_group) 

278 direction_rsh_result_list.append(direction_rsh_result) 

279 time_rsh_result_list.append(time_rsh_result) 

280 

281 # update org_directions_dict for only new keys in org_directions 

282 for key in org_directions: 

283 org_directions_dict.setdefault(key, org_directions[key]) 

284 

285 # create results object 

286 myresults = SDImportDataResults(mses=results.mses, 

287 reduction_group_list=reduction_group_list, 

288 datatable_prefix=table_prefix, 

289 setjy_results=results.setjy_results, 

290 org_directions=org_directions_dict) 

291 

292 myresults.origin = results.origin 

293 myresults.msglist = msglist 

294 for rsh in direction_rsh_result_list: 

295 myresults.rasterscan_heuristics_results_direction.setdefault(rsh.ms.origin_ms, []).append(rsh) 

296 for rsh in time_rsh_result_list: 

297 myresults.rasterscan_heuristics_results_time.setdefault(rsh.ms.origin_ms, []).append(rsh) 

298 return myresults
