22class MSWrapper(object): 

23 """ 

24 MSWrapper is a wrapper around a NumPy array populated with measurement set 

25 data for a specified scan and spectral window. The MSWrapper can be 

26 filtered on various criteria, e.g, spw, scan, antenna, etc., to narrow the 

27 data to a particular data selection. 

28 

29 The static methods MSWrapper.create_averages_from_ms and 

30 MSWrapper.create_averages_from_combination can also be used to create a new 

31 instance of a MSWrapper object. 

32 """

34 @staticmethod 

35 def create_averages_from_ms(filename, scan, spw, memlim, perantave=True): 

36 """ 

37 Create a new MSWrapper for the specified scan and spw. 

38 

39 Reading in raw measurement set data can be a very memory-intensive 

40 process, so data selection is deliberately limited to one scan and one 

41 spw at a time. 

42 

43 :param filename: measurement set filename 

44 :param scan: integer scan ID 

45 :param spw: integer spw ID 

46 :param memlim: Limit for memory for data buffer use, given in Gigabytes 

47 :param perantave: Compute per antenna average (default True) 

48 :type perantave: bool 

49 :return: MSWrapper instance 

50 """ 

51 # Method added as part of PIPE-687 

52 LOG.trace('MSWrapper.create_averages_from_ms(%r, %r, %r, %r, %r)', 

53 filename, scan, spw, memlim, perantave) 

54 

55 # Epsilon used to avoid division by zero in sigma calculations 

56 epsilon = 1.e-6 

57 

58 # put memory limit in bytes 

59 memlim = int(memlim * (1024.0 ** 3)) 

60 

61 # Initialize necessary column name lists 

62 col_names = ['antenna1', 'antenna2', 'flag', 'time', 'corrected_data'] 

63 # Data type for these columns 

64 col_dtypes = [ 

65 np.dtype('int32'), 

66 np.dtype('int32'), 

67 np.dtype('bool'), 

68 np.dtype('float64'), 

69 np.dtype('complex128') 

70 ] 

71 # Data size per element for each column, in bytes 

72 col_dsizes = np.array([c.itemsize for c in col_dtypes], dtype='int32') 

73 # Is polarization a dimension of the column? 1=True, 0=False 

74 poldim = np.array([0, 0, 1, 0, 1], dtype='int32') 

75 # Is frequency a dimension of the column? 1=True, 0=False 

76 freqdim = np.array([0, 0, 1, 0, 1], dtype='int32') 

77 # Total size of the whole piece of data should be: 

78 # Size = nrows*Sum( dsizes*(npols*poldim+1)*(nchan*freqdim+1) )(over number of colnames) 

79 

80 # commented out from copied PL code 

81 with casa_tools.MSReader(filename) as openms: 

82 

83 # select data for this scan 

84 data_selection = {"scan": str(scan), "spw": str(spw)} 

85 openms.msselect(data_selection) 

86 md = openms.metadata() 

87 antennaids = md.antennaids() 

88 nant = len(antennaids) 

89 nbl = nant * (nant - 1) / 2 + nant 

90 nrows = openms.nrow(selected=True) 

91 ntstamps = nrows / nbl 

92 

93 axis_info = openms.getdata(['axis_info']) 

94 # Create frequency axes to be introduced in output V 

95 corr_axis = axis_info['axis_info']['corr_axis'] 

96 freq_axis = axis_info['axis_info']['freq_axis'] 

97 

98 npol = len(corr_axis) 

99 nchan = len(freq_axis['chan_freq']) 

100 

101 # Data size per row and for the whole scan 

102 rowdatasize = np.sum(col_dsizes * (npol * poldim + 1) * (nchan * freqdim + 1)) 

103 scandatasize = nrows * rowdatasize 

104 # Calculate the number of iteration that need to be done to go over 

105 # the entire piece of data for this scan,spw 

106 niter = int(np.ceil(1.0 * scandatasize / memlim)) 

107 nrowsbuffer = int(np.floor(1.0 * memlim / rowdatasize)) 

108 LOG.debug('Scan {0:d} has {1:d} rows ({2:.3f} Gb), memory limit is set to {3:.3f} Gb'.format(scan, nrows, 1.0*scandatasize/(1024.0**3), 1.0*memlim/(1024.0**3))) 

109 LOG.debug('reading data in {0:d} chunks of {1:d} rows'.format(niter, nrowsbuffer)) 

110 

111 norm_sigma_tavg = np.sqrt(ntstamps * (nant - 1.0)) 

112 norm_sigma_favg = np.sqrt(nchan) 

113 

114 #arrays to sum over so we can average later 

115 t_real_sum = {ant: np.ma.zeros((npol, nchan), dtype=np.dtype('float64')) for ant in antennaids} 

116 t_imag_sum = {ant: np.ma.zeros((npol, nchan), dtype=np.dtype('float64')) for ant in antennaids} 

117 t_real_sq_sum = {ant: np.ma.zeros((npol, nchan), dtype=np.dtype('float64')) for ant in antennaids} 

118 t_imag_sq_sum = {ant: np.ma.zeros((npol, nchan), dtype=np.dtype('float64')) for ant in antennaids} 

119 n_t_points = {ant: np.ma.zeros((npol, nchan), dtype=np.dtype('float64')) for ant in antennaids} 

120 

121 f_avg = {ant: None for ant in antennaids} 

122 f_sigma = {ant: None for ant in antennaids} 

123 

124 #Variable to store starting time 

125 tstart = 1.e12 

126 

127 openms.iterinit(maxrows=nrowsbuffer) 

128 do_next = openms.iterorigin() 

129 

130 while do_next: 

131 raw_data = openms.getdata(col_names) 

132 # Create maked array of corrected and masked data 

133 # data has axis order pol->channel->time. Swap order to a more natural time->pol->channel 

134 data = np.ma.MaskedArray( 

135 data=raw_data['corrected_data'].swapaxes(0, 2).swapaxes(1, 2), 

136 dtype=np.dtype('complex128'), 

137 mask=raw_data['flag'].swapaxes(0, 2).swapaxes(1, 2) 

138 ) 

139 tstart = min(tstart, np.min(raw_data['time'])) 

140 # Iterate over antennas, selecting rows of data for averaging 

141 for antenna in antennaids: 

142 sel = np.logical_xor(raw_data['antenna1'] == antenna, raw_data['antenna2'] == antenna) 

143 seldatareal = data[sel].real 

144 seldataimag = data[sel].imag 

145 # If the per antenna adjustment is selected, invert the sign of the imaginary part when the 

146 # antenna is in position 2 

147 if perantave: 147 ↛ 153line 147 didn't jump to line 153 because the condition on line 147 was always true

148 ant2sel = raw_data['antenna2'][sel] 

149 sel2 = np.where(ant2sel == antenna)[0] 

150 for j in sel2: 

151 seldataimag[j, :, :] *= -1.0 

152 # Accumulate the sum of data averaged over time and baselines (MS rows) 

153 t_r_sum = seldatareal.sum(axis=0) 

154 t_i_sum = seldataimag.sum(axis=0) 

155 t_r_sq_sum = np.square(seldatareal).sum(axis=0) 

156 t_i_sq_sum = np.square(seldataimag).sum(axis=0) 

157 

158 #No need to accumulate the sum of data averaged over frequency 

159 f_r_avg = seldatareal.mean(axis=2) 

160 f_i_avg = seldataimag.mean(axis=2) 

161 f_ant_avg = (f_r_avg + 1j*f_i_avg).swapaxes(0,1) 

162 

163 f_r_sq_sum=np.square(np.ma.array([seldatareal[:,:,i]-f_r_avg for i in np.arange(seldatareal.shape[-1])])).sum(axis=0) 

164 f_r_sigma = np.sqrt(f_r_sq_sum/nchan).swapaxes(0,1) 

165 

166 f_i_sq_sum=np.square(np.ma.array([seldataimag[:,:,i]-f_i_avg for i in np.arange(seldataimag.shape[-1])])).sum(axis=0) 

167 f_i_sigma = np.sqrt(f_i_sq_sum/nchan).swapaxes(0,1)#(s,pol)->(pol,s) 

168 f_ant_sigma = (f_r_sigma + 1j* f_i_sigma)/norm_sigma_favg 

169 

170 # Perform the the stacking to existing sums with np.ma.sum() to avoid 

171 # propagating masking over the cumulative sum 

172 t_real_sum[antenna] = np.ma.MaskedArray([t_real_sum[antenna], t_r_sum]).sum(axis=0) 

173 t_imag_sum[antenna] = np.ma.MaskedArray([t_imag_sum[antenna], t_i_sum]).sum(axis=0) 

174 t_real_sq_sum[antenna] = np.ma.MaskedArray([t_real_sq_sum[antenna], t_r_sq_sum]).sum(axis=0) 

175 t_imag_sq_sum[antenna] = np.ma.MaskedArray([t_imag_sq_sum[antenna], t_i_sq_sum]).sum(axis=0) 

176 # Perform the the stacking to existing sums with np.ma.sum() to avoid 

177 # propagating masking over the cumulative sum 

178 if f_avg[antenna] is not None: 178 ↛ 179line 178 didn't jump to line 179 because the condition on line 178 was never true

179 f_avg[antenna] = np.ma.append(f_avg[antenna],f_ant_avg,axis=1) 

180 else: 

181 f_avg[antenna] = f_ant_avg 

182 if f_sigma[antenna] is not None: 182 ↛ 183line 182 didn't jump to line 183 because the condition on line 182 was never true

183 f_sigma[antenna] = np.ma.append(f_sigma[antenna],f_ant_sigma,axis=1) 

184 else: 

185 f_sigma[antenna] = f_ant_sigma 

186 n_t_points[antenna] = np.ma.MaskedArray([n_t_points[antenna], data[sel].count(axis=0)]).sum(axis=0) 

187 # Jump to next chunk of data 

188 do_next = openms.iternext() 

189 

190 # For pixels with no unmasked data accumulated, make sure those 

191 # pixels are filled with some epsilon dummy value 

192 #On the same iteration, get the maximum length of freq averages and use that length as masked array size 

193 f_avg_max_length = 0 

194 

195 for antenna in antennaids: 

196 zeroselt = n_t_points[antenna].data < 1.0 #if no t_points were counted, antenna should be flagged.should it not be masked already? 

197 n_t_points[antenna].data[zeroselt] = epsilon 

198 n_t_points[antenna].mask += zeroselt 

199 

200 f_avg_l = f_avg[antenna].shape[1] 

201 if f_avg_l>f_avg_max_length: 

202 f_avg_max_length = f_avg_l 

203 

204 # Iterate over antennas, now calculating the mean and sigma of the data 

205 # creating the variable V to be returned as output 

206 dtype = average_visibility_dtype(npol,nchan,f_avg_max_length) 

207 visibilities=np.ma.empty((0,), dtype=dtype) 

208 for antenna in antennaids: 

209 #compute avgs 

210 t_avg = (t_real_sum[antenna] + 1j*t_imag_sum[antenna]) / n_t_points[antenna] 

211 t_sigma_real = np.ma.sqrt((t_real_sq_sum[antenna] - np.square(t_real_sum[antenna]) / n_t_points[antenna]) / (n_t_points[antenna] - 1.0)) 

212 t_sigma_imag = np.ma.sqrt((t_imag_sq_sum[antenna] - np.square(t_imag_sum[antenna]) / n_t_points[antenna]) / (n_t_points[antenna] - 1.0)) 

213 # apply final formula from sigma values 

214 t_sigma = (t_sigma_real + 1j * t_sigma_imag) / norm_sigma_tavg 

215 

216 v=np.ma.empty((1,), dtype=dtype) 

217 v['antenna'] = antenna 

218 v['t_avg'] = t_avg 

219 v['t_sigma'] = t_sigma 

220 v['f_avg'] = f_avg[antenna] 

221 v['f_sigma'] = f_sigma[antenna] 

222 v['flagged'] = n_t_points[antenna].mask.all() #flagg antenna if all points were flagged. 

223 visibilities = np.ma.concatenate((visibilities,v),axis=0) 

224 

225 #Set up time axis 

226 int_axis = np.array([i for i in range(int(ntstamps)) for ant in range(nant-1)]) 

227 time_axis = tstart + 1.0*int_axis 

228 

229 return MSWrapper(filename, scan, spw, None, corr_axis, freq_axis, int_axis, time_axis, V=visibilities)

231 @staticmethod 

232 def create_averages_from_combination(mswlist,antennaids): 

233 """ 

234 Calculate and return the average MSWrapper of the list of MSWrapper objects given as input. 

235 The 'sigma' column get filled with the standard error of the mean, by adding inverse squared variances 

236 of each of the objects in the list. 

237 

238 :mswlist: Python List of MSWrapper objects to combine. All of them must be from the same 

239 dataset and SPW, otherwise a dummy object is returned. 

240 :return: MSWrapper object 

241 """ 

242 # Get data from first MSWrapper element of the list, scan will be the list of scans 

243 # discard raw data if present 

244 nscans = len(mswlist) 

245 scan = [mswlist[idxscan].scan for idxscan in range(nscans)] 

246 filename = mswlist[0].filename 

247 spw = mswlist[0].spw 

248 corr_axis = mswlist[0].corr_axis 

249 freq_axis = mswlist[0].freq_axis 

250 int_axis = mswlist[0].int_axis 

251 time_axis = mswlist[0].time_axis 

252 nant = len(antennaids) 

253 npol = len(corr_axis) 

254 nchan = len(freq_axis['chan_freq']) 

255 

256 f_avg_max_length = mswlist[0].V['f_avg'].shape[2] 

257 

258 dtype = average_visibility_dtype(npol,nchan,f_avg_max_length) 

259 visibilities=np.ma.empty((0,), dtype=dtype) 

260 

261 # Average data 

262 for ant in range(nant): 

263 t_data = [mswlist[idxscan].V['t_avg'][ant, :, :] for idxscan in range(nscans)] 

264 t_mean = np.ma.mean(t_data, axis=0) 

265 t_invsigmasqreal = [1.0 / (mswlist[idxscan].V['t_sigma'][ant, :, :].real ** 2) for idxscan in range(nscans)] 

266 t_invsigmasqimag = [1.0 / (mswlist[idxscan].V['t_sigma'][ant, :, :].imag ** 2) for idxscan in range(nscans)] 

267 t_sigmameanreal = 1.0 / np.ma.sqrt(np.ma.sum(t_invsigmasqreal, axis=0)) 

268 t_sigmameanimag = 1.0 / np.ma.sqrt(np.ma.sum(t_invsigmasqimag, axis=0)) 

269 t_sigma = t_sigmameanreal + 1.j * t_sigmameanimag 

270 

271 flags = [mswlist[idxscan].V['flagged'][ant] for idxscan in range(nscans)] 

272 

273 dtype = average_visibility_dtype(npol,nchan,f_avg_max_length) 

274 

275 v=np.ma.empty((1,), dtype=dtype) 

276 v['antenna'] = ant 

277 v['t_avg'] = t_mean 

278 v['t_sigma'] = t_sigma 

279 v['f_avg'] = None #f_mean 

280 v['f_sigma'] = None #f_sigma 

281 v['flagged'] = all(flags) #check this 

282 visibilities = np.ma.concatenate((visibilities,v),axis=0) 

283 

284 return MSWrapper(filename, scan, spw, None, corr_axis, freq_axis, 

285 int_axis, time_axis, V=visibilities)
