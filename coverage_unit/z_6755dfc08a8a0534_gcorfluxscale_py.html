342 """ 

343 Check that the given caltable is well-formed so that a 'fluxscale' will 

344 run successfully on it, by checking that the caltable contains results 

345 for the reference and transfer field(s). Log a warning if fields are 

346 missing. 

347 

348 Args: 

349 caltable: path to caltable to evaluate 

350 ms: MeasurementSet domain object 

351 reference: string with name(s) of reference field(s) 

352 transfer: string with names of transfer fields 

353 """ 

354 # Get the ids of the reference source and transfer calibrator source(s). 

355 ref_fieldid = {field.id for field in ms.fields if field.name in reference.split(',')} 

356 transfer_fieldids = {field.id for field in ms.fields if field.name in transfer.split(',')} 

357

360 fieldids = table.getcol('FIELD_ID') 

361

365 LOG.warning('%s contains ambiguous reference calibrator field names' % os.path.basename(caltable)) 

366 if not fieldids.issuperset(transfer_fieldids): 

367 LOG.warning('%s does not contain results for all transfer calibrators' % os.path.basename(caltable))

369 def _derive_ants_to_use(self, refant): 

370 inputs = self.inputs 

371 

372 # Resolved source heuristics. 

373 # Needs improvement if users start specifying the input antennas. 

374 # For the time being force minblperant to be 2 instead of None to 

375 # avoid ACA and Tsys flagging issues. 

376 allantenna = inputs.antenna 

377 minblperant = 2 

378 

379 if inputs.hm_resolvedcals == 'automatic': 

380 

381 # Get the antennas to be used in the gaincals, limiting 

382 # the range if the reference calibrator is resolved. 

383 refant0 = refant.split(',')[0] # use the first refant 

384 resantenna, uvrange = heuristics.fluxscale.antenna(ms=inputs.ms, refsource=inputs.reference, refant=refant0, 

385 peak_frac=inputs.peak_fraction) 

386 

387 # Do nothing if the source is unresolved. 

388 # If the source is resolved but the number of 

389 # antennas equals the total number of antennas 

390 # use all the antennas but pass along the uvrange 

391 # limit. 

392 if resantenna == '' and uvrange == '': 

393 pass 

394 else: 

395 nant = len(inputs.ms.antennas) 

396 nresant = len(resantenna.split(',')) 

397 if nresant >= nant: 

398 resantenna = allantenna 

399 else: 

400 resantenna = allantenna 

401 uvrange = inputs.uvrange 

402 

403 # Do a phase-only gaincal on the flux calibrator using a 

404 # restricted set of antennas 

405 if resantenna == '': 

406 filtered_refant = refant 

407 else: # filter refant if resolved calibrator or antenna selection 

408 resant_list = resantenna.rstrip('&').split(',') 

409 filtered_refant = str(',').join([ant for ant in refant.split(',') if ant in resant_list]) 

410 

411 return allantenna, filtered_refant, minblperant, resantenna, uvrange

413 def _derive_calvis_flux(self): 

414 """ 

415 Derive calibrated visibility fluxes. 

416 

417 To compute the "calibrated" fluxes, this method will "temporarily" 

418 apply the existing calibration tables, including the new phase and 

419 amplitude caltables created earlier during the hifa_gfluxscale task. 

420 

421 First, create a back-up of the MS flagging state, then run an applycal 

422 for the necessary intents and fields. Next, compute the calibrated 

423 visibility fluxes. Finally, always restore the back-up of the MS 

424 flagging state, to undo any flags that were propagated from the applied 

425 caltables. 

426 

427 :return: commonfluxresults.FluxCalibrationResults containing the 

428 calibrated visibility fluxes and uncertainties. 

429 """ 

430 inputs = self.inputs 

431 

432 # Identify fields and spws to derive calibrated vis for. 

433 transfer_fields = inputs.ms.get_fields(task_arg=inputs.transfer) 

434 sci_spws = set(inputs.ms.get_spectral_windows(science_windows_only=True)) 

435 transfer_fieldids = {str(field.id) for field in transfer_fields} 

436 spw_ids = {str(spw.id) for field in transfer_fields for spw in field.valid_spws.intersection(sci_spws)} 

437 

438 # Create back-up of MS flagging state. 

439 LOG.info('Creating back-up of flagging state') 

440 flag_backup_name = 'before_gfluxscale_calvis' 

441 task = casa_tasks.flagmanager(vis=inputs.vis, mode='save', versionname=flag_backup_name) 

442 self._executor.execute(task) 

443 

444 # Run computation of calibrated visibility fluxes in try/finally to 

445 # ensure that the MS are always restored, even in case of an exception. 

446 try: 

447 # Apply all caltables registered in the callibrary in the local 

448 # context to the MS. 

449 LOG.info('Applying pre-existing caltables and preliminary phase-up and amplitude caltables.') 

450 acinputs = applycal.IFApplycalInputs(context=inputs.context, vis=inputs.vis, field=inputs.transfer, 

451 intent=inputs.transintent, flagsum=False, flagbackup=False) 

452 actask = applycal.SerialIFApplycal(acinputs) 

453 self._executor.execute(actask) 

454 

455 # Initialize result. 

456 result = commonfluxresults.FluxCalibrationResults(inputs.vis) 

457 

458 # Compute the mean calibrated visibility flux for each field and 

459 # spw and add as flux measurement to the final result. 

460 for fieldid in transfer_fieldids: 

461 for spwid in spw_ids: 

462 mean_flux, std_flux = mstools.compute_mean_flux(self.inputs.ms, fieldid, spwid, self.inputs.transintent) 

463 if mean_flux: 

464 flux = domain.FluxMeasurement(spwid, mean_flux, origin=ORIGIN) 

465 flux.uncertainty = domain.FluxMeasurement(spwid, std_flux, origin=ORIGIN) 

466 result.measurements[fieldid].append(flux) 

467 finally: 

468 # Restore the MS flagging state. 

469 LOG.info('Restoring back-up of flagging state.') 

470 task = casa_tasks.flagmanager(vis=inputs.vis, mode='restore', versionname=flag_backup_name) 

471 self._executor.execute(task) 

472 

473 return result

621 def _do_phasecals(self, all_ants: str, restr_ants: str, refant: str, minblperant: int, 

622 uvrange: str) -> List[GaincalResults]: 

623 # Collect phase cal results for merging into context. 

624 phase_results = [] 

625 

626 # Identify unique transfer intents, whether PHASE/CHECK are present 

627 # among the transfer intents, and what non-PHASE/CHECK intents are 

628 # present. 

629 trans_intents = set(self.inputs.transintent.split(',')) 

630 pc_intents = {'CHECK', 'PHASE'} & trans_intents 

631 non_pc_intents = trans_intents - pc_intents 

632 

633 # PIPE-1154: identify set of all calibrator intents that are not PHASE 

634 # / CHECK; these impact which fields are in subsequent phase 

635 # calibrations. 

636 amp_intent = set(self.inputs.refintent.split(',')) 

637 exclude_intents = amp_intent | non_pc_intents 

638 

639 # Compute phase caltable for the flux calibrator, using restricted set 

640 # of antennas. 

641 phase_results.append(self._do_phasecal_for_amp_calibrator(restr_ants, refant, minblperant, uvrange, 

642 non_pc_intents)) 

643 

644 # PIPE-1154: compute phase caltable(s) with optimal parameters for 

645 # PHASE and/or CHECK fields that do not cover any of the other 

646 # calibrator intents. 

647 phase_results.extend(self._do_phase_for_phase_check_no_overlap(pc_intents, exclude_intents, all_ants, refant)) 

648 

649 # PIPE-1490: for PHASE fields that do cover other calibrator intents, 

650 # create a separate solve to ensure those solutions are put in a 

651 # separate caltable. 

652 phase_results.extend(self._do_phase_for_phase_with_overlap(exclude_intents, all_ants, refant)) 

653 

654 # PIPE-1154: for the remaining calibrator intents, compute phase 

655 # solutions using full set of antennas. 

656 if non_pc_intents: 

657 phase_results.extend(self._do_phasecal_for_other_calibrators(non_pc_intents, all_ants, refant)) 

658 

659 # Accept all phase cal results into the local context to register the 

660 # newly created phase caltables in the callibrary of the local context, 

661 # so that these phase solutions will be used in pre-apply during 

662 # upcoming amplitude solves. 

663 for result in phase_results: 

664 result.accept(self.inputs.context) 

665 

666 return phase_results

708 def _do_phasecal_for_amp_calibrator(self, antenna: str, refant: str, minblperant: int, uvrange: str, 

709 non_pc_intents: set) -> GaincalResults: 

710 """ 

711 Worker method to compute the phase calibration for the amplitude 

712 calibrator. 

713 

714 Retrieve optimal gaincal parameters from SpW maps in MS, preferentially 

715 for AMPLITUDE intent, but fall-back to BANDPASS intent in cases where 

716 amplitude calibrator == bandpass calibrator (PIPE-2499). 

717 

718 Args: 

719 antenna: A comma-delimited string specifying the antenna names or 

720 ids to be used. 

721 refant: A string specifying the reference antenna(s) to use. 

722 minblperant: Minimum number of baselines required per antenna in 

723 phase solve. 

724 uvrange: String specifying whether and how to select data by 

725 length in the phase solve. 

726 non_pc_intents: set of non phase/check calibrator intents, that 

727 resulting caltable should be registered to as well. 

728 

729 Returns: 

730 GaincalResults object. 

731 """ 

732 inputs = self.inputs 

733 

734 # Get optimal phase solution parameters for the amplitude calibrator 

735 # based on spw mapping info in MS. By default, retrieve these for the 

736 # amplitude intent (aka inputs.refintent). 

737 intent_for_param = inputs.refintent 

738 # PIPE-2499: however, if the amplitude calibrator field was also used as 

739 # the bandpass calibrator, then assume hifa_spwphaseup will only have 

740 # stored SpwMapping for BANDPASS intent (having skipped AMPLITUDE) and 

741 # therefore retrieve parameters for BANDPASS intent. 

742 if any("BANDPASS" in fld.intents for fld in inputs.ms.get_fields(inputs.reference)): 

743 intent_for_param = "BANDPASS" 

744 combine, gaintype, interp, solint, spwmap = self._get_phasecal_params(intent_for_param, inputs.reference) 

745 

746 # Compute phase caltable for the amplitude calibrator. 

747 LOG.info(f'Compute phase gaincal table for flux calibrator (intent={inputs.refintent},' 

748 f' field={inputs.reference}).') 

749 phase_result = self._do_gaincal(field=inputs.reference, intent=inputs.refintent, gaintype=gaintype, calmode='p', 

750 combine=combine, solint=solint, antenna=antenna, uvrange=uvrange, 

751 minsnr=inputs.minsnr, refant=refant, minblperant=minblperant, spwmap=spwmap, 

752 interp=interp) 

753 

754 # PIPE-1831: update the CalApplication to add the other non-phase/check 

755 # calibrator intents as valid intents that this phase caltable can be 

756 # applied to. This is necessary for datasets where the amplitude 

757 # calibrator field(s) also cover one or more of the "other" (non-phase/ 

758 # non-check) calibrator intents. Without this, any subsequent gaincal on 

759 # this field would split the call for this field into separate ones for 

760 # AMP and other intents, leading to undesired duplicate solutions. 

761 # This does assume that the AMPLITUDE *scans* on the amplitude 

762 # calibrator field(s) also covered those "other" calibrator intents; 

763 # this is not always the case, see discussion on PIPE-2822. 

764 if phase_result.pool: 

765 original_calapp = phase_result.pool[0] 

766 intents_str = ",".join({original_calapp.intent} | non_pc_intents) 

767 modified_calapp = callibrary.copy_calapplication(original_calapp, intent=intents_str) 

768 phase_result.pool[0] = modified_calapp 

769 phase_result.final[0] = modified_calapp 

770 

771 return phase_result

866 def _do_phasecal_for_other_calibrators(self, intents: set, antenna: str, refant: str) -> list[GaincalResults]: 

867 """ 

868 Compute the phase calibration for the "other" (not: phase, check, or 

869 amplitude) calibrators, typically bandpass, diffgain, and/or 

870 polarization intents. 

871 

872 Args: 

873 intents: set of intents to perform phase calibration for. 

874 antenna: A comma-delimited string specifying the antenna names or 

875 ids to be used. 

876 refant: A string specifying the reference antenna(s) to use. 

877 

878 Returns: 

879 List of GaincalResults objects. 

880 """ 

881 inputs = self.inputs 

882 phase_results = [] 

883 

884 # Identify fields that cover the provided intents. Note that 

885 # inputs.transfer will skip any field that already covers the intent 

886 # inputs.refintent (the amplitude calibrator), as that is covered by a 

887 # separate gaincal. 

888 fields = [f for f in inputs.ms.get_fields(inputs.transfer) if f.intents.intersection(intents)] 

889 

890 # PIPE-2499: for each field, check whether it is used for multiple 

891 # calibrator intents, and handle these as special cases. 

892 for field in fields: 

893 # Matching intents in current field. 

894 fld_intents = field.intents.intersection(intents) 

895 fld_intents_str = ",".join(fld_intents) 

896 LOG.info(f'Compute phase gaincal table for other calibrators (intent={fld_intents_str},' 

897 f' field={field.name}).') 

898 

899 # If this field is used both as diffgain and bandpass calibrator, 

900 # then assume that hifa_spwphaseup will only have stored SpwMapping 

901 # for BANDPASS intent (having skipped DIFFGAINSRC and DIFFGAINREF) 

902 # and therefore retrieve parameters for BANDPASS intent, and solve 

903 # for all intents at once. 

904 # Note: it assumed there is no support for band-to-band polarization 

905 # observations, therefore if DIFFGAIN is present, there cannot be 

906 # any of the POL* intents (but if there were, those would get solved 

907 # for here). 

908 if "DIFFGAINSRC" in fld_intents and "BANDPASS" in fld_intents: 

909 combine, gaintype, interp, solint, spwmap = self._get_phasecal_params('BANDPASS', field.name) 

910 phase_results.append(self._do_gaincal(field=field.name, intent=fld_intents_str, gaintype=gaintype, 

911 calmode='p', combine=combine, solint=solint, antenna=antenna, 

912 uvrange='', minsnr=self.inputs.minsnr, refant=refant, 

913 spwmap=spwmap, interp=interp)) 

914 # If this field is used as the diffgain calibrator, with no overlap 

915 # with bandpass calibrator, then assume that hifa_spwphaseup will 

916 # have stored separate SpwMapping info for DIFFGAINREF and 

917 # DIFFGAINSRC and create separate solves for those. It is assumed 

918 # here that if DIFFGAINSRC is present, DIFFGAINREF must be present 

919 # as well. It is further assumed that there is no support for 

920 # band-to-band polarization, so this field should not have also POL* 

921 # intents (but if there were, then those POL* intents would not get 

922 # solved). 

923 elif "DIFFGAINSRC" in fld_intents and "BANDPASS" not in fld_intents: 

924 for dg_intent in {"DIFFGAINREF", "DIFFGAINSRC"}: 

925 combine, gaintype, interp, solint, spwmap = self._get_phasecal_params(dg_intent, field.name) 

926 phase_results.append(self._do_gaincal(field=field.name, intent=dg_intent, gaintype=gaintype, 

927 calmode='p', combine=combine, solint=solint, antenna=antenna, 

928 uvrange='', minsnr=self.inputs.minsnr, refant=refant, 

929 spwmap=spwmap, interp=interp)) 

930 # For all other cases, use all intents of current field to retrieve 

931 # optimal parameters and compute phase solutions. At present, this 

932 # is expected to handle the following cases: 

933 # 

934 # - field is a bandpass calibrator (no overlap with polarization): 

935 # In this case, a matching SpwMapping should be present in MS, so 

936 # it would use the optimal parameters for BANDPASS. 

937 # 

938 # - field is a polarization calibrator, with or without overlap with 

939 # bandpass: 

940 # In this case, because it is a polarization calibrator, it is 

941 # required to use the default gaincal parameters. hifa_spwphaseup 

942 # currently explicitly does not derive optimal parameters for any 

943 # field covering polarization intents, so therefore the look-up of 

944 # optimal parameters for current field will find not find a 

945 # matching SpwMapping, and thus it should use default phasecal 

946 # parameters. 

947 else: 

948 combine, gaintype, interp, solint, spwmap = self._get_phasecal_params(fld_intents_str, field.name) 

949 phase_results.append(self._do_gaincal(field=field.name, intent=fld_intents_str, gaintype=gaintype, 

950 calmode='p', combine=combine, solint=solint, antenna=antenna, 

951 uvrange='', minsnr=self.inputs.minsnr, refant=refant, 

952 spwmap=spwmap, interp=interp)) 

953 

954 return phase_results

956 def _get_phasecal_params(self, intent: str, field: str) -> tuple[str, str, str | None, str, list[int]]: 

957 """ 

958 Retrieve the optimal parameters to use for phase calibration for a given 

959 single intent and single field. These parameters are retrieved from the 

960 SpW maps stored in the MS object for the MS currently being processed. 

961 

962 Args: 

963 intent: intent to retrieve phase cal parameters for. 

964 field: field name to retrieve phase cal parameters for. 

965 

966 Returns: 

967 5-tuple containing: 

968 * combine parameter ('' or 'spw') 

969 * gaintype parameter ('G' or 'T') 

970 * interpolation parameter (None, 'linearPD,linear', or 'linear,linear') 

971 * solution interval parameter 

972 * spwmap: List representing a spectral window map that specifies 

973 which SpW IDs should be re-mapped / combined. 

974 """ 

975 inputs = self.inputs 

976 ms = inputs.ms 

977 

978 # By default, no spw mapping or combining, no interp, gaintype='G', and 

979 # using input solint. 

980 combine = '' 

981 gaintype = 'G' 

982 interp = None 

983 solint = inputs.phaseupsolint 

984 spwmap = [] 

985 

986 # Try to fetch spwmapping info from MS for requested intent and field. 

987 spwmapping = ms.spwmaps.get((intent, field), None) 

988 

989 # If a mapping was found, use the spwmap, and update combine and interp 

990 # depending on whether it is a combine spw mapping. 

991 if spwmapping: 

992 spwmap = spwmapping.spwmap 

993 solint = spwmapping.solint 

994 gaintype = spwmapping.gaintype 

995 

996 # If the spwmap is for combining spws, then override combine and 

997 # interp accordingly. 

998 if spwmapping.combine: 

999 combine = 'spw' 

1000 interp = 'linearPD,linear' 

1001 else: 

1002 # PIPE-1154: when using a phase up spw mapping, ensure that 

1003 # interp = 'linear,linear'; though this may need to be changed 

1004 # in the future, see PIPEREQ-85. 

1005 interp = 'linear,linear' 

1006 

1007 return combine, gaintype, interp, solint, spwmap

1114 def _flag_ampcal(self, caltable: str) -> List[str]: 

1115 # Get fields and SpWs to evaluate. 

1116 fields = self.inputs.ms.get_fields(name=','.join([self.inputs.transfer, self.inputs.reference])) 

1117 scispws = self.inputs.ms.get_spectral_windows() 

1118 

1119 # Create an antenna id-to-name translation dictionary. 

1120 antenna_id_to_name = {ant.id: ant.name for ant in self.inputs.ms.antennas if ant.name.strip()} 

1121 

1122 # Evaluate each field separately. 

1123 flagcmds = [] 

1124 for field in fields: 

1125 # Retrieve unflagged amplitudes for current field from amplitude 

1126 # caltable. 

1127 with casa_tools.TableReader(caltable) as table: 

1128 with contextlib.closing(table.query(f"FIELD_ID == {field.id}")) as subtable: 

1129 idx_unflagged = np.where(subtable.getcol('FLAG') == 0) 

1130 amplitudes = np.abs(subtable.getcol('CPARAM')[idx_unflagged]) 

1131 

1132 # Compute median and MAD for current field. 

1133 median = np.median(amplitudes) 

1134 mad = adopted.MAD(amplitudes) 

1135 

1136 # Evaluate each SpW separately. 

1137 for spw in scispws: 

1138 # Skip evaluation if no data are available for current field and 

1139 # SpW; this occurs for example with BandToBand datasets. 

1140 if spw not in field.valid_spws: 

1141 continue 

1142 

1143 # Retrieve unflagged amplitudes, timestamps, and antennas for 

1144 # current field and SpW from amplitude caltable. 

1145 with casa_tools.TableReader(caltable) as table: 

1146 taql = f"SPECTRAL_WINDOW_ID == {spw.id} && FIELD_ID == {field.id}" 

1147 with contextlib.closing(table.query(taql)) as subtable: 

1148 # Identify unflagged data. Note: CPARAM and FLAG are of 

1149 # shape [pol, channel, row], while TIME and ANTENNA1 

1150 # are of shape [row], hence using 3rd index for latter. 

1151 flags = subtable.getcol('FLAG') 

1152 # If this SpW is already fully flagged, skip it. 

1153 if np.all(flags): 

1154 continue 

1155 idx_unflagged = np.where(flags == 0) 

1156 amplitudes = np.abs(subtable.getcol('CPARAM')[idx_unflagged]) 

1157 times = subtable.getcol('TIME')[idx_unflagged[2]] 

1158 antennas = subtable.getcol('ANTENNA1')[idx_unflagged[2]] 

1159 

1160 # Compute sigma and identify outliers. 

1161 sigma = np.abs(amplitudes - median) / mad 

1162 idx_to_flag = np.where(sigma > self.inputs.amp_outlier_sigma)[0] 

1163 

1164 # Generate flagging commands for outlier timestamps. 

1165 # Note: '&&&' is appended to the antenna name to restrict to 

1166 # flagging auto-correlation baselines for that antenna, and to 

1167 # ensure that the antenna gets flagged even if it is the refant 

1168 # (PIPE-2155). 

1169 for idx in idx_to_flag: 

1170 start = casa_tools.quanta.time(casa_tools.quanta.quantity(times[idx] - 0.5, 's'), form=['ymd']) 

1171 end = casa_tools.quanta.time(casa_tools.quanta.quantity(times[idx] + 0.5, 's'), form=['ymd']) 

1172 flagcmds.append(f"mode='manual' antenna='{antenna_id_to_name[antennas[idx]]}&&&' spw='{spw.id}'" 

1173 f" field='{field.id}' timerange='{start[0]}~{end[0]}'" 

1174 f" reason='QA2:gfluxscale_amp_time_sigma={sigma[idx]:.6f}'") 

1175 

1176 # Apply flags. 

1177 if flagcmds: 

1178 flagsetterinputs = FlagdataSetter.Inputs(context=self.inputs.context, table=caltable, inpfile=[]) 

1179 flagsettertask = FlagdataSetter(flagsetterinputs) 

1180 flagsettertask.flags_to_set(flagcmds) 

1181 self._executor.execute(flagsettertask) 

1182 

1183 # Return flag commands for rendering in weblog. 

1184 return flagcmds

1349def calc_averages_per_field(results): 

1350 """ 

1351 Return a compiled and averaged flux calibrations per field. 

1352 

1353 :param results: 

1354 :return: 

1355 """ 

1356 averages = collections.defaultdict(list) 

1357 for field_name, measurement_structs in results.items(): 

1358 spw_names = {spw_name for _, spw_name, _ in measurement_structs} 

1359 for spw_name in spw_names: 

1360 measurements_for_spw = [measurement for _, name, measurement in measurement_structs 

1361 if name == spw_name] 

1362 if len(measurements_for_spw) == 0: 

1363 continue 

1364 

1365 mean = reduce(operator.add, measurements_for_spw) / len(measurements_for_spw) 

1366 

1367 # copy the uncertainty if there's only one measurement, 

1368 # otherwise calculate the standard error of the mean. 

1369 if len(measurements_for_spw) == 1: 

1370 m = measurements_for_spw[0] 

1371 unc_I = m.uncertainty.I 

1372 unc_Q = m.uncertainty.Q 

1373 unc_U = m.uncertainty.U 

1374 unc_V = m.uncertainty.V 

1375 else: 

1376 JY = measures.FluxDensityUnits.JANSKY 

1377 unc_I = stats.sem([float(m.I.to_units(JY)) for m in measurements_for_spw]) 

1378 unc_Q = stats.sem([float(m.Q.to_units(JY)) for m in measurements_for_spw]) 

1379 unc_U = stats.sem([float(m.U.to_units(JY)) for m in measurements_for_spw]) 

1380 unc_V = stats.sem([float(m.V.to_units(JY)) for m in measurements_for_spw]) 

1381 

1382 # floats are interpreted as Jy, so we don't need to convert 

1383 # SEM values 

1384 mean.uncertainty = FluxMeasurement(spw_name, unc_I, Q=unc_Q, U=unc_U, V=unc_V, origin=ORIGIN) 

1385 

1386 averages[field_name].append((spw_name, mean)) 

1387 

1388 return averages

1392 """ 

1393 Copy a flux result dict, remapping the target spectral window in 

1394 the original result to a new measurement set. 

1395 

1396 This function makes a copy of a dict of flux calibration results 

1397 (keys=field names, values=FluxMeasurements), remapping the spectral 

1398 window target in the results to the corresponding spectral window 

1399 in the target measurement set. 

1400 

1401 :param context: pipeline context 

1402 :param vis: name of the measurement set to remap spws to 

1403 :param field_measurements: flux calibrations to adopt 

1404 :type field_measurements: dict with format {str: [FluxMeasurements]} 

1405 :return: flux results remapped to measurement set 

1406 :rtype: dict with format {str: [FluxMeasurements]} 

1407 """

1409 science_spws = ms.get_spectral_windows(science_windows_only=True) 

1410 spw_names_to_id = {spw.name: spw.id for spw in science_spws} 

1411 # spw names must uniquely identify a science spw, otherwise we 

1412 # can't create a correct spw ID mapping 

1413 assert (len(spw_names_to_id) == len(science_spws)) 

1414 

1415 d = {field_name: [copy_flux_measurement(m, spw_id=spw_names_to_id[spw_name]) 

1416 for spw_name, m in measurements if spw_name in spw_names_to_id] 

1417 for field_name, measurements in field_measurements.items()} 

1418 

1419 return d
