409 def _derive_phase_to_target_check_mapping(ms: MeasurementSet) -> Dict[str, Set]: 

410 """ 

411 Derive mapping between PHASE calibrator fields (by name) and 

412 corresponding fields (by name) with TARGET / CHECK intent that these 

413 PHASE calibrators should calibrate. 

414 

415 PIPE-1154: This heuristic is intended for ALMA observing, and assumes 

416 that the first scan of a TARGET / CHECK field is always preceded by a 

417 scan of the corresponding PHASE calibrator. This method further assumes 

418 that scan IDs increase sequentially with observing time. 

419 

420 Args: 

421 ms: MeasurementSet to derive mapping for. 

422 

423 Returns: 

424 Dictionary of PHASE field names (key) and set of names of 

425 corresponding TARGET/CHECK fields (value). 

426 """ 

427 # Get the PHASE field names. 

428 phase_fields = [f.name for f in ms.get_fields(intent='PHASE')] 

429 

430 # Initialize the mapping for each PHASE calibrator field. 

431 mapping = {f: set() for f in phase_fields} 

432 

433 # Get IDs of PHASE intent scans. 

434 phase_scan_ids = [s.id for s in ms.get_scans(scan_intent='PHASE')] 

435 

436 for intent in ['CHECK', 'TARGET']: 

437 # Get field names for current intent. 

438 fields = [f.name for f in ms.get_fields(intent=intent)] 

439 

440 for field in fields: 

441 # Get ID of first scan for current field with current intent. 

442 first_scan_id = ms.get_scans(field=field, scan_intent=intent)[0].id 

443 

444 # PIPE-1154: in standard ALMA observing, each first scan of a 

445 # field with TARGET or CHECK intent should be preceded by a 

446 # scan of its corresponding PHASE calibrator. 

447 # Identify PHASE intent scans that preceded the first scan. 

448 preceding_phase_scan_ids = [i for i in phase_scan_ids if i < first_scan_id] 

449 if preceding_phase_scan_ids: 

450 # Pick nearest in time PHASE intent scan as the match, and 

451 # identify name of corresponding field. 

452 matching_phase_scan_id = max(preceding_phase_scan_ids) 

453 else: 

454 # Identify PHASE intent scans that followed the first scan. 

455 following_phase_scan_ids = [i for i in phase_scan_ids if i > first_scan_id] 

456 if following_phase_scan_ids: 

457 # As a fall-back, pick nearest in time PHASE intent 

458 # scan after first field scan, but raise warning. 

459 matching_phase_scan_id = min(following_phase_scan_ids) 

460 LOG.warning(f"{ms.basename}: no PHASE scans found prior to the first scan for field {field}" 

461 f" ({intent}), will match nearest PHASE scan that was taken after.") 

462 else: 

463 matching_phase_scan_id = None 

464 LOG.warning(f"{ms.basename}: no PHASE scans found prior or after first scan for field {field}" 

465 f" ({intent}).") 

466 

467 # If a matching PHASE scan was found, then update mapping to 

468 # link the corresponding PHASE field to current field. 

469 if matching_phase_scan_id: 

470 matching_phase_field = [f.name for f in ms.get_scans(scan_id=matching_phase_scan_id)[0].fields][0] 

471 mapping[matching_phase_field].add(field) 

472 

473 return mapping

877 def _do_snrtest(self, intent: str, field: str, spws: list[SpectralWindow]) -> SNRTestResult: 

878 """ 

879 Run gaincal SNR task to perform SNR test for specified intent and 

880 field. 

881 

882 Args: 

883 intent: intent for which to perform SNR test. 

884 field: field for which to perform SNR test. 

885 spws: list of spectral window objects for which to perform SNR test. 

886 

887 Returns: 

888 SnrTestResult that characterises the results of the gaincal SNR test 

889 """ 

890 # Simplify inputs. 

891 inputs = self.inputs 

892 

893 task_args = { 

894 'output_dir': inputs.output_dir, 

895 'vis': inputs.vis, 

896 'field': field, 

897 'intent': intent, 

898 'spw': ','.join(str(spw.id) for spw in spws), 

899 'bwedgefrac': inputs.bwedgefrac, 

900 'hm_nantennas': inputs.hm_nantennas, 

901 'maxfracflagged': inputs.maxfracflagged, 

902 } 

903 task_inputs = gaincalsnr.GaincalSnrInputs(inputs.context, **task_args) 

904 gaincalsnr_task = gaincalsnr.GaincalSnr(task_inputs) 

905 result = self._executor.execute(gaincalsnr_task) 

906 

907 # PIPE-2499: based on the intent being analysed, set whether to use 

908 # "scan time" or "integration time" as the reference time, and select 

909 # corresponding values for SNRs and the SNR limit. 

910 # The weaker calibrators CHECK and PHASE will use an SNR limit of 32 

911 # (phasesnr) and scan-based values. The other, brighter, calibrators 

912 # typically use an SNR limit of 10 (intphasesnr). 

913 if intent in WEAK_CALIBRATOR_INTENTS: 

914 snrs = result.snrs 

915 snrlimit = inputs.phasesnr 

916 ref_times_to_use = result.scantimes 

917 else: 

918 snrs = result.snrsint 

919 snrlimit = inputs.intphasesnr 

920 ref_times_to_use = result.inttimes 

921 

922 # Initialize and populate outputs. 

923 goodsnrs, ref_times, int_times = [], [], [] 

924 for i, snr in enumerate(snrs): 

925 if snr is None: 

926 goodsnrs.append(None) 

927 ref_times.append(None) 

928 int_times.append(None) 

929 else: 

930 goodsnrs.append(snr >= snrlimit) 

931 ref_times.append(ref_times_to_use[i]) 

932 int_times.append(result.inttimes[i]) 

933 

934 return SNRTestResult( 

935 spw_ids=result.spwids, 

936 snr_values=snrs, 

937 is_good_snr=goodsnrs, 

938 reference_times=ref_times, 

939 integration_times=int_times 

940 )

942 def _do_combined_snr_test(self, spwlist: list, perspwsnr: list, spwmap: list) -> dict: 

943 """ 

944 Compute combined SNRs from the "per-SpW SNR". Grouping of SpWs is 

945 specified by input parameter spwmap. 

946 

947 For each grouped SpWs, combined SNR is calculated by: 

948 combined SNR = numpy.linalg.norm(list of per SpW SNR in a group) 

949 

950 Args: 

951 spwlist: List of spw IDs to calculate combined SNR 

952 perspwsnr: List of SNRs of each SpW 

953 spwmap: List representing a spectral window map that specifies 

954 which SpW IDs should be combined. 

955 

956 Returns: 

957 Dictionary containing for a given reference SpW the corresponding 

958 mapped SpWs and combined SNR. 

959 """ 

960 LOG.info("Start combined SpW SNR test") 

961 LOG.debug('- spwlist to analyze: {}'.format(spwlist)) 

962 LOG.debug('- per SpW SNR: {}'.format(perspwsnr)) 

963 LOG.debug('- spwmap = {}'.format(spwmap)) 

964 

965 # Initialize return object. 

966 combined_snrs = {} 

967 

968 # Filter reference SpW IDs of each group. 

969 unique_mappedspw = {spwmap[spwid] for spwid in spwlist} 

970 for mappedspwid in unique_mappedspw: 

971 snrlist = [] 

972 combined_idx = [] 

973 

974 # only consider SpW IDs in spwlist for combination 

975 for i in range(len(spwlist)): 

976 spwid = spwlist[i] 

977 if spwmap[spwid] == mappedspwid: 

978 snr = perspwsnr[i] 

979 if snr is None: 

980 continue 

981 snrlist.append(snr) 

982 combined_idx.append(i) 

983 

984 if not snrlist: 

985 LOG.error('No SpW with valid SNR values; cannot calculate the combined SNR') 

986 return {} 

987 

988 # calculate combined SNR from per spw SNR 

989 combined_snr = numpy.linalg.norm(snrlist) 

990 combined_spws = [spwlist[j] for j in combined_idx] 

991 LOG.info('Reference SpW ID = {} (Combined SpWs = {}) : Combined SNR = {}' 

992 ''.format(mappedspwid, str(combined_spws), combined_snr)) 

993 # For current reference SpW, store list of combined SpWs and 

994 # the combined SNR. 

995 combined_snrs[str(mappedspwid)] = (combined_spws, combined_snr) 

996 

997 return combined_snrs

999 def _do_gaincal(self, caltable: str | None = None, field: str | None = None, intent: str | None = None, 

1000 gaintype: str | None = None, solint: str | None = None, combine: str | None = None, 

1001 minblperant: int | None = None, minsnr: int | None = None) -> GaincalResults: 

1002 """ 

1003 Runs gaincal worker task separately for each SpectralSpec present 

1004 among the requested SpWs, each appending to the same caltable. 

1005 

1006 The CalApplications in the result are modified to set calwt to False. 

1007 

1008 Args: 

1009 caltable: name of output caltable 

1010 field: field selection string 

1011 intent: intent selection string 

1012 gaintype: gain type to use 

1013 solint: solution interval to use 

1014 combine: selects whether to combine SpWs 

1015 minblperant: minimum baselines per antenna 

1016 minsnr: minimum SNR 

1017 

1018 Returns: 

1019 Results object from gaincal worker task. 

1020 """ 

1021 inputs = self.inputs 

1022 ms = inputs.ms 

1023 

1024 # Identify which science spws were selected by inputs parameter. 

1025 request_spws = ms.get_spectral_windows(task_arg=inputs.spw) 

1026 

1027 # Identify which scans covered the requested intent, field, and any of 

1028 # the requested spws. 

1029 targeted_scans = ms.get_scans(scan_intent=intent, spw=inputs.spw, field=field) 

1030 

1031 # Among the requested spws, identify which have a scan among the 

1032 # targeted scans. 

1033 scan_spws = {spw for scan in targeted_scans for spw in scan.spws if spw in request_spws} 

1034 

1035 # Create a separate phase solution caltable for each SpectralSpec 

1036 # grouping of SpWs and collect the corresponding CalApplications from 

1037 # the task results. Each caltable should have a unique filename since 

1038 # the filename includes the SpW selection. 

1039 calapps_pool, calapps_final = [], [] 

1040 for spectral_spec, tuning_spw_ids in utils.get_spectralspec_to_spwid_map(scan_spws).items(): 

1041 tuning_spw_str = ','.join([str(i) for i in sorted(tuning_spw_ids)]) 

1042 LOG.info('Processing spectral spec {}, spws {}'.format(spectral_spec, tuning_spw_str)) 

1043 

1044 scans_with_data = ms.get_scans(spw=tuning_spw_str, scan_intent=inputs.intent) 

1045 if not scans_with_data: 

1046 LOG.info('No data to process for spectral spec {}. Continuing...'.format(spectral_spec)) 

1047 continue 

1048 

1049 # added in case not passed - PL2024 was hardcoded 

1050 if not solint: 

1051 solint = 'inf' 

1052 

1053 # Initialize gaincal inputs. 

1054 task_args = { 

1055 'output_dir': inputs.output_dir, 

1056 'vis': inputs.vis, 

1057 'caltable': caltable, 

1058 'field': field, 

1059 'intent': intent, 

1060 'spw': tuning_spw_str, 

1061 'solint': solint, 

1062 'gaintype': gaintype, 

1063 'calmode': 'p', 

1064 'minsnr': minsnr, 

1065 'combine': combine, 

1066 'refant': inputs.refant, 

1067 'minblperant': minblperant, 

1068 } 

1069 task_inputs = gtypegaincal.GTypeGaincalInputs(inputs.context, **task_args)

1275 def _compute_solint( 

1276 self, spwids: list, snrs: list, ref_times: list, int_times: list, intent: str, mappingmode: str 

1277 ) -> tuple[str, str, float]: 

1278 """ 

1279 Compute the optimal solution interval and gaintype. 

1280 

1281 Computes the optimal values to use for solution interval and gaintype 

1282 in phase-up gaincal solutions. The ideal solution interval is a single 

1283 integration time interval (solint='int'), but this method uses the SNR 

1284 estimates provided to ensure that sufficient SNR is achieved, or whether 

1285 some amount of time averaging (i.e. solint > 'int') is necessary. 

1286 

1287 spwids, snrs, ref_times, and int_times are all passed from the SNR test 

1288 function, where SNRs are based on the reference time (ref_times), and 

1289 int_times is provided as the integration time (for each SpW). 

1290 

1291 The mappingmode indicates what type of spw mapping is used in the 

1292 phase-up solve: 

1293 

1294 * single: uses no mapping (each spw mapped to itself); in this case, 

1295 the value of the lowest (worst) SNR will be used. 

1296 * mapping: uses spw re-mapping; in this case, the highest SNR will be 

1297 used, as low-SNR SpWs will be re-mapped to this highest SNR SpW. 

1298 * combine: uses spw combination; in this case, linearalg is used to 

1299 compute the combined SNR that is then used to govern the solint. 

1300 

1301 Args: 

1302 spwids: List of the spectral window IDs. 

1303 snrs: List of the SNR per SpW, calculated for the reference time. 

1304 ref_times: List of the reference time per SPW, in minutes. 

1305 int_times: List of the integration time per SPW, in minutes. 

1306 intent: Intent to compute solint for. 

1307 mappingmode: Type of spw mapping to be used in phase solve, with 

1308 options: 'single', 'mapping', 'combine'. 

1309 

1310 Returns: 

1311 3-tuple containing: 

1312 * Solution interval to use, as string (e.g. "int", or "10.0s") 

1313 * Gaintype to use ("G" or "T") 

1314 * SNR threshold used 

1315 """ 

1316 inputs = self.inputs 

1317 quanta = casa_tools.quanta 

1318 

1319 # Set default SNR limit based on intent. This may get overridden further 

1320 # below, e.g. case of optimal solint for bright calibrators. 

1321 snr_threshold_used = self._snr_limit_for_intent(intent) 

1322 

1323 # Restrict the input SpWs, SNRs, and times to the SpWs-to-use. 

1324 # 

1325 # Bright AMPLITUDE or BANDPASS calibrators in Band-to-Band datasets will 

1326 # have scans in both the (high-freq) diffgain source SpWs and the 

1327 # (low-freq) diffgain reference SpWs, and subsequent phase solves for 

1328 # these calibrators will be done in a single call for all those SpWs. 

1329 # For these calibrators, use only the diffgain source SpWs, expected to 

1330 # have low SNR, to compute the optimal solution interval. 

1331 if inputs.ms.is_band_to_band and intent in {'AMPLITUDE', 'BANDPASS'}: 

1332 dgsrc_spwids = [spw.id for spw in inputs.ms.get_spectral_windows(intent='DIFFGAINSRC')] 

1333 to_keep = [idx for idx, spwid in enumerate(spwids) if spwid in dgsrc_spwids] 

1334 # For all other data, loop over the spectral specs and identify the 

1335 # SpectralSpec with the lowest-usable-SNR SpW in it; this SpectralSpec 

1336 # and its SpWs (and corresponding SNR values) will be used in the 

1337 # subsequent evaluation of best solint / gaintype. 

1338 # Standard datasets will typically contain a single spectral spec, but 

1339 # spectral scans or multi-tuning datasets will contain multiple spectral 

1340 # specs. 

1341 else: 

1342 scispws = inputs.ms.get_spectral_windows() 

1343 to_keep = [] 

1344 snr_min = float('inf') 

1345 spectralspec_to_spw_ids = utils.get_spectralspec_to_spwid_map(scispws) 

1346 

1347 # Identify the spws with the lowest SNR that's still usable. This runs two 

1348 # passes through the data: 

1349 # first pass: only consider SNRs above the threshold 

1350 # second pass: consider all SNRs if nothing was found in the first pass 

1351 for above_threshold in (True, False): 

1352 for spwids_in_spec in spectralspec_to_spw_ids.values(): 

1353 # Gets indices of SpWs that are in current spectral spec 

1354 spw_index = [i for i, spwid in enumerate(spwids) if spwid in spwids_in_spec] 

1355 # and their corresponding SNR values 

1356 spectralspec_snrs = [snrs[i] for i in spw_index] 

1357 # filter out SNRs below threshold for first pass 

1358 if above_threshold: 

1359 spectralspec_snrs = [s for s in spectralspec_snrs if s > snr_threshold_used] 

1360 # find the minimum and the corresponding spw to keep 

1361 if spectralspec_snrs: 

1362 min_snr = min(spectralspec_snrs) 

1363 if min_snr < snr_min: 

1364 to_keep, snr_min = spw_index, min_snr 

1365 # if we found an SNR, we're done - otherwise, loop again with above_threshold=False 

1366 if to_keep: 

1367 break 

1368 

1369 # Filter SNRs and times for SpWs to keep. 

1370 # PIPE-2499: for reference and integration time, it is assumed this is 

1371 # the same across all SpWs, so pick the first element as representative, 

1372 # and convert these times from minutes to seconds. 

1373 snrs = [snrs[idx] for idx in to_keep] 

1374 int_time = int_times[to_keep[0]] * 60. 

1375 ref_time = ref_times[to_keep[0]] * 60. 

1376 

1377 # Select what SNR (among SNRs for all considered SpWs) to let govern the 

1378 # optimal solution interval based on what type of SpW mapping mode will 

1379 # be used in subsequent phase-up solves. 

1380 match mappingmode: 

1381 case 'single': 

1382 # If no mapping is used (each spw mapped to itself), then use the 

1383 # lowest (worst) SNR to govern the optimal solint. 

1384 snr_to_use = min(snrs) 

1385 case 'mapping': 

1386 # Use lowest SNR of all 'good' SNRs, i.e. those above snr_threshold_used. 

1387 # As 'mapping' is only set when there is least one good SNR, we do not need 

1388 # to provide a default or handle the ValueError that would be raised by taking 

1389 # min() of an empty list 

1390 snr_to_use = min([snr for snr in snrs if snr > snr_threshold_used]) 

1391 case 'combine': 

1392 # If using SpW combination, then compute the Euclidean norm of the 

1393 # SNR values to represent the combined SNR. 

1394 snr_to_use = numpy.linalg.norm(snrs) 

1395 case _: 

1396 raise ValueError(f"Invalid mappingmode: {mappingmode}") 

1397 

1398 # Set SNR for integration time and required SNR based on the intent. 

1399 if intent in WEAK_CALIBRATOR_INTENTS: 

1400 # For the potentially weaker calibrators, scale the SNR thresholds. 

1401 int_snr = numpy.sqrt(int_time/ref_time) * snr_to_use 

1402 req_snr = numpy.sqrt(int_time/ref_time) * inputs.phasesnr 

1403 else: 

1404 # No scaling for the other calibrators (BANDPASS, DIFFGAIN, ...). 

1405 int_snr = snr_to_use 

1406 req_snr = inputs.intphasesnr 

1407 

1408 # Compute the required solint by scaling the integration time by the 

1409 # ratio of required SNR over integration-time-based SNR. 

1410 req_solint = int_time * (req_snr/int_snr)**2 

1411 # By default, assume that the phase-up will use gaintype='G', i.e. solve 

1412 # for the standard complex polarization-specific gain. 

1413 gaintype = "G" 

1414 

1415 # If the required solint after rounding would be at/below the 

1416 # integration time (i.e. < 1.5 integration time), then 'int' is already 

1417 # the optimal (lowest) solint to use, so return early with this (and 'G' 

1418 # as the corresponding default gaintype). 

1419 if req_solint < 1.5 * int_time: 

1420 return 'int', gaintype, snr_threshold_used 

1421 

1422 # If the required solint (with default gaintype) is above 'int', and the 

1423 # phase-up will use SpW combination, then prefer to use gaintype "T", 

1424 # i.e. solving across polarizations. Since this improves the SNR, 

1425 # re-check whether with gaintype='T' the required solint would fall at 

1426 # or below 'int'. 

1427 if mappingmode == 'combine': 

1428 gaintype = 'T' 

1429 # Scale the integration-time based SNR threshold, assuming that with 

1430 # gaintype='T' the signal is combined from at least 2 polarizations; 

1431 # and re-compute the required solint with this scaled int-time SNR. 

1432 int_snr = int_snr * numpy.sqrt(2) 

1433 req_solint = int_time * (req_snr/int_snr)**2 

1434 # If the required solint with gaintype='T' after rounding is 

1435 # at/below the integration time, then return early with this as the 

1436 # optimal solint. 

1437 if req_solint < 1.5 * int_time: 

1438 return 'int', gaintype, snr_threshold_used 

1439 

1440 # If the required solint is definitely above the integration time, even 

1441 # after considering the options of using SpW combination and 

1442 # gaintype='T' then proceed to compute the optimal solint above 'int', 

1443 # using different approaches for the weaker "CHECK" and "PHASE" 

1444 # calibrators vs. all other (assumed bright) calibrators. 

1445 if intent in WEAK_CALIBRATOR_INTENTS: 

1446 # For CHECK and PHASE calibrators, the required SNR will have been 

1447 # based on scan times (passed along as the reference time). 

1448 scan_time = ref_time 

1449 

1450 # If the required solint is more than half the scan time, then cap 

1451 # the optimal solint to a maximum of half the scan time. 

1452 if req_solint > scan_time / 2.: 

1453 LOG.info(f"{inputs.ms.basename}, intent={intent}: setting the optimal gaincal solint to half the scan" 

1454 f" time.") 

1455 solint = scan_time / 2. 

1456 # Otherwise, the required solint is larger than the integration time 

1457 # but less than half the scan time, so proceed to find the best 

1458 else: 

1459 # Compute how many integration times fit in the required solint 

1460 # and in the scan time. The results should be >= 2, as the case 

1461 # of 1 (after rounding) would have resulted in an early return 

1462 # in a preceding step. 

1463 req_solint_in_int = round_half_up(req_solint / int_time) 

1464 max_solints_in_scan = round_half_up(scan_time / int_time) 

1465 

1466 # Identify candidate solints as multiples of the integration 

1467 # time, between 2x int and half of the scan time, that also work 

1468 # as an exact division of the scan time, i.e. resulting in 

1469 # exact equal solution intervals with no remainder. 

1470 time_factors = [t for t in range(2, int(max_solints_in_scan / 2)) 

1471 if max_solints_in_scan % t == 0] 

1472 valid_factors = [t for t in time_factors if t >= req_solint_in_int] 

1473 

1474 # If any valid integer multiple candidates were found, then use 

1475 # the lowest factor to set the solution interval. 

1476 if valid_factors: 

1477 solint = min(valid_factors) * int_time 

1478 # Otherwise, revert to PL2024 heuristic and set solint to a 

1479 # quarter of the scan time, without forcing to nearest integer 

1480 # multiple of integration time. 

1481 else: 

1482 LOG.info(f"{inputs.ms.basename}, intent={intent}: unable to find optimal number of integrations" 

1483 f" within a scan of the SNR requirement; setting the gaincal solint to 1/4 scan time.") 

1484 solint = scan_time / 4. 

1485 

1486 # Optimal solint determination for all other (i.e. bright) calibrators. 

1487 else: 

1488 # Set solint to nearest integer multiple of the integration time. 

1489 solint = round_half_up(req_solint / int_time) * int_time 

1490 

1491 # If any of the fields for this bright calibrator are used as the 

1492 # flux calibrator, then use the lowest allowed integration-time 

1493 # based SNR limit instead, to minimize the solint and associated 

1494 # decoherence error that would impact flux scaling (PIPE-2499). 

1495 if any("AMPLITUDE" in f.intents for f in inputs.ms.get_fields(intent=intent)): 

1496 LOG.info(f"{inputs.ms.basename}: the optimal solution interval found for intent {intent} is larger than" 

1497 f" 'int', but this intent is shared with the AMPLITUDE intent on at least one field, and the" 

1498 f" amplitude calibrator requires the shortest viable solint. Re-computing optimal solint based" 

1499 f" on reduced SNR limit of {inputs.intphasesnrmin} ('intphasesnrmin').") 

1500 

1501 # Re-compute the required solint based on re-scaling from 

1502 # normal integration-time based SNR threshold to the minimum 

1503 # integration-time based SNR threshold. First, propagate the 

1504 # SNR scaling from the working 'req_solint' (i.e. unrounded) 

1505 req_solint = req_solint * (inputs.intphasesnrmin / inputs.intphasesnr) ** 2 

1506 # now round it according to a unit of integration time 

1507 solint = round_half_up(req_solint / int_time) * int_time 

1508 snr_threshold_used = inputs.intphasesnrmin 

1509 

1510 # If this adjusted solint after rounding would be at/below the 

1511 # integration time (i.e. < 1.5 integration time), then return 

1512 # early with 'int' as the optimal solint. 

1513 if solint < 1.5 * int_time: 

1514 return 'int', gaintype, snr_threshold_used 

1515 

1516 # If the optimal solint exceeds the maximum solint threshold, then 

1517 # set the final solint to this maximum threshold, rounded to nearest 

1518 # integer multiple of the integration time. 

1519 if solint >= inputs.phaseupmaxsolint: 

1520 LOG.warning(f"{inputs.ms.basename}, intent={intent}: optimal solint {solint:.3f}s exceeds the maximum" 

1521 f" allowed limit of {inputs.phaseupmaxsolint}s, and will be capped to this limit (rounded" 

1522 f" to nearest integer multiple of integration time). Phase-up in subsequent Pipeline stages" 

1523 f" will have low solution SNR.") 

1524 solint = round_half_up(inputs.phaseupmaxsolint / int_time) * int_time 

1525 

1526 # Finally, convert the optimal solint to a string, and use "int" if 

1527 # the optimal solint was below the integration time. 

1528 if solint <= int_time: 

1529 solint = 'int' 

1530 else: 

1531 solint = quanta.tos(quanta.quantity(solint, 's'), 3) 

1532 

1533 return solint, gaintype, snr_threshold_used

1587 def _compute_snr_from_gaincal( 

1588 self, 

1589 snr_result: SNRTestResult, 

1590 field: str, 

1591 intent: str, 

1592 low_snr_threshold: float = None, 

1593 catalogue_snr_multiplier: float = None, 

1594 gaintable_snr_multiplier: float = None 

1595 ): 

1596 """ 

1597 Estimates the Signal-to-Noise Ratio (SNR) by generating and analyzing a gaincal 

1598 calibration table. Updates the SNR values in the provided SnrTestResult object 

1599 based on the analysis results. 

1600 

1601 This function implements the SNR estimation logic described in PIPE-2505: 

1602 - For SPWs with SNR below threshold: applies a scaling factor (catalogue_snr_multiplier) 

1603 to avoid calibration failures 

1604 - For SPWs with SNR above threshold: computes new SNR estimates using gaincal analysis, 

1605 applying a scaling factor to the measured values. 

1606 

1607 Parameters: 

1608 snr_result: SnrTestResult 

1609 Object containing SNR test results, including spw_ids and snr_values that 

1610 will be updated based on the analysis. 

1611 field: str 

1612 Field identifier to use for the gaincal calculation. 

1613 intent: str 

1614 Intent identifier to use for the gaincal calculation. 

1615 low_snr_threshold: float, optional 

1616 Threshold below which SPWs are considered low SNR and gaincal is not attempted. 

1617 Default is -1 (corresponds to 'X' in PIPE-2505), forcing gaincal 

1618 for all SPWs. 

1619 catalogue_snr_multiplier: float, optional 

1620 Scaling factor applied to SNR values derived from the flux catalogue. 

1621 Default is 0.75 (corresponds to 'Y' in PIPE-2505). 

1622 gaintable_snr_multiplier: float, optional 

1623 Scaling factor applied to SNR values measured from the temporary gaintable. 

1624 Default is 1.0 (corresponds to 'Z' in PIPE-2505). 

1625 

1626 Returns: 

1627 None 

1628 The function updates the snr_result object in place. 

1629 """ 

1630 if snr_result.has_no_snrs: 

1631 return 

1632 

1633 # these variables can't be arg defaults as it would make them impossible to change post-import 

1634 if low_snr_threshold is None: 

1635 low_snr_threshold = LOW_SNR_THRESHOLD 

1636 if catalogue_snr_multiplier is None: 

1637 catalogue_snr_multiplier = CATALOGUE_SNR_MULTIPLIER 

1638 if gaintable_snr_multiplier is None: 

1639 gaintable_snr_multiplier = GAINTABLE_SNR_MULTIPLIER 

1640 

1641 # dict to map spw IDs to corrected SNR values 

1642 snr_corrections = {spw_id: snr for spw_id, snr in zip(snr_result.spw_ids, snr_result.snr_values) 

1643 if snr is not None} 

1644 

1645 # identify spws above/below the low-SNR threshold (='X' in PIPE-2505 spec) 

1646 low_snr_spws, high_snr_spws = self._classify_spws_by_snr(snr_corrections, low_snr_threshold) 

1647 

1648 # running gaincal minsnr=2 would result in failure for low-SNR spws. For these 

1649 # low-SNR spws, set the estimated SNR to Y * SNR 

1650 self._update_snr_for_low_snr_spws(snr_corrections, low_snr_spws, low_snr_threshold, catalogue_snr_multiplier) 

1651 

1652 # For the remaining high SNR windows, generate a G caltable and set the 

1653 # estimated SNR to Z * median SNR, as measured from the caltable 

1654 if high_snr_spws: 

1655 caltable_filename = self._generate_gain_caltable(field, intent, high_snr_spws) 

1656 self._update_snr_for_high_snr_spws(snr_corrections, high_snr_spws, caltable_filename, gaintable_snr_multiplier) 

1657 

1658 snr_limit = self._snr_limit_for_intent(intent) 

1659 self._update_snr_result(snr_result, snr_corrections, snr_limit)

1677 def _classify_spws_by_snr(snr_corrections: dict[int, float], threshold: float) -> tuple[set[int], set[int]]: 

1678 """ 

1679 Classifies spectral windows (SPWs) into two sets based on their Signal-to-Noise 

1680 Ratio (SNR) compared to a provided threshold. This method identifies and 

1681 categorizes SPWs with low SNR and high SNR, returning them as two separate sets. 

1682 

1683 Parameters: 

1684 snr_corrections (dict[int, float]): A dictionary mapping SPW IDs to their 

1685 corresponding SNR values. 

1686 threshold (float): The SNR threshold for classification. SPWs with SNRs 

1687 lower than this value are categorized as low SNR, while the rest are 

1688 categorized as high SNR. 

1689 

1690 Returns: 

1691 tuple[set[int], set[int]]: A tuple containing two sets: 

1692 - The first set contains the SPW IDs of SPWs with SNR lower than the 

1693 threshold (low SNR). 

1694 - The second set contains the SPW IDs of SPWs with SNR equal to or 

1695 higher than the threshold (high SNR). 

1696 """ 

1697 low_snr_spws = {spw_id for spw_id, snr in snr_corrections.items() if snr < threshold} 

1698 high_snr_spws = set(snr_corrections.keys()) - low_snr_spws 

1699 return low_snr_spws, high_snr_spws

1701 def _update_snr_for_low_snr_spws( 

1702 self, 

1703 snr_corrections: dict[int, float], 

1704 low_snr_spws: set[int], 

1705 threshold: float, 

1706 multiplier: float 

1707 ) -> None: 

1708 """ 

1709 Handles low signal-to-noise ratio (SNR) spectral windows (spws) by applying a 

1710 multiplier to their SNR values. This method modifies the provided 

1711 `snr_corrections` dictionary in place. 

1712 

1713 Parameters: 

1714 snr_corrections (dict[int, float]): A dictionary mapping spw IDs to their 

1715 SNR values. The SNR values are updated for spws that meet the low-SNR 

1716 condition. 

1717 low_snr_spws (set[int]): A set of spw IDs that are identified as having 

1718 low SNR values (below the threshold). 

1719 threshold (float): The SNR threshold used for identifying spws as low-SNR 

1720 spws. 

1721 multiplier (float): The multiplication factor applied to the SNR values of 

1722 low-SNR spws. 

1723 

1724 Returns: 

1725 None 

1726 """ 

1727 for spw_id in sorted(low_snr_spws): 

1728 old_snr = snr_corrections[spw_id] 

1729 snr_corrections[spw_id] *= multiplier 

1730 LOG.info(f"Estimated SNR for {self.inputs.vis} spw {spw_id} is below threshold " 

1731 f"{threshold}. Skipping gaincal; setting SNR for combine heuristics to " 

1732 f"{snr_corrections[spw_id]:.3f} = {multiplier} * {old_snr:.3f}.")

1765 def _update_snr_for_high_snr_spws( 

1766 self, 

1767 snr_corrections: dict[int, float], 

1768 high_snr_spws: set[int], 

1769 caltable_filename: str, 

1770 multiplier: float 

1771 ) -> None: 

1772 """ 

1773 Processes high Signal-to-Noise Ratio (SNR) spectral windows (spws) and updates 

1774 the SNR corrections dictionary with computed or default values based on the 

1775 filtered calibrated table data. 

1776 

1777 Parameters: 

1778 snr_corrections (dict[int, float]): A dictionary to store SNR corrections 

1779 for each spectral window where the keys are the spectral window IDs, 

1780 and the values are the corresponding corrections. 

1781 high_snr_spws (set[int]): A set containing IDs of the spectral windows 

1782 determined to have high SNR values. 

1783 caltable_filename (str): filename of the caltable to analyse 

1784 multiplier (float): A multiplier factor applied to the median SNR values 

1785 for applying a scaling adjustment to the estimated SNR. 

1786 

1787 Returns: 

1788 None 

1789 This function updates the snr_corrections dict in place. 

1790 """ 

1791 try: 

1792 caltable = CaltableWrapperFactory.from_caltable(caltable_filename) 

1793 except OSError: 

1794 LOG.info(f'Caltable for {self.inputs.vis} is missing, most likely due to zero' 

1795 f'solutions. Estimated SNR set to zero for all spws.') 

1796 for spw_id in high_snr_spws: 

1797 snr_corrections[spw_id] = 0 

1798 return 

1799 

1800 for spw in high_snr_spws: 

1801 try: 

1802 snr_data = caltable.filter(spw=spw).data['SNR'] 

1803 except KeyError: 

1804 LOG.info(f'No SNR data present in temporary gain table for {self.inputs.vis} ' 

1805 f'spw {spw}. Setting estimated SNR to zero.') 

1806 snr_corrections[spw] = 0 

1807 continue 

1808 

1809 # Get number of correlations for this SpW. 

1810 corr_type = commonhelpermethods.get_corr_products(self.inputs.ms, spw) 

1811 ncorrs = len(corr_type) 

1812 is_single_polarisation = ncorrs == 1 

1813 

1814 if is_single_polarisation: 

1815 # For single pol, one column will be populated with zeroes or nulls. 

1816 # Identify which column holds data for the median SNR calculation 

1817 idx_for_pol = commonhelpermethods.get_pol_id(self.inputs.ms, spw, corr_type[0]) 

1818 # note the use of numpy.median over numpy.ma.median: we WANT to include 

1819 # the masked/flagged values in the median calculation. To avoid triggering 

1820 # the UserWarning that results from operating on a masked array, we operate 

1821 # on data 

1822 # See https://open-jira.nrao.edu/browse/PIPE-2505?focusedId=237663&page=com.atlassian.jira.plugin.system.issuetabpanels:comment-tabpanel#comment-237663 

1823 median_snr = numpy.median(snr_data.data[:, idx_for_pol]) 

1824 else: 

1825 # otherwise, calculate the median over all pols. As obove, including masked values 

1826 median_snr = numpy.median(snr_data.data) 

1827 

1828 if median_snr in (numpy.inf, numpy.nan): 

1829 LOG.info(f'No valid data in temporary gain table for {self.inputs.vis} ' 

1830 f'spw {spw}. Setting estimated SNR to zero.') 

1831 snr_corrections[spw] = 0 

1832 else: 

1833 snr_corrections[spw] = float(median_snr * multiplier) 

1834 LOG.info(f'Based on a temporary gain table, calculated SNR for {self.inputs.vis} ' 

1835 f'spw {spw} = {median_snr:.3f}. Setting SNR for combine heuristics to ' 

1836 f'{snr_corrections[spw]:.3f} = {multiplier} * {median_snr:.3f}')

1841 snr_corrections. The function maps each spectral window's (SPW) ID from 

1842 snr_result to a corrected SNR value using the snr_corrections dictionary. If

1845 

1846 Parameters:

1849 snr_corrections (dict[int, float]): A dictionary mapping SPW IDs (int) to

1855 None

1862 

1863 snr_result.snr_values = new_snrs 

1864 snr_result.is_good_snr = good_snrs
