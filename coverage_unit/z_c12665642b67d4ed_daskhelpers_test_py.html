80@skip_if_no_dask 

81def test_parallel_processing(dask_client: Client, sample_data_dir: str) -> None: 

82 """Test parallel processing with Dask using delayed. 

83 

84 Args: 

85 dask_client: The Dask client fixture. 

86 sample_data_dir: The path to the sample data directory. 

87 """ 

88 import dask 

89 

90 @dask.delayed 

91 def load_array(filename: str) -> np.ndarray: 

92 """Load a numpy array from file. 

93 

94 Args: 

95 filename: The path to the numpy array file. 

96 

97 Returns: 

98 np.ndarray: The loaded numpy array. 

99 """ 

100 return np.load(filename) 

101 

102 @dask.delayed 

103 def process_array(array: np.ndarray) -> Dict[str, float]: 

104 """Process an array with a CPU-intensive operation. 

105 

106 Args: 

107 array: The numpy array to process. 

108 

109 Returns: 

110 Dict[str, float]: A dictionary containing statistics of the processed array. 

111 """ 

112 # Simulate CPU-intensive work 

113 result = np.sin(array) * np.cos(array) * np.sqrt(np.abs(array)) 

114 return { 

115 'mean': float(np.mean(result)), 

116 'std': float(np.std(result)), 

117 'min': float(np.min(result)), 

118 'max': float(np.max(result)), 

119 } 

120 

121 # Create computation graph 

122 computation_tasks: List[Dict[str, float]] = [] 

123 for i in range(5): 

124 filename = f'{sample_data_dir}/array_{i}.npy' 

125 array = load_array(filename) 

126 stats = process_array(array) 

127 computation_tasks.append(stats) 

128 

129 # Execute all tasks in parallel 

130 results: List[Dict[str, float]] = dask.compute(*computation_tasks) 

131 

132 # Verify results 

133 assert len(results) == 5 

134 for result in results: 

135 assert isinstance(result, dict) 

136 assert 'mean' in result 

137 assert 'std' in result 

138 assert 'min' in result 

139 assert 'max' in result 

140 # Values should be within expected ranges for sin*cos*sqrt operations 

141 assert -0.5 <= result['mean'] <= 0.5 

142 assert result['min'] >= -0.5 

143 assert result['max'] <= 0.5

147def test_parallel_map(dask_client: Client, sample_data_dir: str) -> None: 

148 """Test parallel map operation with Dask bag. 

149 

150 Args: 

151 dask_client: The Dask client fixture. 

152 sample_data_dir: The path to the sample data directory. 

153 """ 

154 import dask.bag as db 

155 

156 # List all numpy files 

157 files: List[str] = [f'{sample_data_dir}/array_{i}.npy' for i in range(5)] 

158 

159 # Create a bag from the file list 

160 file_bag = db.from_sequence(files) 

161 

162 # Define operations 

163 def calculate_stats(filename: str) -> Dict[str, Any]: 

164 """Load a file and calculate statistics. 

165 

166 Args: 

167 filename: The path to the file. 

168 

169 Returns: 

170 Dict[str, Any]: A dictionary containing file statistics. 

171 """ 

172 arr = np.load(filename) 

173 transformed = np.exp(-arr) * np.cos(arr) 

174 return { 

175 'file': os.path.basename(filename), 

176 'mean': float(np.mean(transformed)), 

177 'sum': float(np.sum(transformed)), 

178 'count': len(arr), 

179 } 

180 

181 # Apply the function to each file in parallel 

182 result_bag = file_bag.map(calculate_stats) 

183 results: List[Dict[str, Any]] = result_bag.compute() 

184 

185 # Verify results 

186 assert len(results) == 5 

187 for result in results: 

188 assert 'file' in result 

189 assert 'mean' in result 

190 assert 'sum' in result 

191 assert 'count' in result 

192 assert result['count'] == 500000

195@skip_if_no_dask 

196def test_parallel_task_performance(dask_client: Client, sample_data_dir: str) -> None: 

197 """Test and measure performance of parallel vs sequential execution. 

198 

199 This test compares the performance of parallel and sequential execution of a 

200 CPU-intensive task. It measures the execution time for both and prints 

201 the speedup. Note that the speedup is not asserted because it depends on 

202 hardware and system load. 

203 

204 Args: 

205 dask_client: The Dask client fixture. 

206 sample_data_dir: The path to the sample data directory. 

207 """ 

208 import dask 

209 

210 # Define a CPU-intensive task 

211 def intensive_task(seed: int) -> float: 

212 """CPU-intensive calculation. 

213 

214 Args: 

215 seed: Seed for random number generation. 

216 

217 Returns: 

218 float: The mean of the calculation results. 

219 """ 

220 np.random.seed(seed) 

221 x = np.random.random(1000000) 

222 for _ in range(5): 

223 x = np.sin(x) + np.cos(x**2) + np.sqrt(np.abs(x)) 

224 return np.mean(x) 

225 

226 # Sequential execution 

227 start_time = time.time() 

228 sequential_results: List[float] = [intensive_task(i) for i in range(10)] 

229 sequential_time = time.time() - start_time 

230 

231 # Parallel execution with Dask 

232 start_time = time.time() 

233 delayed_tasks = [dask.delayed(intensive_task)(i) for i in range(10)] 

234 parallel_results: List[float] = dask.compute(*delayed_tasks) 

235 parallel_time = time.time() - start_time 

236 

237 # Verify results match 

238 for seq, par in zip(sequential_results, parallel_results): 

239 assert abs(seq - par) < 1e-10 

240 

241 # Check performance improvement 

242 # Note: This may not always pass on single-core systems or very loaded systems 

243 print(f'Sequential time: {sequential_time:.2f}s') 

244 print(f'Parallel time: {parallel_time:.2f}s') 

245 print(f'Speedup: {sequential_time / parallel_time:.2f}x') 

246 

247 # We don't assert on speedup as it depends on hardware 

248 # but we can check that the parallel version works 

249 assert len(parallel_results) == 10
