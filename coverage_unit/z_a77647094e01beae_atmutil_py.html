32def init_at(at: casatools.atmosphere, humidity: float = 20.0,

36 """ 

37 Initialize atmospheric profile and spectral window setting. 

38 

39 Initialize atmospheric profile and spectral window setting in CASA 

40 atmosphere tool using input antenna site parameters, the center 

41 frequency, and the width of a frequency range. 

42 

43 Args: 

44 at: CASA atmosphere tool instance to initialize. 

45 humidity: The relative humidity at the ground (unit: %). 

46 temperature: The temperature at the ground (unit: K). 

47 pressure: The pressure at the ground (unit: mbar). 

48 atmtype: An AtmType enum that defines a type of atmospheric profile. 

49 altitude: The altitude of antenna site to calculate atmospheric

52 nchan: Number of channels in the frequency range.

57 temperature=myqa.quantity(temperature, 'K'), 

58 altitude=myqa.quantity(altitude, 'm'), 

59 pressure=myqa.quantity(pressure, 'mbar'), 

60 atmType=atmtype) 

61 fwidth = nchan * resolution 

62 at.initSpectralWindow(nbands=1, 

63 fCenter=myqa.quantity(fcenter, 'GHz'), 

64 fWidth=myqa.quantity(fwidth, 'GHz'), 

65 fRes=myqa.quantity(resolution, 'GHz'))

117def calc_airmass(elevation: float = 45.0) -> float: 

118 """ 

119 Calculate the relative airmass of a given elevation angle. 

120 

121 Args: 

122 elevation: An angle of elevation (unit: degree). 

123 

124 Returns: 

125 The relative airmass to the one at zenith. 

126 """ 

127 return 1.0 / math.cos(math.radians(90.0 - elevation))

130def calc_transmission(airmass: float, dry_opacity: Union[float, np.ndarray], 

131 wet_opacity: Union[float, np.ndarray]) -> Union[float, np.ndarray]: 

132 """ 

133 Calculate total atmospheric transmission. 

134 

135 Calculate total atmospheric transmission from the zenith opacities and 

136 relative airmass. 

137 

138 Args: 

139 airmass: The relative airmass to the zenith one. 

140 dry_opacity: The integrated zenith dry opacity. 

141 wet_opacity: The integrated zenith wet opacity. 

142 

143 Returns: 

144 The atmospheric transmission. 

145 """ 

146 return np.exp(-airmass * (dry_opacity + wet_opacity))

183def _test(pwv: float = 1.0, elevation: float = 45.0) -> np.ndarray: 

184 """ 

185 Calculate atmospheric transmission and generate a plot. 

186 

187 Calculate atmospheric transmission of a given PWV and elevation angle. 

188 The default parameter values of init_at function are used to initialize 

189 atmospheric and spectral window settings. A plot of atmospheric 

190 transmission, wet and dry zenith opacities of each channel of the spectral 

191 window is also generated. 

192 

193 Args: 

194 pwv: The zenith water vapor column for forward radiative transfer 

195 calculation (unit: mm). 

196 elevation: The angle of elevation (unit: degree). 

197 

198 Returns: 

199 An array of atmospheric transmission of each channel of the spectral window. 

200 """ 

201 myat = casa_tools.atmosphere 

202 myqa = casa_tools.quanta 

203 init_at(myat) 

204 myat.setUserWH2O(myqa.quantity(pwv, 'mm')) 

205 frequency = myqa.getvalue(myqa.convert(myat.getSpectralWindow(), 'GHz')) 

206 

207 airmass = calc_airmass(elevation) 

208 

209 dry_opacity = get_dry_opacity(myat) 

210 wet_opacity = get_wet_opacity(myat) 

211 transmission = calc_transmission(airmass, dry_opacity, wet_opacity) 

212 

213 plot(frequency, dry_opacity, wet_opacity, transmission) 

214 

215 return transmission

218def plot(frequency: np.ndarray, dry_opacity: np.ndarray, 

219 wet_opacity: np.ndarray, transmission: np.ndarray): 

220 """ 

221 Generate a plot of atmospheric transmission, wet and dry opacities. 

222 

223 Generate a twin axes plot of atmospheric transmission, wet and dry 

224 opacities by matplotlib. 

225 

226 Args: 

227 frequency: An array of frequency values. 

228 dry_opacity: The integrated dry opacity at each frequency. 

229 wet_opacity: The integrated wet opacity at each frequency. 

230 transmission: The atmospheric transmission at each frequency. 

231 """ 

232 plt.clf() 

233 a1 = plt.gcf().gca() 

234 plt.plot(frequency, dry_opacity, label='dry') 

235 plt.plot(frequency, wet_opacity, label='wet') 

236 plt.legend(loc='upper left', bbox_to_anchor=(0., 0.5)) 

237 a2 = a1.twinx() 

238 a2.yaxis.set_major_formatter(plt.NullFormatter()) 

239 a2.yaxis.set_major_locator(plt.NullLocator()) 

240 plt.gcf().sca(a2) 

241 plt.plot(frequency, transmission, 'm-') 

242 M = transmission.min() 

243 Y = 0.8 

244 ymin = (M - Y) / (1.0 - Y) 

245 ymax = transmission.max() + (1.0 - transmission.max()) * 0.1 

246 plt.ylim([ymin, ymax])

249def get_spw_spec(vis: str, spw_id: int) -> Tuple[float, int, float]: 

250 """ 

251 Calculate spectral setting of a spectral window. 

252 

253 Calculate the center frequency, number of channels, and channel resolution 

254 of a spectral window in a MeasurementSet. The values can be passed to 

255 init_at function to initialize spectral window setting in atmosphere tool. 

256 

257 Args: 

258 vis: Path to MeasurementSet. 

259 spw_id: A spectral window ID to select. 

260 

261 Returns: 

262 A three element tuple of the center frequency in GHz, number of 

263 channels, and resolution in GHz. 

264 """ 

265 with casa_tools.TableReader(os.path.join(vis, 'SPECTRAL_WINDOW')) as mytb: 

266 nrow = mytb.nrows() 

267 if spw_id < 0 or spw_id >= nrow: 

268 raise RuntimeError('spw_id {} is out of range'.format(spw_id)) 

269 nchan = mytb.getcell('NUM_CHAN', spw_id) 

270 chan_freq = mytb.getcell('CHAN_FREQ', spw_id) 

271 

272 center_freq = (chan_freq.min() + chan_freq.max()) / 2.0 

273 resolution = chan_freq[1] - chan_freq[0] 

274 

275 # Hz -> GHz 

276 toGHz = 1.0e-9 

277 center_freq *= toGHz 

278 resolution *= toGHz 

279 

280 return center_freq, nchan, resolution

283def get_median_elevation(vis: str, antenna_id: int) -> float: 

284 """ 

285 Calculate the median elevation of pointing directions of an antenna. 

286 

287 Calculate the median elevation of pointing directions of an antenna in a 

288 MeasurementSet. The pointing directions are obtained from the DIRECTION 

289 column in POINTING subtable. Only supports DIRECTION column in AZELGEO 

290 coordinate frame and the unit in radian. 

291 

292 Args: 

293 vis: Path to MeasurementSet. 

294 antenna_id: The antenna ID to select. 

295 

296 Returns: 

297 The median of elevation of selected antenna (unit: degree). 

298 Returns 45.0 if DIRECTION is not in AZELGEO or the value is invalid 

299 i.e. not within (0.0, 90.0]. 

300 

301 Raises: 

302 RuntimeError: An error when DIRECTION column has unsupported coodinate 

303 unit. 

304 """ 

305 with casa_tools.TableReader(os.path.join(vis, 'POINTING')) as mytb: 

306 tsel = mytb.query('ANTENNA_ID == {}'.format(antenna_id)) 

307 # default elevation 

308 elevation = 45.0 

309 try: 

310 if tsel.nrows() > 0: 

311 colkeywords = tsel.getcolkeywords('DIRECTION') 

312 if colkeywords['MEASINFO']['Ref'] == 'AZELGEO': 

313 if not (colkeywords['QuantumUnits'] == 'rad').all(): 

314 raise RuntimeError('The unit must be radian. Got {}'.format(str(colkeywords['QuantumUnits']))) 

315 elevation_list = tsel.getcol('DIRECTION')[1][0] 

316 median_elevation = math.degrees(np.median(elevation_list)) 

317 

318 # check if the value is in reasonable range 

319 if 0.0 < median_elevation and median_elevation <= 90.0: 

320 elevation = median_elevation 

321 else: 

322 LOG.attention( 

323 f'Encountered invalid median elevation value, {median_elevation:.4g}deg. ' 

324 f'Use {elevation:.2g}deg instead.' 

325 ) 

326 finally: 

327 tsel.close() 

328 

329 return elevation

332def get_representative_elevation(vis, antenna_id: int) -> float: 

333 """Return representative elevation value computed from phasecenter. 

334 

335 Representative elevation value is defined as the median elevation 

336 of the first science field during the observation. 

337 If there are multiple observations (OBSERVATION rows) in the MS, 

338 start/end times of the observation are determined by the min/max 

339 of time ranges of all observations. 

340 

341 Args: 

342 vis: Path to MeasurementSet. 

343 antenna_id: The antenna ID to select. 

344 

345 Returns: 

346 Elevation value in degree. 

347 """ 

348 myqa = casa_tools.quanta 

349 myme = casa_tools.measures 

350 

351 # collect necessary information using msmd tool 

352 with casa_tools.MSMDReader(vis) as mymsmd: 

353 mposition = mymsmd.antennaposition(antenna_id) 

354 target_field = mymsmd.fieldsforintent('OBSERVE_TARGET*')[0] 

355 time_range = mymsmd.timerangeforobs(0) 

356 for i in range(1, mymsmd.nobservations()): 

357 another_time_range = mymsmd.timerangeforobs(i) 

358 if myqa.lt(another_time_range['begin']['m0'], time_range['begin']['m0']): 

359 time_range['begin'] = another_time_range['begin'] 

360 if myqa.gt(another_time_range['end']['m0'], time_range['end']['m0']): 

361 time_range['end'] = another_time_range['end'] 

362 start_time = myqa.convert(time_range['begin']['m0'], 's')['value'] 

363 end_time = myqa.convert(time_range['end']['m0'], 's')['value'] 

364 mdirection = mymsmd.phasecenter(target_field) 

365 

366 # convert phasecenter to AZEL and return elevation in degree 

367 def _elevation_at(time_in_sec: float) -> float: 

368 myme.done() 

369 mepoch = myme.epoch('UTC', myqa.quantity(time_in_sec, 's')) 

370 myme.doframe(mepoch) 

371 myme.doframe(mposition) 

372 azel = myme.measure(mdirection, 'AZELGEO') 

373 myme.done() 

374 elevation = myqa.convert(azel['m1'], 'deg')['value'] 

375 return elevation 

376 

377 # compute elevation value at 1 min (=60sec) interval 

378 # and take median of computed list of elevations 

379 elevation_list = [_elevation_at(t) for t in np.arange(start_time, end_time, 60)] 

380 median_elevation = np.median(elevation_list) 

381 

382 return median_elevation

385def get_altitude(vis: str) -> float: 

386 """Return observatory-dependent altitude in meter. 

387 

388 This method gets observatory name from MS, and return 

389 appropriate altitude value depending on the observatory 

390 name. 

391 

392 Args: 

393 vis: Path to MeasurementSet. 

394 

395 Returns: 

396 Altitude value in meter. It only supports ALMA, VLA, and NRO. 

397 Otherwise, the method returns default value, 5000m. 

398 """ 

399 with casa_tools.MSMDReader(vis) as mymsmd: 

400 observatory = mymsmd.observatorynames()[0] 

401 

402 if observatory.find('ALMA') != -1 or observatory.find('ACA') != -1: 

403 altitude = 5059 

404 elif observatory.find('VLA') != -1: 

405 altitude = 2124 

406 elif observatory.find('NRO') != -1 or observatory.find('Nobeyama') != -1: 

407 altitude = 1300 

408 else: 

409 # default value is 5000m 

410 altitude = 5000 

411 

412 return altitude

415def get_transmission(vis: str, antenna_id: int = 0, spw_id: int = 0, 

416 doplot: bool = False) -> Tuple[np.ndarray, np.ndarray]: 

417 """ 

418 Calculate atmospheric transmission of an antenna and spectral window. 

419 

420 Calculate the atmospheric transmission of a given spectral window for an 

421 elevation angle corresponding to pointings of a given antenna in a 

422 MeasurementSet. 

423 

424 Args: 

425 vis: Path to MeasurementSet. 

426 spw_id: A spectral window ID. 

427 antenna_id: The antenna ID. 

428 doplot: If True, plot the atmospheric transmission and opacities. 

429 

430 Returns:

433 frequency. 

434 """

440def get_transmission_for_range(vis: str, center_freq: float, nchan: int, resolution: float, antenna_id: int = 0, doplot: bool = False) -> Tuple[np.ndarray, np.ndarray]: 

441 """ 

442 Calculate atmospheric transmission covering a range of frequency. 

443 

444 A number of channels, a resolution of a frequency range, and 

445 the median of elevations in all pointings of the selected antenna 

446 are used to calculate the atmospheric transmission. 

447 The atmospheric profile is constructed by default site parameters 

448 of the function, init_at. 

449 The median of zenith water vapor column (pwv) is used to calculate 

450 the transmission. 

451 

452 Args: 

453 vis: Path to MeasurementSet. 

454 center_freq: Center frequency of the frequency range (unit: GHz). 

455 nchan: Number of channels in the frequency range. 

456 resolution: Frequency resolution (unit: GHz/ch). 

457 antenna_id: The antenna ID. 

458 doplot: If True, plot the atmospheric transmission and opacities. 

459 

460 Returns: 

461 A tuple of 2 arrays. The first one is an array of frequencies in the 

462 unit of GHz, and the other is the atmospheric transmission at each 

463 frequency. 

464 """ 

465 # first try computing elevation value from phasecenter 

466 # if it fails, inspect POINTING table 

467 try: 

468 elevation = get_representative_elevation(vis, antenna_id) 

469 except Exception: 

470 elevation = get_median_elevation(vis, antenna_id) 

471 

472 # get median PWV using Todd's script 

473 if get_table_nrow(vis+'/ASDM_CALATMOSPHERE') or get_table_nrow(vis+'/ASDM_CALWVR'): 

474 # PIPE-2212: Ensure that PWV is checked only if the relevant subtable exists and is not empty. 

475 # This prevents open table tool istance from being left behind by adopted.getMedianPWV(). 

476 pwv, _ = adopted.getMedianPWV(vis=vis) 

477 else: 

478 # fallback to pwv=0.0 

479 pwv = 0.0 

480 

481 altitude = get_altitude(vis) 

482 

483 myat = casa_tools.atmosphere 

484 myqa = casa_tools.quanta 

485 init_at(myat, fcenter=center_freq, nchan=nchan, resolution=resolution, 

486 altitude=altitude) 

487 myat.setUserWH2O(myqa.quantity(pwv, 'mm')) 

488 

489 airmass = calc_airmass(elevation) 

490 dry_opacity = get_dry_opacity(myat) 

491 wet_opacity = get_wet_opacity(myat) 

492 transmission = calc_transmission(airmass, dry_opacity, wet_opacity) 

493 frequency = myqa.convert(myat.getSpectralWindow(0), "GHz")['value'] 

494 

495 if doplot: 

496 plot(frequency, dry_opacity, wet_opacity, transmission) 

497 

498 myat.done() 

499 return frequency, transmission
