381 def _configureFlagRule(self): 

382 """A private method to convert input parameters to FlagRuleDictionary""" 

383 d = {'TsysFlag': (self.flag_tsys, [self.tsys_thresh]), 

384 'RmsPreFitFlag': (self.flag_prfr, [self.prfr_thresh]), 

385 'RmsPostFitFlag': (self.flag_pofr, [self.pofr_thresh]), 

386 'RmsExpectedPreFitFlag': (self.flag_prfre, [self.prfre_thresh]), 

387 'RmsExpectedPostFitFlag': (self.flag_pofre, [self.pofre_thresh]), 

388 'RunMeanPreFitFlag': (self.flag_prfrm, [self.prfrm_thresh, self.prfrm_nmean]), 

389 'RunMeanPostFitFlag': (self.flag_pofrm, [self.pofrm_thresh, self.pofrm_nmean])} 

390 keys = ['Threshold', 'Nmean'] 

391 for k, v in d.items(): 

392 (b, p) = v 

393 if b == True: 

394 self.activateFlagRule(k) 

395 for i in range(len(p)): 

396 self.FlagRuleDictionary[k][keys[i]] = p[i] 

397 elif b == False: 

398 self.deactivateFlagRule(k) 

399 else: 

400 raise RuntimeError("Invalid flag operation definition for %s" % k) 

401 

402 def activateFlagRule(self, key): 

403 """Activates a flag type specified by the input parameter in FlagRuleDictionary""" 

404 if key in self.FlagRuleDictionary: 

405 self.FlagRuleDictionary[key]['isActive'] = True 

406 else: 

407 raise RuntimeError('Error: %s not in predefined Flagging Rules' % key) 

408 

409 def deactivateFlagRule(self, key): 

410 """Deactivates a flag type specified by the input parameter in FlagRuleDictionary""" 

411 if key in self.FlagRuleDictionary: 

412 self.FlagRuleDictionary[key]['isActive'] = False 

413 else: 

414 raise RuntimeError('Error: %s not in predefined Flagging Rules' % key)
