36 @vdp.VisDependentProperty 

37 def intent(self): 

38 # if the spw was set, look to see which intents were observed in that 

39 # spectral window and return the intent based on our order of 

40 # preference: BANDPASS, AMPLITUDE, PHASE 

41 preferred_intents = ('BANDPASS', 'PHASE', 'AMPLITUDE') 

42 if self.spw: 

43 for spw in self.ms.get_spectral_windows(self.spw): 

44 for intent in preferred_intents: 

45 if intent in spw.intents: 

46 if intent != preferred_intents[0]: 

47 LOG.warning('%s spw %s: %s not present, %s used instead' % 

48 (os.path.basename(self.vis), spw.id, 

49 preferred_intents[0], intent)) 

50 return intent 

51 

52 # spw was not set, so look through the spectral windows 

53 for intent in preferred_intents: 

54 for spw in self.ms.spectral_windows: 

55 if intent in spw.intents: 

56 if intent != preferred_intents[0]: 

57 LOG.warning('%s %s: %s not present, %s used instead' % 

58 (os.path.basename(self.vis), spw.id, preferred_intents[0], 

59 intent)) 

60 return intent 

61 

62 # current fallback - return an empty intent 

63 return ''
