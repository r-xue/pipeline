210def get_fluxinfo(ms, fieldnamelist, intent, spwidlist): 

211 """Retrieve the fluxes of selected sources from the pipeline context 

212 as a function of spw id and return the results in a dictionary indexed 

213 by spw id. 

214 

215 The input parameters 

216 ms: The pipeline context ms object 

217 fieldnamelist: The list of field names to be selected 

218 intent: The intent of the fields to be selected 

219 spwlist: The list of spw ids to be selected 

220 

221 The output flux dictionary fluxdict 

222 

223 The flux dictionary key and value 

224 key: The spw id value: The source flux dictionary 

225 

226 The source flux dictionary keys and values 

227 key: 'field_name' value: The name of the source, e.g. '3C286' 

228 key: 'flux' value: The flux of the source in Jy, e.g. 1.53 

229 """ 

230 

231 # Initialize the flux dictionary as an ordered dictionary 

232 fluxdict = collections.OrderedDict() 

233 LOG.info('Finding sources fluxes') 

234 

235 # Loop over the science spectral windows 

236 for spwid in spwidlist: 

237 

238 # Get the spectral window object. 

239 try: 

240 spw = ms.get_spectral_window(spwid) 

241 except KeyError: 

242 continue 

243 

244 # Loop over field names. There is normally only one. 

245 for fieldname in fieldnamelist: 

246 

247 # Get fields associated with the name and intent. 

248 # There should be only one. If there is more 

249 # than one pick the first field. 

250 fields = ms.get_fields(name=fieldname, intent=intent) 

251 if len(fields) <= 0: 

252 continue 

253 field = fields[0] 

254 

255 # Check for flux densities 

256 if len(field.flux_densities) <= 0: 

257 continue 

258 

259 # Find the flux for the spw, sorted by origin 

260 try: 

261 flux = min([fd for fd in field.flux_densities if fd.spw_id == spw.id], 

262 key=lambda f: PREFERRED_ORIGIN_ORDER.get(f.origin, 1e9)) 

263 except ValueError: 

264 # no flux for this spectral window 

265 continue 

266 

267 I, _, _, _ = flux.casa_flux_density 

268 fluxdict[spw.id] = collections.OrderedDict() 

269 fluxdict[spw.id]['field_name'] = fieldname 

270 fluxdict[spw.id]['flux'] = I 

271 LOG.info('Setting flux for field {} spw {} to {:0.4f} Jy'.format(fieldname, spw.id, I)) 

272 

273 return fluxdict

303 def get_scans_for_field_intent(msobj, fieldname_list, scan_intent): 

304 """ 

305 Return a list of scan object that matches an intent and 

306 the list of field names 

307 

308 Inputs: 

309 msobj: Measurementset object 

310 fieldname_list: a list of field names 

311 scan_intent: a Pipeline scan intent string 

312 Retruns: a list of scan objects 

313 """ 

314 # Get the list of unique field names 

315 fieldset = set(fieldname_list) 

316 

317 # Get scans with the intent associated with the field name list 

318 scans = [] 

319 for scan in msobj.get_scans(scan_intent=scan_intent): 

320 # Remove scans not associated with the input field names 

321 scanfieldset = {field.name for field in scan.fields} 

322 if len(fieldset.intersection(scanfieldset)) == 0: 

323 continue 

324 scans.append(scan) 

325 return scans 

326 ###### End: helper function 

327 

328 # Get atmospheric scans associated with the field name list 

329 atmscans = get_scans_for_field_intent(ms, fieldnamelist, 'ATMOSPHERE') 

330 

331 # PIPE-2658: if no atmospheric scans were found for field of current 

332 # intent, then use the Tsyscal heuristic for generating a mapping of 

333 # "intent-to-Tsys-gainfield", and try to use this identify appropriate Tsys 

334 # field and corresponding atmospheric scans. 

335 if not atmscans: 

336 intent_to_tsysfield_map = get_intent_to_tsysfield_map(ms, is_single_dish=False) 

337 tsysfield = intent_to_tsysfield_map.get(intent, '') 

338 # If no match was found, or the Tsys field was set to "nearest", then 

339 # this approach cannot work. 

340 if tsysfield and tsysfield != 'nearest': 

341 atmscans = get_scans_for_field_intent(ms, [tsysfield], 'ATMOSPHERE') 

342 

343 # If no atmospheric scans were found, and the intent specifies a phase 

344 # calibrator or check source, then try to find atmospheric scans associated 

345 # with the science target fields. 

346 if not atmscans and intent in ['CHECK', 'PHASE']: 

347 fieldlist = [f.name for f in ms.get_fields(intent='TARGET')] 

348 if fieldlist: 

349 atmscans = get_scans_for_field_intent(ms, fieldlist, 'ATMOSPHERE') 

350 

351 # If still no atmospheric scans were found, and the intent specifies a 

352 # check source, then try to find atmospheric scans associated with the 

353 # phase calibrator fields. 

354 if not atmscans and intent == 'CHECK': 

355 fieldlist = [f.name for f in ms.get_fields(intent='PHASE')] 

356 if fieldlist: 

357 atmscans = get_scans_for_field_intent(ms, fieldlist, 'ATMOSPHERE') 

358 

359 # Still no atmospheric scans found 

360 # Return 

361 if not atmscans: 

362 return tsysdict 

363 

364 # Get the scans associated with the field name list and intent 

365 obscans = get_scans_for_field_intent(ms, fieldnamelist, intent) 

366 

367 # No data scans found 

368 if not obscans: 

369 return tsysdict 

370 

371 # Loop over the science spws 

372 for spwid in spwidlist: 

373 

374 # Get spectral window 

375 try: 

376 spw = ms.get_spectral_window(spwid) 

377 except: 

378 continue 

379 

380 # Find best atmospheric spw 

381 # This dictionary is created only if the spw id is valid 

382 ftsysdict = collections.OrderedDict() 

383 for atmscan in atmscans: 

384 

385 # Get field name 

386 scanfieldlist = [field.name for field in atmscan.fields] 

387 fieldname = scanfieldlist[0] 

388 

389 # Get tsys spws and spw ids 

390 scanspwlist = [scanspw for scanspw in list(atmscan.spws) 

391 if scanspw.num_channels not in ms.exclude_num_chans] 

392 scanspwidlist = [scanspw.id for scanspw in list(atmscan.spws) 

393 if scanspw.num_channels not in ms.exclude_num_chans] 

394 

395 # Match the Tsys spw to the science spw 

396 # Match first by id then by frequency 

397 bestspwid = None 

398 if spw.id in scanspwidlist: 

399 # bestspwid = scanspw.id 

400 bestspwid = spw.id 

401 else: 

402 mindiff = sys.float_info.max 

403 for scanspw in scanspwlist: 

404 if spw.band != scanspw.band: 

405 continue 

406 if spw.baseband != scanspw.baseband: 

407 continue 

408 diff = abs(spw.centre_frequency.value - scanspw.centre_frequency.value) 

409 if diff < mindiff: 

410 bestspwid = scanspw.id 

411 mindiff = diff 

412 

413 # No spw match found 

414 if bestspwid is None: 

415 continue 

416 

417 # Create dictionary entry based on first scan matched. 

418 ftsysdict['tsys_field_name'] = fieldname 

419 ftsysdict['intent'] = intent 

420 

421 # Pick the first obs scan following the Tsys scan 

422 # This should deal with the shared phase / science target 

423 # scans 

424 for obscan in obscans: 

425 if obscan.id > atmscan.id: 

426 ftsysdict['snr_scan'] = obscan.id 

427 break 

428 

429 # PIPE-1154: if no scan for field name list and intent occurred 

430 # after the Tsys scan, then fall back to picking the most recent 

431 # scan that occurred before the Tsys scan (assuming obscans are 

432 # sorted chronologically). 

433 if 'snr_scan' not in ftsysdict: 

434 for obscan in obscans: 

435 if obscan.id < atmscan.id: 

436 ftsysdict['snr_scan'] = obscan.id 

437 

438 ftsysdict['tsys_scan'] = atmscan.id 

439 ftsysdict['tsys_spw'] = bestspwid 

440 break 

441 

442 # Update the spw dictionary 

443 if ftsysdict: 

444 LOG.info(' Matched spw %d to a Tsys spw %d' % (spwid, bestspwid)) 

445 tsysdict[spwid] = ftsysdict 

446 else: 

447 LOG.warning(' Cannot match spw %d to a Tsys spw in MS %s' % (spwid, ms.basename)) 

448 

449 return tsysdict

478def get_mediantemp(ms, tsys_spwlist, scan_list, antenna='', temptype='tsys'): 

479 """Get median Tsys, Trx, or Tsky temperatures as a function of spw and return 

480 a dictionary 

481 

482 Input parameters 

483 ms: The pipeline measurement set object 

484 tsys_spwlist: The list of Tsys spw ids, e.g. [9,11,13,15] 

485 scan_list: The list of associated observation scan numbers, e.g. [4,8] 

486 antenna: The antenna selectionm '' for all antennas, or a single antenna id or name 

487 temptype: The temperature type 'tsys' (default), 'trx' or 'tsky' 

488 

489 The output dictionary 

490 

491 The median temperature dictionary keys and values 

492 key: the spw id value: The median Tsys temperature in degrees K 

493 """ 

494 # PIPE-775: Output the call to the function. The second and third arguments (tsys_spwlist and scan_list) 

495 # should have the same length 

496 LOG.trace("Called get_mediantemp({}, {}, {}, antenna='{}', temptype='{}')".format( 

497 ms, tsys_spwlist, scan_list, antenna, temptype)) 

498 

499 # Initialize 

500 medtempsdict = collections.OrderedDict() 

501 LOG.info('Estimating Tsys temperatures') 

502 

503 # Temperature type must be one of 'tsys' or 'trx' or 'tsky' 

504 if temptype not in ['tsys', 'trx', 'tsky']: 

505 return medtempsdict 

506 

507 # Get list of unique scan ids. 

508 unique_scans = sorted(set(scan_list)) 

509 

510 # Get the associated spws for each scan id 

511 scans_spws = {scan: [spw.id for spw in ms.get_scans(scan_id=scan)[0].spws] for scan in unique_scans} 

512 LOG.debug("Scan spws: {}".format(scans_spws)) 

513 

514 # Determine the start and end times for each unique scan 

515 begin_scan_times = [] 

516 end_scan_times = [] 

517 for scan in unique_scans: 

518 reqscan = ms.get_scans(scan_id=scan) 

519 if not reqscan: 

520 LOG.warning('Cannot find observation scan %d in MS %s' % (scan, ms.basename)) 

521 return medtempsdict 

522 start_time = reqscan[0].start_time 

523 end_time = reqscan[0].end_time 

524 begin_scan_times.append(start_time) 

525 end_scan_times.append(end_time) 

526 LOG.debug('scan %d start %s end %s' % (scan, start_time, end_time)) 

527 

528 # Get the syscal table meta data. 

529 with casa_tools.TableReader(os.path.join(ms.name, 'SYSCAL')) as table: 

530 

531 # Get the antenna ids 

532 tsys_antennas = table.getcol('ANTENNA_ID') 

533 if len(tsys_antennas) < 1: 

534 LOG.warning('The SYSCAL table is blank in MS %s' % ms.basename) 

535 return medtempsdict 

536 

537 # Get columns and tools needed to understand the tsys times 

538 time_colkeywords = table.getcolkeywords('TIME') 

539 time_unit = time_colkeywords['QuantumUnits'][0] 

540 time_ref = time_colkeywords['MEASINFO']['Ref'] 

541 mt = casa_tools.measures 

542 qt = casa_tools.quanta 

543 

544 # Get time and intervals 

545 tsys_times = table.getcol('TIME') 

546 tsys_intervals = table.getcol('INTERVAL') 

547 

548 # Compute the time range of validity for each tsys measurement 

549 # Worry about memory efficiency later 

550 # PIPE-775: This considers the time to be the central time 

551 tsys_start_times = tsys_times - 0.5 * tsys_intervals 

552 tsys_end_times = tsys_start_times + tsys_intervals 

553 

554 # Get the spw ids 

555 tsys_spws = table.getcol('SPECTRAL_WINDOW_ID') 

556 tsys_uniqueSpws = np.unique(tsys_spws) 

557 

558 # Create a scan id array and populate it with zeros 

559 scanids = np.zeros(len(tsys_start_times), dtype=np.int32) 

560 

561 # Determine if a tsys measurement matches the scan interval 

562 # If it does set the scan to the scan id 

563 nmatch = 0

583 LOG.warning('No SYSCAL table row matches for scans %s tsys spws %s in MS %s' % 

584 (unique_scans, tsys_spwlist, ms.basename))

586 else: 

587 LOG.info(' SYSCAL table row matches for scans %s Tsys spws %s %d / %d' %

597 for spw, scan in zip(tsys_spwlist, scan_list): 

598 

599 # If no Tsys data skip to the next window 

600 if spw not in tsys_uniqueSpws: 

601 LOG.warning('Tsys spw %d is not in the SYSCAL table for MS %s' % 

602 (spw, ms.basename)) 

603 continue 

604 # return medtempsdict

606 # Loop over the rows

608 with casa_tools.TableReader(os.path.join(ms.name, 'SYSCAL')) as table:

626 LOG.info(" Median Tsys %s value for Tsys spw %2d = %.1f K" % (temptype, spw, medtempsdict[spw]))

628 LOG.warning(' No Tsys data for spw %d scan %d in MS %s' % (spw, scan, ms.basename))

630 # Return median temperature per spw and scan. 

631 return medtempsdict

954def compute_gaincalsnr(ms, spwlist, spw_dict, intent, edge_fraction): 

955 """Compute the gain to signal-to-noise given the spw list and the spw 

956 dictionary. 

957 

958 This code assumes that the science spws are observed in both the 

959 calibrator and the science target. 

960 

961 Args: 

962 ms: The pipeline context MeasurementSet object. 

963 spwlist: List of spw ids. 

964 spw_dict: Spw dictionary (join from spwid, tsys, flux, temp, obs dicts). 

965 intent: Intent of source. 

966 edge_fraction: Fraction of the edge that is flagged. 

967 

968 Returns: 

969 The output SNR dictionary containing: 

970 key: spw id (int) value: the pre-averaging parameter dictionary for spwid. 

971 

972 The pre-averaging parameter dictionary keys and values: 

973 key: 'band' value: The ALMA receiver band 

974 key: 'frequency_Hz' value: The frequency of the spw in Hz 

975 key: 'bandwidth' value: The bandwidth of the spw in Hz 

976 key: 'nchan_total' value: The total number of channels 

977 key: 'chanwidth_Hz' value: The median channel width in Hz 

978 

979 key: 'tsys_spw' value: The tsys spw id as an integer 

980 key: 'median_tsys' value: The median tsys value 

981 

982 key: 'flux_Jy' value: The flux of the source in Jy 

983 key: 'scantime_minutes' value: The (scan) exposure time in minutes 

984 key: 'inttime_minutes' value: The integration time in minutes 

985 key: 'sensitivity_per_scan_mJy' value: The sensitivity per scan in mJy 

986 key: 'sensitivity_per_int_mJy' value: The sensitivity per integration in mJy 

987 key: 'snr_per_scan' value: The snr per scan 

988 key: 'snr_per_int' value: The snr per integration 

989 """ 

990 # Initialize the output solution interval dictionary 

991 snr_dict = collections.OrderedDict() 

992 

993 maxEffectiveBW = 2.0e9 * (1.0 - 2.0 * edge_fraction) 

994 

995 # PIPE-788: retrieve the fraction of flagged data 

996 scans = set(np.hstack([spw_dict[spwid]['snr_scans'] for spwid in spwlist])) 

997 flag_task = casa_tasks.flagdata(vis=ms.name, scan=','.join(map(str, list(scans))), mode='summary') 

998 flag_result = flag_task.execute() 

999 

1000 # PIPE-2499: log whether combine and solint will be set based on integration 

1001 # time or scan time. 

1002 if intent in {'AMPLITUDE', 'BANDPASS', 'DIFFGAINREF', 'DIFFGAINSRC'}: 

1003 LOG.info(f"{ms.basename}: for intent {intent}, the integration time {spw_dict[spwlist[0]]['integrationtime']:.2f}" 

1004 f" (min) will govern combine and solint calculations.") 

1005 else: 

1006 # The exptime is added across all scans for given intent, so dividing by 

1007 # nr. of scans to report scan time. 

1008 LOG.info(f"{ms.basename}: for intent {intent}, the scan time" 

1009 f" {spw_dict[spwlist[0]]['exptime'] / len(spw_dict[spwlist[0]]['snr_scans']):.2f} (min) will govern" 

1010 f" combine and solint calculations.") 

1011 

1012 for spwid in spwlist: 

1013 # Add entry to output SNR dictionary for spw. 

1014 snr_dict[spwid] = collections.OrderedDict() 

1015 

1016 # Science spw info 

1017 # Channel information probably not required 

1018 snr_dict[spwid]['band'] = spw_dict[spwid]['band'] 

1019 snr_dict[spwid]['frequency_Hz'] = spw_dict[spwid]['bandcenter'] 

1020 snr_dict[spwid]['bandwidth'] = spw_dict[spwid]['bandwidth'] 

1021 snr_dict[spwid]['nchan_total'] = spw_dict[spwid]['nchan'] 

1022 snr_dict[spwid]['chanwidth_Hz'] = spw_dict[spwid]['chanwidths'] 

1023 

1024 # Tsys spw info 

1025 snr_dict[spwid]['tsys_spw'] = spw_dict[spwid]['tsys_spw'] 

1026 snr_dict[spwid]['median_tsys'] = spw_dict[spwid]['median_tsys'] 

1027 

1028 # Info on integration time and exposure time. 

1029 snr_dict[spwid]['inttime_minutes'] = spw_dict[spwid]['integrationtime'] 

1030 snr_dict[spwid]['scantime_minutes'] = spw_dict[spwid]['exptime'] / len(spw_dict[spwid]['snr_scans']) 

1031 

1032 # Determine the receiver band 

1033 bandidx = ALMA_BANDS.index(spw_dict[spwid]['band']) 

1034 

1035 # Compute the various generic SNR factors 

1036 if spw_dict[spwid]['median_tsys'] <= 0.0: 

1037 relativeTsys = 1.0 

1038 LOG.info('Spw %d Tsys <= 0K in MS; %s assuming nominal Tsys for SNR calculation', spwid, ms.basename) 

1039 else: 

1040 relativeTsys = spw_dict[spwid]['median_tsys'] / ALMA_TSYS[bandidx] 

1041 nbaselines = spw_dict[spwid]['num_7mantenna'] + spw_dict[spwid]['num_12mantenna'] - 1 

1042 arraySizeFactor = np.sqrt(ALMA_FIDUCIAL_NUM_ANTENNAS * (ALMA_FIDUCIAL_NUM_ANTENNAS-1) / 2.0 / nbaselines) 

1043 if spw_dict[spwid]['num_7mantenna'] == 0: 

1044 areaFactor = 1.0 

1045 elif spw_dict[spwid]['num_12mantenna'] == 0: 

1046 areaFactor = (12.0 / 7.0) ** 2 

1047 else: 

1048 # general case: eq. 6 in arXiv:2306.07420 

1049 areaFactor = (spw_dict[spwid]['num_12mantenna'] + spw_dict[spwid]['num_7mantenna'] * (12./7)**2) / \ 

1050 (spw_dict[spwid]['num_12mantenna'] + spw_dict[spwid]['num_7mantenna']) 

1051 polarizationFactor = np.sqrt(2.0) 

1052 

1053 # SNR computation 

1054 # Compute common bandwidth factor. 

1055 bandwidthFactor = np.sqrt(ALMA_FIDUCIAL_BANDWIDTH / min(spw_dict[spwid]['bandwidth'], maxEffectiveBW)) 

1056 # PIPE-788: multiply the exposure time by the fraction of unflagged data 

1057 flagFactor = 1.0 / np.sqrt(1 - flag_result['spw'][str(spwid)]['flagged'] / flag_result['spw'][str(spwid)]['total']) 

1058 

1059 # Compute sensitivity scaled by integration time and add to output dict. 

1060 timeFactorInt = np.sqrt(ALMA_FIDUCIAL_EXP_TIME / spw_dict[spwid]['integrationtime']) 

1061 factorInt = relativeTsys * timeFactorInt * arraySizeFactor * \ 

1062 areaFactor * bandwidthFactor * polarizationFactor * flagFactor 

1063 sensitivityInt = ALMA_SENSITIVITIES[bandidx] * factorInt 

1064 snr_dict[spwid]['sensitivity_per_int_mJy'] = sensitivityInt 

1065 

1066 # Compute sensitivity scaled by scan time and add to output dict. 

1067 timeFactorScan = np.sqrt(ALMA_FIDUCIAL_EXP_TIME / (spw_dict[spwid]['exptime'] / len(spw_dict[spwid]['snr_scans']))) 

1068 factorScan = relativeTsys * timeFactorScan * arraySizeFactor * \ 

1069 areaFactor * bandwidthFactor * polarizationFactor * flagFactor 

1070 sensitivityScan = ALMA_SENSITIVITIES[bandidx] * factorScan 

1071 snr_dict[spwid]['sensitivity_per_scan_mJy'] = sensitivityScan 

1072 

1073 # If valid flux info is available for the current SpW, then store this 

1074 # and corresponding SNR info. 

1075 if 'flux' in spw_dict[spwid]: 

1076 # Store flux in Jansky 

1077 snr_dict[spwid]['flux_Jy'] = spw_dict[spwid]['flux'] 

1078 # SNR info, integration-time based and scan-based; convert flux to mJy 

1079 flux_mJy = spw_dict[spwid]['flux'] * 1000.0 

1080 snr_dict[spwid]['snr_per_int'] = flux_mJy / sensitivityInt 

1081 snr_dict[spwid]['snr_per_scan'] = flux_mJy / sensitivityScan 

1082 

1083 _log_sensitivity_info(spwid, snr_dict, 'int', has_flux=True) 

1084 _log_sensitivity_info(spwid, snr_dict, 'scan', has_flux=True) 

1085 else: 

1086 snr_dict[spwid]['flux_Jy'] = None 

1087 snr_dict[spwid]['snr_per_int'] = None 

1088 snr_dict[spwid]['snr_per_scan'] = None 

1089 

1090 _log_sensitivity_info(spwid, snr_dict, 'int', has_flux=False) 

1091 _log_sensitivity_info(spwid, snr_dict, 'scan', has_flux=False) 

1092 

1093 return snr_dict

1144 """Compute the optimal bandpass frequency solution intervals given the spw list 

1145 and the spw dictionary. 

1146 

1147 The input parameters 

1148 spwlist The list of spw ids 

1149 spw_dict The spw dictionary 

1150 reqPhaseupSnr The requested phaseup SNR 

1151 minBpNintervals The minimum number of phase up time intervals, e.g. 2 

1152 reqBpSnr The requested bandpass SNR 

1153 minBpNchan The minimum number of bandpass channel solutions, e.g. 8 

1154 

1155 The output solution interval dictionary keys and values - passed by compute_bpsolint:

1164

1171 key: 'sensitivity_per_integration_mJy' value: The sensitivity per integration in mJy 

1172 key: 'snr_per_channel' value: The signal-to-noise per channel 

1173 key: 'sensitivity_per_channel_mJy' value: The sensitivity in mJy per channel 

1174 

1175 key: 'bpsolint' value: The frequency solint in MHz 

1176 key: 'nchan_bpsolint' value: The total number of solint channels

1184 if evenbpsolints:

1205 

1206 # PIPE-408: do not continue if there are no unflagged baselines for current spw; this will cause this spw 

1207 # to be absent from the solution interval dictionary that is returned. 

1208 if nbaselines < 0: 

1209 LOG.warning("Cannot compute optimal bandpass frequency solution interval for spw {} in MS {}; no (unflagged)"

1222 ntotant

1227 pubandwidthFactor = np.sqrt(8.0e9 / spw_dict[spwid]['bandwidth']) 

1228 pufactor = relativeTsys * putimeFactor * arraySizeFactor * \ 

1229 areaFactor * pubandwidthFactor * polarizationFactor 

1230 pusensitivity = ALMA_SENSITIVITIES[bandidx] * pufactor 

1231 snrPerIntegration = spw_dict[spwid]['flux'] * 1000.0 / pusensitivity

1279 tooFewIntervals = True 

1280 asterisks = '***' 

1281 else: 

1282 tooFewIntervals = False

1286 spwid,

1288 solint_dict[spwid]['phaseup_solint'], 

1289 solInts, 

1290 reqPhaseupSnr)) 

1291 solint_dict[spwid]['nphaseup_solutions'] = solInts 

1292 if tooFewIntervals: 

1293 LOG.warning('%s Spw %d would have fewer than %d time intervals in its solution in MS %s' % 

1294 (asterisks, spwid, minBpNintervals, ms.basename))

1296 # Bandpass solution 

1297 # Determine frequency interval in MHz 

1298 # 

1299 # Get number of channels. 

1300 if requiredChannels > 1.0: 

1301 if evenbpsolints: 

1302 nchan = evenChannels 

1303 else:

1305 else: 

1306 nchan = 1

1310 # * the frequency interval is passed with limited precision (typically 

1311 # in MHz with 6 decimals, i.e. a precision of Hz)

1346 spwid, 

1347 solint_dict[spwid]['bandwidth']*1.0e-6, 

1348 solint_dict[spwid]['bpsolint'], 

1349 solChannels, 

1350 reqBpSnr)) 

1351 solint_dict[spwid]['nbandpass_solutions'] = solChannels
