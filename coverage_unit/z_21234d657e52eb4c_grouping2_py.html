16class GroupByPosition2(api.Heuristic): 

17 """Grouping by RA/DEC position.""" 

18 

19 def calculate(self, ra: np.ndarray, dec: np.ndarray, r_combine: Angle, r_allowance: Angle) -> Tuple[Dict, List]: 

20 """Group data by RA/DEC position. 

21 

22 Divides data into groups by their positions in two 

23 dimensional space which are given by ra and dec. 

24 Groups data within the circle with the radius of 

25 r_combine. Only position difference larger than 

26 r_allowance are regarded as significant. 

27 For r_combine and r_allowance, specified values are 

28 interpreted to be in degree unless their units are 

29 explicitly given. 

30 

31 Args: 

32 ra: 

33 List of R.A. 

34 dec: 

35 List of DEC. 

36 r_combine: 

37 Data inside r_combine will be grouped together. 

38 r_allowance: 

39 Data inside r_allowance are assumed to be the same position. 

40 

41 Returns: 

42 Two-tuple containing information on group membership 

43 (PosDict) and boundaries between groups (PosGap). 

44 

45 PosDict is a dictionary whose keys are indices for 

46 ra and dec. Values of PosDict are the list which 

47 contains different value depending on whether the 

48 position specified by the index is reference data 

49 for the group or not. If k is reference index, 

50 PosDict[k] lists indices for group member 

51 ([ID1, ID2,..., IDN]). Otherwise, PosDict[k] is 

52 [-1, m] where m is the index to reference data. 

53 

54 PosGap is a list of gaps in terms of array indices 

55 for ra and dec ([IDX1, IDX2,...,IDXN]). Length of 

56 PosGap is (number of groups) - 1. 

57 """ 

58 qa = casa_tools.quanta 

59 if isinstance(r_combine, dict): 

60 # r_combine should be quantity 

61 CombineRadius = qa.convert(r_combine, 'deg')['value'] 

62 else: 

63 CombineRadius = r_combine 

64 if isinstance(r_allowance, dict): 

65 # r_allowance should be quantity 

66 AllowanceRadius = qa.convert(r_allowance, 'deg')['value'] 

67 else: 

68 AllowanceRadius = r_allowance 

69 

70 ThresholdAR = AllowanceRadius * AllowanceRadius 

71 CombineDiameter = 2.0 * CombineRadius 

72 

73 # 2009/2/10 Quicker Method 

74 Nrows = len(ra) 

75 PosDict = {} 

76 SelectDict = {} 

77 MinRA = ra.min() 

78 MinDEC = dec.min() 

79 # Calculate the lattice position (sRA, sDEC) for each pointings 

80 # Create list of pointings (dictionary) for each lattice position 

81 for x in range(Nrows): 

82 sRA = int((ra[x] - MinRA) / CombineDiameter) 

83 sDEC = int((dec[x] - MinDEC) / CombineDiameter) 

84 if sRA not in SelectDict: 

85 SelectDict[sRA] = {} 

86 if sDEC not in SelectDict[sRA]: 

87 SelectDict[sRA][sDEC] = [x] 

88 else: 

89 SelectDict[sRA][sDEC].append(x) 

90 

91 # Create PosDict 

92 # make a list of spectra inside each lattice grid 

93 # store the list in PosDict[i] where 'i' is the smallest row number in the list 

94 # Other spectra have a reference to 'i' 

95 LOG.debug('SelectDict.keys() : %s' % list(SelectDict.keys())) 

96 for sRA in SelectDict: 

97 LOG.debug('len(SelectDict[%s].keys()) : %s' % (sRA, len(list(SelectDict[sRA].keys())))) 

98 for sDEC in SelectDict[sRA]: 

99 PosDict[SelectDict[sRA][sDEC][0]] = SelectDict[sRA][sDEC] 

100 if len(SelectDict[sRA][sDEC]) != 1: 

101 for x in SelectDict[sRA][sDEC][1:]: 

102 PosDict[x] = [-1, SelectDict[sRA][sDEC][0]] 

103 del SelectDict 

104 

105 # Calculate thresholds to determine gaps 

106 DeltaP = np.sqrt((ra[1:] - ra[:-1])**2.0 + (dec[1:] - dec[:-1])**2.0) 

107 DeltaQ = np.take(DeltaP, np.nonzero(DeltaP > ThresholdAR)[0]) 

108 if len(DeltaQ) != 0: 

109 ThresholdP = np.median(DeltaQ) * 10.0 

110 else: 

111 ThresholdP = 0.0 

112 LOG.info('threshold:%s deg' % ThresholdP) 

113 

114 # List rows which distance from the previous position is larger than the threshold 

115 PosGap = [] 

116 for x in range(1, Nrows): 

117 if DeltaP[x - 1] > ThresholdP: 

118 PosGap.append(x) 

119 LOG.info('Position Gap %s deg at row=%d' % (DeltaP[x-1], x)) 

120 

121 # Print information 

122 if len(PosGap) == 0: 

123 PosGapMsg = 'Found no position gap' 

124 else: 

125 PosGapMsg = 'Found %d position gap(s)' % len(PosGap) 

126 LOG.info(PosGapMsg) 

127 

128 #print '\nPosGap', PosGap 

129 #print '\nPosDict', PosDict 

130 return PosDict, PosGap

136 def calculate(self, timebase: Sequence[Real], time_diff: Sequence[Real]) -> Tuple[List, List]: 

137 """Group data by time sequence. 

138 

139 Divides data into groups by their difference (time_diff). 

140 Two groups are defined based on "small" and "large" gaps, 

141 which are internally computed by ThresholdForGroupByTime 

142 heuristic. The time_diff is generated from timebase in 

143 most of the cases. The timebase contains all time stamps 

144 and time_diff is created from selected time stamps in 

145 other case. 

146 

147 Args: 

148 timebase: 

149 base list of time stamps for threshold estimation 

150 time_diff: 

151 difference from the previous time stamp 

152 

153 Returns: 

154 Two-tuple containing information on group membership 

155 (TimeTable) and boundaries between groups (TimeGap). 

156 

157 TimeTable is the "list-of-list" whose items are the set 

158 of indices for each group. TimeTable[0] is the groups 

159 separaged by "small" gap while TimeTable[1] is for 

160 groups separated by "large" gap. They are used for 

161 baseline subtraction (hsd_baseline) and subsequent 

162 flagging (hsd_blflag). 

163 

164 TimeTable: 

165 [[[ismall00,...,ismall0M],[...],...,[ismallX0,...,ismallXN]], 

166 [[ilarge00,...,ilarge0P],[...],...,[ilargeY0,...,ilargeYQ]]] 

167 TimeTable[0]: separated by small gaps 

168 TimeTable[1]: separated by large gaps 

169 

170 TimeGap is the list of indices which indicate boundaries 

171 for "small" and "large" gaps. These are used for plotting. 

172 

173 TimeGap: [[rowX1, rowX2,...,rowXN], [rowY1, rowY2,...,rowYN]] 

174 TimeGap[0]: small gap 

175 TimeGap[1]: large gap 

176 """ 

177 LOG.info('Grouping by Time...') 

178 

179 # Threshold for grouping 

180 h = ThresholdForGroupByTime() 

181 (Threshold1, Threshold2) = h(timebase) 

182 

183 TimeTable = [[], []] 

184 SubTable1 = [0] 

185 SubTable2 = [0] 

186 TimeGap = [[], []] 

187 

188 # Detect small and large time gaps 

189 for index in range(len(time_diff)): 

190 indexp1 = index + 1 

191 if time_diff[index] <= Threshold1: 

192 SubTable1.append(indexp1) 

193 else: 

194 TimeTable[0].append(SubTable1) 

195 SubTable1 = [indexp1] 

196 TimeGap[0].append(indexp1) 

197 LOG.info('Small Time Gap: %s sec at row=%d' % (time_diff[index], indexp1)) 

198 if time_diff[index] <= Threshold2: 

199 SubTable2.append(indexp1) 

200 else: 

201 TimeTable[1].append(SubTable2) 

202 SubTable2 = [indexp1] 

203 TimeGap[1].append(indexp1) 

204 LOG.info('Large Time Gap: %s sec at row=%d' % (time_diff[index], indexp1)) 

205 

206 if len(SubTable1) > 0: TimeTable[0].append(SubTable1) 

207 if len(SubTable2) > 0: TimeTable[1].append(SubTable2) 

208 del SubTable1, SubTable2 

209 

210 # print information 

211 if len(TimeGap[0]) == 0: 211 ↛ 212line 211 didn't jump to line 212 because the condition on line 211 was never true

212 TimeGapMsg = 'Found no time gap' 

213 LOG.info(TimeGapMsg) 

214 else: 

215 TimeGapMsg1 = 'Found %d small time gap(s)' % len(TimeGap[0]) 

216 TimeGapMsg2 = 'Found %d large time gap(s)' % len(TimeGap[1]) 

217 LOG.info(TimeGapMsg1) 

218 LOG.info(TimeGapMsg2) 

219 

220 #print '\nTimeGap', TimeGap 

221 #print '\nTimeTable', TimeTable 

222 return TimeTable, TimeGap

227class ThresholdForGroupByTime(api.Heuristic): 

228 """Estimate thresholds for large and small time gaps.""" 

229 

230 def calculate(self, timebase: Sequence[Real]) -> Tuple[List, List]: 

231 """Estimate thresholds for large and small time gaps. 

232 

233 Estimate thresholds for large and small time gaps using 

234 base list of time stamps. Threshold for small time gap, 

235 denoted as Threshold1, is computed from a median value 

236 of nonzero time differences multiplied by five, i.e., 

237 

238 dt = timebase[1:] - timebase[:-1] 

239 Threhold1 = 5 * np.median(dt[dt != 0]) 

240 

241 where timebase is assumed to be np.ndarray. Threshold 

242 for large time gap, denoted as Threshold2, is computed 

243 from a median value of time differences larger than 

244 Threshold1 mutiplied by five, i.e., 

245 

246 Threshold2 = 5 * np.median( 

247 dt[np.logical_and(dt != 0, dt > Threshold1)] 

248 ) 

249 

250 Args: 

251 timebase: 

252 base list of time stamps for threshold estimation 

253 

254 Returns: 

255 Two-tuple of threshold values for small and large 

256 time gaps, respectively. 

257 """ 

258 ArrayTime = np.array(timebase) 

259 

260 # 2009/2/5 adapted for multi beam which assumes to have identical time stamps 

261 # identical time stamps are rejected before determining thresholds 

262 # DeltaT: difference from the previous time stamp 

263 DeltaT = ArrayTime[1:] - ArrayTime[:-1] 

264 DeltaT1 = np.take(DeltaT, np.nonzero(DeltaT)[0]) 

265 Threshold1 = np.median(DeltaT1) * 5.0 

266 DeltaT2 = np.take(DeltaT1, np.nonzero(DeltaT1 > Threshold1)[0]) 

267 if len(DeltaT2) > 0: 267 ↛ 270line 267 didn't jump to line 270 because the condition on line 267 was always true

268 Threshold2 = np.median(DeltaT2) * 5.0 

269 else: 

270 Threshold2 = Threshold1 

271 

272 # print information 

273 LOG.info('Threshold1 = %s sec' % Threshold1) 

274 LOG.info('Threshold2 = %s sec' % Threshold2) 

275 LOG.info('MaxDeltaT = %s sec' % DeltaT1.max()) 

276 LOG.info('MinDeltaT = %s sec' % DeltaT1.min()) 

277 

278 return Threshold1, Threshold2
