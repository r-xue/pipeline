99 def plot(self, spw_ids=None, antenna_ids=None, antenna_names=None): 

100 # these function arguments are used for debugging the plot routines, 

101 # so we can call plot for a particular troublesome spw/antenna. 

102 ms = self._ms 

103 plothelper = self._plothelper 

104 

105 # get the spw IDs common to all caltables.. 

106 all_spw_ids = [set(data.spw) for _, data in self.data] 

107 common_spw_ids = reduce(lambda x, y: x.intersection(y), all_spw_ids) 

108 # .. filter to match those specified as function arguments.. 

109 if spw_ids is not None: 

110 common_spw_ids = common_spw_ids.intersection(set(spw_ids)) 

111 # .. before converting to domain objects 

112 spws = [spw for spw in ms.spectral_windows 

113 if spw.id in common_spw_ids] 

114 

115 # Do the same for antenna IDs, finding those common to all caltables.. 

116 all_antenna_ids = [set(data.antenna) for _, data in self.data] 

117 common_antenna_ids = reduce(lambda x, y: x.intersection(y), all_antenna_ids) 

118 # .. filtering to match those specified as function arguments.. 

119 if antenna_ids is not None: 

120 common_antenna_ids = common_antenna_ids.intersection(set(antenna_ids)) 

121 if antenna_names is not None: 

122 named_antenna_ids = {ant.id for ant in ms.antennas if ant.name in antenna_names} 

123 common_antenna_ids = common_antenna_ids.intersection(named_antenna_ids) 

124 # .. before converting to domain objects 

125 antennas = [ant for ant in ms.antennas 

126 if ant.id in common_antenna_ids] 

127 

128 assert len(spws) > 0, 'No common spws to plot in %s' % utils.commafy([d.filename for d in self.data]) 

129 assert len(antennas) > 0, 'No common antennas to plot in %s' % utils.commafy([d.filename for d in self.data]) 

130 

131 plots = [] 

132 for spw in spws: 

133 for antenna_group in plothelper.group_antennas(antennas): 

134 plots.append(self.get_plot_wrapper(spw, self._scans, antenna_group)) 

135 

136 return [p for p in plots if p is not None]

360 annotation='FLAGGED'): 

361 """ 

362 Plot flagged data. 

363 

364 data -- the CaltableWrapper for the data selection

368 idx_start = max(masked_slice.start-1, 0)

381 rect = matplotlib.patches.Rectangle((start, 0), width=width, height=1, 

382 transform=trans, color='#EEEEEE', 

383 alpha=0.2) 

384 

385 axis.add_patch(rect) 

386 

387 if annotate: 

388 axis.text(start + width/2, 0.5, annotation, color='k', transform=trans, 

389 size=9, ha='center', va='center', rotation=90)
