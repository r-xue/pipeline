102 def prepare(self) -> resultobjects.TsyscalResults: 

103 inputs = self.inputs 

104 

105 # make a note of the current inputs state before we start fiddling 

106 # with it. This origin will be attached to the final CalApplication. 

107 origin = callibrary.CalAppOrigin(task=Tsyscal, inputs=inputs.to_casa_args()) 

108 

109 # construct the Tsys cal file 

110 gencal_args = inputs.to_casa_args() 

111 gencal_job = casa_tasks.gencal(caltype='tsys', **gencal_args) 

112 self._executor.execute(gencal_job) 

113 

114 tsys_table = gencal_args['caltable'] 

115 

116 LOG.todo('tsysspwmap heuristic re-reads measurement set!') 

117 LOG.todo('tsysspwmap heuristic won\'t handle missing file') 

118 nospwmap, spwmap = tsysspwmap(ms=inputs.ms, tsystable=tsys_table, channel_tolerance=inputs.chantol) 

119 

120 calfrom_defaults = dict(caltype='tsys', spwmap=spwmap, interp='linear,linear') 

121 

122 is_single_dish = utils.contains_single_dish(inputs.context) 

123 calapps = get_calapplications(inputs.ms, tsys_table, calfrom_defaults, origin, spwmap, is_single_dish) 

124 

125 return resultobjects.TsyscalResults(pool=calapps, unmappedspws=nospwmap)

127 def analyse(self, result: resultobjects.TsyscalResults) -> resultobjects.TsyscalResults: 

128 # double-check that the caltable was actually generated 

129 on_disk = [ca for ca in result.pool if ca.exists()] 

130 result.final[:] = on_disk

132 missing = [ca for ca in result.pool if ca not in on_disk] 

133 result.error.clear()

146def get_calapplications(ms: MeasurementSet, tsys_table: str, calfrom_defaults: dict, origin: callibrary.CalAppOrigin, 

147 spw_map: list, is_single_dish: bool) -> list[callibrary.CalApplication]: 

148 """ 

149 Get a list of CalApplications that apply a Tsys caltable to a measurement 

150 set using the gainfield mapping defined in CAS-12213. 

151 

152 Note: this function only provides the gainfield argument for the CalFrom 

153 constructor. Any other required CalFrom constructor arguments should be 

154 provided to this function via the calfrom_defaults parameter. 

155 

156 Args: 

157 ms: MeasurementSet to apply calibrations to. 

158 tsys_table: name of Tsys table. 

159 calfrom_defaults: dict of CalFrom constructor arguments. 

160 origin: CalOrigin for the created CalApplications. 

161 spw_map: Tsys SpW map. 

162 is_single_dish: boolean declaring if current MS is for Single-Dish. 

163 

164 Returns: 

165 List of CalApplications. 

166 """ 

167 # Get the map of intent:gainfield 

168 soln_map = get_intent_to_tsysfield_map(ms, is_single_dish) 

169 

170 # Create the static dict of calfrom arguments. Only the 'gainfield' argument changes from calapp to calapp; the 

171 # other arguments remain unchanged. 

172 calfrom_args = dict(calfrom_defaults) 

173 calfrom_args['gaintable'] = tsys_table 

174 

175 # get the mapping of field ID to unambiguous identifier for more user friendly logs 

176 field_id_to_identifier = utils.get_field_identifiers(ms) 

177 

178 # create a domain object mapping of science spw to Tsys spw 

179 domain_spw_map = {ms.spectral_windows[i]: ms.spectral_windows[j] for i, j in enumerate(spw_map)} 

180 

181 # Now loop through the MS intents, creating a specific Tsys registration for each intent. 

182 calapps = [] 

183 for intent in ms.intents: 

184 # get the preferred Tsys gainfield for this intent, falling back to 'nearest' if not specified 

185 gainfield = soln_map.get(intent, 'nearest') 

186 

187 # The CASA callibrary cannot handle registrations with multiple fldmap fields, e.g., fldmap='1,2'. 

188 if ',' in gainfield: 

189 LOG.info('Calculating workaround for CASA callibrary fldmap incompatibility: ' 

190 '{} intent -> fldmap={!r}'.format(intent, gainfield)) 

191 # get the fields for this non-Tsys intent 

192 fields_with_intent = ms.get_fields(intent=intent) 

193 # get the Tsys fields that were to be applied in the fldmap 

194 tsys_fields = ms.get_fields(task_arg=gainfield) 

195 

196 # holds mapping of field,spw -> Tsys field 

197 field_to_tsys_field = collections.defaultdict(dict) 

198 

199 # For the fields for the current non-Tsys intent, we'll emulate the 

200 # CASA gainfield='nearest' option by selecting the spatially 

201 # closest Tsys field with the same tuning. 

202 me = casa_tools.measures 

203 qa = casa_tools.quanta 

204 for non_tsys_field in fields_with_intent: 

205 non_tsys_direction = non_tsys_field.mdirection 

206 

207 # For multi-tuning EBs, the spws may need different gainfield arguments. For example, in 

208 # uid://A002/Xcf3a9c/X3a3d, the phase cal has no Tsys scans and was observed with three different 

209 # tunings. Each tuning is distinct and was used for a different target field. Each target field *does* 

210 # have a Tsys scan. Due to the lack of Tsys scans on the phase cal, the target fields are selected as 

211 # the fallback gainfields (e.g., # gainfield='A,B,C'), which due to the CASA callibrary 

212 # limitation we now need to boil down to the closest target that was observed using the same 

213 # spectral setup. 

214 

215 # get the spws used to observe with intent XXX for this non-Tsys field 

216 non_tsys_scans = ms.get_scans(scan_intent=intent, field=non_tsys_field.id) 

217 non_tsys_spws = {dd.spw for scan in non_tsys_scans for dd in scan.data_descriptions} 

218 # filter out non-science windows, leaving just the spws for this non-Tsys field 

219 non_tsys_spws = [spw for spw in non_tsys_spws if spw.type in ('TDM', 'FDM')] 

220 

221 for non_tsys_spw in sorted(non_tsys_spws, key=attrgetter('id')): 

222 # check to see if there's data for the field/intent/spw combination 

223 scans = ms.get_scans(scan_intent=intent, spw=non_tsys_spw.id, field=non_tsys_field.id) 

224 if not scans: 

225 # If there's no data, there's no calibration required. 

226 continue 

227 

228 # for each spw used for the intent=XXX scan, find the Tsys field that was observed using the Tsys 

229 # spw which is mapped to this science spw 

230 tsys_spw_for_spw = domain_spw_map[non_tsys_spw] 

231 tsys_fields_with_required_spw = [f for f in tsys_fields if tsys_spw_for_spw in f.valid_spws] 

232 

233 LOG.debug('Candidate Tsys fields for spw {}: {}' 

234 ''.format(non_tsys_spw.id, ','.join(f.name for f in tsys_fields_with_required_spw))) 

235 

236 separations = [] 

237 for tsys_field in tsys_fields_with_required_spw: 

238 tsys_direction = tsys_field.mdirection 

239 separation = me.separation(tsys_direction, non_tsys_direction) 

240 separation_degs = qa.getvalue(qa.convert(separation, 'deg'))[0] 

241 separations.append((tsys_field, separation_degs)) 

242 

243 if not separations: 

244 msg = ('Failed Tsys calibration for {} spw {}: no Tsys scan with the same tuning identified' 

245 ''.format(non_tsys_field.name, non_tsys_spw.id)) 

246 LOG.error(msg) 

247 continue 

248 

249 closest = min(separations, key=itemgetter(1))[0] 

250 LOG.info('Tying {} field #{} spw #{} to closest Tsys field {} spw #{}'.format( 

251 intent, field_id_to_identifier[non_tsys_field.id], non_tsys_spw.id, 

252 field_id_to_identifier[closest.id], tsys_spw_for_spw.id) 

253 ) 

254 field_to_tsys_field[non_tsys_field][non_tsys_spw.id] = closest 

255 

256 # create a CalTo specifically for the intent fields with their selected Tsys field 

257 for non_tsys_field, spw_to_tsys_field in field_to_tsys_field.items(): 

258 field_arg = field_id_to_identifier[non_tsys_field.id] 

259 for spw, tsys_field in spw_to_tsys_field.items(): 

260 gainfield_arg = field_id_to_identifier[tsys_field.id] 

261 calto = callibrary.CalTo(vis=ms.name, intent=intent, spw=spw, field=field_arg) 

262 calfrom = callibrary.CalFrom(gainfield=gainfield_arg, **calfrom_args) 

263 calapp = callibrary.CalApplication(calto, calfrom, origin) 

264 calapps.append(calapp) 

265 

266 else: 

267 LOG.info('Setting Tsys gainfield={!r} for {} data in {}'.format(gainfield, intent, ms.basename)) 

268 

269 # With gainfield set appropriately, construct the CalApplication and add it to the results 

270 calto = callibrary.CalTo(vis=ms.name, intent=intent) 

271 calfrom = callibrary.CalFrom(gainfield=gainfield, **calfrom_args) 

272 calapp = callibrary.CalApplication(calto, calfrom, origin) 

273 calapps.append(calapp) 

274 

275 return calapps
