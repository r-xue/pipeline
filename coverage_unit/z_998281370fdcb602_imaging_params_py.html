71class Parameters: 

72 """Abstract class of Parameter object. 

73 

74 Note: _immutable_parameters is the list which is set some parameter names of a class inherit Parameter. 

75 All parameters in _immutable_parameters are set immutable by freeze(). 

76 """ 

77 

78 # List immutable-able parameter names in it at a child class 

79 _immutable_parameters = [] 

80 _immutable_prefix = '__immutable_' 

81 

82 def __init__(self) -> None: 

83 """Initialize an instance.""" 

84 self.__dict__[self._immutable_prefix] = False 

85 # initialize immutable-able parameters. 

86 # the name of parameters must be started with an alphanumeric character. 

87 for _name in self._immutable_parameters: 

88 if re.match(r'^[a-zA-Z0-9].\w*$', _name): 

89 self.__dict__[f'{self._immutable_prefix}{_name}'] = True 

90 else: 

91 raise ValueError('invalid parameter name') 

92 

93 def is_immutable(self, _name: str='') -> bool: 

94 """Check itself or an attribute whether immutable or not. 

95 

96 Args: 

97 _name (str): Attribute name. If it is a default value, the method returns 

98 a boolean whether the instance itself is immutable or not. 

99 

100 Returns: 

101 True if it is immutable. 

102 """ 

103 return self.__dict__.get(f'{self._immutable_prefix}{_name}', False) 

104 

105 def freeze(self) -> None: 

106 """Set the instance immutable.""" 

107 self.__dict__[self._immutable_prefix] = True 

108 

109 @debugwrapper(msg='setattr') 

110 def __setattr__(self, _name: str, _value: Any) -> None: 

111 """Override object.__setattr__(). 

112 

113 Args: 

114 _name (str): Attribute name 

115 _value (Any): Attribute Object 

116 

117 Raises: 

118 NotImplementedError: raises if _name is immutable. 

119 """ 

120 if self.is_immutable() and self.is_immutable(_name): 

121 raise NotImplementedError(f'attribute {_name} is immutable.') 

122 super.__setattr__(self, _name, _value)
