29 def update_mako_context(self, ctx, context, results): 

30 """ 

31 Updates the Mako context appropriate to the results created by a 

32 Wvrgcalflag task. 

33 

34 Args: 

35 ctx: the mako context for weblog page 

36 context: the pipeline Context 

37 results: the Wvrgcalflag results object 

38 """ 

39 plots_dir = os.path.join(context.report_dir, 'stage%d' % results.stage_number) 

40 if not os.path.exists(plots_dir): 

41 os.mkdir(plots_dir) 

42 

43 applications = [] 

44 flag_plots = {} 

45 metric_plots = {} 

46 phase_offset_summary_plots = {} 

47 phase_offset_detail_plots = {} 

48 baseline_summary_plots = {} 

49 baseline_detail_plots = {} 

50 wvrinfos = {} 

51 ms_non12 = [] 

52 ms_toofew12 = [] 

53 for result in results: 

54 vis = os.path.basename(result.inputs['vis']) 

55 ms = context.observing_run.get_ms(vis) 

56 

57 # If there's no WVR data (empty dataresult) 

58 if not result.flaggerresult.dataresult: 

59 # Check if this MS only contains data for non-WVR antennas. 

60 if len([a for a in ms.antennas if a.diameter != 12.0]) == len(ms.antennas): 

61 ms_non12.append(os.path.basename(vis)) 

62 elif result.too_few_wvr: 

63 ms_toofew12.append(os.path.basename(vis)) 

64 continue 

65 

66 applications.extend(self.get_wvr_applications(result.flaggerresult.dataresult)) 

67 try: 

68 wvrinfos[vis] = result.flaggerresult.dataresult.wvr_infos 

69 except AttributeError: 

70 pass 

71 

72 # collect flagging metric plots from result 

73 if result.flaggerresult.dataresult.qa_wvr.view: 

74 plotter = image.ImageDisplay() 

75 plots = plotter.plot(context, result.flaggerresult.dataresult.qa_wvr, reportdir=plots_dir, 

76 prefix='qa', change='WVR') 

77 # sort plots by spw 

78 metric_plots[vis] = sorted(plots, key=lambda plot: int(plot.parameters['spw'])) 

79 

80 if result.flaggerresult.view: 

81 flag_plotter = image.ImageDisplay() 

82 plots = flag_plotter.plot(context, result.flaggerresult, reportdir=plots_dir, 

83 prefix='flag', change='Flagging') 

84 # sort plots by spw 

85 flag_plots[vis] = sorted(plots, key=lambda plot: int(plot.parameters['spw'])) 

86 

87 if result.flaggerresult.dataresult.qa_wvr.nowvr_result: 

88 # generate the phase offset summary plots 

89 phase_offset_summary_plotter = display.WVRPhaseOffsetSummaryPlot(context, result.flaggerresult) 

90 phase_offset_summary_plots[vis] = phase_offset_summary_plotter.plot() 

91 

92 # generate the per-antenna phase offset plots 

93 phase_offset_plotter = display.WVRPhaseOffsetPlot(context, result.flaggerresult) 

94 phase_offset_detail_plots[vis] = phase_offset_plotter.plot() 

95 

96 baseline_plotter = display.WVRPhaseVsBaselineChart(context, result.flaggerresult) 

97 baseline_detail_plots[vis] = baseline_plotter.plot() 

98 

99 # PIPE-851: get the first scan for the QA intent(s) with SpW 

100 # matching the gaintable SpW. 

101 qa_intent = set(result.flaggerresult.dataresult.inputs['qa_intent'].split(',')) 

102 if not result.flaggerresult.dataresult.qa_wvr.qa_spw: 

103 # If no valid QA SpW was set for this data result, then no 

104 # valid matching scans can be found. 

105 qa_scan = '' 

106 else: 

107 qa_spw = int(result.flaggerresult.dataresult.qa_wvr.qa_spw) 

108 qa_scan = sorted(scan.id for scan in ms.scans if not qa_intent.isdisjoint(scan.intents) 

109 for scan_spw in scan.spws if scan_spw.id == qa_spw)[0] 

110 # scan parameter on plot is comma-separated string 

111 qa_scan = str(qa_scan) 

112 LOG.trace('Using scan %s for phase vs baseline summary plots', qa_scan) 

113 baseline_summary_plots[vis] = [p for p in baseline_detail_plots[vis] 

114 if qa_scan in set(p.parameters['scan'].split(','))] 

115 

116 weblog_dir = os.path.join(context.report_dir, 

117 'stage%s' % results.stage_number) 

118 # render plots for all EBs in one page 

119 phase_offset_subpages = {} 

120 baseline_summary_subpages = {} 

121 metric_subpages = {} 

122 flag_subpages = {} 

123 for d, plotter_cls, subpages in ( 

124 (phase_offset_detail_plots, WvrgcalflagPhaseOffsetPlotRenderer, phase_offset_subpages), 

125 (baseline_detail_plots, WvrgcalflagPhaseOffsetVsBaselinePlotRenderer, baseline_summary_subpages), 

126 (metric_plots, WvrcalflagMetricPlotsRenderer, metric_subpages), 

127 (flag_plots, WvrgcalflagFlagPlotRenderer, flag_subpages)): 

128 if d: 

129 all_plots = list(utils.flatten([v for v in d.values()])) 

130 renderer = plotter_cls(context, results, all_plots) 

131 with renderer.get_file() as fileobj: 

132 fileobj.write(renderer.render()) 

133 # redirect the subpages to the master page 

134 for vis in d: 

135 subpages[vis] = renderer.path 

136 

137 # add the PlotGroups to the Mako context. The Mako template will parse 

138 # these objects in order to create links to the thumbnail pages we 

139 # just created 

140 ctx.update({ 

141 'applications': applications, 

142 'wvrinfos': wvrinfos, 

143 'flag_plots': flag_plots, 

144 'flag_subpages': flag_subpages, 

145 'metric_plots': metric_plots, 

146 'metric_subpages': metric_subpages, 

147 'phase_offset_summary_plots': phase_offset_summary_plots, 

148 'phase_offset_subpages': phase_offset_subpages, 

149 'baseline_summary_plots': baseline_summary_plots, 

150 'baseline_summary_subpages': baseline_summary_subpages, 

151 'dirname': weblog_dir 

152 }) 

153 

154 # tell the user not to panic for non-12m dishes missing WVR  

155 if ms_non12: 

156 msg = ('WVR data are not expected for %s, which %s not observed ' 

157 'using 12m antennas.' 

158 '' % (utils.commafy(ms_non12, quotes=False, conjunction='or'), 

159 'was' if len(ms_non12) == 1 else 'were')) 

160 ctx['alerts_info'] = [msg] 

161 elif ms_toofew12: 

162 msg = ("No WVR correction computed for {}, which {} observed" 

163 " with too few 12m antennas." 

164 "".format(utils.commafy(ms_toofew12, quotes=False, conjunction='or'), 

165 'was' if len(ms_toofew12) == 1 else 'were')) 

166 ctx['alerts_info'] = [msg] 

167 

168 # Phase vs time for the overview plot should be for the widest window 

169 # at the highest frequency 

170 # spws = sorted(ms.spectral_windows, 

171 # key=operator.attrgetter('bandwidth', 'centre_frequency')) 

172 # overview_spw = spws[-1] 

173 

174 @staticmethod 

175 def get_wvr_applications(result): 

176 applications = [] 

177 

178 interpolated = utils.commafy(result.wvrflag, False) 

179 

180 # define a closure that adds a wvrapplication for each calapplication 

181 # unless the applications list already contains one for that  

182 # ms/caltable combination 

183 def collect(calapps, accept): 

184 for calapp in calapps: 

185 ms = os.path.basename(calapp.vis) 

186 gaintable = os.path.basename(calapp.gaintable) 

187 

188 a = WvrApplication(ms, gaintable, interpolated, accept) 

189 

190 if not any([r for r in applications 

191 if r.ms == ms and r.gaintable == gaintable]): 

192 applications.append(a) 

193 

194 collect(result.final, True) 

195 collect(result.pool, False) 

196 

197 return applications
