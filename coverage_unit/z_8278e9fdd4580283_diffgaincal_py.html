207 def _assess_if_spw_combination_is_necessary(self, intent: str, spw: list[int], force_for_spwmapmodes: list[str], 

208 scan_groups: list[str] | None = None) -> bool: 

209 """ 

210 Assess whether spectral window combination is necessary based on 

211 `hm_spwmapmode`, the SpwMapping for given intent, and/or whether SpWs 

212 would be missing in final gaintable. 

213 

214 Args: 

215 intent: Intent to assess. 

216 spw: List of Spectral Window IDs to assess. 

217 force_for_spwmapmodes: List of values for `hm_spwmapmode` for which 

218 to force SpW combination. 

219 scan_groups: Optional, list of group(s) of scans to assess. 

220 

221 Returns: 

222 Boolean declaring whether to use SpW combination. 

223 """ 

224 inputs = self.inputs 

225 

226 # Force SpW combination if hm_spwmapmode matches any of the specified 

227 # "forced" modes, which can include any of the allowed values for the 

228 # input parameter hm_spwmapmode *except* for hm_spwmapmode='auto' 

229 # ('auto' is separately evaluated below). 

230 if inputs.hm_spwmapmode in force_for_spwmapmodes: 

231 return True 

232 

233 # For automatic hm_spwmapmode, check whether SpW combination is 

234 # required based on SpwMapping or in case the resulting caltable would 

235 # contain any missing/fully flagged SpWs. 

236 combine_spw = False 

237 if inputs.hm_spwmapmode == 'auto': 

238 # Assess the SpwMapping for diffgain calibrator for whether SpW 

239 # combination is required. 

240 for field in inputs.ms.get_fields(intent=intent): 

241 combine_spw = self._assess_spw_combine_based_on_spwmapping_and_snr( 

242 intent, field.name, spw, scan_groups=scan_groups) 

243 # Assessment is made in a loop over all diffgain calibrator 

244 # fields, and as soon as SpW combination is needed for one, it 

245 # is required for all. In practice, it is currently expected 

246 # that band-to-band datasets only have a single diffgain 

247 # calibrator. Should this change in operations in the future, 

248 # then it should be revisited whether to apply a single SpW 

249 # combination strategy to all diffgain calibrators and their 

250 # corresponding phase calibrator sources. 

251 if combine_spw: 

252 break 

253 

254 # If no SpW combination is needed based on SpwMapping, then generate 

255 # a temporary (diagnostic) phase caltable without SpW combination, 

256 # and check whether any SpWs would be missing / fully flagged; if 

257 # so, then force SpW combination after all. 

258 if not combine_spw: 

259 # If provided groups of scans, then do combination over scan. 

260 combine = '' if scan_groups is None else 'scan' 

261 diag_result = self._do_phasecal(intent=intent, spw=spw, combine=combine) 

262 bad_spws = self._assess_spws_in_gaintable(diag_result, combine_spw, spw, scan_groups=scan_groups) 

263 if bad_spws: 

264 combine_spw = True 

265 

266 return combine_spw

268 def _assess_spw_combine_based_on_spwmapping_and_snr(self, intent: str, field: str, spwids: list[int], 

269 scan_groups: list[str] | None = None) -> bool: 

270 """ 

271 Assess whether spectral window combination is necessary based on 

272 spectral window mapping and scaling estimated SNR (for solint) to full 

273 aggregate exposure time of scans. 

274 

275 Args: 

276 intent: Intent to assess; should be DIFFGAINREF or DIFFGAINSRC. 

277 field: Field to assess. 

278 spwids: Spectral window IDs to assess SNR for in case SpwMapping 

279 specified to use spectral window combination. 

280 scan_groups: Optional: list of scan groups to assess. 

281 

282 Returns: 

283 Boolean declaring whether to use spectral window combination. 

284 """ 

285 # Try to retrieve SpW mapping info from MS for requested intent and 

286 # field. 

287 spwmapping: SpwMapping | None = self.inputs.ms.spwmaps.get((intent, field), None) 

288 

289 # If no SpW mapping info is available, then the default is to use no SpW 

290 # combination. 

291 if spwmapping is None: 

292 return False 

293 

294 # Otherwise, proceed to assess the info in the SpW mapping info to 

295 # determine whether to use SpW combination: 

296 

297 # Track whether SpWs were re-mapped or combined. 

298 spws_mapped_or_combined = False 

299 # If SpWs are combined, then use input spectral windows for SNR 

300 # assessment. 

301 if spwmapping.combine: 

302 spws_mapped_or_combined = True 

303 spwids_mapped = spwids 

304 LOG.info(f"{self.inputs.ms.basename}: SpwMapping for intent {intent} is using spectral window combination," 

305 f" re-assessing estimated SNR based on full exposure times of corresponding scans.") 

306 # Otherwise, check if any SpWs are re-mapped, and select those SpWs 

307 # for SNR assessment. 

308 else: 

309 spwids_mapped = [idx for idx, spwid in enumerate(spwmapping.spwmap) if idx != spwid] 

310 if spwids_mapped: 

311 spws_mapped_or_combined = True 

312 LOG.info(f"{self.inputs.ms.basename}: SpwMapping for intent {intent} is using spectral window mapping," 

313 f" re-assessing estimated SNR based on full exposure times of corresponding scans.") 

314 

315 # If SpWs are to be re-mapped or combined, proceed to assess if SpW 

316 # combination would still be needed after scaling estimated SNR from 

317 # solint to full exposure time (scan time * nr. scans). 

318 if spws_mapped_or_combined: 

319 # Retrieve scan and integration time. 

320 scantime, inttime = self._get_scan_and_integration_time(self.inputs.ms, intent, field, spwids) 

321 

322 # Retrieve solution interval time in seconds from SpW mapping. 

323 if spwmapping.solint == 'int': 

324 solint = inttime 

325 else: 

326 # Assume solint was recorded as string in units of seconds, 

327 # convert to float with CASA quanta. 

328 solint = casa_tools.quanta.quantity(spwmapping.solint)['value'] 

329 

330 # Get the number of scans from groups of scans if provided, but 

331 # otherwise assume this to be a single scan. 

332 # It is assumed that the band-to-band observation uses the same 

333 # number of scans in each scan group, so the first scan group is 

334 # taken as representative of how long a scan group should be. 

335 nr_scans = len(scan_groups[0]) if scan_groups else 1 

336 

337 # Compute SNR scale factor, to scale estimated SNR for current 

338 # solint to expected SNR for exposure time. 

339 snr_scale_factor = np.sqrt(scantime * nr_scans / solint) 

340 

341 # Loop over SpWs in SNR info and check each re-mapped/combined 

342 # SpW whether the scaled estimated SNR would (still) be below 

343 # the minimum required SNR threshold. 

344 spwids_mapped = [str(s) for s in spwids_mapped] 

345 for spwid, snr in spwmapping.snr_info: 

346 if spwid in spwids_mapped and snr * snr_scale_factor < spwmapping.snr_threshold_used: 

347 LOG.info(f"{self.inputs.ms.basename}, intent {intent}: estimated SNR is below threshold for good" 

348 f" solutions ({spwmapping.snr_threshold_used}) for at least one of the re-mapped" 

349 f" / combined SpWs, will use SpW combination.") 

350 return True 

351 

352 # If this is reached, then it is expected that no SpW combination is 

353 # necessary based on info in SpwMapping. 

354 return False

356 def _assess_spws_in_gaintable(self, gaincal_results: common.GaincalResults, combine_spw: bool, spw: list[int], 

357 scan_groups: list[str] | None = None) -> str: 

358 """ 

359 If no SpW combination was expected to be used, then this method assesses 

360 the resulting temporary phase caltable for missing SpWs, fully flagged 

361 SpWs, and/or SpWs with large fraction of flagged scans, and returns 

362 these as a single QA message. If this message is not empty, then SpW 

363 combination is deemed necessary after all. 

364 

365 Args: 

366 gaincal_results: List of gaincal worker task results representing 

367 the temporary phase caltable. 

368 combine_spw: boolean declaring whether SpW combination is to be used. 

369 spw: SpW IDs to consider. 

370 scan_groups: Groups of scans to consider (impacts nr. of scan 

371 solutions expected in caltable). 

372 

373 Returns: 

374 String message described what SpWs in temporary caltable have issues. 

375 """ 

376 # If SpW combination was already required, return early; no need to 

377 # assess whether flagged/missing data in gaintable would necessitate SpW 

378 # combination. 

379 if combine_spw: 

380 return '' 

381 

382 inputs = self.inputs 

383 

384 # Retrieve caltable and which intent it was derived for. 

385 caltable = gaincal_results.final[0].gaintable 

386 intent = gaincal_results.inputs['intent'] 

387 

388 # Retrieve SpW ID, scan, and flagging columns from the caltable. 

389 with casa_tools.TableReader(caltable) as table: 

390 tbl_spwids = table.getcol("SPECTRAL_WINDOW_ID") 

391 tbl_flags = table.getcol("FLAG") 

392 tbl_scanids = table.getcol("SCAN_NUMBER") 

393 

394 # Proceed with 3 separate tests, keeping track of what SpWs are missing, 

395 # or have too few scan solutions, or too much data flagged. 

396 bad_spws = [] 

397 

398 # First check if all requested SpWs are present in the gaintable. 

399 missing_spwids = ','.join(str(s) for s in spw if s not in tbl_spwids) 

400 if missing_spwids: 

401 LOG.info(f"{caltable}: missing solutions for SpW(s) {missing_spwids}.") 

402 bad_spws.append(f"SpW(s) {missing_spwids} have no solutions.") 

403 

404 # Secondly, check if the number of scan solutions in the caltable is 

405 # significantly lower (set by threshold parameters) than the expected 

406 # nr. of scan solutions, which can happen if scans are entirely flagged 

407 # in prior stages. 

408 # 

409 # To do so, first determine number of scan solutions expected in 

410 # caltable. If no scan groups were provided, then expect 1 solution per 

411 # scan; otherwise expect 1 solution per group (combining over scan). 

412 if scan_groups is None: 

413 n_exp_scan_solns = len(self.inputs.ms.get_scans(scan_intent=intent)) 

414 else: 

415 n_exp_scan_solns = len(scan_groups) 

416 # And then determine for each SpW ID whether the nr. of scan solutions 

417 # compared to expected number is acceptable. 

418 spw_with_too_many_missing_scans = [] 

419 for uniq_spwid in sorted(set(tbl_spwids)): 

420 # Determine unique scans for current SpW. 

421 scanids_for_spwid = {tbl_scanids[idx] for idx, spwid in enumerate(tbl_spwids) if spwid == uniq_spwid} 

422 

423 # Compute fraction of missing scan solutions in table. 

424 missing_scan_ratio = 1.0 - float(len(scanids_for_spwid)) / n_exp_scan_solns 

425 

426 # If fraction of missing scans exceeds the limit, then mark this SpW 

427 # as bad. 

428 if missing_scan_ratio > inputs.missing_scans_frac_limit: 

429 spw_with_too_many_missing_scans.append(uniq_spwid) 

430 

431 # If any SpW has too many missing scans: 

432 if spw_with_too_many_missing_scans: 

433 bad_spws.append(f"SpW(s) {','.join(str(s) for s in spw_with_too_many_missing_scans)} have too many missing" 

434 f" scans.") 

435 

436 # Thirdly, assess for each SpW whether the fraction of flagged data is 

437 # above the maximum threshold. 

438 spw_with_too_much_flagging = [] 

439 for uniq_spwid in sorted(set(tbl_spwids)): 

440 # Get indices in caltable data corresponding to current SpW. 

441 idx_spw = np.where(tbl_spwids == uniq_spwid)[0] 

442 

443 # Get number of correlations (polarizations) for this SpW. 

444 corr_type = commonhelpermethods.get_corr_products(inputs.ms, uniq_spwid) 

445 ncorrs = len(corr_type) 

446 

447 # For single-polarization data, the caltable is expected to still 

448 # contain at least 2 columns, so identify which polarization index 

449 # to use. For multi-pol data, assess all polarizations. 

450 if ncorrs == 1: 

451 pol_to_assess = [commonhelpermethods.get_pol_id(inputs.ms, uniq_spwid, corr_type[0])] 

452 else: 

453 pol_to_assess = list(range(len(tbl_flags))) 

454 

455 # Separately assess each polarization, but as soon as 1 pol has too 

456 # much flagged data, then consider SpW bad. 

457 for idx_pol in pol_to_assess: 

458 # Compute fraction of flagged rows for current polarization and 

459 # SpW. Numpy treats True as 1 and False as 0, so can use mean. 

460 flag_ratio = np.mean(tbl_flags[idx_pol, 0, idx_spw]) 

461 # If fraction of flagged data in even just one polarization 

462 # exceeds the limit, then mark this SpW as bad, and continue 

463 # with next SpW. 

464 if flag_ratio > inputs.flagging_frac_limit: 

465 spw_with_too_much_flagging.append(uniq_spwid) 

466 break 

467 

468 # If any SpW has too much flagged data: 

469 if spw_with_too_much_flagging: 

470 bad_spws.append(f"SpW(s) {','.join(str(s) for s in spw_with_too_much_flagging)} have too high fraction of" 

471 f" flagged data.") 

472 

473 # Return as single string. 

474 bad_spws_message = ' '.join(bad_spws) 

475 return bad_spws_message

515 def _do_phasecal(self, intent: str, spw: list[int], combine: str | None = None) -> common.GaincalResults: 

516 """ 

517 Local method to perform phase gaincal with pre-defined values for 

518 parameters shared among all gaincal steps in this task. 

519 

520 If combination over scan is required, then this method identifies the 

521 scan groups and calls gaincal for each scan group, ensuring solutions 

522 are appended to the same final caltable. 

523 

524 Args: 

525 intent: intent selection to use. 

526 spw: spectral windows to use. 

527 combine: axis to combine solutions over. 

528 

529 Returns: 

530 GaincalResults 

531 """ 

532 # SpW IDs as string for CASA gaincal. 

533 spwids_str = ','.join(str(s) for s in spw) 

534 

535 # If combine is not specified or not combining over scans, then perform 

536 # single gaincal for given intent, spw, and combine. 

537 if combine is None or 'scan' not in combine: 

538 result = self._do_gaincal(intent=intent, spw=spwids_str, combine=combine) 

539 # Otherwise, perform separate gaincal for each scan group: 

540 else: 

541 # Determine scan groups for intent. 

542 scan_groups = self._get_scan_groups(intent=intent) 

543 

544 # Run gaincal for the first scan group. 

545 result = self._do_gaincal(intent=intent, spw=spwids_str, combine=combine, scan=scan_groups[0]) 

546 

547 # Only extract and reuse the caltable if there is more than one scan group. 

548 if len(scan_groups) > 1: 

549 caltable = result.inputs['caltable'] 

550 

551 # Loop the remaining scan groups - specifically diffgain_onsource 'b2b offset' solve. 

552 # Should only be one 'more' as current operations as of 2024 use two diffgain 'blocks'. 

553 for scan_group in scan_groups[1:]: 

554 # PIPE-2915: check if there was a gaintable made in the first instance, then append 

555 # If the first scan group solve did not get made (possibly due to flags), 

556 # we use the next group to make a fresh gaintable. 

557 # Note that in operations there is likely only ever two scan groups. 

558 table_exists = os.path.exists(caltable) 

559 

560 if not table_exists: 

561 LOG.info('Diffgain B2B offset caltable not found for initial scan group') 

562 

563 # If table_exists is True, we append. If False, we start a fresh solve. 

564 result = self._do_gaincal( 

565 intent=intent, 

566 spw=spwids_str, 

567 combine=combine, 

568 scan=scan_group, 

569 caltable=caltable, 

570 append=table_exists, 

571 ) 

572 return result

574 def _do_phasecal_for_diffgain_reference(self) -> tuple[common.GaincalResults, dict[IntentField, SpwMapping]]: 

575 """ 

576 Compute phase gain solutions for the diffgain reference intent and 

577 register the resulting caltable in the local task context to be applied 

578 to the diffgain on-source intent. 

579 

580 Returns: 

581 2-tuple containing: 

582 * GaincalResults for diffgain reference. 

583 * Updated SpW mapping dictionary; can be empty if no updates were necessary. 

584 """ 

585 intent = 'DIFFGAINREF' 

586 

587 # Select which spectral windows to act on. 

588 spwids = [spw.id for spw in self.inputs.ms.get_spectral_windows(intent=intent)] 

589 

590 # Assess if SpW combination is necessary. 

591 combine_spw = self._assess_if_spw_combination_is_necessary( 

592 intent, spwids, force_for_spwmapmodes=['all', 'both', 'reference']) 

593 

594 # With choice made of whether SpW combination is necessary, now run 

595 # phase calibration for the diffgain reference intent scans. 

596 combine = 'spw' if combine_spw else '' 

597 phasecal_result = self._do_phasecal(intent=intent, spw=spwids, combine=combine) 

598 

599 # Check phase caltable for missing / fully flagged SpW, turn into QA 

600 # message, and attach message to phase GaincalResults. 

601 phasecal_result.qa_message = self._assess_spws_in_gaintable(phasecal_result, combine_spw, spwids) 

602 

603 # Update SpW mapping for phase calibrator based on whether SpW 

604 # combination is necessary for the diffgain reference. 

605 updated_spwmaps = self._update_spwmap_for_phase(self.inputs.ms, combine_spw) 

606 

607 # If a caltable was created, then create a modified CalApplication to 

608 # correctly register the caltable. 

609 if phasecal_result.pool: 

610 # Retrieve the diffgain spectral windows, and create SpW mapping to 

611 # re-map diffgain on-source SpWs to diffgain reference SpWs. 

612 dg_refspws = self.inputs.ms.get_spectral_windows(intent='DIFFGAINREF') 

613 dg_srcspws = self.inputs.ms.get_spectral_windows(intent='DIFFGAINSRC') 

614 dg_spwmap = update_spwmap_for_band_to_band([], dg_refspws, dg_srcspws, combine_spw) 

615 

616 # Define CalApplication overrides to specify how the diffgain 

617 # reference phase solutions should be registered in the callibrary. 

618 # Note: solutions derived for the diffgain reference intent are 

619 # registered here to be applied to the diffgain on-source intent and 

620 # SpWs with a corresponding SpW mapping. 

621 calapp_overrides = { 

622 'calwt': False, 

623 'intent': 'DIFFGAINSRC', 

624 'interp': 'linearPD,linear', 

625 'spw': ','.join(str(spw.id) for spw in dg_srcspws), 

626 'spwmap': dg_spwmap, 

627 } 

628 

629 # There should be only 1 caltable, so replace the CalApp for that 

630 # one. 

631 modified_calapp = callibrary.copy_calapplication(phasecal_result.pool[0], **calapp_overrides) 

632 phasecal_result.pool[0] = modified_calapp 

633 phasecal_result.final[0] = modified_calapp 

634 

635 # Register these diffgain reference phase solutions into local 

636 # context to ensure that these are used in pre-apply when computing 

637 # phase solutions for the diffgain on-source intent. 

638 phasecal_result.accept(self.inputs.context) 

639 

640 return phasecal_result, updated_spwmaps

642 def _do_phasecal_for_diffgain_onsource(self) -> common.GaincalResults | None: 

643 """ 

644 Compute phase gain solutions for the diffgain on-source intent. Gain 

645 solutions are created separately for groups of scans, using 

646 combine='scan', and appended to a single caltable. Each scan group is 

647 intended to comprise scans taken closely together in time (typically 

648 alternated with the diffgain reference scans), and these groups are 

649 typically done before and after the science target scans. In practice, 

650 scan groups are identified as scans with IDs separated by less than 4. 

651 

652 The resulting caltable is immediately registered with the callibrary in 

653 the local task context (to ensure pre-apply in the subsequent 

654 computation of residual offset). For final acceptance of this caltable 

655 in the top-level callibrary, the CalApplication is updated to ensure the 

656 resulting caltable will be applied to the science target or check 

657 source. 

658 

659 Returns: 

660 GaincalResults for the diffgain on-source phase solutions caltable, 

661 or None if no diffgain scan groups are identified. 

662 """ 

663 intent = 'DIFFGAINSRC' 

664 

665 # Determine scan groups for diffgain on-source intent. Exit early if no 

666 # scan groups could be identified. 

667 scan_groups = self._get_scan_groups(intent=intent) 

668 if not scan_groups: 

669 LOG.warning(f"{self.inputs.ms.basename}: no scan groups found for diffgain on-source intent.") 

670 return None 

671 

672 # Select which spectral windows to act on. 

673 spwids = [spw.id for spw in self.inputs.ms.get_spectral_windows(intent=intent)] 

674 

675 # Assess if SpW combination is necessary. 

676 combine_spw = self._assess_if_spw_combination_is_necessary( 

677 intent, spwids, force_for_spwmapmodes=['all', 'offset'], scan_groups=scan_groups) 

678 

679 # With choice made of whether SpW combination is necessary, now run 

680 # phase calibration for the diffgain on-source intent scans. Note, for 

681 # the diffgain on-source intent, solves are done separately per-scan, 

682 # using combination over the scan. 

683 combine = 'scan,spw' if combine_spw else 'scan' 

684 phasecal_result = self._do_phasecal(intent=intent, spw=spwids, combine=combine) 

685 

686 # Check phase caltable for missing / fully flagged SpW, turn into QA 

687 # message, and attach message to phase GaincalResults. 

688 phasecal_result.qa_message = self._assess_spws_in_gaintable(phasecal_result, combine_spw, spwids, 

689 scan_groups=scan_groups) 

690 

691 # If a caltable was created, then create a modified CalApplication to 

692 # correctly register the caltable. 

693 if phasecal_result.pool: 

694 # If SpW combination was used for the diffgain on-source, then 

695 # compute a corresponding SpW map. 

696 if combine_spw: 

697 dg_srcspws = self.inputs.ms.get_spectral_windows(intent=intent) 

698 dg_spwmap = combine_spwmap(dg_srcspws) 

699 else: 

700 dg_spwmap = [] 

701 

702 # Prior to registering this caltable into the local context, set 

703 # overrides in the CalApplication: 

704 calapp_overrides = { 

705 'calwt': False, 

706 'intent': intent, 

707 'spwmap': dg_spwmap, 

708 } 

709 modified_calapp = callibrary.copy_calapplication(phasecal_result.pool[0], **calapp_overrides) 

710 phasecal_result.pool[0] = modified_calapp 

711 phasecal_result.final[0] = modified_calapp 

712 

713 # Register the diffgain on-source phase solutions into local context 

714 # to ensure that these are used in pre-apply when computing the 

715 # residual phase offsets (diagnostic) caltable. 

716 phasecal_result.accept(self.inputs.context) 

717 

718 # Update the CalApplication again, this time to ensure this diffgain 

719 # phase offsets caltable will be applied to the science target 

720 # and/or check source. 

721 calapp_overrides = { 

722 'calwt': False, 

723 'intent': 'TARGET,CHECK', 

724 'spwmap': dg_spwmap, 

725 } 

726 modified_calapp = callibrary.copy_calapplication(phasecal_result.pool[0], **calapp_overrides) 

727 phasecal_result.pool[0] = modified_calapp 

728 phasecal_result.final[0] = modified_calapp 

729 

730 return phasecal_result

762 def _get_scan_groups(self, intent: str) -> list[str]: 

763 """ 

764 Return list of scan groups associated with given intent, where each 

765 group are scans separated by less than 4 in ID. 

766 

767 E.g., if the scans for given intent are: 

768 '5,7,9,11,79,81,83' 

769 Then this will return: 

770 ['5,7,9,11', '79,81,83'] 

771 

772 Returns: 

773 List of groups of scans. 

774 """ 

775 # Get IDs of DIFFGAIN scans for given intent. 

776 dg_scanids = sorted(scan.id for scan in self.inputs.ms.get_scans(scan_intent=intent)) 

777 

778 # Group scans separated in ID by less than 4. 

779 scan_groups = [','.join(map(str, group)) 

780 for group in np.split(dg_scanids, np.where(np.diff(dg_scanids) > 3)[0] + 1)] 

781 

782 return scan_groups

784 @staticmethod 

785 def _update_spwmap_for_phase(ms: MeasurementSet, combine_spw: bool) -> dict[IntentField, SpwMapping]: 

786 # If SpW combination is not necessary, then no need for updated 

787 # SpW mapping. 

788 if not combine_spw: 

789 return {} 

790 

791 # Otherwise, check whether there exists a SpwMapping for the PHASE 

792 # calibrator; if this was already specifying to use SpW combination, 

793 # then no need for an updated SpwMapping. 

794 phase_spwmappings = {k: v for k, v in ms.spwmaps.items() if k.intent == 'PHASE'} 

795 if not phase_spwmappings or any(spwmapping.combine for spwmapping in phase_spwmappings.values()): 

796 return {} 

797 

798 # Otherwise, proceed to create a new SpW map for the PHASE calibrator 

799 # that is appropriate for SpW combination. 

800 # 

801 # First create a SpW map for combining the (low frequency) diffgain 

802 # reference SpWs. 

803 dg_refspws = ms.get_spectral_windows(intent='DIFFGAINREF') 

804 spwmap = combine_spwmap(dg_refspws) 

805 # Then, update the SpW map for band-to-band, adding in the (high 

806 # frequency) diffgain science SpWs. 

807 dg_srcspws = ms.get_spectral_windows(intent='DIFFGAINSRC') 

808 spwmap = update_spwmap_for_band_to_band(spwmap, dg_refspws, dg_srcspws, combine=True) 

809 

810 # Update the SpwMapping for the PHASE calibrators. 

811 for intfld, spwmapping in phase_spwmappings.items(): 

812 # Retrieve SNR estimates for diffgain reference SpWs, compute their 

813 # estimated combined SNR. 

814 dg_refspwids = [str(s.id) for s in dg_refspws] 

815 combined_snr = np.linalg.norm([snr for spwid, snr in spwmapping.snr_info 

816 if spwid in dg_refspwids]) 

817 # Create new SNR info dictionary with the combined SNR entry, and 

818 # add the SNR info for individual SpWs from the original SpwMapping. 

819 snr_info = {f"Combined ({', '.join(dg_refspwids)})": combined_snr} 

820 snr_info.update(spwmapping.snr_info) 

821 

822 # Store as a new SpwMapping with updated values for combine, spwmap, 

823 # and the SNR info. 

824 phase_spwmappings[intfld] = spwmapping._replace(combine=True, spwmap=spwmap, snr_info=snr_info) 

825 

826 return phase_spwmappings

829 def _get_scan_and_integration_time(ms: MeasurementSet, intent: str, field: str, spwids: list[int]) -> tuple[float, float]: 

830 """ 

831 Retrieve the scan (exposure) and integration time, in seconds, used for 

832 given intent and field in given measurement set. 

833 

834 Args: 

835 ms: measurement set to use. 

836 intent: intent to retrieve times for. 

837 field: field to retrieve times for. 

838 spwids: spectral window ids to retrieve times for. 

839 

840 Returns: 

841 2-Tuple containing: 

842 - scan (exposure) time in seconds for given ms, intent, field. 

843 - integration time in seconds for given ms, intent, field. 

844 """ 

845 # Get scans for given intent, field, spwids. 

846 scans = ms.get_scans(scan_intent=intent, field=field, spw=spwids) 

847 

848 # Assume that the integration time and scan time are the same for all 

849 # scans and all SpWs for given intent, so use the first scan and first 

850 # spectral window as the representative ones. 

851 int_time = scans[0].mean_interval(spwids[0]).total_seconds() 

852 scan_time = scans[0].exposure_time(spwids[0]).total_seconds() 

853 

854 return scan_time, int_time
