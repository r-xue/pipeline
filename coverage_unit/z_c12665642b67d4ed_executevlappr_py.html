26def executeppr(pprXmlFile: str, importonly: bool = True, loglevel: str = 'info', 

27 plotlevel: str = 'summary', interactive: bool = True) -> None: 

28 """ 

29 Runs Pipeline Processing Request (PPR). 

30 

31 Executes pipeline tasks based on instructions described in pprXmlFile. 

32 

33 Args: 

34 pprXmlFile: A path to PPR file. 

35 importonly: Whether or not to indicate to stop processing after 

36 importing data. If True, execution of PPR stops after 

37 h*_importdata stage. The parameter has no effect if there is no 

38 h*_importdata stage in PPR. 

39 loglevel: A logging level. Available levels are, 'critical', 'error', 

40 'warning', 'info', 'debug', 'todo', and 'trace'. 

41 plotlevel: A plot level. Available levels are, 'all', 'default', and 

42 'summary' 

43 interactive: If True, print pipeline log to STDOUT. 

44 """ 

45 # save existing context to disk 

46 save_existing_context() 

47 

48 # Useful mode parameters 

49 echo_to_screen = interactive 

50 workingDir = None 

51 

52 try: 

53 # Decode the processing request 

54 casa_tools.post_to_log("Analyzing pipeline processing request ...", echo_to_screen=echo_to_screen) 

55 info, structure, relativePath, intentsDict, asdmList, procedureName, commandsList = \ 

56 _getFirstRequest(pprXmlFile) 

57 processing_intents = _getProcessingIntents(intentsDict, procedureName) 

58 

59 # Set the directories 

60 if 'SCIPIPE_ROOTDIR' in os.environ: 

61 workingDir = os.path.join(os.path.expandvars('$SCIPIPE_ROOTDIR'), relativePath, 'working') 

62 rawDir = os.path.join(os.path.expandvars('$SCIPIPE_ROOTDIR'), relativePath, 'rawdata') 

63 else: 

64 # PIPE-2093: if $SCIPIPE_ROOTDIR doesn't exist, we likely run in a local dev/test environment. 

65 # Then we will override the typical production workingDir/rawDIR values that are traditionally 

66 # constructed from $SCIPIPE_ROOTDIR and the PPR <RelativePath> field. Note that we assume that 

67 # any executeppr call here happens inside the "working/" directory. 

68 workingDir = os.path.abspath(os.path.join('..', 'working')) 

69 rawDir = os.path.abspath(os.path.join('..', 'rawdata')) 

70 

71 # Get the pipeline context 

72 context = cli.h_init(loglevel=loglevel, plotlevel=plotlevel, processing_intents=processing_intents) 

73 

74 except Exception: 

75 casa_tools.post_to_log("Beginning pipeline run ...", echo_to_screen=echo_to_screen) 

76 casa_tools.post_to_log("For processing request: " + pprXmlFile, echo_to_screen=echo_to_screen) 

77 traceback.print_exc(file=sys.stdout) 

78 errstr = traceback.format_exc() 

79 casa_tools.post_to_log(errstr, echo_to_screen=echo_to_screen) 

80 casa_tools.post_to_log("Terminating procedure execution ...", echo_to_screen=echo_to_screen) 

81 errorfile = utils.write_errorexit_file(workingDir, 'errorexit', 'txt') 

82 return 

83 

84 # Request decoded, starting run. 

85 casa_tools.post_to_log("Beginning pipeline run ...", echo_to_screen=echo_to_screen) 

86 casa_tools.post_to_log("For processing request: " + pprXmlFile, echo_to_screen=echo_to_screen) 

87 

88 # Check for common error conditions. 

89 if relativePath == "": 

90 casa_tools.post_to_log(" Undefined relative data path", echo_to_screen=echo_to_screen) 

91 casa_tools.post_to_log("Terminating pipeline execution ...", echo_to_screen=echo_to_screen) 

92 errorfile = utils.write_errorexit_file(workingDir, 'errorexit', 'txt') 

93 return 

94 elif len(asdmList) < 1: 

95 casa_tools.post_to_log(" Empty ASDM list", echo_to_screen=echo_to_screen) 

96 casa_tools.post_to_log("Terminating pipeline execution ...", echo_to_screen=echo_to_screen) 

97 errorfile = utils.write_errorexit_file(workingDir, 'errorexit', 'txt') 

98 return 

99 elif len(commandsList) < 1: 

100 casa_tools.post_to_log(" Empty commands list", echo_to_screen=echo_to_screen) 

101 casa_tools.post_to_log("Terminating pipeline execution ...", echo_to_screen=echo_to_screen) 

102 errorfile = utils.write_errorexit_file(workingDir, 'errorexit', 'txt') 

103 return 

104 

105 # List project summary information 

106 casa_tools.post_to_log("Project summary", echo_to_screen=echo_to_screen) 

107 for item in info: 

108 casa_tools.post_to_log(" " + item[1][0] + item[1][1], echo_to_screen=echo_to_screen) 

109 ds = dict(info) 

110 context.project_summary = project.ProjectSummary( 

111 proposal_code=ds['proposal_code'][1], 

112 proposal_title='unknown', 

113 piname='unknown', 

114 observatory=ds['observatory'][1], 

115 telescope=ds['telescope'][1]) 

116 

117 # List project structure information 

118 casa_tools.post_to_log("Project structure", echo_to_screen=echo_to_screen)

120 casa_tools.post_to_log(" " + item[1][0] + item[1][1], echo_to_screen=echo_to_screen) 

121 

122 context.project_structure = project.ProjectStructure( 

123 ppr_file=pprXmlFile, 

124 recipe_name=procedureName) 

125 

126 # Create performance parameters object 

127 context.project_performance_parameters = _getPerformanceParameters(intentsDict) 

128 

129 # Print the relative path 

130 casa_tools.post_to_log("Directory structure", echo_to_screen=echo_to_screen) 

131 casa_tools.post_to_log(" Working directory: " + workingDir, echo_to_screen=echo_to_screen) 

132 casa_tools.post_to_log(" Raw data directory: " + rawDir, echo_to_screen=echo_to_screen) 

133 

134 # Construct the ASDM list 

135 casa_tools.post_to_log("Number of ASDMs: " + str(len(asdmList)), echo_to_screen=echo_to_screen) 

136 files = [] 

137 sessions = [] 

138 defsession = 'session_1' 

139 for asdm in asdmList: 

140 session = defsession 

141 sessions.append(session) 

142 files.append(os.path.join(rawDir, asdm[1])) 

143 casa_tools.post_to_log(" Session: " + session + " ASDM: " + asdm[1], echo_to_screen=echo_to_screen) 

144 

145 # Paths for all these ASDM should be the same 

146 # Add check for this ? 

147 

148 # Beginning execution 

149 casa_tools.post_to_log("\nStarting procedure execution ...\n", echo_to_screen=echo_to_screen) 

150 casa_tools.post_to_log("Procedure name: " + procedureName + "\n", echo_to_screen=echo_to_screen) 

151 

152 # Names of import tasks that need special treatment: 

153 import_tasks = ('h_importdata', 'hifa_importdata', 'hifv_importdata') 

154 restore_tasks = ('h_restoredata', 'hifv_restoredata') 

155 breakpoint_task = 'breakpoint' 

156 # Loop over the commands 

157 for command in commandsList: 

158 

159 # Get task name and arguments lists. 

160 pipeline_task_name = command[0] 

161 task_args = command[1] 

162 casa_tools.set_log_origin(fromwhere=pipeline_task_name) 

163 

164 # PIPE-2388: Skipping breakpoint command 

165 if pipeline_task_name == breakpoint_task: 

166 casa_tools.post_to_log("Found {}, not performing any action.".format(pipeline_task_name), echo_to_screen=echo_to_screen) 

167 continue 

168 

169 # Execute the command 

170 casa_tools.post_to_log("Executing command ..." + pipeline_task_name, echo_to_screen=echo_to_screen) 

171 try: 

172 pipeline_task = cli.get_pipeline_task_with_name(pipeline_task_name) 

173 

174 # List parameters 

175 for keyword, value in task_args.items(): 

176 casa_tools.post_to_log(" Parameter: " + keyword + " = " + repr(value), echo_to_screen=echo_to_screen) 

177 

178 # For import/restore tasks, set vis and session explicitly (not inferred from context). 

179 if pipeline_task_name in import_tasks or pipeline_task_name in restore_tasks: 

180 task_args['vis'] = files 

181 task_args['session'] = sessions 

182 

183 # If spectral mode is set to True, skip the Hanning task. 

184 spectral_mode = intentsDict.get('SPECTRAL_MODE', False) 

185 if spectral_mode and pipeline_task_name == 'hifv_hanning': 

186 casa_tools.post_to_log("SPECTRAL_MODE=True. Hanning smoothing will not be executed.") 

187 continue 

188 

189 results = pipeline_task(**task_args) 

190 casa_tools.post_to_log('Results ' + str(results), echo_to_screen=echo_to_screen) 

191 

192 if importonly and pipeline_task_name in import_tasks: 

193 casa_tools.post_to_log("Terminating execution after running " + pipeline_task_name, 

194 echo_to_screen=echo_to_screen) 

195 break 

196 

197 except Exception: 

198 # Log message if an exception occurred that was not handled by 

199 # standardtask template (not turned into failed task result). 

200 casa_tools.post_to_log("Unhandled error in executevlappr while running pipeline task {}" 

201 "".format(pipeline_task_name), echo_to_screen=echo_to_screen) 

202 errstr = traceback.format_exc() 

203 casa_tools.post_to_log(errstr, echo_to_screen=echo_to_screen) 

204 errorfile = utils.write_errorexit_file(workingDir, 'errorexit', 'txt') 

205 break 

206 

207 # Stop execution if result is a failed task result or a list 

208 # containing a failed task result. 

209 tracebacks = utils.get_tracebacks(results) 

210 if len(tracebacks) > 0: 

211 # Save the context 

212 cli.h_save() 

213 

214 casa_tools.set_log_origin(fromwhere='') 

215 

216 errorfile = utils.write_errorexit_file(workingDir, 'errorexit', 'txt') 

217 previous_tracebacks_as_string = "{}".format("\n".join([tb for tb in tracebacks])) 

218 raise exceptions.PipelineException(previous_tracebacks_as_string) 

219 

220 # Save the context 

221 cli.h_save() 

222 

223 casa_tools.post_to_log("Terminating procedure execution ...", echo_to_screen=echo_to_screen) 

224 

225 casa_tools.set_log_origin(fromwhere='') 

226 

227 return

230# Return the intents list, the ASDM list, and the processing commands 

231# for the first processing request. There should in general be only 

232# one but the schema permits more. Generalize later if necessary. 

233def _getFirstRequest(pprXmlFile: 'XmlObject') -> Union[Tuple[list, str, dict, list, list], Tuple[list, list, str, dict, list, str, list]]: 

234 # Initialize 

235 info = [] 

236 relativePath = "" 

237 intentsDict = {} 

238 commandsList = [] 

239 asdmList = [] 

240 

241 # Turn the XML file into an object 

242 pprObject = _getPprObject(pprXmlFile=pprXmlFile) 

243 

244 # Count the processing requests. 

245 numRequests = _getNumRequests(pprObject=pprObject) 

246 if numRequests <= 0: 

247 casa_tools.post_to_log("Terminating execution: No valid processing requests") 

248 return info, relativePath, intentsDict, asdmList, commandsList 

249 elif numRequests > 1: 

250 casa_tools.post_to_log("Warning: More than one processing request") 

251 casa_tools.post_to_log('Number of processing requests: ', numRequests) 

252 

253 # Get brief project summary 

254 info = _getProjectSummary(pprObject) 

255 

256 # Get project structure. Set to empty list for VLA. 

257 structure = [] 

258 

259 # Get the intents dictionary 

260 numIntents, intentsDict = _getIntents(pprObject=pprObject, requestId=0, numRequests=numRequests) 

261 casa_tools.post_to_log('Number of intents: {}'.format(numIntents)) 

262 casa_tools.post_to_log('Intents dictionary: {}'.format(intentsDict)) 

263 

264 # Get the commands list 

265 procedureName, numCommands, commandsList = _getCommands(pprObject=pprObject, requestId=0, numRequests=numRequests) 

266 casa_tools.post_to_log('Number of commands: {}'.format(numCommands)) 

267 casa_tools.post_to_log('Commands list: {}'.format(commandsList)) 

268 

269 # Count the scheduling block sets. Normally there should be only 

270 # one although the schema allows multiple sets. Check for this 

271 # condition and process only the first. 

272 numSbSets = _getNumSchedBlockSets(pprObject=pprObject, requestId=0, numRequests=numRequests) 

273 if numSbSets <= 0: 

274 casa_tools.post_to_log("Terminating execution: No valid scheduling block sets") 

275 return info, relativePath, intentsDict, asdmList, commandsList 

276 elif numSbSets > 1: 

277 casa_tools.post_to_log("Warning: More than one scheduling block set") 

278 casa_tools.post_to_log('Number of scheduling block sets: {}'.format(numSbSets)) 

279 

280 # Get the ASDM list 

281 relativePath, numAsdms, asdmList = _getAsdmList(pprObject=pprObject, numSbSets=numSbSets, numRequests=numRequests) 

282 casa_tools.post_to_log('Relative path: {}'.format(relativePath)) 

283 casa_tools.post_to_log('Number of Asdms: {}'.format(numAsdms)) 

284 casa_tools.post_to_log('ASDM list: {}'.format(asdmList)) 

285 

286 return info, structure, relativePath, intentsDict, asdmList, procedureName, commandsList

303# Given the pipeline processing request object return the number of processing 

304# requests. For EVLA this should always be 1 but check. Assume a single scheduling block 

305# per processing request. 

306def _getNumRequests(pprObject: 'XmlObject') -> int: 

307 ppr_prequests = pprObject.SciPipeRequest.ProcessingRequests 

308 

309 numRequests = 0 

310 

311 # Try single element / single scheduling block first. 

312 try: 

313 relative_path = ppr_prequests.ProcessingRequest.DataSet.RelativePath.getValue() 

314 numRequests = 1 

315 return numRequests 

316 except Exception: 

317 pass 

318 

319 # Next try multiple processing requests / single scheduling block 

320 search = 1 

321 while search: 

322 try: 

323 relative_path = ppr_prequests.ProcessingRequest[numRequests].DataSet.RelativePath.getValue() 

324 numRequests = numRequests + 1 

325 except Exception: 

326 search = 0 

327 if numRequests > 0: 

328 return numRequests 

329 else: 

330 pass 

331 

332 # Return the number of requests. 

333 return numRequests
