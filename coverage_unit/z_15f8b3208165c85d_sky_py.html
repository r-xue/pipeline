49def plotfilename(image, reportdir, collapseFunction=None, stokes=None): 

50 """Generate a filename for a plot based on image and other parameters. 

51 

52 Args: 

53 image (str): Path to the image file. 

54 reportdir (str): Directory where the report will be saved. 

55 collapseFunction (Optional[str]): Collapse function used, if any. 

56 stokes (Optional[str]): Stokes parameter, if specified. 

57 

58 Returns: 

59 str: The generated filename for the plot.  

60  

61 Note: 

62 The Stokes suffix is only attached when explicitly specified (PIPE-1410).  

63 """ 

64 

65 name_elements = [os.path.basename(image)] 

66 name_elements.extend(filter(None, [collapseFunction, stokes, 'sky', 'png'])) 

67 name = os.path.join(reportdir, filenamer.sanitize('.'.join(name_elements), valid_chars=_valid_chars)) 

68 

69 return name

81 def plot_per_stokes(self, *args, stokes_list: Optional[List[str]] = None, **kwargs) -> List: 

82 """Plot sky images from a CASA image file with multiple Stokes planes. 

83 

84 Args: 

85 *args: Positional arguments to be passed to self.plot(). 

86 stokes_list (Optional[List[str]]): List of Stokes parameters to plot. 

87 If None, all available Stokes planes will be plotted. 

88 **kwargs: Keyword arguments to be passed to self.plot(). 

89 

90 Returns: 

91 List: A list of plot objects, one for each Stokes parameter. 

92 

93 Description: 

94 This method generates one plot per Stokes parameter for a given image file. 

95 It uses self.plot() to create each individual plot. 

96 

97 Note: 

98 PIPE-1401: A new keyword argument 'stokes' is added to SkyDisplay.plot(), 

99 SkyDisplay._plot_panel(), etc. By default (stokes=None), the behavior 

100 before PIPE-1401 is preserved: no additional stokes-related suffix in .png 

101 or 'stokes' key in the plot wrapper object. 

102 """ 

103 if stokes_list is None: 

104 stokes_list = get_stokes(args[1]) 

105 

106 return [self.plot(*args, stokes=stokes, **kwargs) for stokes in stokes_list]

108 def _get_default_dpi(self, context): 

109 """Get the default DPI (dots per inch) for imaging plots. 

110 

111 Args: 

112 context: The Pipeline context object. 

113 

114 Returns: 

115 float or None: The default DPI value. 400.0 for VLA imaging plots, 

116 self._dpi if set, or None otherwise. 

117 

118 Description: 

119 This method determines the default DPI for imaging plots based on 

120 the context and instance settings. It prioritizes the instance's 

121 _dpi attribute if set, otherwise uses context-based logic. 

122 

123 Note: 

124 For VLA hif_makeimages sky plots, the default is 400 DPI (PIPE-1083). 

125 """ 

126 

127 # class instance default takes precedence over the context-based default 

128 if self._dpi is not None: 

129 return self._dpi 

130 

131 last_result = context.results[-1] 

132 

133 # PIPE-1083: when making VLA/SS hif_makeimages sky plot, we default to 400 dpi 

134 if last_result.taskname == 'hif_makeimages': 

135 if last_result.results: 

136 first_result = last_result.results[0] 

137 if first_result.imaging_mode in ('VLA', 'EVLA', 'JVLA') or first_result.imaging_mode.startswith('VLASS'): 

138 return 400.0 

139 

140 # PIPE-1083: when making sky plots for the tasks/stages below, we default to 400 dpi. 

141 # hif_makermsimages, hif_makecutoutsimages, and hifv_pbcor 

142 if last_result.taskname in ('hif_makermsimages', 'hif_makecutoutimages', 'hifv_pbcor'): 

143 return 400.0 

144 

145 return None

147 def _get_vla_band(self, context, miscinfo) -> str | None: 

148 """Get the VLA band string, only for VLA aggregated cont imaging.""" 

149 last_result = context.results[-1] 

150 

151 # Check if we should use band notation based on task and imaging mode 

152 use_band_notation = ( 

153 # hif_makeimage VLA-PI cont imaging sky plots 

154 ( 

155 last_result.taskname == 'hif_makeimages' 

156 and last_result.results[0].imaging_mode in {'VLA', 'EVLA', 'JVLA'} 

157 and miscinfo.get('specmode') == 'cont' 

158 ) 

159 or 

160 # hifv_pbcor VLA-PI cont imaging sky plots 

161 ( 

162 last_result.taskname == 'hifv_pbcor' 

163 and context.imaging_mode is None 

164 and miscinfo.get('specmode') == 'cont' 

165 ) 

166 ) 

167 

168 if not use_band_notation: 

169 return None 

170 

171 # Get measurement set and band mapping - VLA has only 1 MS 

172 ms = context.observing_run.get_measurement_sets()[0] 

173 spw2band = ms.get_vla_spw2band() 

174 virtspw_str = miscinfo.get('virtspw', '') 

175 if not virtspw_str: 

176 return None 

177 

178 bands = {spw2band[spw_id] for spw in virtspw_str.split(',') if (spw_id := int(spw)) in spw2band} 

179 

180 # VLA imaging happens per-band, typically resulting in single-element set 

181 return ','.join(bands) if bands else None

183 def plot(self, context, imagename, reportdir, intent=None, collapseFunction='mean', 

184 stokes: Optional[str] = None, vmin=None, vmax=None, mom8_fc_peak_snr=None, 

185 maskname=None, dpi=None, **imshow_args): 

186 """Plot sky images from a image file.""" 

187 

188 if not imagename: 

189 return [] 

190 

191 if vmin is not None and vmax is not None: 

192 imshow_args['norm'] = plt.Normalize(vmin, vmax, clip=True) 

193 

194 # The dpi input from a method call takes precedence over the class/context-based default. 

195 if dpi is not None: 

196 dpi_savefig = dpi 

197 else: 

198 dpi_savefig = self._get_default_dpi(context) 

199 

200 plotfile, coord_names, field, band = self._plot_panel(context, reportdir, imagename, collapseFunction=collapseFunction, 

201 stokes=stokes, 

202 mom8_fc_peak_snr=mom8_fc_peak_snr, 

203 maskname=maskname, dpi=dpi_savefig, **imshow_args) 

204 

205 # field names may not be unique, which leads to incorrectly merged 

206 # plots in the weblog output. As a temporary fix, change to field + 

207 # intent - which is better but again, not guaranteed unique. 

208 if intent: 

209 field = f'{field!s} ({intent!s})' 

210 

211 with casa_tools.ImageReader(imagename) as image: 

212 miscinfo = image.miscinfo() 

213 

214 parameters = {k: miscinfo[k] for k in ['virtspw', 'pol', 'field', 'datatype', 'type', 'iter'] if k in miscinfo} 

215 parameters['ant'] = None 

216 parameters['band'] = band 

217 parameters['moment'] = collapseFunction 

218 if isinstance(stokes, str): 

219 # PIPE-1401: only save the 'stokes' keyword when it was explicitly requested. 

220 parameters['stokes'] = stokes 

221 try: 

222 parameters['prefix'] = os.path.basename(imagename).split('.')[0] 

223 except: 

224 parameters['prefix'] = None 

225 

226 plot = logger.Plot(plotfile, x_axis=coord_names[0], 

227 y_axis=coord_names[1], field=field, 

228 parameters=parameters) 

229 

230 return plot

232 def _collapse_image(self, imagename, collapseFunction='mean', stokes: Optional[str] = None): 

233 """Collapse an image along the spectral axis.""" 

234 

235 stokes_present = get_stokes(imagename) 

236 if stokes not in stokes_present: 

237 stokes_select = stokes_present[0] 

238 else: 

239 stokes_select = stokes 

240 

241 with casa_tools.ImageReader(imagename) as image: 

242 collapsed = image.collapse(function=collapseFunction, axes=2, stokes=stokes_select) 

243 data = collapsed.getchunk(dropdeg=True) 

244 mask = np.invert(collapsed.getchunk(getmask=True, dropdeg=True)) 

245 mdata = np.ma.array(data, mask=mask) 

246 collapsed.done() 

247 return mdata

249 @matplotlibrc_formal 

250 def _plot_panel(self, context, reportdir, imagename, collapseFunction='mean', 

251 stokes: Optional[str] = None, mom8_fc_peak_snr=None, 

252 maskname=None, dpi=None, **imshow_args): 

253 """Method to plot a map.""" 

254 plotfile = plotfilename(image=os.path.basename(imagename), 

255 reportdir=reportdir, collapseFunction=collapseFunction, stokes=stokes) 

256 LOG.info('Plotting %s to %s', imagename, plotfile) 

257 

258 stokes_present = get_stokes(imagename) 

259 if stokes not in stokes_present: 

260 stokes_select = stokes_present[0] 

261 # PIPE-1401: plot mask sky images for different stokes plane even when the mask file has a single Stokes plane. 

262 # note: the fallback is required for vlass-se-cube because the user input mask is from vlass-se-cont with only Stokes=I. 

263 if isinstance(stokes, str): 

264 LOG.warning('Stokes %s is requested, but only Stokes=%s is present.', stokes, stokes_present) 

265 LOG.warning('We will try to create a plot with a fallback of Stokes=%s.', stokes_present) 

266 else: 

267 LOG.info( 

268 'No Stokes selection is specified, we will use the first present Stokes plane: Stokes=%s.', 

269 stokes_select, 

270 ) 

271 else: 

272 stokes_select = stokes 

273 LOG.info(f'Stokes={stokes_select} is selected.') 

274 

275 if not os.path.exists(plotfile) or self.overwrite: 

276 with casa_tools.ImageReader(imagename) as image: 

277 _, _, _, nchan = image.shape() 

278 if nchan == 1: 

279 # Use Stokes-axis slicing instead of frequency collapsing 

280 rgtool = casa_tools.regionmanager 

281 region = rgtool.frombcs(csys=image.coordsys().torecord(), shape=image.shape(), 

282 stokes=stokes_select, stokescontrol='a') 

283 collapsed = image.subimage(region=region) 

284 else: 

285 try: 

286 if collapseFunction == 'center': 

287 collapsed = image.collapse(function='mean', chans=str( 

288 image.summary()['shape'][3]//2), stokes=stokes_select, axes=3) 

289 elif collapseFunction == 'mom0': 

290 # image.collapse does not have a true "mom0" option. "sum" is close, but the 

291 # scaling is different. 

292 # TODO: Switch the whole _plot_panel method to using immoments(?) Though the 

293 # downside is that images can no longer be made just in memory. They 

294 # always have to be written to disk. 

295 tmpfile = f'{os.path.basename(imagename)}_mom0_tmp.img' 

296 casa_tasks.immoments(imagename=imagename, moments=[ 

297 0], outfile=tmpfile, stokes=stokes_select).execute() 

298 assert os.path.exists(tmpfile) 

299 collapsed = image.newimagefromimage(infile=tmpfile) 

300 shutil.rmtree(tmpfile) 

301 elif collapseFunction == 'mom8': 

302 collapsed = image.collapse(function='max', stokes=stokes_select, axes=3) 

303 else: 

304 # Note: in case 'max' and non-pbcor image a moment 0 map was written to disk 

305 # in the past. With PIPE-558 this is done in hif/tasks/tclean.py tclean._calc_mom0_8() 

306 collapsed = image.collapse(function=collapseFunction, stokes=stokes_select, axes=3) 

307 except Exception as ex: 

308 LOG.info(ex) 

309 traceback_msg = traceback.format_exc() 

310 LOG.debug(traceback_msg) 

311 # All channels flagged or some other error. Make collapsed zero image. 

312 collapsed_new = image.newimagefromimage(infile=imagename) 

313 collapsed_new.set(pixelmask=True, pixels='0') 

314 collapsed = collapsed_new.collapse(function='mean', stokes=stokes_select, axes=3) 

315 collapsed_new.done() 

316 else: 

317 collapsed = casa_tools.image 

318 collapsed.open(imagename) 

319 

320 cs = collapsed.coordsys() # needs to explicitly close later 

321 coord_names = cs.names() 

322 miscinfo = collapsed.miscinfo() 

323 

324 vla_cont_band = self._get_vla_band(context, miscinfo) 

325 if vla_cont_band is not None: 

326 # VLA-specmode='cont' only, not triggered for ALMA, VLASS. 

327 miscinfo['band'] = vla_cont_band 

328 else: 

329 # Use the reference-frequencey (in Hz) as the fallback value of the 'band' key 

330 # This key is only used for VLASS and VLA so will not affect ALMA 

331 miscinfo['band'] = cs.referencevalue(format='n')['numeric'][3] 

332 

333 field_name = miscinfo.get('field', None) 

334 band_name = miscinfo.get('band', None) 

335 

336 # early abort: don't replot if a file with the required name already exists 

337 if os.path.exists(plotfile) and not self.overwrite: 

338 LOG.info('plotfile already exists: %s', plotfile) 

339 # done with ia/cs tool instances 

340 collapsed.done() 

341 cs.done() 

342 return plotfile, coord_names, field_name, band_name 

343 

344 cs.setunits(type='direction', value='arcsec arcsec') 

345 coord_units = cs.units() 

346 coord_refs = cs.referencevalue(format='s') 

347 brightness_unit = collapsed.brightnessunit() 

348 beam_rec = collapsed.restoringbeam() 

349 if 'major' in beam_rec: 

350 cqa = casa_tools.quanta 

351 bpa = cqa.convert(beam_rec['positionangle'], 'deg')['value'] 

352 bmaj = cqa.convert(beam_rec['major'], 'arcsec')['value'] 

353 bmin = cqa.convert(beam_rec['minor'], 'arcsec')['value'] 

354 beam = [bmaj, bmin, bpa] 

355 else: 

356 beam = None 

357 

358 # otherwise do the plot 

359 data = collapsed.getchunk() 

360 mask = np.invert(collapsed.getchunk(getmask=True)) 

361 collapsed.done() 

362 shape = np.shape(data) 

363 data = data.reshape(shape[0], shape[1]) 

364 mask = mask.reshape(shape[0], shape[1]) 

365 mdata = np.ma.array(data, mask=mask) 

366 

367 # get tl/tr corner positions in offset 

368 blc = cs.torel(cs.toworld([-0.5, -0.5, 0, 0]))['numeric'] 

369 trc = cs.torel(cs.toworld([shape[0]-0.5, shape[1]-0.5, 0, 0]))['numeric'] 

370 

371 # remove any incomplete matplotlib plots, if left these can cause weird errors 

372 plt.close('all') 

373 

374 # PIPE-2843: disable constrained layout to prevent potential bad interaction with AnnotationBbox 

375 # Also see: https://github.com/matplotlib/matplotlib/issues/24453 

376 fig, ax = plt.subplots(figsize=self.figsize, constrained_layout=False) 

377 

378 # plot data 

379 if 'cmap' not in imshow_args: 

380 # matplotlib Colormap has its own __copy__ implementation: 

381 # https://github.com/matplotlib/matplotlib/blob/v3.3.x/lib/matplotlib/colors.py#L616 

382 imshow_args['cmap'] = copy.copy(plt.cm.jet) 

383 imshow_args['cmap'].set_bad('k', 1.0) 

384 im = ax.imshow(mdata.T, interpolation='nearest', origin='lower', aspect='equal', 

385 extent=[blc[0], trc[0], blc[1], trc[1]], **imshow_args) 

386 

387 if maskname is not None and os.path.exists(maskname): 

388 mdata_mask = self._collapse_image(maskname) 

389 ax.contour(mdata_mask.T, [0.99], origin='lower', colors='white', linewidths=0.7, 

390 extent=[blc[0], trc[0], blc[1], trc[1]]) 

391 

392 ax.axis('image') 

393 lims = ax.axis() 

394 

395 # make ticks and labels white 

396 for line in ax.xaxis.get_ticklines() + ax.yaxis.get_ticklines(): 

397 line.set_color('white') 

398 for labels in ax.xaxis.get_ticklabels() + ax.yaxis.get_ticklabels(): 

399 labels.set_fontsize(0.5 * labels.get_fontsize()) 

400 

401 # colour bar 

402 if self.exclude_desc: 

403 height = '80%' 

404 else: 

405 height = '50%' 

406 cax = inset_axes( 

407 ax, 

408 width='4%', 

409 height=height, 

410 loc='center left', 

411 bbox_to_anchor=(1.01, 0.0, 1, 1), 

412 bbox_transform=ax.transAxes, 

413 ) 

414 cb = plt.colorbar(im, cax=cax) 

415 fontsize = 8 

416 cb.ax.tick_params(labelsize=fontsize) 

417 cb.set_label(brightness_unit, fontsize=fontsize) 

418 cb.ax.yaxis.set_offset_position('left') 

419 

420 # plot beam 

421 if beam is not None: 

422 beam_patch = mpatches.Ellipse((lims[0] + 0.1 * (lims[1]-lims[0]), lims[2] + 0.1 * (lims[3]-lims[2])), 

423 width=beam[1], height=beam[0], 

424 linestyle='solid', edgecolor='yellow', fill=False, 

425 angle=-beam[2]) 

426 ax.add_patch(beam_patch) 

427 

428 # add xy labels 

429 ax.set_xlabel(f'{coord_names[0]} ({coord_units[0]})') 

430 ax.set_ylabel(f'{coord_names[1]} ({coord_units[1]})') 

431 

432 # Add the color text box 

433 if not self.exclude_desc: 

434 # image reference pixel 

435 yoff = -0.15 

436 xoff = 0.0 

437 yoff = SkyDisplay.plottext(cb.ax, xoff, yoff, 'Reference Position:', 40) 

438 for i, k in enumerate(coord_refs['string']): 

439 # note: the labels present the reference value at individual axes of the collapsed image 

440 # https://casadocs.readthedocs.io/en/latest/api/tt/casatools.image.html#casatools.image.image.collapse 

441 name_abbreviations = { 

442 'Right Ascension': ' R.A.', 

443 'Declination': ' Dec.', 

444 } 

445 coord_name = coord_names[i] 

446 for name, abbr in name_abbreviations.items(): 

447 coord_name = coord_name.replace(name, abbr) 

448 yoff = SkyDisplay.plottext(cb.ax, xoff, yoff, f'{coord_name}: {k}', 40, mult=0.8) 

449 # if peaksnr is available for the mom8_fc image, include it in the plot 

450 if 'mom8_fc' in imagename and mom8_fc_peak_snr is not None: 

451 yoff = 1.15 

452 SkyDisplay.plottext(cb.ax, xoff, yoff, 'Peak SNR: {:.5f}'.format(mom8_fc_peak_snr), 40) 

453 

454 mode_texts = { 

455 'mean': 'mean', 

456 'mom0': 'integ. line int. (mom0)', 

457 'max': 'peak line int. (mom8)', 

458 'mom8': 'peak line int. (mom8)', 

459 'center': 'center slice' 

460 } 

461 

462 image_info = {'display': mode_texts[collapseFunction]} 

463 image_info.update(miscinfo) 

464 # PIPE-2708: improve the virtspw key formatting 

465 if image_info.get('virtspw') is not None: 

466 image_info['virtspw'] = utils.find_ranges(image_info['virtspw']) 

467 

468 type_mapping = { 

469 'flux': 'pb', 

470 'mom0_fc': 'Line-free Moment 0', 

471 'mom8_fc': 'Line-free Moment 8' 

472 } 

473 

474 if 'type' in image_info: 

475 image_info['type'] = type_mapping.get(image_info['type'], image_info['type']) 

476 

477 if isinstance(image_info.get('band', None), str): 

478 # Currently only used for VLA specmode='cont' imaging results 

479 key_color = [('type', 'k'), 

480 ('display', 'r'), 

481 ('field', 'k'), 

482 ('band', 'k'), 

483 ('pol', 'k'), 

484 ('iter', 'k')] 

485 else: 

486 key_color = [('type', 'k'), 

487 ('display', 'r'), 

488 ('field', 'k'), 

489 ('virtspw', 'k'), 

490 ('pol', 'k'), 

491 ('iter', 'k')] 

492 labels = [TextArea('%s:%s' % (key, image_info[key]), textprops=dict(color=color)) 

493 for key, color in key_color 

494 if image_info.get(key) is not None] 

495 

496 txt = HPacker(children=labels, align="baseline", pad=0, sep=7) 

497 bbox = AnnotationBbox(txt, xy=(0.5, 1.04), 

498 xycoords='axes fraction', 

499 frameon=True, 

500 box_alignment=(0.5, 0.0), 

501 pad=0.3) 

502 ax.add_artist(bbox) 

503 

504 # PIPE-997: plot a 41pix-wide PSF inset if the image is larger than 41*3 

505 if miscinfo.get('type', None) == 'psf': 

506 # use the same vmin/vmax as the full-size plot. 

507 if 'norm' not in imshow_args: 

508 vmin, vmax = im.get_clim() 

509 imshow_args['norm'] = plt.Normalize(vmin, vmax, clip=True) 

510 self._plot_psf_inset(ax, mdata, imshow_args, beam=beam, cs=cs) 

511 

512 # save the image 

513 fig.savefig(plotfile, bbox_inches='tight', dpi=dpi, pil_kwargs={'optimize': True}) 

514 plt.close(fig) 

515 

516 if not os.path.exists(plotfile): 

517 # PIPE-2022: Generate a warning if the PNG file is missing. The 

518 # message is caught by a local logging handler for the weblog. 

519 LOG.warning(f'Plot {plotfile} is missing on disk') 

520 

521 # done with ia.coordsys() 

522 cs.done() 

523 

524 return plotfile, coord_names, field_name, band_name

526 def _plot_psf_inset(self, ax, mdata, imshow_args, beam=None, cs=None): 

527 """Plot the PSF inset panel.""" 

528 

529 npix_inset_half = 20 

530 maxfrac_inset = 1./3 

531 npix_inset = npix_inset_half*2.+1 

532 shape = mdata.shape 

533 if npix_inset < shape[0]*maxfrac_inset or npix_inset < shape[1]*maxfrac_inset: 

534 x0 = shape[0]//2-npix_inset_half 

535 x1 = shape[0]//2+npix_inset_half+1 

536 y0 = shape[1]//2-npix_inset_half 

537 y1 = shape[1]//2+npix_inset_half+1 

538 mdata_sub = mdata[x0:x1, y0:y1] 

539 axinset = ax.inset_axes(bounds=[0.60, 0.02, 0.38, 0.38])

569 def plottext(ax, xoff, yoff, text, maxchars, mult=1.0, line_spacing=0.04): 

570 """Plots text on a Matplotlib Axes object with automatic line breaks. 

571 

572 This function is designed to be robust and uses standard library 

573 tools for text wrapping. 

574 

575 Args: 

576 ax (matplotlib.axes.Axes): The Axes object to plot the text on. 

577 xoff (float): World x-coordinate where text is to start (in axes coordinates, 0-1). 

578 yoff (float): World y-coordinate where text is to start (in axes coordinates, 0-1). 

579 text (str): The text string to display. 

580 maxchars (int): The maximum number of characters before a newline is inserted. 

581 mult (float, optional): Factor by which the text fontsize is to be multiplied. 

582 Defaults to 1.0. 

583 line_spacing (float, optional): Vertical spacing between lines in axes coordinates. 

584 Defaults to 0.04. 

585 

586 Returns: 

587 float: The final y-coordinate after plotting the text. 

588 """ 

589 lines = textwrap.wrap(text, width=maxchars) 

590 

591 current_yoff = yoff 

592 

593 for line in lines: 

594 ax.text( 

595 xoff, 

596 current_yoff, 

597 line, 

598 va='top', 

599 fontsize=mult * 8, transform=ax.transAxes, 

600 clip_on=False, 

601 ) 

602 current_yoff -= line_spacing * mult 

603 

604 current_yoff -= line_spacing * mult 

605 

606 return current_yoff
