103 def _identify_best_refant(self, session_name, vislist, nant): 

104 """ 

105 Identify best reference antenna for specified list of measurement sets. 

106 

107 First, the reference antenna lists of all measurement sets are combined 

108 into a single ranked list. 

109 

110 Secondly, the 'nant' highest ranked antennas are tested to see whether 

111 CASA gaincal refant heuristics would accept the antenna or instead 

112 would pick a different refant for one or more solutions. For this test, 

113 a phase caltable is created with refant set to a specific ant, and this 

114 phase caltable is evaluated to count how many rows have "non-zero" 

115 phases (absolute phase above a threshold) for the specified antenna. 

116 These "non-zero" phases are taken as an indirect indication that CASA 

117 gaincal overrode the specified antenna for one or more rows (e.g. spw). 

118 

119 If an antenna is found to cause no non-zero phase deviations for 

120 any MS, then it is immediately returned as the highest ranking 

121 "perfect" reference antenna. 

122 

123 Otherwise, once the 'nant' highest ranking antennas (or all available 

124 ants, whichever is lowest nr.) are evaluated for suitability, then 

125 the antenna with the lowest total number of non-zero phases is chosen 

126 as the final "best" reference antenna for this session. 

127 

128 If the measurement sets have only a single reference antenna in 

129 common, then this antenna is immediately returned as the "best" 

130 reference antenna, without running the "non-zero" phase evaluation. 

131 

132 If the measurement sets have no reference antennas in common, then an 

133 empty string is returned for "best" reference antenna. 

134 

135 :param session_name: name of session 

136 :type session_name: str 

137 :param vislist: list of measurement sets 

138 :type vislist: list [vis, vis, ...] 

139 :param nant: maximum number of antennas to evaluate phases for 

140 :type nant: int 

141 :return: name of best reference antenna 

142 :rtype: str 

143 """ 

144 # Create ranked refant list. 

145 refants = self._create_combined_refant_list(session_name, vislist) 

146 

147 # Handle the edge cases of MSes having only 0 or 1 refant in common. 

148 nrefants = len(refants) 

149 if nrefants == 0: 

150 LOG.warning("Measurement sets for session \"{}\" have no reference antennas in common, cannot determine a" 

151 " final best reference antenna.".format(session_name)) 

152 return '' 

153 elif nrefants == 1: 

154 LOG.warning("Measurement sets for session \"{}\" have only one reference antennas in common ({}), which" 

155 " will be set as the final best reference antenna.".format(session_name, refants[0])) 

156 return refants[0] 

157 

158 # If there are multiple candidate refants in common, then continue with 

159 # evaluating the ranked list of refants based on non-zero phases. 

160 n_nonzero = collections.defaultdict(int) 

161 

162 # If the maximum number of antennas to evaluate phases for is less than 

163 # one, then no best refant can be found. 

164 if nant < 1: 

165 LOG.warning("Unable to find best reference antenna, maximum number of antennas to evaluate phases for is" 

166 " set too low: {}".format(nant)) 

167 return '' 

168 

169 # Run phase evaluation for specified nr. of antennas or nr. of 

170 # antennas in common, whichever is lowest. 

171 for iant in range(min(nant, nrefants)): 

172 ant = refants[iant] 

173 LOG.info("Session \"{}\": running phase evaluation heuristics for candidate antenna {}" 

174 "".format(session_name, ant)) 

175 

176 # Run a gaincal for each vis with refant set to the current ant. 

177 gcal_result = self._create_phase_caltables(vislist, ant) 

178 

179 # Evaluate caltables to detect if refant changed. 

180 n_nonzero_phase = self._evaluate_phase_caltables(gcal_result, ant, self.inputs.phase_threshold) 

181 

182 # In the (unlikely) case that no refant changes (i.e. no non-zero 

183 # phases) were detected for any MS, then this ant is the highest 

184 # ranking "perfect" reference antenna, and there is no need to 

185 # evaluate any further down the list. 

186 if not any(n_nonzero_phase.values()): 

187 LOG.info("Session \"{}\": no phase outlier rows found for candidate antenna {}, so will select" 

188 " this as the highest ranking best reference antenna.".format(session_name, ant)) 

189 return ant 

190 

191 # Store total nr. of non-zero phases for this antenna. 

192 n_nonzero[ant] = sum(n_nonzero_phase.values()) 

193 LOG.info("Session \"{}\": number of phase outlier rows found for candidate antenna {}: {}" 

194 "".format(session_name, ant, n_nonzero[ant])) 

195 

196 # From evaluated antennas, pick the one with the lowest total number of non-zero phase values; 

197 best_refant = sorted(n_nonzero, key=n_nonzero.get)[0] 

198 

199 # If the best choice of reference antenna still resulted in non-zero 

200 # phases, then log a warning. 

201 if n_nonzero[best_refant] != 0: 

202 LOG.warning("Session \"{}\": final choice of best reference antenna ({}) resulted in {} phase outliers." 

203 "".format(session_name, best_refant, n_nonzero[best_refant])) 

204 

205 return best_refant

207 def _create_combined_refant_list(self, session_name, vislist): 

208 """ 

209 Combines the ranked reference antenna lists for all measurement sets 

210 specified in 'vislist' to return as a single ranked reference antenna 

211 list. 

212 

213 :param session_name: name of session 

214 :type session_name: str 

215 :param vislist: list of measurement sets 

216 :type vislist: list [vis, vis, ...] 

217 :return: list of reference antennas 

218 :rtype: list[str] 

219 """ 

220 # If there is just one MS, then return its reference antenna list. 

221 nvis = len(vislist) 

222 if nvis == 1: 

223 LOG.info("Session \"{}\" has only one measurement set ({}), continuing with its reference antenna list for" 

224 " evaluation of best reference antenna.".format(session_name, os.path.basename(vislist[0]))) 

225 return self.inputs.context.observing_run.get_ms(vislist[0]).reference_antenna.split(',') 

226 

227 # Otherwise, continue with combining the multiple refant lists. 

228 

229 # Sort vislist by start time of observation; this will allow for 

230 # ordering by highest ranking in first observed MS, in cases where 

231 # other sorting (further below) has resulted in equal ranking. 

232 vislist = sorted(vislist, key=lambda v: self.inputs.context.observing_run.get_ms(v).start_time['m0']['value']) 

233 

234 # Collect refant list for all MSes. 

235 refant_lists = [self.inputs.context.observing_run.get_ms(vis).reference_antenna.split(',') for vis in vislist] 

236 

237 # Create dictionary of antennas, mapping each antenna to a list of 

238 # their per-MS ranking as refant. If an ant was 1st in the refant list, 

239 # it is assigned rank 1; 2nd in list is rank 2, # etc... 

240 refant_ranks = collections.defaultdict(list) 

241 for refant_list in refant_lists: 

242 for idx, ant in enumerate(refant_list): 

243 refant_ranks[ant].append(idx+1) 

244 

245 # For each ant in common among all MSes, retrieve/compute the following 

246 # information: 

247 # 1. name of ant 

248 # 2. the cross-product of their per-MS ranks 

249 # 3. the standard deviation of their per-MS ranks 

250 # 4. their refant rank for the first observed MS 

251 # (assumed to be first element in list, due to earlier sorting) 

252 antinfo = [(ant, reduce(lambda x, y: x * y, refant_ranks[ant]), np.std(refant_ranks[ant]), refant_ranks[ant][0]) 

253 for ant in refant_ranks if len(refant_ranks[ant]) == nvis] 

254 

255 # Create the final list of ranked reference antennas, after sorting 

256 # consecutively by cross-product of per-MS ranks, standard deviation 

257 # of per-MS ranks, and refant ranking for first-observed MS. 

258 refants_ranked = sorted(antinfo, key=operator.itemgetter(1, 2, 3)) 

259 

260 # Report the combined ranked antenna list. 

261 refant_str = ["{} ({}, {}, {})".format(*ant) for ant in refants_ranked] 

262 LOG.info("Ranked antenna list for session \"{}\", listed as 'antenna (cross-product of per-MS ranking, stdev" 

263 " of per-MS ranks, ranking for first observed MS)': {}".format(session_name, ', '.join(refant_str))) 

264 

265 # Return a list of just the antenna names. 

266 return [ant[0] for ant in refants_ranked]

268 def _create_phase_caltables(self, vislist, refant): 

269 """

272 

273 :param vislist: list of measurement sets 

274 :type vislist: list [vis, vis, ...]

279 container = vdp.InputsContainer(gaincal.GTypeGaincal, self.inputs.context, vis=vislist, intent="PHASE",

282 return self._executor.execute(gaincal_task)

284 def _evaluate_phase_caltables(self, results, antenna, threshold): 

285 """ 

286 Evaluate each phase caltable (one per measurement set) in specified 

287 'results' to determine for specified 'antenna' how many rows have 

288 absolute phases above the specified 'threshold'. 

289 

290 :param results: gaincal.GTypeGaincal Results 

291 :param antenna: name of antenna to check for non-zero phase outliers 

292 :type antenna: str 

293 :param threshold: threshold for detecting "non-zero" phase outliers 

294 :type threshold: int or float 

295 :return: dictionary with total number of non-zero phase outliers for 

296 each vis that has a caltable in results. 

297 """ 

298 nonzero_phase_per_vis = collections.defaultdict() 

299 for result in results: 

300 if result.final: 

301 calapp = result.final[0] 

302 

303 # Convert antenna name to ID. 

304 ms = self.inputs.context.observing_run.get_ms(calapp.vis) 

305 antenna_id = ms.get_antenna(antenna)[0].id 

306 

307 # Test whether antenna appears in caltable. 

308 with casa_tools.TableReader(calapp.gaintable) as tb: 

309 ants = tb.getcol('ANTENNA1') 

310 if antenna_id not in ants: 

311 LOG.warning("Antenna {} ({}) not found in caltable {}" 

312 "".format(antenna, antenna_id, os.path.basename(calapp.gaintable))) 

313 return 

314 

315 # Determine non-zero phases for each spw, and add to total. 

316 nonzero = 0 

317 for spw in calapp.spw.split(','): 

318 # Retrieve phase data from caltable for specified antenna and current spw. 

319 with casa_tools.TableReader(calapp.gaintable) as tb: 

320 taql = "ANTENNA1 == {} && SPECTRAL_WINDOW_ID == {}".format(antenna_id, spw) 

321 with contextlib.closing(tb.query(taql)) as subtb: 

322 phases = subtb.getcol('CPARAM') 

323 

324 # Add number of outlier phases to running total of "non-zero" phases. 

325 nonzero += len(np.where(np.abs(np.angle(phases, deg=True)) > threshold)[0]) 

326 

327 # Store total number of non-zero phase outliers for current vis. 

328 nonzero_phase_per_vis[calapp.vis] = nonzero 

329 

330 return nonzero_phase_per_vis
