151 def get_plot_wrapper(self): 

152 figfile = self.get_figfile() 

153 vis = os.path.basename(self.result.inputs['vis']) 

154 wrapper = logger.Plot(figfile, 

155 x_axis='Baseline length (m)', 

156 y_axis='Phase RMS (deg)', 

157 parameters={'vis': vis, 

158 'spw': self.spw, 

159 'scan': self.scan, 

160 'field': self.field, 

161 'desc': 'Baseline length vs. Phase RMS'}) 

162 

163 if not os.path.exists(figfile): 

164 LOG.trace('Phase RMS structure plot for vis %s not found. Creating new ' 

165 'plot: %s' % (vis, figfile)) 

166 try: 

167 self.create_plot() 

168 except Exception as ex: 

169 LOG.error('Could not create Phase RMS structure plot for' 

170 ' vis %s' % vis) 

171 LOG.exception(ex)

173 plt.clf()
