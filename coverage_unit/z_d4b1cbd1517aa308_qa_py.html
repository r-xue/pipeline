192 """

227 M4FACTORS = { 

228 'amp_vs_freq.intercept': 100.0, 

229 'amp_vs_freq.slope': 100.0 * 2.0, 

230 'phase_vs_freq.intercept': 100.0 / 180.0, 

231 'phase_vs_freq.slope': 100.0 * 2.0 / 180.0

233 QAEVALF_MIN = 0.33 

234 QAEVALF_MAX = 1.0

237 def __init__(self, ms: MeasurementSet, intents: list[str], outliers: list[Outlier]): 

238 # Save basic data and MeasurementSet domain object 

239 self.ms = ms 

240 

241 # Create the relevant vector array for QA scores evaluation, and save them 

242 self.outliers = outliers 

243 self.noutliers = len(outliers) 

244 self.metricnames = np.array([list(o.reason)[0].replace('gt90deg_offset_','') for o in outliers]) 

245 self.gt90degoffset = np.array([('gt90deg_offset' in list(o.reason)[0]) for o in outliers]) 

246 self.metricscores = np.array([np.abs(o.num_sigma) for o in outliers]) 

247 self.delta_phys = np.array([np.abs(o.delta_physical) for o in outliers]) 

248 self.is_amp_sym_off = np.array([o.amp_freq_sym_off for o in outliers]) 

249 self.metricthresholds = np.array([self.SCORE_THRESHOLDS[m] if m in self.SCORE_THRESHOLDS.keys() else 9999.0 for m in self.metricnames]) 

250 self.mtratio = self.metricscores/self.metricthresholds 

251 self.scan = np.array([list(o.scan)[0] for o in outliers]) 

252 self.spw = np.array([list(o.spw)[0] for o in outliers]) 

253 self.intent = np.array([list(o.intent)[0] for o in outliers]) 

254 self.ant = np.array([list(o.ant)[0] for o in outliers]) 

255 self.pol = np.array([list(o.pol)[0] for o in outliers]) 

256 # prototype operated on intents in ms, including all intents for scans with multiple intents 

257 self.allintents = frozenset(intents).intersection(ms.intents) 

258 self.long_msg = 'EVALUATE_TO_GET_LONGMSG' 

259 self.short_msg = 'EVALUATE_TO_GET_SHORTMSG' 

260 # Initialize metrics/data dictionary 

261 self.qascoremetrics = {} 

262 

263 for intent in self.allintents: 

264 self.qascoremetrics[intent] = {} 

265 for spw in ms.get_spectral_windows(intent=','.join(intents)): 

266 self.qascoremetrics[intent][spw.id] = {} 

267 for metric in self.SCORE_THRESHOLDS: 

268 self.qascoremetrics[intent][spw.id][metric] = {}

285 # Get data selection from QA score 

286 selscan = np.array(list(qascore.applies_to.scan)) 

287 selspw = np.array(list(qascore.applies_to.spw)) 

288 selintent = np.array(list(qascore.applies_to.intent)) 

289 selant = np.array(list(qascore.applies_to.ant)) 

290 # QA scores retain the full metric ID, whereas the ID in this 

291 # function's score dicts are stripped of the gt90deg_offset prefix. 

292 # Ideally we'd refactor the metric ID so that gt90deg_offset is a 

293 # subcomponent of the ID (e.g., phase_vs_freq.intercept.gt90deg_offset 

294 # or similar), but that's too risky a change at this stage so we just 

295 # strip the gt90deg_offset prefix. 

296 selmetric = qascore.origin.metric_name.replace('gt90deg_offset_','') 

297 

298 # Case of no data selected as outlier for this metric, 

299 # fill values with default values for non-outlier QA scores 

300 if len(selscan) == 0 and len(selspw) == 0 and len(selintent) == 0 and len(selant) == 0: 

301 d = dict(significance=0.0, is_amp_sym_off=False, outliers=False) 

302 for s_dict in [v for v in self.qascoremetrics.values() if isinstance(v, dict)]: 

303 for m_dict in [v for v in s_dict.values() if isinstance(v, dict)]: 

304 for m in m_dict.values(): 

305 m.update(d) 

306 s_dict['subscore'] = 1.0 

307 self.qascoremetrics['finalscore'] = 1.0 

308 self.long_msg = qascore.longmsg 

309 self.short_msg = qascore.shortmsg 

310 return self.qascoremetrics['finalscore']

312 testspw = lambda x: x in selspw 

313 testscan = lambda x: x in selscan 

314 testintent = lambda x: x in selintent 

315 testant = lambda x: x in selant 

316 basesel = np.array(list(map(testspw, self.spw))) & np.array(list(map(testscan, self.scan))) & np.array(list(map(testintent, self.intent))) & np.array(list(map(testant, self.ant))) 

317 

318 for i in selintent: 

319 for s in selspw: 

320 for m in mlist: 

321 # For this metric, select the pool of outliers from the "applies_to" attribute 

322 sel = (basesel & (self.metricnames == m) & (self.mtratio > 1.0)) 

323 nsel = np.sum(sel) 

324 if nsel > 0: 

325 idxmax = np.argsort(self.metricscores[sel])[-1] 

326 # Get ratio Metric/Threshold for maximum value -> significance 

327 self.qascoremetrics[i][s][m]['significance'] = self.mtratio[sel][idxmax] 

328 # Generate message for this max outlier 

329 thismaxoutlieridx = np.arange(self.noutliers)[sel][idxmax]

331 thisqamsg = QAMessage(self.ms, thismaxoutlier, reason=list(thismaxoutlier.reason)[0]) 

332 self.qascoremetrics[i][s][m]['long_msg'] = thisqamsg.full_message 

333 self.qascoremetrics[i][s][m]['short_msg'] = thisqamsg.short_message 

334 # copy the boolean is_amp_sym_offset from this QA scores 

335 self.qascoremetrics[i][s][m]['is_amp_sym_off'] = self.is_amp_sym_off[sel][idxmax] 

336 self.qascoremetrics[i][s][m]['outliers'] = True 

337 else: 

338 metric_axes, outlier_description, extra_description = REASONS_TO_TEXT[m] 

339 # Correct capitalisation as we'll prefix the metric with 'No ' 

340 metric_axes = metric_axes.lower() 

341 self.qascoremetrics[i][s][m]['short_msg'] = 'No {} outliers'.format(metric_axes) 

342 self.qascoremetrics[i][s][m]['long_msg'] = 'No {} {} detected for {}'.format(metric_axes, outlier_description, self.ms.basename) 

343 self.qascoremetrics[i][s][m]['significance'] = 0.0 

344 self.qascoremetrics[i][s][m]['is_amp_sym_off'] = False 

345 self.qascoremetrics[i][s][m]['outliers'] = False 

346 

347 longmsgsubscores = np.array([self.qascoremetrics[i][s][selmetric]['long_msg'] for s in selspw])

361 # it needs to be symmetric for any outlier in the pool. 

362 is_amp_sym_off_all = all(is_amp_sym_off_subscores)

368 thisminscore = self.INTENT_MINSCORE['AMP_SYM_OFFSET'] 

369 else: 

370 thisminscore = self.INTENT_MINSCORE[i]

372 self.qascoremetrics[i]['subscore'] = max(thisminscore, auxqascore) 

373 else: 

374 self.qascoremetrics[i]['subscore'] = 1.0 

375

377 finalset = [self.qascoremetrics[i]['subscore'] for i in selintent] 

378 if len(finalset) > 0: 

379 self.qascoremetrics['finalscore'] = min(finalset) 

380 else: 

381 self.qascoremetrics['finalscore'] = 1.0 

382 # Generate summary line 

383 if len(selintent) == 1: 

384 self.long_msg = self.qascoremetrics[selintent[0]]['long_msg'] 

385 self.short_msg = self.qascoremetrics[selintent[0]]['short_msg'] 

386 elif len(selintent) == 0: 

387 self.long_msg = '' 

388 self.short_msg = '' 

389 else: 

390 LOG.info('Multiple intents for this QAscore: %s', qascore) 

391 self.long_msg = '' 

392 self.short_msg = '' 

393

397def get_qa_scores( 

398 ms: MeasurementSet, 

399 export_outliers: bool, 

400 outlier_score: float, 

401 flag_all: bool, 

402 export_mswrappers: bool, 

403 output_path: Optional[Path] = Path(''), 

404 memory_gb: Optional[float] = MEMORY_CHUNK_SIZE, 

405) -> dict[WebLogLocation, list[pqa.QAScore]]: 

406 """ 

407 Calculate amp/phase vs freq and time outliers for an EB and convert to QA scores. 

408 

409 This is the key entry point for applycal QA metric calculation. It 

410 delegates to the detailed metric implementation in ampphase_vs_freq_qa.py 

411 and ampphase_vs_time_qa.py to detect outliers, 

412 converting the outlier descriptions to normalised QA 

413 scores. 

414 """ 

415 # all outlier scores objects will be saved here 

416 all_scores = [] 

417 

418 # if there are any average visibilities saved, they are in buffer_folder 

419 buffer_path = output_path / 'databuffer' 

420 if export_mswrappers and not os.path.exists(buffer_path): 

421 os.makedirs(buffer_path, exist_ok=True) 

422 

423 # Define intents that need to be processed avoiding intents with repeated scans 

424 intents2proc = qa_utils.get_intents_to_process(ms, INTENTS) 

425 outliers = [] 

426 for intent in intents2proc: 

427 LOG.debug('Processing intent %s', intent) 

428 outliers_for_intent = score_all_scans( 

429 ms, intent, 

430 flag_all=flag_all, 

431 memory_gb=memory_gb, 

432 buffer_path=buffer_path, 

433 export_mswrappers=export_mswrappers 

434 ) 

435 outliers.extend(outliers_for_intent) 

436 

437 if export_outliers: 

438 debug_path = 'applycalQA_outliers.txt' 

439 with open(debug_path, 'a') as debug_file: 

440 debug_file.write(f'AMPLITUDE_SLOPE_THRESHOLD: {ampphase_vs_freq_qa.AMPLITUDE_SLOPE_THRESHOLD}\n') 

441 debug_file.write(f'AMPLITUDE_SLOPE_PHYSICAL_THRESHOLD: {ampphase_vs_freq_qa.AMPLITUDE_SLOPE_PHYSICAL_THRESHOLD}\n') 

442 debug_file.write(f'AMPLITUDE_INTERCEPT_THRESHOLD: {ampphase_vs_freq_qa.AMPLITUDE_INTERCEPT_THRESHOLD}\n') 

443 debug_file.write(f'AMPLITUDE_INTERCEPT_PHYSICAL_THRESHOLD: {ampphase_vs_freq_qa.AMPLITUDE_INTERCEPT_PHYSICAL_THRESHOLD}\n') 

444 debug_file.write(f'PHASE_SLOPE_THRESHOLD: {ampphase_vs_freq_qa.PHASE_SLOPE_THRESHOLD}\n') 

445 debug_file.write(f'PHASE_SLOPE_PHYSICAL_THRESHOLD: {ampphase_vs_freq_qa.PHASE_SLOPE_PHYSICAL_THRESHOLD}\n') 

446 debug_file.write(f'PHASE_INTERCEPT_THRESHOLD: {ampphase_vs_freq_qa.PHASE_INTERCEPT_THRESHOLD}\n') 

447 debug_file.write(f'PHASE_INTERCEPT_PHYSICAL_THRESHOLD: {ampphase_vs_freq_qa.PHASE_INTERCEPT_PHYSICAL_THRESHOLD}\n') 

448 

449 for i,o in enumerate(outliers): 

450 # Filter doubles from sources with multiple intents 

451 duplicate_entry = any( 

452 o.vis == outliers[j].vis and 

453 o.scan == outliers[j].scan and 

454 o.spw == outliers[j].spw and 

455 o.ant == outliers[j].ant and 

456 o.pol == outliers[j].pol and 

457 o.reason == outliers[j].reason and 

458 o.num_sigma == outliers[j].num_sigma and 

459 o.delta_physical == outliers[j].delta_physical and 

460 o.amp_freq_sym_off == outliers[j].amp_freq_sym_off 

461 for j in range(i) 

462 ) 

463 if duplicate_entry: 

464 continue 

465 

466 str_components = [ 

467 outlier_attr_to_str(o, attr, ms) 

468 for attr in 

469 ('vis', 'scan', 'spw', 'ant', 'pol', 'reason', 'num_sigma', 'delta_physical', 'amp_freq_sym_off') 

470 ] 

471 msg = ' '.join(c for c in str_components if c != '') 

472 debug_file.write(f'{msg}\n') 

473 

474 # convert outliers to QA scores 

475 scores_for_intent = outliers_to_qa_scores(ms, outliers, outlier_score) 

476 all_scores.extend(scores_for_intent) 

477 

478 # Get summary QA scores 

479 qaevalf = QAScoreEvalFunc(ms, INTENTS, outliers) 

480 final_scores = summarise_scores(all_scores, ms, qaevalf) 

481 

482 return final_scores

485class QAMessage: 

486 """ 

487 QAMessage constructs a user-friendly QA message for an Outlier. 

488 

489 The QAMessage instance has two attributes, full_message and short_message, 

490 that are of interest. full_message holds the text to be used when the 

491 message is the first to be printed. short_message holds the text to be 

492 used when this message is to be appended to the text of other QAMessages. 

493 Naturally, this assumes the calling code only concatenates messages 

494 that originate from the same reason. 

495 """ 

496 

497 def __init__(self, ms: MeasurementSet, outlier: Outlier, reason: str): 

498 metric_axes, outlier_description, extra_description = REASONS_TO_TEXT[reason] 

499 

500 # convert pol=0,1 to pol=XX,YY 

501 # corr axis should be the same for all windows so just pick the first 

502 if outlier.pol: 

503 spw_id = list(outlier.spw)[0] 

504 corr_axes = dict(enumerate(ms.get_data_description(spw=spw_id).corr_axis)) 

505 corrs = sorted([corr_axes[c] for c in outlier.pol]) 

506 corr_msg = ','.join(corrs) 

507 else: 

508 corr_msg = '' 

509 

510 vis = utils.commafy(sorted(outlier.vis), quotes=False) 

511 intent_msg = f' {utils.commafy(sorted(outlier.intent), quotes=False)} calibrator' if outlier.intent else '' 

512 spw_msg = f' spw {utils.find_ranges(outlier.spw)}' if outlier.spw else '' 

513 if -1 in outlier.scan: 

514 # outlier.scan can be a set or a tuple 

515 tmp_scan = set(copy.deepcopy(outlier.scan)) 

516 tmp_scan.remove(-1) 

517 scan_msg = f' scan {",".join(list(filter(None, ["all", utils.find_ranges(tmp_scan)])))}' 

518 else: 

519 scan_msg = f' scan {utils.find_ranges(outlier.scan)}' if outlier.scan else '' 

520 

521 # convert ant=1,3,5 to ant=DV03,CM05,CM08 etc. 

522 ant_names = sorted([ant.name 

523 for ant_id in outlier.ant 

524 for ant in ms.antennas 

525 if ant.id == ant_id]) 

526 

527 if len(ant_names) == len(ms.antennas): 

528 ant_names = ['all antennas'] 

529 ant_msg = f' {",".join(ant_names)}' if ant_names else '' 

530 corr_msg = f' {corr_msg}' if corr_msg else '' 

531 

532 # TODO bifurcate this method for Outliers and TargetDataSelections 

533 # TargetDataSelections and Outliers have very similar interfaces but 

534 # are not quite identical, leading to code like below that needs to 

535 # check variable types. 

536 if isinstance(outlier, Outlier): 

537 num_sigma_msg = '{0:.3f}'.format(outlier.num_sigma) 

538 delta_physical_msg = '{0:.3f}'.format(outlier.delta_physical) 

539 amp_freq_sym_off_msg = 'Y' if outlier.amp_freq_sym_off else 'N' 

540 significance_msg = f'; n_sig={num_sigma_msg}; d_phys={delta_physical_msg}; ampsymoff={amp_freq_sym_off_msg}' 

541 else: 

542 significance_msg = '' 

543 

544 short_msg = f'{metric_axes} {outlier_description}' 

545 full_msg = f'{short_msg} for {vis}{intent_msg}{spw_msg}{ant_msg}{corr_msg}{scan_msg}{significance_msg}{extra_description}' 

546 

547 self.short_message = short_msg 

548 self.full_message = full_msg

551def outliers_to_qa_scores(ms: MeasurementSet, 

552 outliers: list[ampphase_vs_freq_qa.Outlier], 

553 outlier_score: float) -> list[pqa.QAScore]: 

554 """ 

555 Convert a list of consolidated Outliers into a list of equivalent 

556 QAScores. 

557 

558 The MeasurementSet argument is required to convert antenna IDs to antenna 

559 names. 

560 

561 All generated QAScores will be assigned the numeric score given in 

562 outlier_score. 

563 

564 :param ms: MeasurementSet domain object for the DataSelections 

565 :param outliers: list of Outliers 

566 :param outlier_score: score to assign to generated QAScores 

567 :return: 

568 """ 

569 hashable = [] 

570 for outlier in outliers: 

571 # convert ['amp.slope','amp.intercept'] into 'amp.slope,amp.intercept', etc. 

572 hashable.append(ampphase_vs_freq_qa.Outlier(vis=outlier.vis, 

573 intent=outlier.intent, 

574 scan=outlier.scan, 

575 spw=outlier.spw, 

576 ant=outlier.ant, 

577 pol=outlier.pol, 

578 num_sigma=outlier.num_sigma, 

579 delta_physical=outlier.delta_physical, 

580 amp_freq_sym_off=outlier.amp_freq_sym_off, 

581 reason=','.join(sorted(outlier.reason)))) 

582 reasons = {outlier.reason for outlier in hashable} 

583 

584 qa_scores = [] 

585 for reason in reasons: 

586 outliers_for_reason = [outlier for outlier in hashable if outlier.reason == reason] 

587 if not outliers_for_reason: 

588 continue 

589 

590 for outlier in outliers_for_reason: 

591 msgs = QAMessage(ms, outlier, reason=outlier.reason) 

592 

593 applies_to = pqa.TargetDataSelection(vis=outlier.vis, scan=outlier.scan, 

594 intent=outlier.intent, spw=outlier.spw, 

595 ant=outlier.ant, pol=outlier.pol) 

596 

597 score = pqa.QAScore(outlier_score, longmsg=msgs.full_message, shortmsg=msgs.short_message, 

598 applies_to=applies_to, hierarchy=reason, 

599 weblog_location=pqa.WebLogLocation.ACCORDION) 

600 score.origin = pqa.QAOrigin(metric_name=reason, 

601 metric_score=outlier.num_sigma, 

602 metric_units='sigma deviation from reference fit') 

603 qa_scores.append(score) 

604 

605 return qa_scores

616def in_casa_format(data_selections: DataSelectionToScores) -> DataSelectionToScores: 

617 """ 

618 Restate data selections in concise CASA format. 

619 """ 

620 # we need a neat way to format each message from the indices. This dict 

621 # states how each field should be formatted 

622 formatters = { 

623 'vis': lambda l: ','.join(l), 

624 'intent': lambda l: ','.join(l),

636 new_ds = DataSelection(**formatted_args) 

637 formatted[new_ds] = scores 

638 return formatted

641def summarise_scores( 

642 all_scores: list[pqa.QAScore], 

643 ms: MeasurementSet, 

644 qaevalf: QAScoreEvalFunc, 

645) -> dict[pqa.WebLogLocation, list[pqa.QAScore]]: 

646 """ 

647 Process a list of QAscores, replacing the detailed and highly specific 

648 input scores with compressed representations intended for display in the 

649 web log accordion, and even more generalised summaries intended for 

650 display as warning banners. 

651 """ 

652 # list to hold the final QA scores: non-combined hidden scores, plus the 

653 # summarised (and less specific) accordion scores and banner scores 

654 scores_by_location: dict[pqa.WebLogLocation, list[pqa.QAScore]] = {} 

655 

656 # we don't want the non-combined scores reported in the web log. They're 

657 # useful for the QA report written to disk, but for the web log the 

658 # individual scores will be aggregated into general, less specific QA 

659 # scores. 

660 hidden_scores = copy.deepcopy(all_scores) 

661 for score in hidden_scores: 

662 score.weblog_location = pqa.WebLogLocation.HIDDEN 

663 scores_by_location[pqa.WebLogLocation.HIDDEN] = hidden_scores 

664 

665 # JH update to spec for PIPE-477: 

666 # 

667 # After looking at the current messages, I have come to the conclusion 

668 # that its really not necessary for the message to relate whether the 

669 # outlier is slope or offset or both, or that the pol is XX or YY or both, 

670 # since these are easily discerned once the antenna and scan are known. So 

671 # that level of detail can be suppressed to keep the number of accordion 

672 # messages down. I have changed the example in the description accordingly. 

673 

674 # Rescoring must happen BEFORE score aggregation, as the scoring algorithm 

675 # makes a distinction between metrics of different origins. This also needs 

676 # to happen *before* the banner scores are compiled, as further dimension 

677 # erasure leads to complaints from the QA evaluation function for score in 

678 # processed_scores: 

679 for score in all_scores: 

680 adjusted_score = qaevalf(score) 

681 score.score = adjusted_score 

682 # Note that we adopt the score but not the QAScoreEvalFunc long/short 

683 # messages, as those messages refer to a single vis/spw/ant/intent 

684 # selection. Instead, we prefer to keep the original aggregated 

685 # message that applies to the union data selection. 

686 

687 processed_scores = [] 

688 # Collect scores. The phase scores (normal and > 90 deg offset) should 

689 # get just one single 1.0 score in case of no outliers. 

690 for metric_root, metric_regex in [ 

691 ('amp_vs_freq', '^amp_vs_freq'), 

692 ('phase_vs_freq', '^(phase_vs_freq|gt90deg_offset_phase_vs_freq)') 

693 ]: 

694 # erase just the polarisation dimension for accordion messages, 

695 # leaving the messages specific enough to identify the plot that 

696 # caused the problem 

697 discard = ['pol'] 

698 msgs = combine_scores(all_scores, metric_root, metric_regex, discard, ms, pqa.WebLogLocation.ACCORDION) 

699 num_scores = len(msgs) 

700 processed_scores.extend(msgs) 

701 

702 # add a single 1.0 accordion score for metrics that generated no outlier 

703 if num_scores == 0: 

704 metric_axes, outlier_description, _ = REASONS_TO_TEXT[metric_root] 

705 # Correct capitalisation as we'll prefix the metric with 'No ' 

706 metric_axes = metric_axes.lower() 

707 short_msg = 'No {} outliers'.format(metric_axes) 

708 long_msg = 'No {} {} detected for {}'.format(metric_axes, outlier_description, ms.basename) 

709 score = pqa.QAScore(1.0, 

710 longmsg=long_msg, 

711 shortmsg=short_msg, 

712 hierarchy=metric_root, 

713 weblog_location=pqa.WebLogLocation.ACCORDION, 

714 applies_to=pqa.TargetDataSelection(vis={ms.basename})) 

715 score.origin = pqa.QAOrigin(metric_name=metric_root, 

716 metric_score=0, 

717 metric_units='number of outliers') 

718 processed_scores.append(score) 

719 

720 # scores destined for the weblog accordion 

721 accordion_scores: list[QAScore] = [] 

722 

723 # PIPE-1770 spec: 

724 # For INTENT=CHECK, suppress or leave out the QA messages that identify 

725 # outliers by antenna & scan (i.e., so it is only reports outlier 

726 # type/spw for each ms). 

727 check_scores = [s for s in processed_scores if 'CHECK' in s.applies_to.intent] 

728 for metric_root, metric_regex in [ 

729 ('amp_vs_freq', '^amp_vs_freq'), 

730 ('phase_vs_freq', '^(phase_vs_freq|gt90deg_offset_phase_vs_freq)') 

731 ]: 

732 msgs = combine_scores(check_scores, metric_root, metric_regex, ['ant', 'scan'], ms, pqa.WebLogLocation.ACCORDION) 

733 accordion_scores.extend(msgs) 

734 

735 # PIPE-1770 spec: 

736 # For other intents, suppress the fully aggregated QA message (since it 

737 # is a duplication now that we have combined the QA messages that used 

738 # to be shown at the top and bottom of the page 

739 # 

740 # Suppress fully-aggregated scores means do not run them through 

741 # combine_scores, i.e., just add them to the final results as-is. 

742 non_check_scores = [s for s in processed_scores if 'CHECK' not in s.applies_to.intent] 

743 accordion_scores.extend(non_check_scores) 

744 

745 scores_by_location[pqa.WebLogLocation.ACCORDION] = accordion_scores 

746 

747 # JH request from 8/4/20: 

748 # 

749 # The one thing I'd like to ask is to list the accordion messages ordered 

750 # by {ms; intent; spw} so that they appear in "figure order" (currently 

751 # they seem to be ordered by {ms; intent; scan} 

752 # 

753 for destination, unsorted_scores in scores_by_location.items(): 

754 sorted_scores = sorted(unsorted_scores, key=lambda score: (sorted(score.applies_to.vis), 

755 sorted(score.applies_to.intent), 

756 sorted(score.applies_to.spw), 

757 sorted(score.applies_to.scan))) 

758 scores_by_location[destination] = sorted_scores 

759 

760 return scores_by_location

763def combine_scores(all_scores: list[pqa.QAScore], 

764 hierarchy_root: str, 

765 hierarchy_regex: str, 

766 discard: list[str], 

767 ms: MeasurementSet, 

768 location: pqa.WebLogLocation) -> list[pqa.QAScore]: 

769 """ 

770 Combine and summarise a list of QA scores. 

771 

772 QA scores that share a base metric type and/or differ only in the data 

773 selection dimensions given in discard are aggregated and summarised 

774 together. 

775 """ 

776 all_scores = copy.deepcopy(all_scores) 

777 

778 # We're going to merge slope and offset outliers together into a single 

779 # record. We do not need to group or partition the input data by QA score 

780 # as every outlier gets the same score. If that condition changes, the 

781 # algorithm below will need to change to group and process by score value 

782 # too. 

783 

784 # create a filter function to leave the scores we want to process 

785 filter_fn = lambda qa_score: bool(re.match(hierarchy_regex, qa_score.hierarchy)) 

786 

787 # get all QA scores generated by a '<metric> vs X' algorithm 

788 scores_for_metric = [o for o in all_scores if filter_fn(o)] 

789 

790 # create a data structure that map data selections to the scores that 

791 # apply to them 

792 ds = map_data_selection_to_scores(scores_for_metric) 

793 

794 # Combine data selections that differ only by polarisation. This erases 

795 # the distinction between pol=0 and pol=1, placing QA scores in a single 

796 # data selection that spans all polarisations (other data selection 

797 # dimensions aside). 

798 no_pols = discard_dimension(ds, discard) 

799 

800 # hierarchically merge adjacent data selections 

801 merged_scores = compress_data_selections(no_pols, DataSelection._fields) 

802 

803 # filter out to leave one QA score, the score with the highest-valued 

804 # metric, in the list. This also resets the QAScore.applies_to to match 

805 # the data selection. 

806 as_min = take_min_as_representative(merged_scores) 

807 

808 # we can now discard the DataSelection keys, as the QAScore has now been 

809 # updated with the correct data selection 

810 qa_scores = [score for score_list in as_min.values() for score in score_list] 

811 

812 # rewrite the score messages 

813 for qa_score in qa_scores: 

814 # QAMessage takes an Outlier namedtuple, but TargetDataSelection 

815 # shares enough of the Outlier interface (vis, spw, scan, intent, 

816 # etc.) that we can pass QAMessage directly without converting to 

817 # an Outlier 

818 msgs = QAMessage(ms, qa_score.applies_to, reason=hierarchy_root) 

819 qa_score.shortmsg = msgs.short_message 

820 qa_score.longmsg = msgs.full_message 

821 qa_score.hierarchy = hierarchy_root 

822 qa_score.weblog_location = location 

823 

824 return qa_scores

827def take_min_as_representative(to_merge: DataSelectionToScores) -> DataSelectionToScores: 

828 """ 

829 Filter out all but the worst score per data selection. 

830 

831 Note that this function also rewrites QAScore.applies_to to match the data 

832 selection the QA score applies to. 

833 

834 This function operates on a dict that maps DataSelections to list of QA 

835 scores. For each list, it discards all but the worst score as determined 

836 by the metric. The selection will be biased towards metrics whose value 

837 distribution tends higher than other metrics, but it seem the best we can 

838 do. 

839 """ 

840 result: DataSelectionToScores = {} 

841 for ds, all_scores_for_ds in to_merge.items(): 

842 # get score with worst metric 

843 scores_and_metrics = [(1-o.score, o.origin.metric_score, o) for o in all_scores_for_ds] 

844 

845 # PIPE-634: hif_applycal crashes with TypeError: '>' not supported 

846 # between instances of 'QAScore' and 'QAScore' 

847 # 

848 # When the score and metric score are equal, max starts comparing the 

849 # QAScores themselves, which fails as the comparison operators are 

850 # not implemented. From the perspective of a 'worst score' calculation 

851 # the scores are equal so it doesn't matter which one we take. Hence, 

852 # we can supply an ordering function which simply excludes the QAScore 

853 # object from the calculation. 

854 def omit_qascore_instance(t): 

855 return t[0], t[1] 

856 

857 worst_score = max(scores_and_metrics, key=omit_qascore_instance)[2] 

858 

859 c = copy.deepcopy(worst_score) 

860 c.applies_to = pqa.TargetDataSelection(**ds._asdict()) 

861 

862 result[ds] = [c] 

863 

864 return result

867def discard_dimension(to_merge: DataSelectionToScores, attrs: Iterable[str]) -> DataSelectionToScores: 

868 """ 

869 Aggregate QA scores held in one or more DataSelection dimensions, 

870 discarding data selection indices for those dimensions. 

871 

872 This function discards DataSelection dimensions. Say four QA scores were 

873 registered, one each to spws 16, 18, 20, and 22. Calling this function 

874 with attrs=['spw'] would combine those scores into a single DataSelection 

875 with spw='', i.e., spw data selection is left unspecified. 

876 """ 

877 new_dsts = {} 

878 for data_selection, qa_scores in to_merge.items(): 

879 new_attrs = {attr: tuple() for attr in attrs} 

880 new_ds = data_selection._replace(**new_attrs) 

881 

882 # note that the QA scores themselves still have the data selection 

883 # specifiers in their .applies_to. For example, when asked to discard 

884 # pol, the DataSelection keys in the the result object would have 

885 # pol='' but the list of QA scores held as a value for that key would 

886 # still have pol=0 or pol=1, etc. 

887 if new_ds in new_dsts: 

888 new_dsts[new_ds].extend(qa_scores) 

889 else: 

890 new_dsts[new_ds] = copy.deepcopy(qa_scores) 

891 return new_dsts

894def map_data_selection_to_scores(scores: Iterable[pqa.QAScore]) -> DataSelectionToScores: 

895 """

909def compress_data_selections(to_merge: DataSelectionToScores, 

910 attrs_to_merge: Reversible[str]) -> DataSelectionToScores: 

911 """ 

912 Combine adjacent data selections to give a new data structure that 

913 expresses the same data selection but in a more compressed form. 

914 

915 A data selection applies over various dimensions: spw, field, pol, etc.. 

916 This function merges data selections hierarchically, identifying adjacent 

917 data selections per data selection dimension given in attrs_to_merge, and 

918 concatenating those adjacent dimension indices together. 

919 """ 

920 to_merge = copy.deepcopy(to_merge) 

921 

922 # We can identify data selections that apply to the same data except for a 

923 # particular field by creating a new tuple that omits that field and 

924 # sorting/grouping on the reduced tuple. This function is used to create 

925 # the reduced tuple that 'ignores' the specified field(s) 

926 def get_keyfunc(cols_to_ignore: list[str]): 

927 def keyfunc(ds: DataSelection): 

928 return tuple(getattr(ds, field) for field in ds._fields if field not in cols_to_ignore) 

929 return keyfunc 

930 

931 # hierarchical merging of tuple fields, grouping/merging in reverse order of 

932 # DataSelection fields. This has the effect of merging DataSelections from 

933 # the bottom up, e.g., first merge data selections that differ only in pol, 

934 # then merge data selections that differ only in ant, etc. 

935 keys_to_merge = to_merge.keys() 

936 for attr in reversed(attrs_to_merge): 

937 key_func = get_keyfunc([attr]) 

938 keys_to_merge = sorted(keys_to_merge, key=key_func) 

939 to_add = {} 

940 to_del = [] 

941 for k, g in itertools.groupby(keys_to_merge, key_func): 

942 # k is the data selection minus the ignored field, while g 

943 # iterates over the data selections matching k that differ only in 

944 # the ignored field. These selections in g can be merged. 

945 group = list(g) 

946 # convert from ((1, ), (2, ), (5, )) to (1, 2, 5) 

947 merged_vals = tuple(set(itertools.chain(*(getattr(g, attr) for g in group)))) 

948 

949 # we now need to reconstruct the full data selection tuple from the 

950 # reduced tuple in k plus the values we've just merged. We do this 

951 # by created a dict of DataSelection named arguments for everything 

952 # except the field we've merged... 

953 d = {o: p for p, o in zip(k, (f for f in DataSelection._fields if f != attr))} 

954 # ... and then adding the merged field with the merged value 

955 d[attr] = merged_vals 

956 

957 merged_ds = DataSelection(**d) 

958 to_add[merged_ds] = list(itertools.chain(*[to_merge[o] for o in group])) 

959 to_del.extend(group) 

960 

961 for k in to_del: 

962 del to_merge[k] 

963 to_merge.update(to_add) 

964 

965 # replace the original unmerged data with our merged selections and 

966 # we're ready to go round again for the next field 

967 keys_to_merge = list(to_merge.keys()) 

968 

969 return to_merge

972def outlier_attr_to_str(o: Outlier, attr: str, ms: MeasurementSet) -> str: 

973 """ 

974 Convert an outlier attribute to a string representation. 

975 

976 This function takes an Outlier object and an attribute name, and returns a 

977 string representation of the attribute's value. It handles different 

978 attribute types, including iterables and non-iterables, and formats them 

979 appropriately. 

980 

981 @param o: The Outlier object. 

982 @param attr: The name of the attribute to convert. 

983 @param ms: MeasurementSet object used to convert antenna IDs to names 

984 

985 @return :A string representation of the attribute's value. 

986 """ 

987 raw_val = getattr(o, attr) 

988 try: 

989 str_val = ','.join(map(str, sorted(raw_val))) 

990 except TypeError: 

991 # not an iterable 

992 str_val = str(raw_val) 

993 

994 match attr: 

995 case 'ant': 

996 # use antenna names to match the weblog 

997 str_val = ','.join(sorted(a.name for a in ms.get_antenna(str_val))) 

998 case 'num_sigma': 

999 str_val = f'{float(str_val):.1f}' 

1000 case 'delta_physical': 

1001 str_val = f'{float(str_val):.3f}' 

1002 case 'scan': 

1003 str_val = 'all' if str_val == '-1' else str_val
