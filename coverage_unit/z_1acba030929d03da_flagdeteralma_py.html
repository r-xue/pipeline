175 # 

176 # Note: at present, February 2020, ALMA correlators produce 

177 # spectral windows whose bandwidths are quantized: 

178 # 

179 # Baseline correlator (BLC) widths:

220 def get_fracspw(self, spw): 

221 # From T. Hunter on PIPE-425: in early ALMA Cycles, the ACA 

222 # correlator's frequency profile synthesis (fps) algorithm produced TDM 

223 # spws that had 64 channels in full-polarisation, 124 channels in dual 

224 # pol, and 248 channels in single-pol. 

225 # TODO: find out whether it should be 62 (as in code) or 64 (as per above comment) 

226 # 

227 # By comparison, the baseline correlator (BLC) standard values are 128 

228 # channels for dual pol, and 256 channels for single pol, and in more 

229 # recent cycles, the ACA TDM spws agree in channel number and bandwidth 

230 # with the BLC TDM spws. 

231 # 

232 # The following override is preserved to handle the old "FPS" generated 

233 # ACA TDM spws, for which a different threshold is used. 

234 if spw.num_channels in (62, 124, 248): 

235 return self.inputs.fracspwfps 

236 else: 

237 return self.inputs.fracspw

239 def verify_spw(self, spw): 

240 # Override the default verifier: 

241 # - first run the default verifier 

242 # - then run extra test to skip flagging of TDM windows 

243 super().verify_spw(spw) 

244 

245 # Test whether the spw is TDM or FDM. If it is FDM, then raise a 

246 # ValueError. From T. Hunter on PIPE-425: 

247 # TDM spectral windows have either 4*64 channels in full polarisation, 

248 # 2*128 for dual polarisation, or 1*256 channels for single 

249 # polarisation; i.e., in all cases, a TDM spw has ncorr*nchans = 256. 

250 # 

251 # By comparison, the smallest number of channels in an FDM spectral 

252 # window is 4*120 (full pol), 2*240 (dual-pol), or 1*480 (single-pol) 

253 # when online binning is set to 16. 

254 # 

255 # Based on this, it is assumed that any spw with ncorr*nchans <= 256 is 

256 # TDM, and any spw with ncorr*nchans > 256 is FDM. 

257 dd = self.inputs.ms.get_data_description(spw=spw) 

258 ncorr = len(dd.corr_axis) 

259 if ncorr * spw.num_channels > 256: 

260 raise ValueError('Spectral window {} is an FDM spectral window, skipping the TDM edge flagging' 

261 'heuristics.'.format(spw.id))

285 def _get_edgespw_cmds(self): 

286 # Run default edge channel flagging first. 

287 to_flag = super()._get_edgespw_cmds() 

288 

289 # Loop over the spectral windows, generate a flagging command for each 

290 # spw in the ms. Calling get_spectral_windows() with no arguments 

291 # returns just the science windows, which is exactly what we want. 

292 for spw in self.inputs.ms.get_spectral_windows(): 

293 try: 

294 # Test whether this spw should be evaluated for edge channel 

295 # flagging. 

296 self.verify_spw(spw) 

297 except ValueError: 

298 # If we get here, then the spw verification failed. Proceed 

299 # with alternate edge channel flagging. 

300 LOG.debug('Proceeding with FDM edge channel flagging heuristics for spw {}' 

301 ''.format(spw.id)) 

302 

303 # CAS-5231: FDM edge channel flagging heuristic for spectral 

304 # windows whose bandwidth exceeds 1875 MHz. 

305 threshold = measures.Frequency(1875, measures.FrequencyUnits.MEGAHERTZ) 

306 if spw.bandwidth > threshold: 

307 to_flag.extend(self._get_fdm_edgespw_cmds(spw, threshold)) 

308 

309 # PIPE-425: run a separate edge channel flagging heuristic for 

310 # ACA spectral windows with shorter bandwidth (defined above), 

311 # to ensure that channels tuned too close to the edge of the 

312 # baseband are flagged. Assume that the ACA spw bandwidths are 

313 # stored in MHz. 

314 # 

315 # TODO: in CASA 6.1, the correlator type should start to be 

316 # propagated from ASDM to MS. Once available, this test could 

317 # be future-proofed (w.r.t. possible new correlators) by 

318 # explicitly checking whether the spw is an ACA spw 

319 # (e.g., self.inputs.ms.correlator_name == 'ALMA_ACA') 

320 spw_bw_in_mhz = spw.bandwidth.to_units(otherUnits=measures.FrequencyUnits.MEGAHERTZ) 

321 if spw_bw_in_mhz in self._aca_edge_flag_thresholds.keys(): 

322 to_flag.extend(self._get_aca_edgespw_cmds(spw, self._aca_edge_flag_thresholds[spw_bw_in_mhz])) 

323 

324 return to_flag

326 def _get_fdm_edgespw_cmds(self, spw, threshold): 

327 """ 

328 FDM spectral window edge flagging heuristic. 

329 

330 Returns a list containing a flagging command that will flag all channels 

331 that lie beyond "threshold/2" from the center frequency. 

332 

333 For example: if the threshold is 1875 MHz, this will flag any channels 

334 +- 937.5 MHz from the center frequency. 

335 

336 :param spw: spectral window to evaluate 

337 :param threshold: bandwidth threshold 

338 :return: list containing flagging command as string 

339 :rtype: list[str] 

340 """ 

341 LOG.debug('Bandwidth greater than {} for spw {}. Proceeding with flagging all channels beyond +-{} from the' 

342 ' center frequency.'.format(str(threshold), spw.id, str(threshold / 2.0))) 

343 

344 cen_freq = spw.centre_frequency 

345 

346 # channel range lower than threshold 

347 lo_freq = cen_freq - spw.bandwidth / 2.0 

348 hi_freq = cen_freq - threshold / 2.0 

349 minchan_lo, maxchan_lo = spw.channel_range(lo_freq, hi_freq) 

350 

351 # upper range higher than threshold 

352 lo_freq = cen_freq + threshold / 2.0 

353 hi_freq = cen_freq + spw.bandwidth / 2.0 

354 minchan_hi, maxchan_hi = spw.channel_range(lo_freq, hi_freq) 

355 

356 # Append to flag list 

357 # Clean up order of channel ranges high to low 

358 chan1 = '{0}~{1}'.format(minchan_lo, maxchan_lo) 

359 chan2 = '{0}~{1}'.format(minchan_hi, maxchan_hi) 

360 chans = sorted([chan1, chan2]) 

361 cmd = '{0}:{1};{2}'.format(spw.id, chans[0], chans[1]) 

362 to_flag = [cmd] 

363 

364 return to_flag

366 def _get_aca_edgespw_cmds(self, spw, threshold): 

367 """ 

368 ACA FDM spectral window edge flagging heuristic. 

369 

370 Return a list containing a flagging command that will flag all channels 

371 that lie too close to the edge of the baseband. 

372 

373 :param spw: spectral window to evaluate 

374 :param threshold: threshold frequency range used to determine whether 

375 spectral window channels are too close to edge of baseband 

376 :return: list containing flagging command as string 

377 :rtype: list[str] 

378 """ 

379 LOG.debug('Spectral window {} is an ACA FDM spectral window. Proceeding with flagging channels' 

380 ' that are too close to the baseband edge.'.format(spw.id)) 

381 

382 # For the given spectral window, identify the corresponding SQLD 

383 # spectral window(s) with TARGET intent taken in same baseband. 

384 bb_spws = [s for s in self.inputs.ms.get_spectral_windows(science_windows_only=False) 

385 if s.baseband == spw.baseband and s.type == 'SQLD' and 'TARGET' in s.intents] 

386 

387 # If no baseband spw could be identified, add the spw to the list of 

388 # missing basebands and return with no new flagging commands. 

389 if not bb_spws: 

390 self.missing_baseband_spws.append(spw.id) 

391 return [] 

392 # If exactly 1 matching baseband spw was found, use that. 

393 elif len(bb_spws) == 1: 

394 bb_spw = bb_spws[0] 

395 # If multiple matches were found within the same baseband, then 

396 # attempt to further match against the same spectral tuning 

397 # (SpectralSpec, PIPE-1991). 

398 else: 

399 bb_spws_same_ss = [s for s in bb_spws if s.spectralspec == spw.spectralspec] 

400 # If exactly 1 match was found with same SpectralSpec, use that. 

401 if len(bb_spws_same_ss) == 1: 

402 bb_spw = bb_spws_same_ss[0] 

403 # Otherwise, log a warning that no unambiguous match could be found, 

404 # and proceed to use first match from original filter. 

405 else: 

406 bb_spw = bb_spws[0] 

407 LOG.warning(f"{self.inputs.ms.basename}: unable to find unambiguous match of baseband SpW for science" 

408 f" SpW {spw.id}, selected first match (SpW {bb_spw.id}).") 

409 

410 # Compute frequency ranges for which any channel that falls within the 

411 # range should be flagged; these ranges are set by the baseband edges 

412 # and the provided threshold. 

413 bb_edges = [measures.FrequencyRange(bb_spw.min_frequency, bb_spw.min_frequency + threshold), 

414 measures.FrequencyRange(bb_spw.max_frequency - threshold, bb_spw.max_frequency)] 

415 

416 # Compute list of channels to flag as those channels that have an 

417 # overlap with either edge range of the baseband. 

418 to_flag = [str(i) for i, ch in enumerate(spw.channels) 

419 if ch.frequency_range.overlaps(other=bb_edges[0]) 

420 or ch.frequency_range.overlaps(other=bb_edges[1])] 

421 

422 # If flagging is needed, turn the list of channels into a list of 

423 # a flagging command with spectral window id, and consolidated channel 

424 # ranges. utils.find_ranges returns comma separated ranges, but for 

425 # multiple channel ranges within a single spw, CASA's flagdata expects 

426 # these to be separated by semi-colon. 

427 if to_flag: 

428 chan_to_flag = utils.find_ranges(to_flag).replace(',', ';') 

429 LOG.attention('{} - Flagging edge channels for ACA spectral window {}, channel(s) {}, due to proximity' 

430 ' to edge of baseband.'.format(self.inputs.ms.basename, spw.id, chan_to_flag)) 

431 to_flag = ['{}:{}'.format(spw.id, chan_to_flag)] 

432 

433 return to_flag

448 """Retrieve the relevant data to extend partial polarization flagging to all the polarizations (see PIPE-1028).

462 

463 # PIPE-1245: restrict evaluation of partial polarization flagging to scans 

464 # that do not cover the following intents: 

465 unwanted_intents = {'ATMOSPHERE', 'FOCUS', 'POINTING'}

467 spw_scan_selections = iter((spw, dd, scan) for spw, dd in zip(spws_ids, datadescids) for scan in scan_ids)

532def get_partialpol_spws(ms_name): 

533 """Obtain the spws and DATA_DESC_IDs required for the Partial Polarization flagging agent. 

534 According to the comments in PIPE-1028 they are all non-WVR/non-SQLD spws with an intent that 

535 starts with OBSERVE_TARGET or CALIBRATE_PHASE (which includes the Science spws). 

536 

537 Note that there is a chance that this function can be refactored using one on the pipeline domain object 

538 functions. If the translation of spw to DATA_DESC_ID is not required, this function may not be needed at all. 

539 

540 :param ms_name: Name of the Measurement Set 

541 :return: List of spws.ids and list of DATA_DESC_IDs (with the same length) 

542 """ 

543 

544 # Note: In all the examples checked, spw id and DATA_DESC_ID are the same, but as this may not be always true and 

545 # Todd uses this translation in his code, both sets of data (spws and DATA_DESC_IDs) are retrieved. 

546 # Note: In the examples checked, non-Science spws do not have usable data. 

547 with casa_tools.MSMDReader(ms_name) as msmd: 

548 spws_fdm_tdm = msmd.almaspws(fdm=True, tdm=True) 

549 spws_observe_target = msmd.spwsforintent('OBSERVE_TARGET*') 

550 spws_calibrate_phase = msmd.spwsforintent('CALIBRATE_PHASE*') 

551 spws = np.intersect1d(np.union1d(spws_observe_target, spws_calibrate_phase), spws_fdm_tdm) 

552 datadescids = [msmd.datadescids(spw=spw)[0] for spw in spws] 

553 return list(spws), datadescids

556def get_partialpol_flag_cmd_params(flags, ant1, ant2, time, interval): 

557 """Get the parameters that identify points where partial polarizations need to be flagged. 

558 This function should be called only if the data presents more than one polarization. 

559 At the moment it only handles data with 3 dimensions (n_pol, n_channels, n_params). 

560 

561 :param flags: numpy array with the flags with shape (n_pol, n_channels, n_params) 

562 :param ant1: numpy array with the antenna1s with shape (n_params, ) 

563 :param ant2: numpy array with the antenna2s with shape (n_params, ) 

564 :param time: numpy array with the times with shape (n_params, ) 

565 :param interval: numpy array with the intervals with shape (n_params, ) 

566 :return: List of dictionaries with the set of params to identify partial polarizations. 

567 The dictionaries contain the keys: 

568 * "ant1" - ID of the antenna1, 

569 * "ant2" - ID of the antenna2, 

570 * "time" - Central time of the 'scan', 

571 * "interval" - Duration of the 'scan', and 

572 * "channels"- a list of numerical values of affected channels that can be compressed later. 

573 :rtype: List[Dict] 

574 """ 

575 shape = np.shape(flags) 

576 # Check: Is there any chance that there are data with only 2 dimensions? 

577 if len(shape) != 3: 577 ↛ 578line 577 didn't jump to line 578 because the condition on line 577 was never true

578 LOG.error("Partial Polarization flagging: Data with no channel information are not handled by the pipeline yet") 

579 n_pol, n_channels, n_params = shape 

580 # Create a table of shape (n_channels, n_params) with the number of pols flagged 

581 folded_flags = np.sum(flags, axis=0) 

582 # Identify where polarization data is partially flagged. 

583 to_extend_idx = (folded_flags > 0) & (folded_flags < n_pol) 

584 # Identify the sets of parameters that have partial polarizations for any 

585 # of the channels. 

586 param_sets_to_check = np.where(np.any(to_extend_idx, axis=0))[0] 

587 params = [] 

588 # Iterate only through the sets of parameters with partial polarizations 

589 for param_set_idx in param_sets_to_check: 

590 # Get the numerical values of the channels that have partial polarizations 

591 channel_sel = list(np.where(to_extend_idx[:, param_set_idx])[0]) 

592 params.append({ 

593 "ant1": ant1[param_set_idx], 

594 "ant2": ant2[param_set_idx], 

595 "time": time[param_set_idx], 

596 "interval": interval[param_set_idx], 

597 "channels": channel_sel 

598 }) 

599 return params

602def convert_params_to_commands(ms, params, ant_id_map=None): 

603 """Convert the identified partial polarization parameters to flagging commands. 

604 

605 :param ms: Measurement Set to get the antenna id map (it can be None if ant_id_map is entered) 

606 :param params: List of dictionaries with the parameters 

607 :param ant_id_map: Dictionary mapping antenna IDs to their names (optional; overrides data from ms) 

608 :return: List of flagging commands 

609 :rtype: List[str] 

610 """ 

611 if ant_id_map is None: 

612 ant_id_map = {ant.id: ant.name for ant in ms.antennas} 

613 commands = [] 

614 for param in params: 

615 # Antennas 

616 antenna_str = "{}&&{}".format(ant_id_map[param["ant1"]], ant_id_map[param["ant2"]]) 

617 # spw and channels 

618 ch_str = '{}:'.format(param["spw"]) 

619 ch_ranges = channel_ranges(param["channels"]) 

620 ch_str += ";".join(["{}~{}".format(down, up) if down != up else "{}".format(down) for down, up in ch_ranges]) 

621 # Times 

622 time_start = casa_tools.quanta.quantity(param["time"] - 0.5 * param["interval"], param["time_unit"]) 

623 time_end = casa_tools.quanta.quantity(param["time"] + 0.5 * param["interval"], param["time_unit"]) 

624 time_str_start = casa_tools.quanta.time(time_start, prec=9, form="ymd")[0] 

625 time_str_end = casa_tools.quanta.time(time_end, prec=9, form="ymd")[0] 

626 time_str = "{}~{}".format(time_str_start, time_str_end) 

627 # Combine 

628 command = "antenna='{}' spw='{}' timerange='{}' reason='partialpol'".format(antenna_str, ch_str, time_str) 

629 commands.append(command) 

630 return commands

633def lowtrans_alma(ms: MeasurementSet, mintransrepspw: float, mintransnonrepspws: float, 

634 max_frac_low_trans: float) -> List[str]: 

635 """ 

636 Create flagging commands to flag science spectral windows with low 

637 atmospheric transmission (PIPE-624). 

638 

639 Args: 

640 ms: Measurement Set to evaluate. 

641 mintransrepspw: Atmospheric transmissivity threshold used to flag 

642 the representative science spectral window when fraction of data 

643 with low transmissivity exceeds "frac_low_trans". 

644 mintransnonrepspws: Atmospheric transmissivity threshold used to flag 

645 the non-representative science spectral window(s) when fraction of 

646 data with low transmissivity exceeds "frac_low_trans". 

647 max_frac_low_trans: Threshold fraction of data with low transmission 

648 at-or-above which a SpW is flagged for low transmission. 

649 

650 Returns: 

651 List of flagging commands. 

652 """ 

653 # Initialize flagging commands. 

654 commands = [] 

655 

656 # Compute the PWV for current MS. If this PWV value is invalid, then skip 

657 # the rest of the heuristic. 

658 pwv, _ = getMedianPWV(vis=ms.name) 

659 if pwv == 1.0 or np.isnan(pwv) or pwv < 0: 

660 LOG.debug(f'Invalid value for PWV ({pwv}) encountered during evaluation of low atmospheric transmission' 

661 f' flagging, no flagging commands generated.') 

662 return commands 

663 

664 # Get list of science scans and science SpWs, and representative SpW. 

665 scans = ms.get_scans(scan_intent='TARGET') 

666 if not scans: 

667 LOG.info('No science target scans found in MS %s, no flagging commands generated.', ms.name) 

668 return commands 

669 

670 scispws = ms.get_spectral_windows() 

671 _, repr_spwid = ms.get_representative_source_spw() 

672 

673 # Compute the mean airmass for each science scan. 

674 airmass_for_scan = {scan.id: get_airmass_for_alma_scan(scan) for scan in scans} 

675 

676 # Initializes atmospheric profile for a defined set of atmospheric 

677 # parameters from PIPE-624, that are common for all scans and SpWs. 

678 myat = casa_tools.atmosphere 

679 atmutil.init_atm(myat, altitude=5059.0, humidity=20.0, temperature=273.0, pressure=563.0, max_altitude=48.0, 

680 delta_p=5.0, delta_pm=1.1, h0=1.0, atmtype=atmutil.AtmType.tropical) 

681 

682 # Evaluate low transmission for each SpW: 

683 for spw in scispws: 

684 # Set transmission threshold based on whether current SpW is the 

685 # representative SpW. 

686 thresh_transm = mintransrepspw if spw.id == repr_spwid else mintransnonrepspws 

687 

688 # Initialize spectral window setting in atmosphere tool for current SpW, 

689 # and update atmospheric profile to set PWV (needs to happen after SpW is 

690 # initialized). 

691 atmutil.init_spw(myat, fcenter=float(spw.mean_frequency.to_units(measures.FrequencyUnits.GIGAHERTZ)), 

692 nchan=spw.num_channels, 

693 resolution=float(spw.bandwidth.to_units(measures.FrequencyUnits.GIGAHERTZ) / spw.num_channels)) 

694 myqa = casa_tools.quanta 

695 myat.setUserWH2O(myqa.quantity(pwv, 'mm')) 

696 

697 # Get wet and dry opacity from atmospheric profile, for channels of 

698 # current SpW. 

699 dry_opacity = atmutil.get_dry_opacity(myat) 

700 wet_opacity = atmutil.get_wet_opacity(myat) 

701 

702 # Get transmission spectrum for opacity profiles for current SpW and 

703 # airmass of each scan. 

704 transm = np.asarray([atmutil.calc_transmission(airmass_for_scan[scan.id], dry_opacity, wet_opacity) 

705 for scan in scans]) 

706 

707 # For collection of transmission spectra for current SpW, assess the 

708 # fraction of data points (channels, scans) that fall below the 

709 # threshold. 

710 n_low_trans = len(np.where(transm < thresh_transm)[0]) 

711 frac_below_thresh = n_low_trans / transm.size 

712 

713 # If the fraction of data points below transmission threshold is equal 

714 # to or higher than the given (fraction) threshold, then generate a 

715 # flagging command to flag the current SpW entirely. 

716 if frac_below_thresh >= max_frac_low_trans: 

717 LOG.info(f"{ms.basename}, SpW {spw.id}: fraction of data with low transmission = {n_low_trans} /" 

718 f" {transm.size} = {frac_below_thresh:.2f}; this is equal/above the max fraction of" 

719 f" {max_frac_low_trans} therefore this SpW will become flagged.") 

720 

721 # Add new flagging command for current SpW. 

722 command = f"mode='manual' spw='{spw.id}' reason='low_transmission'" 

723 commands.append(command) 

724 else: 

725 LOG.info( 

726 f"{ms.basename}, SpW {spw.id}: fraction of data with low transmission = {n_low_trans} / {transm.size} =" 

727 f" {frac_below_thresh:.2f}; this is below the max fraction of {max_frac_low_trans}, therefore" 

728 f" therefore this SpW will not be flagged.") 

729 

730 return commands

733def get_elevation_for_alma_scan(scan, edge): 

734 """Get the elevation for the beginning or the end of an ALMA scan. 

735 

736 Args: 

737 scan: Scan object. 

738 edge: Selects which edge of the scan, either "start" or "end". 

739 

740 Returns: 

741 Astropy quantity corresponding to the elevation. 

742 """ 

743 alma_site = EarthLocation.from_geocentric(x=2225015.30883296, y=-5440016.41799762, z=-2481631.27428014, unit='m') 

744 scan_field = next(iter(scan.fields)) 

745 coords = SkyCoord( 

746 scan_field.mdirection['m0']['value'], # RA 

747 scan_field.mdirection['m1']['value'], # DEC 

748 frame=scan_field.mdirection['refer'].lower(), 

749 unit=(scan_field.mdirection['m0']['unit'], scan_field.mdirection['m1']['unit'])) 

750 if edge == "start": 

751 time = Time(scan.start_time['m0']['value'], format='mjd') 

752 elif edge == "end": 

753 time = Time(scan.end_time['m0']['value'], format='mjd') 

754 else: 

755 raise RuntimeError('The parameter edge should be either "start" or "end".') 

756 coords_altaz = coords.transform_to(AltAz(obstime=time, location=alma_site)) 

757 return coords_altaz.alt

760def get_airmass_for_alma_scan(scan): 

761 """Compute the airmass corresponding to an ALMA observation scan. 

762 

763 Args: 

764 scan: Scan object. 

765 

766 Returns: 

767 Mean airmass of the scan. 

768 """ 

769 start_elevation = get_elevation_for_alma_scan(scan, "start") 

770 end_elevation = get_elevation_for_alma_scan(scan, "end") 

771 start_airmass = atmutil.calc_airmass(start_elevation.deg) 

772 end_airmass = atmutil.calc_airmass(end_elevation.deg) 

773 return (start_airmass + end_airmass)/2.
