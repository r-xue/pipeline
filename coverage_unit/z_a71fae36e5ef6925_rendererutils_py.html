231def sort_row_by(row, axes): 

232 # build primary, secondary, tertiary, etc. axis sorting functions 

233 def f(axis): 

234 def g(plot): 

235 return plot.parameters.get(axis, '') 

236 return g 

237 

238 # create a parameter getter for each axis 

239 accessors = [f(axis.strip()) for axis in axes.split(',')] 

240 

241 # sort plots in row, using a generated tuple (p1, p2, p3, ...) for 

242 # secondary sort 

243 return sorted(row, key=lambda plot: tuple([fn(plot) for fn in accessors]))

246def group_plots(data, axes): 

247 if data is None: 

248 return [] 

249 

250 # build primary, secondary, tertiary, etc. axis sorting functions 

251 def f(axis): 

252 def g(plot): 

253 return plot.parameters.get(axis, '') 

254 return g 

255 

256 keyfuncs = [f(axis) for axis in axes.split(',')] 

257 return _build_rows([], data, keyfuncs)

260def _build_rows(rows, data, keyfuncs, axis: str=''): 

261 # if this is a leaf, i.e., we are in the lowest level grouping and there's 

262 # nothing further to group by, add a new row 

263 if not keyfuncs: 

264 rows.append((data, axis)) 

265 return 

266 

267 # otherwise, this is not the final sorting axis and so proceed to group 

268 # the results starting with the first (or next) axis... 

269 keyfunc = keyfuncs[0] 

270 data = sorted(data, key=keyfunc) 

271 for group_value, items_with_value_generator in itertools.groupby(data, keyfunc): 

272 # convert to list so we don't exhaust the generator 

273 items_with_value = list(items_with_value_generator) 

274 # ... , creating sub-groups for each group as we go 

275 _build_rows(rows, items_with_value, keyfuncs[1:], axis=group_value) 

276 

277 return rows

371def get_relative_url(report_dir: str, stage_dir: str, subpage_dir: str, 

372 allow_nonexistent: bool = True) -> str | None: 

373 """ 

374 Return url to weblog subpage relative to the weblog root path, based on 

375 provided report dir, stage dir, and subpage dir. Check for and remove 

376 common path elements and handle either all relative paths, or all absolute 

377 paths. 

378 

379 If allow_nonexistent (default: True) is set to False, return None when the 

380 constructed path does not exist. 

381 """ 

382 # Check whether weblog stage path contains common path 

383 # with report dir, and if so, determine actual relative path. 

384 stage_cpath = os.path.commonpath([report_dir, stage_dir]) 

385 if stage_cpath: 

386 stage_relpath = os.path.relpath(stage_dir, stage_cpath) 

387 else: 

388 stage_relpath = stage_dir 

389 

390 # Check whether subpage path contains common path with the 

391 # report + weblog stage path, and if so, determine actual 

392 # relative path. 

393 subpage_cpath = os.path.commonpath([os.path.join(report_dir, stage_relpath), subpage_dir]) 

394 if subpage_cpath: 

395 subpage_relpath = os.path.relpath(subpage_dir, subpage_cpath) 

396 else: 

397 subpage_relpath = subpage_dir 

398 

399 # Combine paths. 

400 report_abspath = os.path.abspath(report_dir) 

401 subpage_abspath = os.path.join(report_abspath, stage_relpath, subpage_relpath) 

402 

403 # Return relative url if path exists. 

404 if os.path.exists(subpage_abspath) or allow_nonexistent: 

405 return os.path.relpath(subpage_abspath, report_abspath) 

406 else: 

407 return None

552def get_multiple_line_string(values: Iterable[Any], str_format: str = '{}', separator: str = '<br>') -> str: 

553 """Formats a sequence of values into a single delimited string. 

554 

555 Args:

560
