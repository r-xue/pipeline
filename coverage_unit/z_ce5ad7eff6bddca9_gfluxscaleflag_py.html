261 def prepare(self): 

262 inputs = self.inputs 

263 

264 # Initialize results. 

265 result = GfluxscaleflagResults() 

266 

267 # Store the vis in the result 

268 result.vis = inputs.vis 

269 result.plots = dict() 

270 

271 # create a shortcut to the plotting function that pre-supplies the inputs and context 

272 plot_fn = functools.partial(create_plots, inputs, inputs.context) 

273 

274 # Create back-up of flags. 

275 LOG.info('Creating back-up of "pre-gfluxscaleflag" flagging state') 

276 flag_backup_name_pregfsf = 'before_gfsflag' 

277 task = casa_tasks.flagmanager( 

278 vis=inputs.vis, mode='save', versionname=flag_backup_name_pregfsf) 

279 self._executor.execute(task) 

280 

281 # Create phase caltable(s) and merge into the local context. 

282 self._do_phasecal() 

283 

284 # Create amplitude caltable and merge it into the local context. 

285 # CAS-10491: for scan-based (solint='inf') amplitude solves that 

286 # will be applied to the calibrator, set interp to 'nearest'. 

287 LOG.info('Compute amplitude gaincal table.') 

288 amp_interp = 'nearest,linear' if inputs.solint == 'inf' else 'linear,linear' 

289 self._do_gaincal(intent=inputs.intent, gaintype='T', calmode='a', combine='', solint=inputs.solint, 

290 minsnr=inputs.minsnr, refant=inputs.refant, interp=amp_interp) 

291 

292 # Ensure that any flagging applied to the MS by this applycal are 

293 # reverted at the end, even in the case of exceptions. 

294 try: 

295 # Apply the new caltables to the MS. 

296 LOG.info('Applying phase-up, bandpass, and amplitude cal tables.') 

297 # Apply the calibrations. 

298 callib_map = self._do_applycal(merge=False) 

299 # copy across the vis:callibrary dict to our result. This dict 

300 # will be inspected by the renderer to know if/which callibrary 

301 # files should be copied across to the weblog stage directory 

302 result.callib_map.update(callib_map) 

303 

304 # Make "after calibration, before flagging" plots for the weblog 

305 LOG.info('Creating "after calibration, before flagging" plots') 

306 result.plots['before'] = plot_fn(inputs.intent, suffix='before') 

307 

308 # Run correctedampflag to identify outliers for intents specified in 

309 # intents_for_flagging; let "field" and "spw" be initialized 

310 # automatically based on intents and context. 

311 LOG.info('Running correctedampflag to identify outliers to flag.') 

312 cafinputs = correctedampflag.Correctedampflag.Inputs( 

313 context=inputs.context, vis=inputs.vis, intent=inputs.intent, 

314 antnegsig=inputs.antnegsig, antpossig=inputs.antpossig, 

315 tmantint=inputs.tmantint, tmint=inputs.tmint, tmbl=inputs.tmbl, 

316 antblnegsig=inputs.antblnegsig, antblpossig=inputs.antblpossig, 

317 relaxed_factor=inputs.relaxed_factor, niter=inputs.niter) 

318 caftask = correctedampflag.Correctedampflag(cafinputs) 

319 cafresult = self._executor.execute(caftask) 

320 

321 # If flags were found in the bandpass calibrator... 

322 cafflags = cafresult.flagcmds() 

323 if cafflags: 

324 # Create the "after calibration, after flagging" plots for the weblog. 

325 LOG.info('Creating "after calibration, after flagging" plots') 

326 result.plots['after'] = plot_fn(inputs.intent, suffix='after') 

327 

328 finally: 

329 # Restore the "pre-gfluxscaleflag" backup of the flagging state, to 

330 # undo any flags that were propagated from caltables to the MS by 

331 # the applycal call. 

332 LOG.info('Restoring back-up of "pre-gfluxscaleflag" flagging state.') 

333 task = casa_tasks.flagmanager( 

334 vis=inputs.vis, mode='restore', versionname=flag_backup_name_pregfsf) 

335 self._executor.execute(task) 

336 

337 # If new outliers were identified... 

338 if cafflags: 

339 # Re-apply the newly found flags from correctedampflag. 

340 LOG.info('Re-applying flags from correctedampflag.') 

341 fsinputs = FlagdataSetter.Inputs( 

342 context=inputs.context, vis=inputs.vis, table=inputs.vis, 

343 inpfile=[]) 

344 fstask = FlagdataSetter(fsinputs) 

345 fstask.flags_to_set(cafflags) 

346 fsresult = self._executor.execute(fstask) 

347 

348 # Store flagging task result. 

349 result.cafresult = cafresult 

350 

351 return result

376 def _do_gaincal(self, field: Optional[str] = None, intent: Optional[str] = None, gaintype: str = 'G', 

377 calmode: Optional[str] = None, combine: Optional[str] = None, solint: Optional[str] = None, 

378 minsnr: Optional[float] = None, refant: Optional[str] = None, spwmap: Optional[list] = None, 

379 interp: Optional[str] = None): 

380 inputs = self.inputs 

381 ms = inputs.ms 

382 

383 # Identify which science spws were selected by inputs parameter. 

384 request_spws = ms.get_spectral_windows(task_arg=inputs.spw) 

385 

386 # Identify which scans covered the requested intent, field, and any of 

387 # the requested spws. 

388 targeted_scans = ms.get_scans(scan_intent=intent, spw=inputs.spw, field=field) 

389 

390 # Among the requested spws, identify which have a scan among the 

391 # targeted scans. 

392 scan_spws = {spw for scan in targeted_scans for spw in scan.spws if spw in request_spws} 

393 

394 # Create separate phase caltables for each spectral spec. 

395 for spectral_spec, tuning_spw_ids in utils.get_spectralspec_to_spwid_map(scan_spws).items(): 

396 tuning_spw_str = ','.join([str(i) for i in sorted(tuning_spw_ids)]) 

397 LOG.info('Processing spectral spec {}, spws {}'.format(spectral_spec, tuning_spw_str)) 

398 

399 scans_with_data = ms.get_scans(spw=tuning_spw_str, scan_intent=intent, field=field) 

400 if not scans_with_data: 

401 LOG.info('No data to process for spectral spec {}. Continuing...'.format(spectral_spec)) 

402 continue 

403 

404 # of the fields that we are about to process, does any field have 

405 # multiple intents? 

406 mixed_intents = False 

407 fields_in_scans = {fld for scan in scans_with_data for fld in scan.fields} 

408 singular_intents = frozenset(intent.split(',')) 

409 if len(singular_intents) > 1: 

410 for fld in fields_in_scans: 

411 intents_to_scans = {si: ms.get_scans(scan_intent=si, field=fld.id, spw=tuning_spw_str) 

412 for si in singular_intents} 

413 valid_intents = [k for k, v in intents_to_scans.items() if v] 

414 if len(valid_intents) > 1: 

415 mixed_intents = True 

416 break 

417 

418 if mixed_intents and solint == 'inf': 

419 # multiple items, one for each intent. Each intent will result 

420 # in a separate job. 

421 # Make sure data for the intent exists (PIPE-367) 

422 intents_to_scans = {si: [scan for scan in scans_with_data if si in scan.intents] 

423 for si in singular_intents} 

424 valid_intents = [k for k, v in intents_to_scans.items() if v] 

425 task_intents = valid_intents 

426 

427 else: 

428 # one item, and hence one job, with 'PHASE,BANDPASS,...' 

429 task_intents = [','.join(singular_intents)] 

430 

431 for intent in task_intents: 

432 # Initialize gaincal inputs. 

433 task_inputs = gaincal.GTypeGaincal.Inputs( 

434 inputs.context, 

435 vis=inputs.vis, 

436 field=field, 

437 intent=intent, 

438 spw=tuning_spw_str, 

439 solint=solint, 

440 gaintype=gaintype, 

441 calmode=calmode, 

442 minsnr=minsnr, 

443 combine=combine, 

444 refant=refant, 

445 solnorm=False) 

446 

447 # if we need to generate multiple caltables, make the caltable 

448 # names unique by inserting the intent to prevent them overwriting 

449 # each other 

450 if len(task_intents) > 1: 

451 root, ext = os.path.splitext(task_inputs.caltable) 

452 task_inputs.caltable = '{!s}.{!s}{!s}'.format(root, intent, ext) 

453 

454 # Modify output table filename to append "prelim". 

455 if task_inputs.caltable.endswith('.tbl'): 

456 task_inputs.caltable = task_inputs.caltable[:-4] + '.prelim.tbl' 

457 else: 

458 task_inputs.caltable += '.prelim' 

459 

460 # Initialize and execute gaincal task. 

461 task = gaincal.GTypeGaincal(task_inputs) 

462 result = self._executor.execute(task) 

463 

464 # modify the result so that this caltable is only applied to 

465 # the intent from which the calibration was derived 

466 calapp_overrides = dict(intent=intent) 

467 

468 # Phase solution caltables should be registered with 

469 # calwt=False (PIPE-1154). 

470 if calmode == 'p': 

471 calapp_overrides['calwt'] = False 

472 

473 # Adjust the field if provided. 

474 if field: 

475 calapp_overrides['field'] = field 

476 

477 # Adjust the interp if provided. 

478 if interp: 

479 calapp_overrides['interp'] = interp 

480 

481 # Adjust the spw map if provided. 

482 if spwmap: 

483 calapp_overrides['spwmap'] = spwmap 

484 

485 # Create modified CalApplication and replace CalApp in result 

486 # with this new one. 

487 if result.final: 

488 original_calapp = result.final[0] 

489 modified_calapp = callibrary.copy_calapplication(original_calapp, **calapp_overrides) 

490 result.pool[0] = modified_calapp 

491 result.final[0] = modified_calapp 

492 

493 # Merge the result to the local context to register new caltable 

494 # in local context callibrary. 

495 result.accept(inputs.context)

497 def _do_phasecal(self): 

498 # Note: at present, phaseupsolint is specified as a fixed 

499 # value, defined in inputs. In the future, phaseupsolint may 

500 # need to be set based on exposure times; if so, see discussion 

501 # in CAS-10158 and logic in hifa.tasks.fluxscale.GcorFluxscale. 

502 inputs = self.inputs 

503 

504 # PIPE-2499: create separate phase solutions for each combination of 

505 # input intent and its corresponding fields, while using optimal gaincal 

506 # parameters based on spwmapping registered in the measurement set. 

507 # The phase caltables created are merged into the local context so they 

508 # are pre-applied in the subsequent local amplitude calibration. 

509 for intent in inputs.intent.split(','): 

510 for field in inputs.ms.get_fields(intent=intent): 

511 LOG.info(f'Compute phase gaincal table for intent={intent}, field={field.name}.') 

512 combine, interp, spwmap = self._get_phasecal_params(inputs.ms, intent, field.name) 

513 self._do_gaincal(field=field.name, intent=intent, gaintype='G', calmode='p', combine=combine, 

514 solint=inputs.phaseupsolint, minsnr=inputs.minsnr, refant=inputs.refant, spwmap=spwmap, 

515 interp=interp) 

516 

517 @staticmethod 

518 def _get_phasecal_params(ms: MeasurementSet, intent: str, field: str) -> Tuple[str, Optional[str], list]: 

519 # By default, no spw mapping or combining, no interp. 

520 combine = '' 

521 interp = None 

522 spwmap = [] 

523 

524 # Try to fetch spwmapping info from MS for requested intent and field. 

525 spwmapping = ms.spwmaps.get((intent, field), None) 

526 

527 # If a mapping was found, use the spwmap, and update combine and interp 

528 # depending on whether it is a combine spw mapping. 

529 if spwmapping: 

530 spwmap = spwmapping.spwmap 

531 if spwmapping.combine: 

532 combine = 'spw' 

533 interp = 'linearPD,linear' 

534 

535 return combine, interp, spwmap
