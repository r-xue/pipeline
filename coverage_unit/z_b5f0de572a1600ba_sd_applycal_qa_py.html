17### Dictionary of default Threshold values ### 

18# X-Y_freq_dev: Threshold for devation in ON-source XX-YY polarization data difference

32def data_stats_perchan(msw: mswrapper_sd.MSWrapperSD, filter_order: int = 5, 

33 filter_cutoff: float = 0.01, loworder_to_zero: int = 3): 

34 '''Function that calculates summary statistics per channel for a MSWrapperSD object. This comprises the FFT peak (along 

35 the time dimension) and Pearson correlation coefficient between the XX and YY polarization, also in the time dimension. 

36 These vector are stored in the "data_stats" attribute from the MSWrapper_SD object and are later used by the function 

37 sci_line_det() to determine channels likely to correspond to science line emission. 

38 param: 

39 msw: MSWrapper_SD object for which to calculate the data statistics used for the QA scores calculation. 

40 filter_order, filter_cutoff: (float) parameters to be used with signal.butter highpass filter applied at the beggining. 

41 loworder_to_zero: (int) Number of modes to zero-out at the beggining of the FFT. This is aimed at removing any low order 

42 variation in the time dimension of the data. 

43 peak_modes (list(int)), peak_minsnr (float): Parameters passed to the function scipy.signal.find_peaks_cwt() to search for 

44 peaks in the spectrum (abs(FFT)). 

45 returns: 

46 Dictionary "data_stats" as an attribute of the input MSWrapperSD object, containing two 1D vectors of size equal 

47 to the number of channels of the SPW: 

48 "peak_fft_pwr": The peak of the absolute value of the FFT. 

49 "XYcorr": The Person correlation coefficient between XX and YY. 

50 ''' 

51 

52 #In case of empty object, return dummy arrays 

53 if msw.nrows == 0: 

54 nchan = msw.spw_setup[msw.spw]['nchan'] 

55 msw.data_stats = {'peak_fft_pwr': np.ma.MaskedArray(np.zeros(nchan), mask=np.ones(nchan, dtype=bool)), 

56 'XYcorr': np.ma.MaskedArray(np.zeros(nchan), mask=np.ones(nchan, dtype=bool))} 

57 return 

58 

59 #Handle cases with 2 or 1 polarizations. Full stokes pol is not considered for now, so return dummy array for that. 

60 (npol, nchan, nrows) = np.shape(msw.data) 

61 maskfreq = np.min(msw.data.mask, axis=2) 

62 if npol == 2: 

63 maskfreq1D = maskfreq[0] | maskfreq[1] 

64 elif npol == 1: 

65 maskfreq1D = maskfreq[0] 

66 else: 

67 print('Error: Could not handle npol = '+str(npol)) 

68 msw.data_stats = {'peak_fft_pwr': np.ma.MaskedArray(np.zeros(nchan), mask=np.ones(nchan, dtype=bool)), 

69 'XYcorr': np.ma.MaskedArray(np.zeros(nchan), mask=np.ones(nchan, dtype=bool))} 

70 return 

71 

72 #Selection of skylines 

73 skylinesel = sd_qa_utils.get_skysel_from_msw(msw) 

74 idxskylines = np.arange(nchan)[skylinesel] 

75 

76 #Make a copy of the data and apply the per-chan filter 

77 print('Copying and filtering per channel...') 

78 Fmsw = copy.deepcopy(msw) 

79 #Apply filters per channel 

80 Fmsw.filter(type='chanhighpass', filter_order = filter_order, filter_cutoff = filter_cutoff) 

81 

82 #Get peak of FFT time-spectrum per freq-channel 

83 peak_fft_pwr = [] 

84 XYcorr = [] 

85 nstep = int(np.floor(nchan/10.0)) 

86 full_freqs = np.fft.rfftfreq(nrows) 

87 mask_sections = None 

88 last_chdata_mask = np.zeros(nrows, dtype=bool) 

89 for ch in range(nchan): 

90 if maskfreq1D[ch] or (ch in idxskylines): 

91 peak_fft_pwr.append(0.0) 

92 XYcorr.append(0.0) 

93 maskfreq1D[ch] = True 

94 continue 

95 if (ch%nstep == 0): 

96 print('Processing channel '+str(ch)+'/'+str(nchan)) 

97 chdata = Fmsw.data[:,ch,:] 

98 #Calculate X-Y pols Pearson correlation 

99 if npol == 2: 

100 XYcorr.append(mstats.pearsonr(chdata[0], chdata[1])[0]) 

101 else: 

102 XYcorr.append(0.0) 

103 #Calculate fft_results 

104 if npol == 2: 

105 chdataSum = np.ma.mean(chdata, axis=0) 

106 else: 

107 chdataSum = chdata[0] 

108 #Check is we have a valid masked sections array from lst iteration, if not, set it to None and force recalculate 

109 if (mask_sections is not None) and (not np.all(chdataSum.mask == last_chdata_mask)): 

110 mask_sections = None 

111 absfft_data, mask_sections = sd_qa_utils.abs_rfft_wmask(chdataSum, full_freqfft=full_freqs, mask_sections=mask_sections) 

112 last_chdata_mask = chdataSum.mask 

113 #Zero out constant and neighboring modes 

114 absfft_data[0:loworder_to_zero] = 0.0 

115 #Get peak 

116 peak_fft_pwr.append(np.max(absfft_data)) 

117 peak_fft_pwr = np.ma.MaskedArray(peak_fft_pwr, mask=maskfreq1D) 

118 XYcorr = np.ma.MaskedArray(XYcorr, mask=maskfreq1D) 

119 if npol == 1: 

120 XYcorr.mask = np.ones(nchan, dtype=bool) 

121 

122 #Release memory taken up by the copy of the dataset 

123 del(Fmsw) 

124 

125 #Save results in the original MSWrapper object 

126 msw.data_stats = {'peak_fft_pwr': peak_fft_pwr, 'XYcorr': XYcorr} 

127 

128 return

130def combine_msw_stats(mswCollection: dict) -> dict: 

131 '''Function that return the combination of "data_stats" attributes in a collection of MSWrapperSD objects. 

132 Currently combination occurs just as a channel-to-channel stack, i.e. assuming the LSRK frequency of the 

133 channels of different EBs is the same. 

134 param: 

135 mswCollection: Dictionary with a collection of MSWrapperSD generated by the load_and_stats() function. 

136 returns: 

137 Dictionary "data_stats_comb", which correspond to the combination of "peak_fft_pwr" and "XYcorr" vectors 

138 from the collection of MSWrapperSD, named as "comb_peak_data" and "comb_XYcorr", grouped by science field name and SPW: 

139 data_stats_comb[fieldname][spw] 

140 ''' 

141 

142 #Get arrays to index msw Collection 

143 mswCollectionKeys = list(mswCollection.keys()) 

144 mskeys = (np.array(mswCollectionKeys).T)[0] 

145 mslist = np.unique(mskeys) 

146 idxfirst = {ms: np.where(mskeys == ms)[0][0] for ms in mslist} 

147 fieldnames = np.unique(np.array(sd_qa_utils.flattenlist([mswCollection[mswCollectionKeys[idxfirst[ms]]].spw_setup['fieldname']['*OBSERVE_TARGET#ON_SOURCE*'] for ms in mslist]))) 

148 names_for = {ms: mswCollection[mswCollectionKeys[idxfirst[ms]]].spw_setup['namesfor'] for ms in mslist} 

149 fieldid_for = {ms: {fieldname[0]: fieldid for (fieldid, fieldname) in names_for[ms].items()} for ms in mslist} 

150 antnames_for = {ms: mswCollection[mswCollectionKeys[idxfirst[ms]]].spw_setup['antnames'] for ms in mslist} 

151 #assuming all EBs have the same SPW names, no matter what numbering, and we can take the first one. If not one would need to 

152 #make this more general. SPW list taken is the one form the first SPW 

153 spwlist = mswCollection[mswCollectionKeys[idxfirst[mslist[0]]]].spw_setup['spwlist'] 

154 spwnamelist = mswCollection[mswCollectionKeys[idxfirst[mslist[0]]]].spw_setup['spwnames'] 

155 #Some MSs could have less fields, lets create a list of the MS that do for each field 

156 mslist_for_fieldname = {fieldname: [ms for ms in mslist if fieldname in fieldid_for[ms].keys()] for fieldname in fieldnames} 

157 

158 #Initialize output dictionary with metadata from the combination 

159 data_stats_comb = {'comb_metadata': {'mslist': mslist, 'mskeys': mskeys, 'idxfirst': idxfirst, 'fieldnames': fieldnames, 'names_for': names_for, 'fieldid_for': fieldid_for, 'antnames_for': antnames_for, 'spwlist': spwlist, 'spwnamelist': spwnamelist}} 

160 

161 #Iterate over field and SPW and combine 

162 #Combination done assuming LSRK velocity is being tracked by the online software, so science lines should 

163 #align in channel number. If not, a regrid to a LSRK frame would be necessary. 

164 for fieldname in fieldnames: 

165 data_stats_comb[fieldname] = {} 

166 for spwname in spwnamelist: 

167 comb_peak_data = np.ma.sum([np.ma.sum([mswCollection[(ms, spwname, ant, str(fieldid_for[ms][fieldname]))].data_stats['peak_fft_pwr'] for ant in antnames_for[ms]], axis=0) for ms in mslist_for_fieldname[fieldname]], axis=0) 

168 comb_XYcorr = np.ma.mean([np.ma.mean([mswCollection[(ms, spwname, ant, str(fieldid_for[ms][fieldname]))].data_stats['XYcorr'] for ant in antnames_for[ms]], axis=0) for ms in mslist_for_fieldname[fieldname]], axis=0) 

169 data_stats_comb[fieldname][spwname] = {'comb_peak_data': comb_peak_data, 'comb_XYcorr': comb_XYcorr} 

170 

171 return data_stats_comb

173def sci_line_det(data_stats_comb:dict, sci_threshold:float = 15.0, XYcorr_threshold:float = 15.0, enlarge_box:float = 0.02): 

174 '''Function to create selection vector for science emission line channels, based on the "comb_peak_data" and "comb_XYcorr" 

175 produced by the combine_msw_stats() function. 

176 param: 

177 data_stats_comb: (dict) Data stats combination dictionary, as returned by combine_msw_stats() 

178 sci_threshold: (float) Threshold for the FFT Peak data. 

179 XYcorr_threshold: (float) Threshold for X-Y correlation data. 

180 enlarge_box: (float) Size of the box smoothing applied to resulting selection vector, given as fraction 

181 of the SPW. 

182 returns: 

183 Function modifies the input data_stats_comb variable, adding to this dictionary the selection vectors and summary of the 

184 thresholding algorithm (obtained from the sd_qa_utils.smoothed_sigma_clip function). 

185 ''' 

186 

187 for fieldname in data_stats_comb['comb_metadata']['fieldnames']: 

188 for spwname in data_stats_comb['comb_metadata']['spwnamelist']: 

189 

190 #Select science line channels 

191 fft_results = sd_qa_utils.smoothed_sigma_clip(data_stats_comb[fieldname][spwname]['comb_peak_data'], sci_threshold, mode = 'one-sided') 

192 XYcorr_results = sd_qa_utils.smoothed_sigma_clip(data_stats_comb[fieldname][spwname]['comb_XYcorr'], XYcorr_threshold, mode = 'one-sided', fixedwidth=fft_results['widthmax']) 

193 sci_line_sel = fft_results['outliers'] & XYcorr_results['outliers'] 

194 #Enlarge selection according to enlarge_box parameter 

195 nchan = len(data_stats_comb[fieldname][spwname]['comb_peak_data']) 

196 enlarge_box_pix = int(nchan*enlarge_box) 

197 sci_line_sel = sd_qa_utils.enlargesel(sci_line_sel, enlarge_box_pix) 

198 #Continuum selection 

199 contsel = (~fft_results['outliers']) & (~XYcorr_results['outliers']) 

200 #Store resulting selection vectors back in the data_stats_comb dictionary 

201 data_stats_comb[fieldname][spwname]['sci_line_sel'] = sci_line_sel 

202 data_stats_comb[fieldname][spwname]['cont_sel'] = contsel 

203 data_stats_comb[fieldname][spwname]['fft_results'] = fft_results 

204 data_stats_comb[fieldname][spwname]['XYcorr_results'] = XYcorr_results 

205 

206 return

232def qascorefunc(nsigma: float, score_top: float = 0.67, score_bottom: float = 0.34, nsigma_threshold: float = 3.5, 

233 nsigma_bottom: float = 10.0): 

234 '''Function used to calculate the QA score numerical evaluation as a linear mapping between the number of sigma excess 

235 of the outlier and the QA score value. 

236 param: 

237 nsigma: Number of sigma of detected outlier 

238 score_top: upper range of QA score value to consider 

239 score_bottom: lower range of QA score value to consider 

240 nsigma_threshold: Number of sigmas set as threshold, value for which a QA score turns the color in this range. 

241 nsigma_bottom: Number of sigmas for which a blue QA scores turn bottom value. Must be greater than nsigma_threshold. 

242 ''' 

243 

244 return max(score_top - (score_top - score_bottom)*(nsigma - nsigma_threshold)/(nsigma_bottom - nsigma_threshold), score_bottom)

381 qascore_lowest = 0.34 

382 idx_lowest = k 

383 continue

385 #Detect outliers in ON-data 

386 ondata_results = sd_qa_utils.smoothed_sigma_clip(datadiff, thresholds['X-Y_freq_dev'], mode = 'two-sided') 

387 normdatadiff = ondata_results['normdata'] 

388 outlier_data = ondata_results['outliers'] & (~sci_line_channels)

394 trecdiffX = trecX_results['normdata']

398 trecdiffY = trecY_results['normdata'] 

399 outlier_trecY = trecY_results['outliers'] 

400 nsigma_trecY = trecY_results['snrmax'] 

401 

402 #Pick up Trec outliers that match data outliers 

403 outlier_data_trecX = trecX_results['outliers'] & outlier_data 

404 outlier_data_trecY = trecY_results['outliers'] & outlier_data 

405 

406 #Booleans with outlier cases 

407 has_data_outliers = (nsigma_ondata > thresholds['X-Y_freq_dev']) 

408 has_trecX_outliers = (np.ma.sum(outlier_data_trecX) > 0) 

409 has_trecY_outliers = (np.ma.sum(outlier_data_trecY) > 0) 

410 any_detection = any([has_data_outliers, has_trecX_outliers, has_trecY_outliers]) 

411 

412 #Figure out if data outliers are due to XX or YY polarization, if applicable 

413 pr_x = mstats.pearsonr(dataX, datadiff) 

414 pr_y = mstats.pearsonr(-dataY, datadiff) 

415 #Texts for message 

416 if any_detection and ((pr_x[0] > pr_y[0]) or (has_trecX_outliers and not has_trecY_outliers)): 

417 badpol = 'XX' 

418 badtrec = '. Also detected in the Trec[XX] table.' 

419 elif any_detection and ((pr_x[0] < pr_y[0]) or (not has_trecX_outliers and has_trecY_outliers)): 

420 badpol = 'YY' 

421 badtrec = '. Also detected in the Trec[YY] table.' 

422 elif any_detection and has_trecX_outliers and has_trecY_outliers: 

423 badpol = 'not identified' 

424 badtrec = '. Also detected in the Trec[XX] and Trec[YY] tables.' 

425 else: 

426 pr_x, pr_y = [0.0, 0.0], [0.0, 0.0] 

427 badpol = 'N/A' 

428 

429 #Calculate max of outlier strength relative to range of data for report, use on-data and calcualte 

430 #percentage of deviation relative to RMS 

431 if msw.spw_setup['sensitivityGoalinJy']: 

432 peak_outlier_percent = np.round(100.0*np.abs(ondata_results['datamax'])/msw.spw_setup['sensitivityGoalinJy'], decimals=1) 

433 else: 

434 # unrealistic value to disable second condition in the if statement below 

435 peak_outlier_percent = 120.0 

436 #width detection as percentage of BW 

437 width_bw_percent = 100.0*ondata_results['widthmax']/nchan 

438 

439 #Create objects to create QAscore 

440 reason = 'XX-YY.deviation' 

441 applies_to = pqa.TargetDataSelection(vis={msname}, field={fieldname}, scan={scan}, intent={'*OBSERVE_TARGET#ON_SOURCE*'}, spw={msw.spw}, ant={msw.antenna}, pol={badpol}) 

442 comes_from = pqa.QAOrigin(metric_name=reason, metric_score=nsigma_ondata, metric_units='n-sigma deviation') 

443 

444 #QA score value evaluation 

445 if has_data_outliers and (has_trecX_outliers or has_trecY_outliers) and (peak_outlier_percent >= thresholds['min_freq_dev_percent']): 

446 #Case where actual outliers were found, put a yellow QA score 

447 qascore_value = qascorefunc(nsigma_ondata, score_top = 0.67, score_bottom = 0.34, nsigma_threshold = thresholds['X-Y_freq_dev'], nsigma_bottom = thresholds['nsigma_bottom']) 

448 longmsg = 'XX-YY large deviation outlier in data and Trec table for spw {0:d}, antenna {1:s}, polarization {2:s} in scan {3:s} (field {4:s})'.format(msw.spw, msw.antenna, badpol, str(scan), fieldname) 

449 #longmsg = 'XX-YY deviation outlier ({0:.1f}% of RMS) at {1:.3f}-sigma for spw {2:d}, antenna {3:s}, polarization {4:s} in scan {5:s} (field {6:s}) {7:s}; nsigma Trec: {8:.3f},{9:.3f}; width: {10:.2f}%'.format(peak_outlier_percent, nsigma_ondata, msw.spw, msw.antenna, badpol, str(scan), fieldname, badtrec, nsigma_trecX, nsigma_trecY, width_bw_percent) 

450 #Mark outliers in outlier array for plotting 

451 if scan != 'all': 

452 for row in np.where(msw.scantimesel[scan])[0]: 

453 msw.outliers[:,row] = outlier_data 

454 else: 

455 for row in range(nrows): 

456 msw.outliers[:,row] = outlier_data 

457 elif has_data_outliers and not(has_trecX_outliers or has_trecY_outliers): 

458 #Non outliers case only, put a blue QA score 

459 qascore_value = qascorefunc(nsigma_ondata, score_top = 0.9, score_bottom = 0.68, nsigma_threshold = thresholds['X-Y_freq_dev'], nsigma_bottom = thresholds['nsigma_bottom']) 

460 longmsg = 'XX-YY deviation for spw {0:d}, antenna {1:s}, polarization {2:s} in scan {3:s} (field {4:s})'.format(msw.spw, msw.antenna, badpol, str(scan), fieldname) 

461 #longmsg = 'XX-YY polarization difference ({0:.1f}% of RMS) at {1:.3f}-sigma for spw {2:d}, antenna {3:s}, polarization {4:s} in scan {5:s} (field {6:s}), likely due to atmosphere instability, check deviation mask at baseline subtraction; nsigma Trec: {7:.3f},{8:.3f}; width: {9:.2f}%'.format(peak_outlier_percent, nsigma_ondata, msw.spw, msw.antenna, badpol, str(scan), fieldname, nsigma_trecX, nsigma_trecY, width_bw_percent) 

462 #Mark outliers in outlier array for plotting 

463 if scan != 'all': 

464 for row in np.where(msw.scantimesel[scan])[0]: 

465 msw.outliers[:,row] = outlier_data 

466 else: 

467 for row in range(nrows): 

468 msw.outliers[:,row] = outlier_data 

469 else: 

470 #Case of no outliers and no information 

471 qascore_value = 1.0 

472 longmsg = 'No significant XX-YY polarization difference detected for spw {0:d}, antenna {1:s} in scan {2:s} (field {3:s})'.format(msw.spw, msw.antenna, str(scan), fieldname) 

473 

474 if qascore_value <= qascore_lowest: 

475 qascore_lowest = qascore_value 

476 idx_lowest = k 

477 

478 thisqascore = pqa.QAScore(qascore_value, longmsg=f"{msname}: {longmsg}", shortmsg='XX-YY v/s Frequency deviation', origin=comes_from, applies_to=applies_to, hierarchy=reason) 

479 qascores_scans.append(thisqascore) 

480 

481 analysis[scan] = {'ondata': ondata_results, 'trecX': trecX_results, 'trecY': trecY_results, 

482 'qascore': thisqascore} 

483 

484 #Mark marginalized outlier selections 

485 msw.outlierfreq = (np.max(msw.outliers, axis = 1) > 0) 

486 msw.outliertime = (np.max(msw.outliers, axis = 0) > 0) 

487 

488 #Create summary qascore for this antenna's data 

489 qascoresvalues = np.array([q.score for q in qascores_scans]) 

490 outlierscans = np.array(extscanlist)[(qascoresvalues < 0.67)] 

491 outlierscans_str = ','.join(map(str,outlierscans)) 

492 if idx_lowest >= 0 and idx_lowest < nscans: 

493 qascore = copy.deepcopy(qascores_scans[idx_lowest]) 

494 if len(outlierscans) > 1: 

495 qascore.longmsg = qascore.longmsg.replace('scan '+str(scanlist[idx_lowest]), 'scans '+outlierscans_str) 

496 qascore.applies_to.scan = set(outlierscans) 

497 else: 

498 qascore = copy.deepcopy(qascores_scans[-1]) 

499 

500 detmsg = 'N/A' 

501 if not all([(analysis[scan] is not None) for scan in scanlist]): 

502 snrlist = np.ma.array([analysis[scan]['ondata']['snrmax'] if ((analysis[scan] is not None) and (np.ma.sum(analysis[scan]['ondata']['outliers']) > 0)) else np.nan for scan in scanlist]) 

503 snrlist.mask = ~np.isfinite(snrlist) 

504 idxmaxsnr = np.ma.argmax(snrlist) 

505 scanmax = scanlist[idxmaxsnr] 

506 if not np.ma.is_masked(snrlist[idxmaxsnr]): 

507 detmsg = 'ON-data: SNR={0:.3f},f={1:.3f}GHz,W={2:.3f}MHz({3:.3f}%)'.format(snrlist[idxmaxsnr], freqs[analysis[scanmax]['ondata']['chanmax']], analysis[scanmax]['ondata']['widthmax']*chansep, 100.0*analysis[scanmax]['ondata']['widthmax']/nchan) 

508 

509 #Save analysis data 

510 msw.analysis = analysis 

511 

512 #Plot science line detections 

513 if (plot_sciline == 'always') or ((plot_sciline == 'on-detection') and is_sci_line_det): 

514 sciplotfname = sd_qa_reports.plot_science_det(msw = msw, thresholds = thresholds, plot_output_path = plot_output_path) 

515 

516 #Plot heatmap if QA score < plot_threshold 

517 colorlist = sd_qa_utils.genColorList(len(extscanlist)) 

518 if (idx_lowest >= 0) and (qascore.score <= thresholds['plot_threshold']): 

519 print('Creating plot outputs in '+str(plot_output_path)) 

520 #Make heat map 

521 if msw.data is None:

524 else: 

525 thistsysdata = False

527 tmpmsw = mswrapper_sd.MSWrapperSD.create_from_ms(fname=msw.fname, spw=msw.spw, antenna=msw.antenna, fieldid=msw.fieldid, column='CORRECTED_DATA', onoffsel='ON', spw_setup=msw.spw_setup, attach_tsys_data=thistsysdata) 

528 tmpmsw.filter(type = 'rowmedian') 

529 msw.data = tmpmsw.data 

530 _ = sd_qa_reports.show_heat_XYdiff(msw=msw, plot_output_path=plot_output_path, colorlist=colorlist) 

531 #Make diagnostic plots

539def load_and_stats(msNames: List[str], use_tsys_data: bool = True, sciline_det: bool = True, 

540 buffer_data: bool = False) -> dict: 

541 '''Load a collection of MSWrapper_SD objects, and run a basic filter and statistics on each of them. 

542 Return a dictionary of MSWrapper_SD objects, indexed by tuples in the form (ms, spw, ant, fieldid). 

543 param: 

544 msNames: List of MS names to load. 

545 ondata_filters: List of filters to pass on the ON-source data 

546 use_tsys_data: True/False whether to load CalAtmosphere data from dataset 

547 sciline_det: True/False whether to attempt strong science line detection. The detected channels are passed to outlier_detection() 

548 to exclude them for triggering a false XX-YY deviation. 

549 buffer_data: True/False whether to save/load the MSW collection to disk. This is aimed at easing the 

550 use for repeated execution of the code, mostly testing. 

551 ''' 

552 

553 #Load buffered data from disk, if exists 

554 buf = os.path.join(os.path.dirname(msNames[0]), 'mswCollection.buffer.pkl') 

555 if os.path.exists(buf): 

556 f = open(buf, 'rb') 

557 mswCollection = pickle.load(f) 

558 f.close() 

559 return mswCollection 

560 

561 #Load data and gather statistics 

562 mswCollection = {} 

563 for ms in msNames: 

564 msonly = ms.split('/')[-1] 

565 #Load metadata 

566 spw_setup = sd_qa_utils.getSpecSetup(myms=ms) 

567 #Iterate over SPW, antenna, FieldID 

568 datapieces = product(spw_setup['spwlist'], spw_setup['antnames'], spw_setup['fieldid']['*OBSERVE_TARGET#ON_SOURCE*']) 

569 for datapiece in datapieces: 

570 (spw, ant, fieldid) = datapiece 

571 spwname = spw_setup['spwnames'][spw_setup['spwlist'].index(spw)] 

572 print('Processing MS {0:s}, SPW {1:d}, antenna {2:s}, FieldID {3:d}'.format(ms,spw,ant,fieldid)) 

573 #Load ON-SOURCE data and create MSWrapperSD object 

574 mswCollection[(msonly, spwname, ant, str(fieldid))] = mswrapper_sd.MSWrapperSD.create_from_ms(fname=ms, spw=spw, antenna=ant, fieldid=fieldid, column='CORRECTED_DATA', onoffsel='ON', spw_setup=spw_setup, attach_tsys_data=use_tsys_data) 

575 #Apply preliminary filters to ON-data 

576 print('Applying filter rowmedian...') 

577 mswCollection[(msonly, spwname, ant, str(fieldid))].filter(type = 'rowmedian') 

578 #Gather statistics per channel, if requested 

579 if sciline_det: 

580 data_stats_perchan(mswCollection[(msonly, spwname, ant, str(fieldid))]) 

581 #Clear raw data from MSWrapper object to release memory 

582 mswCollection[(msonly, spwname, ant, str(fieldid))].data = None 

583 

584 if buffer_data: 

585 print('Buffering mswCollection...') 

586 f = open(buf, 'wb') 

587 pickle.dump(mswCollection, f, protocol=2) 

588 f.close() 

589 

590 return mswCollection

592def get_ms_applycal_qascores(msNames: List[str], thresholds: dict = default_thresholds, plot_output_path: str = '.', 

593 use_tsys_data: bool = True, sciline_det: bool = True, 

594 plot_sciline: str = 'on-detection', weblog_output_path: str = '.', 

595 buffer_data: bool = False) -> Tuple[list, list, list]: 

596 '''Function used to obtain applycal X-Y QA score on a list of calibrated MSs, at the 

597 applycal stage of SD pipeline. 

598 param: 

599 msNames: List of MS to process 

600 thresholds: Dictionary containing threshold parameters for ON-source XX-YY deviations, 

601 Trec deviations, minimum deviation level, and QA score scaling. 

602 plot_output_path: (str) Path to save output plots. 

603 use_tsys_data: True/False whether to load CalAtmosphere data from dataset 

604 sciline_det: True/False whether to attempt strong science line detection. The detected channels are passed to outlier_detection() 

605 to exclude them for triggering a false XX-YY deviation. 

606 plot_sciline: (str) String stating whether a diagnostic plot about science line detection should be done. options: 

607 "always"/"on-detection"/"never" 

608 weblog_output_path: (str) Path to save output plots if the plots are intended to 

609 be a part of the pipeline weblog. 

610 buffer_data: True/False whether to buffer the data loaded from the MS into a pickle file. 

611 ''' 

612 

613 #Load data and gather statistics 

614 mswCollection = load_and_stats(msNames, use_tsys_data=use_tsys_data, sciline_det=sciline_det, 

615 buffer_data=buffer_data) 

616 

617 #Combine stats for science line detection, if requested 

618 if sciline_det: 

619 data_stats_comb = combine_msw_stats(mswCollection) 

620 #Run simple science line detection 

621 sci_line_det(data_stats_comb, sci_threshold = default_thresholds['sci_threshold'], XYcorr_threshold = default_thresholds['XYcorr_threshold']) 

622 #put results inside MSWrapper_SD objects data 

623 attach_sci_line_res(mswCollection, data_stats_comb) 

624 

625 #initialize output lists 

626 qascore_list = [] 

627 plots_fnames = [] 

628 qascore_per_scan_list = [] 

629 

630 #Iterate over collection of MSWrapperSD objects 

631 for key, msw in mswCollection.items(): 

632 (ms, spw, ant, fieldid) = key 

633 #Calculate QAscores 

634 (msw_on, qascore, qascores_scans, plotfname) = outlier_detection(msw = msw, thresholds = thresholds, plot_output_path = plot_output_path, plot_sciline = plot_sciline, weblog_output_path = weblog_output_path) 

635 #Collect QA scores 

636 qascore_list.append(qascore) 

637 plots_fnames.append(plotfname) 

638 qascore_per_scan_list.extend(qascores_scans) 

639 

640 if sciline_det: 

641 #Print channel selection to a text output file 

642 scilinestr = sd_qa_utils.sci_line_sel_2str(data_stats_comb) 

643 f = open(plot_output_path + '/scilines_det.txt', 'w') 

644 f.write(scilinestr) 

645 f.close() 

646 

647 return (qascore_list, plots_fnames, qascore_per_scan_list)
