1""" 

2Code in the argmapper module transforms CASA input arguments to pipeline 

3inputs arguments. 

4 

5CASA arguments may be named differently to their corresponding arguments in 

6the pipeline classes. This module defines how those arguments should be 

7renamed so that they are acceptable to the Inputs constructors, and secondly 

8it converts from the CASA concept of null values ('', [], etc.) to the 

9pipeline equivalent. 

10""" 

11import inspect 

12 

13from . import logging 

14from . import task_registry 

15 

16LOG = logging.get_logger(__name__) 

17 

18 

19# _altmapping holds the mapping for CASA arguments that should be given as a 

20# task input argument of a different name. 

21_altmapping = { 

22 'hif_gaincal': {'hm_gaintype': 'mode'}, 

23} 

24 

25 

26def convert_args(pipeline_cls, casa_args, convert_nulls=True): 

27 """ 

28 Convert CASA arguments to pipeline Inputs arguments. 

29 

30 This function converts a dictionary of CASA arguments to the corresponding 

31 dictionary of pipeline Inputs arguments by doing the following: 

32 

33 #. Rename CASA arguments to their pipeline equivalents. 

34 #. Remove any arguments not accepted by the Inputs constructor. 

35 #. Convert any CASA null values to pipeline null values. 

36 

37 :param pipeline_cls: the pipeline Task class 

38 :type pipeline_cls: class 

39 :param casa_args: the dictionary of CASA arguments and values 

40 :type casa_args: dict 

41 :rtype: dict 

42 """ 

43 casa_task = task_registry.get_casa_task(pipeline_cls) 

44 inputs_cls = pipeline_cls.Inputs 

45 

46 # If required, rename CASA pipeline arguments to their pipeline equivalent 

47 if casa_task in _altmapping: 

48 mapping = _altmapping[casa_task] 

49 remapped = {} 

50 for k, v in casa_args.items(): 

51 name = mapping.get(k, k) 

52 if k != name and name in casa_args: 

53 msg = ('Remapping %s for %s inputs overwrites an argument of ' 

54 'the same name' % (name, casa_task)) 

55 LOG.error(msg) 

56 raise AttributeError(msg) 

57 remapped[name] = v 

58 else: 

59 remapped = casa_args 

60 

61 LOG.debug('Original arguments: %s' % casa_args) 

62 LOG.debug('Remapped arguments: %s' % remapped) 

63 

64 # inspect the signature of the Inputs constructor, getting a list of the 

65 # keyword arguments accepted by the constructor 

66 try: 

67 constructor_args = inputs_cls.get_constructor_args() 

68 except AttributeError: 

69 constructor_args = inspect.getfullargspec(inputs_cls.__init__).args 

70 

71 # remove any arguments that are not accepted by the input constructor 

72 accepted = {} 

73 accepted.update((k, v) for k, v in remapped.items() 

74 if k in constructor_args) 

75 

76 # convert any CASA empty string or empty lists to None, thus allowing the 

77 # pipeline to use its default value. 

78 if convert_nulls: 

79 converted = {} 

80 converted.update((k, _convert_null(v)) for k, v in accepted.items()) 

81 else: 

82 converted = accepted 

83 

84 LOG.debug('Final arguments for %s: %s' % (pipeline_cls.__name__, converted)) 

85 return converted

88def _convert_null(val): 

89 # convert empty string to None 

90 if isinstance(val, str): 

91 return None if val in ['', 'context'] else val 

92 

93 # convert empty lists and ['', ''] etc to None 

94 if isinstance(val, list): 

95 no_nulls = [_convert_null(v) for v in val] 

96 nones = [t == None for t in no_nulls] 

97 if all(nones): 

98 return None 

99 else: 

100 return no_nulls 

101 

102 return val

105def task_to_casa(taskname, task_args): 

106 if taskname not in _altmapping: 

107 return task_args 

108 

109 # If required, rename CASA pipeline arguments to their pipeline equivalent 

110 casa_to_task = _altmapping[taskname] 

111 d = dict((v, k) for k, v in casa_to_task.items()) 

112 remapped = {} 

113 for k, v in task_args.items(): 

114 name = d.get(k, k) 

115 remapped[name] = v 

116 

117 return remapped
