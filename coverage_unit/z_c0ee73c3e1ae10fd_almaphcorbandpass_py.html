478 """ 

479 Use SNR results (incl. optimal bandpass solution intervals) to select 

480 the best solution interval and spw combination setting to use in the 

481 phase-up, and the expected phase-up SNR. 

482 

483 Args: 

484 snr_result: dictionary with SNR results for each SpW. 

485 

486 Returns:

488 - phase-up solution interval

490 - expected phase-up SNR

500 # determined. In this case, log a warning, and return early with the 

501 # default input phaseup solint and phaseup combine. 

502 if not snr_result.spwids: 

503 LOG.info(f"{inputs.ms.basename}: no SNR results available, therefore unable to determine optimal phaseup"

514 )

558 # maximum SNR, and store the outcome for this SpectralSpec 

559 # as the corresponding spw index and max SNR. If no valid 

560 # SNRs were found (i.e. all were None), then this 

561 # SpectralSpec and its SpWs are rejected from consideration. 

562 if snrs: 

563 max_snr_spspec.append((spwi, max(snrs))) 

564 

565 # If a maximum SNR was found for at least one SpectralSpec, then 

566 # proceed to identify the SpectralSpec with the lowest max SNR, 

567 # and use its corresponding spw index. 

568 if max_snr_spspec: 

569 spwindex = min(max_snr_spspec, key=lambda x: x[1])[0] 

570 # Otherwise, in the unlikely scenario that no valid SNRs were 

571 # found for any SpW of any SpectralSpec, proceed with an empty 

572 # spw index. 

573 else: 

574 spwindex = [] 

575 

576 # If there are no valid SpWs, or the SNR results are missing optimal 

577 # solint values for all selected SpWs, then no optimal solint can be 

578 # determined. In this case, log a warning, and return early with the 

579 # default input phaseup solint and phaseup combine. 

580 if not spwindex or not any(snr_result.phsolints[i] for i in spwindex): 

581 LOG.info( 

582 f"{inputs.ms.basename}: no SNR results available for any of the expected SpWs, therefore unable" 

583 f" to determine optimal phaseup solint. Reverting to phaseup solint default {inputs.phaseupsolint}." 

584 ) 

585 phase_combine = 'spw' if inputs.hm_phaseup_combine == 'always' else '' 

586 adjustments.append( 

587 SolintAdjustment( 

588 applies_to=TargetDataSelection(vis={inputs.ms.basename}), 

589 original="", 

590 adjusted=inputs.phaseupsolint, 

591 threshold="", 

592 origin=LowSNRPhaseupSolintOrigin.SPWS_MISSING_DATA.value, 

593 reason=f"SNR results not available for any of the expected spws", 

594 ) 

595 ) 

596 return inputs.phaseupsolint, phase_combine, None, adjustments 

597 

598 # Since SNR results are not empty, then use the first available SpW to 

599 # determine the scan integration time as a timedelta object. It is 

600 # assumed that all scans for all SpWs are taken with the exact same 

601 # integration interval. 

602 scans = inputs.ms.get_scans(scan_intent=inputs.intent) 

603 mean_intervals = {scan.mean_interval(snr_result.spwids[0]) 

604 for scan in scans if snr_result.spwids[0] in [spw.id for spw in scan.spws]} 

605 timedelta = mean_intervals.pop() 

606 timedelta_integration_time = timedelta.total_seconds() 

607 

608 # int time is needed to calculate QA score and does not change from here on in 

609 adjustment_cls = functools.partial(SolintAdjustment, integration_time=timedelta_integration_time) 

610 

611 # Determine best solution interval based on SNR results for each SpW 

612 # under consideration. 

613 spwids = [] 

614 phintsnrs = [] 

615 bestsolint = [] 

616 for i in spwindex: 

617 # No solution available for this SpW. 

618 if not snr_result.phsolints[i]: 

619 # This per-spw warning is distinct from the early return on 

620 # L585, which is a global warning for all spws 

621 LOG.warning('No phaseup solint estimate for spw %s in MS %s' % 

622 (snr_result.spwids[i], inputs.ms.basename)) 

623 continue 

624 

625 # Otherwise, keep this SpW under consideration and store its SNR 

626 # estimate. 

627 spwids.append(snr_result.spwids[i]) 

628 phintsnrs.append(snr_result.phintsnrs[i]) 

629 

630 # If the number of phase solutions in the SNR result was below the 

631 # minimum number of phaseup gain solution points: 

632 if snr_result.nphsolutions[i] < inputs.phaseupnsols: 

633 # The solint in the SNR result is not usable, and instead the 

634 # best solint for this SpW is set to achieve the minimum number 

635 # of phase solutions (dividing exposure time by phaseupnsols). 

636 # Note: exposure time in SNR result is a string including unit, 

637 # so using quanta here to get the value as a float. 

638 newsolint = quanta.quantity(snr_result.exptimes[i])['value'] / inputs.phaseupnsols 

639 bestsolint.append(newsolint) 

640 

641 # Depending on whether spw combination is an option, either log 

642 # a message or solint adjustment (=QA warning) that there were 

643 # too few phaseup solution points. 

644 msg = ( 

645 f"{inputs.ms.basename}: phaseup solution for spw {snr_result.spwids[i]} has only" 

646 f" {snr_result.nphsolutions[i]} points; reducing estimated phaseup solint from" 

647 f" {snr_result.phsolints[i]:0.3f}s to {newsolint:0.3f}s." 

648 ) 

649 if inputs.hm_phaseup_combine == "never": 

650 adjustments.append( 

651 adjustment_cls( 

652 applies_to=TargetDataSelection(vis={inputs.ms.basename}, spw={snr_result.spwids[i]}), 

653 original=snr_result.phsolints[i], 

654 adjusted=newsolint, 

655 threshold=inputs.phaseupnsols, 

656 origin=LowSNRPhaseupSolintOrigin.INSUFFICIENT_POINTS.value, 

657 reason=f"Insufficient phase-up solution points for spw {snr_result.spwids[i]}, adjusting solint to {newsolint:0.3f}", 

658 ) 

659 ) 

660 else: 

661 msg += f" However, the option to combine spw will trigger a recalculation of solint." 

662 LOG.info(msg) 

663 # Otherwise, adopt for this SpW the solint from the SNR result. 

664 else: 

665 # Using quanta here to cast to same type as above. 

666 bestsolint.append(quanta.quantity(snr_result.phsolints[i])['value']) 

667 

668 max_bestsolint = max(bestsolint) 

669 min_bestsolint = min(bestsolint) 

670 spw_of_max_bestsolint = spwids[bestsolint.index(max_bestsolint)] 

671 

672 # If the best solution interval for all evaluated SpWs are all 

673 # smaller/equal to the integration time, then the required SNR can be 

674 # reached by setting solint to "int". No SpW combination is necessary 

675 # but may have been explicitly forced by user input. Return early with 

676 # this outcome: 

677 if max_bestsolint <= timedelta_integration_time: 

678 LOG.info(f"{inputs.ms.basename}: the largest optimal solint, {max_bestsolint:.3f}s, is smaller than" 

679 f" the integration time, {timedelta_integration_time:.3f}s, setting solint to 'int'.") 

680 if inputs.hm_phaseup_combine == 'always': 

681 phase_combine = 'spw' 

682 snr_expected = numpy.linalg.norm(phintsnrs) 

683 else: 

684 phase_combine = '' 

685 snr_expected = min(phintsnrs) 

686 

687 adjustments.append( 

688 adjustment_cls( 

689 applies_to=TargetDataSelection(vis={inputs.ms.basename}, spw={spw_of_max_bestsolint}), 

690 original=max_bestsolint, 

691 adjusted="int", 

692 threshold=timedelta_integration_time, 

693 origin=LowSNRPhaseupSolintOrigin.LT_INTEGRATION_TIME.value, 

694 reason=f"Largest optimal solint in spw {spw_of_max_bestsolint} still less than integration time {timedelta_integration_time:.3f}, adjusting solint to 'int'", 

695 ) 

696 ) 

697 return 'int', phase_combine, snr_expected, adjustments 

698 

699 # Otherwise, the maximum best solution interval was above the 

700 # integration time, and it may be possible to use SpW combination to 

701 # improve the SNR. 

702 LOG.info(f"{inputs.ms.basename}: the largest optimal solint, {max_bestsolint:.3f}s, is larger than" 

703 f" the integration time, {timedelta_integration_time:.3f}s.") 

704 

705 # these variables are used to annotate the SolintAdjustment with the 

706 # appropriate reasons and threshold information. 

707 original_solint: float 

708 decision_threshold: float 

709 metric_identifier: str 

710 reason: str 

711 # this will be mutated as required by the metric 

712 applies_to = TargetDataSelection(vis={inputs.ms.basename}) 

713 

714 # If the task input explicitly disabled the option of combining SpWs, 

715 # then stick to the solint selection heuristic from PL2024 and earlier, 

716 # to pick the largest optimal solint that was determined for the SpWs 

717 # but capped to maximum set by inputs.phaseupmaxsolint. 

718 if inputs.hm_phaseup_combine == 'never': 

719 # Combine is explicitly disabled for phase-up. 

720 LOG.info(f"{inputs.ms.basename}: SpW combination for phase-up is explicitly set to 'never' in task inputs" 

721 f" and will therefore not be considered as an option for SNR improvement.") 

722 phaseup_combine = '' 

723 

724 # If the largest optimal solint is greater than the maximum allowed 

725 # value: 

726 if max_bestsolint > inputs.phaseupmaxsolint: 

727 # Set the solution interval to the maximum allowed value, but 

728 # slightly adjust the value to round it to the nearest integer 

729 # multiple of the integration time. Use quanta to turn into a 

730 # string with unit, and restrict to 3 significant digits. 

731 finalfactor = round_up(inputs.phaseupmaxsolint / timedelta_integration_time) 

732 finalsolint = quanta.tos(quanta.quantity(timedelta_integration_time * finalfactor, 's'), 3) 

733 snr_expected = numpy.sqrt(finalfactor) * min(phintsnrs) 

734 LOG.info( 

735 f"{inputs.ms.basename}: the largest optimal solint, {max(bestsolint):.3f}s, is greater than" 

736 f" {inputs.phaseupmaxsolint}s, adjusting to {finalsolint}." 

737 ) 

738 

739 applies_to.spw |= {spw_of_max_bestsolint} 

740 metric_identifier = LowSNRPhaseupSolintOrigin.UNCOMBINED_GT_PHASEUPMAXSOLINT.value 

741 original_solint = max_bestsolint 

742 decision_threshold = inputs.phaseupmaxsolint 

743 reason = f"Largest optimal solint greater than {decision_threshold}s, adjusting to {finalsolint}" 

744 

745 # Identify for which SpWs the derived optimal solint was larger 

746 # than the maximum allowed value, and then warn that those SpWs 

747 # will have a low phaseup SNR. 

748 low_snr_spws = [str(spwid) for bsi, spwid in zip(bestsolint, spwids) if bsi > inputs.phaseupmaxsolint] 

749 if low_snr_spws: 

750 LOG.info( 

751 f"{inputs.ms.basename}: spw(s) {utils.commafy(low_snr_spws, False)} will have a" 

752 f" low phaseup gaincal SNR = {snr_expected}, which is < input phase-up SNR threshold" 

753 f" ({inputs.phaseupsnr})." 

754 ) 

755 

756 # Otherwise, the largest derived optimal solint must be above the 

757 # integration time but below the maximum allowed value: 

758 else: 

759 # Set the solution interval to the largest derived optimal 

760 # solint but slightly adjust the value to round it to the 

761 # nearest integer multiple of the integration time. Use quanta 

762 # to turn into a string with unit, and restrict to 3 significant 

763 # digits. 

764 finalfactor = round_up(max_bestsolint / timedelta_integration_time) 

765 finalsolint = quanta.tos(quanta.quantity(timedelta_integration_time * finalfactor, 's'), 3) 

766 snr_expected = numpy.sqrt(finalfactor) * min(phintsnrs) 

767 

768 metric_identifier = LowSNRPhaseupSolintOrigin.UNCOMBINED_LE_PHASEUPMAXSOLINT.value 

769 original_solint = max_bestsolint 

770 decision_threshold = inputs.phaseupmaxsolint 

771 reason = f"Largest optimal solint less than {decision_threshold}s, adjusting to {finalsolint}" 

772 

773 LOG.info(f"{inputs.ms.basename}: setting solint to {finalsolint}.") 

774 

775 # Otherwise, the optimal derived solint for at least one SpW was larger 

776 # than the integration time, and SpW combination is available as an 

777 # option, so continue with the new PL2025 heuristic (PIPE-2442) to find 

778 # an optimal solution interval for SpW combination, based on aggregate 

779 # bandwidth. 

780 else: 

781 LOG.info(f"{inputs.ms.basename}: combining SpWs to improve phaseup SNR.") 

782 phaseup_combine = 'spw' 

783 

784 # If the smallest optimal solint, to achieve the required SNR, was 

785 # smaller than the integration time for at least one SpW, then the 

786 # SNR will only improve with more bandwidth after combining SpWs, 

787 # so "int" will be the minimum optimal solint for combine='spw'. 

788 if min_bestsolint <= timedelta_integration_time: 

789 LOG.info(f"{inputs.ms.basename}: optimal solint based on aggregate bandwidth is 'int'.") 

790 finalsolint = 'int' 

791 snr_expected = numpy.linalg.norm(phintsnrs) 

792 

793 metric_identifier = LowSNRPhaseupSolintOrigin.COMBINED_MIN_BEST_LE_INTEGRATION_TIME.value 

794 original_solint = min_bestsolint 

795 decision_threshold = timedelta_integration_time 

796 reason = f"Optimal solint based on aggregate bandwidth is {finalsolint}" 

797 

798 # Otherwise, a new optimal solint based on the aggregate bandwidth 

799 # through SpW combination needs to be computed here. 

800 else: 

801 LOG.info(f"{inputs.ms.basename}: computing optimal solint based on aggregate bandwidth.") 

802 

803 # Identify SpW with smallest optimal solint. 

804 spwid_min_solint = spwids[bestsolint.index(min_bestsolint)] 

805 

806 # Determine the aggregate bandwidth for all SpWs originally 

807 # under consideration, including any for which there may not 

808 # have been a valid entry in SNR result, hence use origina 

809 # spwindex instead of spwids. 

810 sum_bw = sum(inputs.ms.get_spectral_window(snr_result.spwids[i]).bandwidth.value for i in spwindex) 

811 

812 # Set bandwidth scaling factor to bandwidth of SpW with the 

813 # smallest optimal solint divided by the aggregate bandwidth of 

814 # the SpWs under consideration. 

815 bwfactor = float(inputs.ms.get_spectral_window(spwid_min_solint).bandwidth.value / sum_bw) 

816 

817 # Set the solint for the SpW combination gaincal by scaling the 

818 # smallest optimal solint with the bandwidth scaling factor, 

819 # and ensuring this is rounded up (ceil) to next nearest integer 

820 # multiple of the integration time. 

821 combfactor = math.ceil(min_bestsolint * bwfactor / timedelta_integration_time) 

822 combinesolint = timedelta_integration_time * combfactor 

823 

824 # Check how the newly determined optimal solint for combining 

825 # SpWs compares to the integration time and the maximum allowed 

826 # solint. 

827 # 

828 # If the optimal combined solint is smaller/equal to the 

829 # integration time, then set the solint to 'int'. 

830 if combinesolint <= timedelta_integration_time: 

831 finalsolint = 'int' 

832 snr_expected = numpy.linalg.norm(phintsnrs) 

833 

834 metric_identifier = LowSNRPhaseupSolintOrigin.COMBINED_LE_INTEGRATION_TIME.value 

835 original_solint = combinesolint 

836 decision_threshold = timedelta_integration_time 

837 reason = f"Optimal combined solint <= integration time ({timedelta_integration_time})" 

838 

839 # If the optimal combined solint is still greater than the 

840 # maximum allowed value (i.e. really low SNR scenario): 

841 elif combinesolint > inputs.phaseupmaxsolint: 

842 # Set the solution interval to the maximum allowed value, 

843 # but slightly adjust the value to round it to the nearest 

844 # integer multiple of the integration time. Use quanta to 

845 # turn into a string with unit, and restrict to 3 

846 # significant digits. 

847 finalfactor = round_up(inputs.phaseupmaxsolint / timedelta_integration_time) 

848 finalsolint = quanta.tos(quanta.quantity(timedelta_integration_time * finalfactor, 's'), 3) 

849 snr_expected = numpy.sqrt(finalfactor) * numpy.linalg.norm(phintsnrs) 

850 LOG.info( 

851 f"{inputs.ms.basename}: the combined spw solution interval, {combinesolint:.3f}s, is" 

852 f" greater than {inputs.phaseupmaxsolint}s, adjusting to {finalsolint}, solution SNR" 

853 f" = {snr_expected:.2f}, < {inputs.phaseupsnr}." 

854 ) 

855 

856 metric_identifier = LowSNRPhaseupSolintOrigin.COMBINED_GT_PHASEUPMAXSOLINT.value 

857 original_solint = combinesolint 

858 decision_threshold = inputs.phaseupmaxsolint 

859 reason = f"Combined solution interval greater than upper limit ({inputs.phaseupmaxsolint})" 

860 

861 # Otherwise, proceed with the optimal combined solint. 

862 else: 

863 # Use quanta to turn into a string with unit, and restrict 

864 # to 3 significant digits. 

865 finalsolint = quanta.tos(quanta.quantity(combinesolint), 3) 

866 snr_expected = numpy.sqrt(combfactor) * numpy.linalg.norm(phintsnrs) 

867 

868 metric_identifier = LowSNRPhaseupSolintOrigin.COMBINED_LT_PHASEUPMAXSOLINT.value 

869 original_solint = combinesolint 

870 decision_threshold = inputs.phaseupmaxsolint 

871 reason = f"Combined solution interval less than upper limit ({inputs.phaseupmaxsolint})" 

872 

873 adjustments.append( 

874 adjustment_cls( 

875 applies_to=applies_to, 

876 original=original_solint, 

877 adjusted=finalsolint, 

878 threshold=decision_threshold, 

879 origin=metric_identifier, 

880 reason=reason, 

881 ) 

882 ) 

883 

884 return finalsolint, phaseup_combine, snr_expected, adjustments

904 )

940 # Compute the smoothed bandpass 

941 def _do_smoothed_bandpass(self) -> BandpassResults | None: 

942 inputs = self.inputs 

943 

944 # Store original values of some parameters. 

945 orig_spw = inputs.spw 

946 orig_solint = inputs.solint 

947 orig_append = inputs.append 

948 

949 try: 

950 # initialize the caltable and list of spws 

951 inputs.caltable = inputs.caltable 

952 spwlist = inputs.ms.get_spectral_windows(orig_spw) 

953 

954 # will hold the CalAppOrigins that record how each CalApp was 

955 # generate. Ideally this would be a list on the CalApp itself, 

956 # but we don't have time to do that right now. 

957 calapp_origins = [] 

958 

959 # Loop through the spw appending the results of each spw 

960 # to the results of the previous one. 

961 for spw in spwlist: 

962 

963 # TDM or FDM 

964 dd = inputs.ms.get_data_description(spw=spw) 

965 if dd is None: 

966 LOG.debug('Missing data description for spw %s' % spw.id) 

967 continue 

968 ncorr = len(dd.corr_axis) 

969 if ncorr not in [1, 2, 4]: 

970 LOG.debug('Wrong number of correlations %s for spw %s' % 

971 (ncorr, spw.id)) 

972 continue 

973 

974 # Smooth if FDM and if it makes sense 

975 if ncorr * spw.num_channels > 256: 

976 if (spw.num_channels // inputs.maxchannels) < 1: 

977 LOG.info(f"{inputs.ms.basename}: Too few channels ({spw.num_channels}) in SpW {spw.id} to use" 

978 f" smoothing (maxchannels={inputs.maxchannels}), reverting to default bandpass solint" 

979 f" {orig_solint}.") 

980 inputs.solint = orig_solint 

981 else: 

982 # PIPE-2036: work-around for potential issue caused by: 

983 # * PL defined solint as a frequency interval, and 

984 # this may correspond to an exact nr. of channels. 

985 # * the frequency interval is passed with limited 

986 # precision (typically in MHz with 6 decimals, i.e. 

987 # a precision of Hz) 

988 # * CASA's bandpass converts the frequency interval 

989 # back to nr. of channels and then take the floor 

990 # 

991 # This can result in e.g. a required nr. of channels of 

992 # 5 corresponding to 4.8828125 MHz but getting passed as 

993 # 4.882812 MHz, then converted back to 4.999999 

994 # channels, and floored to 4. 

995 # 

996 # As a work-around, check whether the frequency interval 

997 # would trigger this, and if so, then round *up* the 

998 # frequency interval to nearest Hz. 

999 bandwidth = spw.bandwidth.to_units(otherUnits=measures.FrequencyUnits.HERTZ) 

1000 newsolint = bandwidth / inputs.maxchannels 

1001 chanwidth = bandwidth / spw.num_channels 

1002 if round(newsolint) / chanwidth < math.floor(newsolint / chanwidth): 

1003 inputs.solint = f"{orig_solint},{round_up(newsolint) * 1.e-6:f}MHz" 

1004 else: 

1005 inputs.solint = f"{orig_solint},{float(newsolint) * 1.e-6:f}MHz" 

1006 LOG.info(f"{inputs.ms.basename}: using smoothed bandpass solint {inputs.solint} for SpW" 

1007 f" {spw.id}.") 

1008 else: 

1009 inputs.solint = orig_solint 

1010 LOG.warning(f"Reverting to default bandpass solint {inputs.solint} for spw {spw.id} in MS" 

1011 f" {inputs.ms.basename}") 

1012 

1013 # PIPE-2116: if requested, set fillgaps to a quarter of the 

1014 # number of channels of current SpW. 

1015 if inputs.hm_auto_fillgaps: 

1016 inputs.fillgaps = int(spw.num_channels / 4) 

1017 else: 

1018 inputs.fillgaps = 0 

1019 

1020 # Compute and append bandpass solution 

1021 inputs.spw = spw.id 

1022 bandpass_task = bandpassmode.BandpassMode(inputs) 

1023 result = self._executor.execute(bandpass_task) 

1024 if os.path.exists(self.inputs.caltable): 

1025 self.inputs.append = True 

1026 self.inputs.caltable = result.final[-1].gaintable 

1027 calapp_origins.extend(result.final[-1].origin) 

1028 

1029 # Reset the calto spw list 

1030 result.pool[0].calto.spw = orig_spw 

1031 if result.final: 

1032 result.final[0].calto.spw = orig_spw 

1033 result.final[0].origin = calapp_origins 

1034 

1035 return result 

1036 

1037 finally: 

1038 inputs.spw = orig_spw 

1039 inputs.solint = orig_solint 

1040 inputs.append = orig_append 

1041 

1042 # Compute the bandpass using SNR estimates 

1043 def _do_snr_bandpass(self, snr_result) -> BandpassResults | None: 

1044 inputs = self.inputs 

1045 quanta = casa_tools.quanta 

1046 

1047 # Store original values of some parameters. 

1048 orig_spw = inputs.spw 

1049 orig_solint = inputs.solint 

1050 orig_append = inputs.append 

1051 

1052 adjustments: list[SolintAdjustment] = [] 

1053 

1054 try: 

1055 # initialize the caltable and list of spws 

1056 inputs.caltable = inputs.caltable 

1057 spwlist = inputs.ms.get_spectral_windows(orig_spw) 

1058 

1059 # will hold the CalAppOrigins that record how each CalApp was 

1060 # generate. Ideally this would be a list on the CalApp itself, 

1061 # but we don't have time to do that right now. 

1062 calapp_origins = [] 

1063 

1064 for spw in spwlist: 

1065 

1066 # TDM or FDM 

1067 dd = inputs.ms.get_data_description(spw=spw) 

1068 if dd is None: 

1069 LOG.debug('Missing data description for spw %s' % spw.id) 

1070 continue 

1071 ncorr = len(dd.corr_axis) 

1072 if ncorr not in [1, 2, 4]: 

1073 LOG.debug('Wrong number of correlations %s for spw %s' % 

1074 (ncorr, spw.id)) 

1075 continue 

1076 

1077 # Find the best solint for that window 

1078 try: 

1079 solindex = snr_result.spwids.index(spw.id) 

1080 except: 

1081 solindex = -1 

1082 

1083 # Use the optimal solution if it is good enough otherwise 

1084 # revert to the default smoothing algorithm 

1085 if solindex >= 0: 

1086 

1087 if snr_result.nbpsolutions[solindex] < inputs.bpnsols: 

1088 # Recompute the solution interval to force the minimum 

1089 # number of solution channels 

1090 factor = 1.0 / inputs.bpnsols 

1091 newsolint = quanta.tos(quanta.mul(snr_result.bandwidths[solindex], factor)) 

1092 inputs.solint = orig_solint + ',' + newsolint 

1093 

1094 original = str(to_frequency(snr_result.bpsolints[solindex])) 

1095 adjusted = f'{orig_solint},{str(to_frequency(newsolint))}' 

1096 

1097 vis = os.path.basename(inputs.vis) 

1098 adjustments.append( 

1099 SolintAdjustment( 

1100 applies_to=TargetDataSelection(vis={vis}, spw={spw.id}), 

1101 original=original, 

1102 threshold=inputs.bpnsols, 

1103 adjusted=adjusted, 

1104 origin=LowSNRPhaseupSolintOrigin.TOO_FEW_CHANNELS.value, 

1105 reason=f'Too few channels: changing recommended bandpass solint from {original} to {adjusted} for spw {spw.id}' 

1106 ) 

1107 ) 

1108 else: 

1109 inputs.solint = orig_solint + ',' + \ 

1110 snr_result.bpsolints[solindex] 

1111 LOG.info('Setting bandpass solint to %s for spw %s' % (inputs.solint, spw.id)) 

1112 

1113 elif ncorr * spw.num_channels > 256: 

1114 LOG.warning(f"{inputs.ms.basename}: no SNR based bandpass solint was found for spw {spw.id}," 

1115 f" reverting to smoothing algorithm.") 

1116 if (spw.num_channels // inputs.maxchannels) < 1: 

1117 LOG.info(f"{inputs.ms.basename}: Too few channels ({spw.num_channels}) in SpW {spw.id} to use" 

1118 f" smoothing (maxchannels={inputs.maxchannels}), reverting to default bandpass solint" 

1119 f" {orig_solint}.") 

1120 inputs.solint = orig_solint 

1121 else: 

1122 # PIPE-2036: work-around for potential issue caused by: 

1123 # * PL defined solint as a frequency interval, and 

1124 # this may correspond to an exact nr. of channels. 

1125 # * the frequency interval is passed with limited 

1126 # precision (typically in MHz with 6 decimals, i.e. 

1127 # a precision of Hz) 

1128 # * CASA's bandpass converts the frequency interval 

1129 # back to nr. of channels and then take the floor 

1130 # 

1131 # This can result in e.g. a required nr. of channels of 

1132 # 5 corresponding to 4.8828125 MHz but getting passed as 

1133 # 4.882812 MHz, then converted back to 4.999999 

1134 # channels, and floored to 4. 

1135 # 

1136 # As a work-around, check whether the frequency interval 

1137 # would trigger this, and if so, then round *up* the 

1138 # frequency interval to nearest Hz. 

1139 bandwidth = spw.bandwidth.to_units(otherUnits=measures.FrequencyUnits.HERTZ) 

1140 newsolint = bandwidth / inputs.maxchannels 

1141 chanwidth = bandwidth / spw.num_channels 

1142 if round(newsolint) / chanwidth < math.floor(newsolint / chanwidth): 

1143 inputs.solint = f"{orig_solint},{round_up(newsolint) * 1.e-6:f}MHz" 

1144 else: 

1145 inputs.solint = f"{orig_solint},{float(newsolint) * 1.e-6:f}MHz" 

1146 LOG.info(f"{inputs.ms.basename}: using smoothed bandpass solint {inputs.solint} for SpW" 

1147 f" {spw.id}.") 

1148 else: 

1149 inputs.solint = orig_solint 

1150 LOG.warning("Reverting to default bandpass solint %s for spw %s in MS %s" % 

1151 (inputs.solint, spw.id, inputs.ms.basename)) 

1152 

1153 # PIPE-2116: if requested, set fillgaps to a quarter of the 

1154 # number of channels of current SpW. 

1155 if inputs.hm_auto_fillgaps: 

1156 inputs.fillgaps = int(spw.num_channels / 4) 

1157 else: 

1158 inputs.fillgaps = 0 

1159 

1160 # Compute and append bandpass solution 

1161 inputs.spw = spw.id 

1162 bandpass_task = bandpassmode.BandpassMode(inputs) 

1163 result = self._executor.execute(bandpass_task) 

1164 if os.path.exists(self.inputs.caltable): 

1165 self.inputs.append = True 

1166 self.inputs.caltable = result.final[-1].gaintable 

1167 calapp_origins.extend(result.final[-1].origin) 

1168 

1169 # Reset the calto spw list 

1170 result.pool[0].calto.spw = orig_spw 

1171 if result.final: 

1172 result.final[0].calto.spw = orig_spw 

1173 result.final[0].origin = calapp_origins 

1174 

1175 result.solint_adjustments.extend(adjustments) 

1176 

1177 return result 

1178 

1179 finally: 

1180 inputs.spw = orig_spw 

1181 inputs.solint = orig_solint 

1182 inputs.append = orig_append

1385def centre_datetime_from_epochs(epoch1, epoch2): 

1386 """ 

1387 Get the time midpoint between two epochs. 

1388 

1389 :param epoch1: epoch 1 

1390 :param epoch2: epoch 2 

1391 :return: time between epoch1 and epoch2 

1392 :rtype: datetime.datetime 

1393 """ 

1394 time1 = utils.get_epoch_as_datetime(epoch1) 

1395 time2 = utils.get_epoch_as_datetime(epoch2) 

1396 # use min & max so it doesn't depend on correct time ordering of 

1397 # arguments

1404 """
