132def tsysspwmap(vis,tsystable,trim=True,relax=False, tsysChanTol=0): 

133 """ 

134 Generate default spwmap for ALMA Tsys, including TDM->FDM associations 

135 Input: 

136 vis the target MeasurementSet  

137 tsystable the input Tsys caltable (w/ TDM Tsys measurements) 

138 trim if True (the default), return minimum-length spwmap; 

139 otherwise the spwmap will be exhaustive and include 

140 the high-numbered (and usually irrelevant) wvr 

141 spws 

142 relax (not yet implemented) 

143 Output: 

144 spw list to use in applycal spwmap parameter for the Tsys caltable 

145 

146 This function takes the Tsys Caltable you wish to apply to a 

147 MeasurementSet and generates a "applycal-ready" spwmap that 

148 provides the appropriate information regarding the transfer 

149 Tsys calibration from TDM spectral windows to FDM spectral 

150 windows. To execute the function: 

151 

152 tsysmap=tsysspwmap(vis='my.ms',tsystable='mytsys.cal') 

153 

154 tsysmap can then be supplied to the applycal spwmap parameter 

155 to ensure proper Tsys calibration application. 

156 

157 """ 

158 

159 localTb = tbtool() 

160 spwMaps = [] 

161 # Get the spectral windows with entries in the solution table 

162 localTb.open(tsystable) 

163 measuredTsysSpw = numpy.unique(localTb.getcol("SPECTRAL_WINDOW_ID")) 

164 localTb.close() 

165 # Get the frequency ranges for the allowed  

166 localTb.open("%s/SPECTRAL_WINDOW" % tsystable) 

167 for i in measuredTsysSpw: 

168 spwMap = SpwMap(i) 

169 chanFreqs = localTb.getcell("CHAN_FREQ",i) 

170 chanWidth = abs(chanFreqs[1]-chanFreqs[0]) 

171 spwMap.chanWidth = chanWidth 

172 spwMap.validFreqRange = [chanFreqs.min()-0.5*chanWidth,\ 

173 chanFreqs.max()+0.5*chanWidth] 

174 spwMaps.append(spwMap) 

175 localTb.close() 

176 # Now loop through the main table's spectral window table 

177 # to map the spectral windows as desired. 

178 localTb.open("%s/SPECTRAL_WINDOW" % vis) 

179 it = localTb.nrows() 

180 localTb.close() 

181 for j in spwMaps : 

182 localTb.open("%s/SPECTRAL_WINDOW" % vis) 

183 j.bbNo = localTb.getcell("BBC_NO",j.calSpwId) 

184 localTb.close() 

185 for i in range(it) : 

186 localTb.open("%s/SPECTRAL_WINDOW" % vis) 

187 chanFreqs = localTb.getcell("CHAN_FREQ",i) 

188 if len(chanFreqs) > 1 : 

189 chanWidth = localTb.getcell("CHAN_WIDTH",i)[0] 

190 freqMin = chanFreqs.min()-0.5*chanWidth 

191 freqMax = chanFreqs.max()+0.5*chanWidth 

192 else : 

193 chanWidth = localTb.getcell("CHAN_WIDTH",i) 

194 freqMin = chanFreqs-0.5*chanWidth 

195 freqMax = chanFreqs+0.5*chanWidth 

196 msSpw = SpwInfo(vis,i) 

197 if j.bbNo == msSpw.values['BBC_NO']: 

198 if freqMin >= j.validFreqRange[0]-tsysChanTol*j.chanWidth and \ 

199 freqMax <= j.validFreqRange[1]+tsysChanTol*j.chanWidth : 

200 j.mapsToSpw.append(i) 

201 localTb.close() 

202 applyCalSpwMap = [] 

203 spwWithoutMatch = [] 

204 localTb.open("%s/SPECTRAL_WINDOW" % vis) 

205 for i in range(it) : 

206 useSpw = None 

207 for j in spwMaps : 

208 if i in j.mapsToSpw : 

209 if useSpw is not None : 

210 if localTb.getcell("BBC_NO") == j.bbNo : 

211 useSpw = j.calSpwId 

212 else : 

213 useSpw = j.calSpwId 

214 if useSpw == None : 

215 useSpw = i 

216 spwWithoutMatch.append(i) 

217 applyCalSpwMap.append(int(useSpw)) 

218 localTb.close() 

219 if len(spwWithoutMatch) != 0: 

220 casalog.post('Found no match for following spw ids: '+str(spwWithoutMatch)) 

221 if trim : 

222 return trimSpwmap(applyCalSpwMap) 

223 else : 

224 return applyCalSpwMap

227def fixsyscaltimes(vis,newinterval=2.0): 

228 """ 

229 Fix TIME,INTERVAL columns in MS SYSCAL subtable 

230 Input: 

231 vis the MS containing the offending SYSCAL subtable 

232 newinterval the interval to use in revised entries 

233 

234 This function is intended to repair MS SYSCAL tables that suffer from 

235 multiple TIME values (over antennas) per Tsys measurement. The gencal 

236 task (mode='tsys' expects all antennas to share the same TIME value 

237 for each Tsys measurement (and this is usually true). The function 

238 finds those measurements that have multiple TIMEs and replaces them 

239 with a common TIME value which takes the value 

240 mean(oldTIME-INTERVAL/2)+newinterval/2. 

241 Usually (always?), oldTIME-INTERVAL/2 is constant over antennas 

242 and represents the physical timestamp of the Tsys measurment. 

243 If the function finds no pathological timestamps, it does not 

244 revise the table. 

245 """ 

246 

247 import pylab as mypl 

248 import math as mymath 

249 myqa=qatool() 

250 mytb=tbtool() 

251 mytb.open(vis+'/SYSCAL',nomodify=False) 

252 

253 spws=mypl.unique(mytb.getcol("SPECTRAL_WINDOW_ID")) 

254 

255 # Nominally, no fix applied 

256 fixapplied=False 

257 

258 for ispw in spws: 

259 st=mytb.query('SPECTRAL_WINDOW_ID=='+str(ispw),name='byspw') 

260 times=st.getcol('TIME') 

261 interval=st.getcol('INTERVAL') 

262 timestamps=times-interval/2 

263 t0=86400.0*mymath.floor(timestamps[0]/86400.0) 

264 

265 utimes=mypl.unique(times-t0) 

266 nT=len(utimes) 

267 utimestamps=mypl.unique(mypl.floor(timestamps)-t0) 

268 nTS=len(utimestamps) 

269 

270 msg='In spw='+str(ispw)+' found '+str(nTS)+' Tsys measurements with '+str(nT)+' TIMEs...' 

271 if nT==nTS: 

272 msg+='OK.' 

273 print(msg) 

274 

275 else: 

276 # If we reach here, then a fix is being applied, and we'll return True 

277 fixapplied=True 

278 

279 msg+=' which is too many, so fixing it:' 

280 print(msg) 

281 

282 for uts in utimestamps: 

283 mask = ((mypl.floor(timestamps))-t0==uts) 

284 uTIMEs=mypl.unique(times[mask]) 

285 nTIMEs=len(uTIMEs) 

286 newtime = mypl.mean(times[mask]-interval[mask]/2) + newinterval/2 

287 msg=' Found '+str(nTIMEs)+' TIMEs at timestamp='+str(myqa.time(str(newtime-newinterval/2)+'s',form='ymd')[0]) 

288 if nTIMEs>1: 

289 msg+=':' 

290 print(msg) 

291 print(' TIMEs='+str([myqa.time(str(t)+'s',form='ymd')[0] for t in uTIMEs])+' --> '+str(myqa.time(str(newtime)+'s',form='ymd')[0])+' w/ INTERVAL='+str(newinterval)) 

292 times[mask]=newtime 

293 interval[mask]=newinterval 

294 st.putcol('TIME',times) 

295 st.putcol('INTERVAL',interval) 

296 else: 

297 msg+='...ok.' 

298 print(msg) 

299 st.close() 

300 mytb.close() 

301 

302 if fixapplied: 

303 print('Some SYSCAL time corrections have been applied.') 

304 else: 

305 print('All SYSCAL times seem correct; no changes required.') 

306 

307 # Return if a fix has been applied 

308 return fixapplied

311def calantsub(incaltable,outcaltable='', 

312 spw='',scan='', 

313 ant='',subant=''): 

314 

315 """ 

316 Substitute cal solutions by antenna 

317 Input: 

318 incaltable Input caltable 

319 outcaltable Output caltable (if '', overwrite result on incaltable) 

320 spw Spectral Window selection (no channel selection permitted) 

321 scan Scan selection 

322 ant Antenna (indices) which need replaced solutions 

323 subant Antenna (indices) with which to replace those in ant 

324 

325 This function provides a means to replace solutions by antenna, 

326 e.g., to substitute one antenna's Tsys spectra with another. 

327 

328 The processing can be limited to specific spectral windows and/or 

329 scans. The spw and scan parameters should be specified in the 

330 standard MS selection manner (comma-separated integers in a string), 

331 except no channel selection is supported. 

332 

333 The ant parameter specifies one or more antenna indices  

334 (comma-separated in a string) for which solutions are to be 

335 replaced. The subant parameter lists the antenna indices 

336 from which the substitute solutions are to be obtained. E.g., 

337 ant='3,5,7',subant='6,8,10' will cause the solutions from 

338 antenna id 6 to be copied to antenna id 5, id 8 to id 5 and id 10 

339 to id 7. The number of antennas specified in ant and subant 

340 must match. 

341  

342 """ 

343 

344 import pylab as mypl 

345 

346 # trap insufficient ant subant specifications 

347 if len(ant)==0 or len(subant)==0: 

348 raise Exception("Must specify at least one ant and subant.") 

349 

350 antlist=ant.split(',') 

351 sublist=subant.split(',') 

352 

353 # trap dumb cases 

354 nant=len(antlist) 

355 nsub=len(sublist) 

356 if nant!=nsub: 

357 raise Exception("Must specify equal number of ant and subant.") 

358 

359 # local tb tool 

360 mytb=tbtool() 

361 

362 # parse selection 

363 selstr='' 

364 if len(spw)>0: 

365 selstr+='SPECTRAL_WINDOW_ID IN ['+spw+']' 

366 if len(scan)>0: 

367 selstr+=' && ' 

368 if len(scan)>0: 

369 selstr+='SCAN_NUMBER IN ['+scan+']' 

370 print("selstr = '"+selstr+"'") 

371 

372 # verify selection (if any) selects non-zero rows 

373 if len(selstr)>0: 

374 mytb.open(incaltable) 

375 st=mytb.query(query=selstr) 

376 nselrows=st.nrows() 

377 st.close() 

378 mytb.close() 

379 if nselrows==0: 

380 raise Exception('Error: scan and/or spw selection selects no rows!') 

381 

382 # manage the output table 

383 if outcaltable=='': 

384 outcaltable=incaltable 

385 print("No outcaltable specified; will overwrite incaltable.") 

386 if outcaltable!=incaltable: 

387 os.system('cp -r '+incaltable+' '+outcaltable) 

388 

389 # open the output table for adjustment 

390 mytb.open(outcaltable,nomodify=False) 

391 

392 stsel=mytb 

393 if len(selstr)>0: 

394 stsel=mytb.query(query=selstr,name='selected') 

395 

396 # cols to substitute: 

397 collist=['TIME','INTERVAL','PARAMERR','SNR','FLAG'] 

398 cols=mytb.colnames() 

399 if cols.count('CPARAM')>0: 

400 collist.append('CPARAM') 

401 else: 

402 collist.append('FPARAM') 

403 

404 # scan list 

405 scans=mypl.unique(stsel.getcol('SCAN_NUMBER')) 

406 

407 print('Found scans = ',scans) 

408 

409 # do one scan at a time 

410 for scan in scans: 

411 st1=stsel.query(query='SCAN_NUMBER=='+str(scan),name='byscan') 

412 spws=mypl.unique(st1.getcol('SPECTRAL_WINDOW_ID')) 

413 

414 print('Scan '+str(scan)+' has spws='+str(spws)) 

415 

416 # do one spw at a time 

417 for ispw in spws: 

418 st2=st1.query(query='SPECTRAL_WINDOW_ID=='+str(ispw),name='byspw'); 

419 

420 for ia in range(nant): 

421 stsub=st2.query(query='ANTENNA1=='+sublist[ia], 

422 name='subant') 

423 stant=st2.query(query='ANTENNA1=='+antlist[ia], 

424 name='ant') 

425 

426 

427 # go to next ant if nothing to do 

428 if stant.nrows()<1: 

429 continue 

430 

431 print(' scan='+str(scan)+' spw='+str(ispw)+' ants: '+str(sublist[ia])+'->'+str(antlist[ia])) 

432 

433 # trap (unlikely?) pathological case 

434 if stsub.nrows()!=stant.nrows(): 

435 raise Exception("In spw "+str(ispw)+" antenna ids "+str(antlist[ia])+" and "+str(sublist[ia])+" have a different number of solutions.") 

436 

437 # substitute values  

438 for col in collist: 

439 stant.putcol(col,stsub.getcol(col)) 

440 

441 stsub.close() 

442 stant.close() 

443 st2.close() 

444 st1.close() 

445 stsel.close() 

446 mytb.close()

669def genImageName(vis='', spw='', field='', imtype='mfs', targettype='sci', stokes='I', mous='', modtext='manual'): 

670 # DPetry, Sept 2017 

671 

672 """ 

673 Generate image name to be used in clean command 

674 following the ALMA archive conventions (see SCIREQ-110, draft recommendations v4.8). 

675 vis = MS to be imaged (to extract object name, SPW frequencies 

676 spw = ID of SPW to be imaged 

677 If several SPWs are imaged at once, enter list of SPWs. 

678 field = ID of field to be imaged (to extract object name). 

679 in case of mosaic, give any ID of field belonging to the mosaic 

680 imtype = image type, 'mfs' (default) - aggregate bandwidth 

681 'cont' - mfs with lines excluded 

682 'cube' - spectrally resolved cube 

683 targettype = 'sci' (default) for science target 

684 'bp' for bandpass calibrator 

685 'ph' for phase calibrator 

686 'chk' for checksource 

687 'amp' for flux calibrator 

688 stokes = 'I' (default) for Stokes I image 

689 any combination of 'I', 'Q', 'U', 'V', 'P', 'A' 

690 mous = '' (default) normally this is filled by the archive. 

691 But if you know the MOUS UID, enter it in the format with "_" 

692 e.g. uid___A001_X888_X66 

693 modtext = 'manual' (default) - this string can be used to add text, 

694 i.e. a modification, to the standard file name. 

695 It is appended at the end of the name. 

696 """ 

697 

698 themous = '' 

699 theobject = '' 

700 thetargettype = '' 

701 targettypes = ['sci', 'bp', 'ph', 'chk', 'amp'] 

702 thespwid = '' 

703 theimtype = '' 

704 imtypes = ['mfs', 'cont', 'cube'] 

705 thestokes = '' 

706 stokess = ['I', 'Q', 'U', 'V', 'IQU', 'IQUV', 'P', 'A'] 

707 themodtext = '' 

708 

709 rval = '' 

710 

711 mymsmd = msmdtool() 

712 

713 # MOUS 

714 if (mous != ''): 

715 themous = mous+'.' 

716 

717 # object 

718 try: 

719 mymsmd.open(vis) 

720 except: 

721 mymsmd.done() 

722 raise Exception("ERROR: problem opening vis") 

723 

724 myfieldnames = mymsmd.fieldnames() 

725 if type(field)==str and (field in myfieldnames): 

726 theobject = field 

727 elif field in range(mymsmd.nfields()): 

728 theobject = myfieldnames[field] 

729 else: 

730 print("ERROR: invalid field: "+field) 

731 print("Valid entries are ", myfieldnames) 

732 print(" or ", range(mymsmd.nfields())) 

733 raise Exception("ERROR: invalid field: "+field) 

734 myspws = mymsmd.spwsforfield(theobject) 

735 mymsmd.close() 

736 

737 # target type 

738 if targettype in targettypes: 

739 thetargettype = targettype 

740 else: 

741 print("ERROR: invalid targettype ", targettype) 

742 print(" Valid entries are ", targettypes) 

743 raise Exception("ERROR: invalid targettype ", targettype) 

744 

745 # spw 

746 if type(spw) != type([]): 

747 spw = [spw] 

748 for myspw in spw: 

749 if myspw in myspws: 

750 if thespwid=='': 

751 thespwid = str(myspw) 

752 else: 

753 thespwid = thespwid+'_'+str(myspw) 

754 else: 

755 if not type(myspw) == int: 

756 print("ERROR: invalid spw: "+str(myspw)+". Valid entries are ", myspws) 

757 raise Exception("ERROR: invalid spw: "+str(myspw)+' Data type must be int.') 

758 else: 

759 print("ERROR: invalid spw: "+str(myspw)+". Valid entries are ", myspws) 

760 raise Exception("ERROR: invalid spw: "+str(myspw)) 

761 

762 # imtype 

763 if imtype in imtypes: 

764 theimtype = imtype 

765 else: 

766 print("ERROR: invalid imtype ", imtype) 

767 print(" Valid entries are ", imtypes) 

768 raise Exception( "ERROR: invalid imtype "+str(imtype)) 

769 

770 # stokes 

771 if stokes in stokess: 

772 thestokes = stokes 

773 else: 

774 print("ERROR: invalid stokes ", stokes) 

775 print(" Valid entries are ", stokess) 

776 raise Exception("ERROR: invalid stokes "+str(stokes)) 

777 

778 # modifying text 

779 

780 if modtext != '': 

781 themodtext = '.'+modtext 

782 if ' ' in themodtext: 

783 raise Exception("ERROR: modtext must not contain spaces") 

784 

785 rval = themous+theobject+'_'+thetargettype+'.spw'+thespwid+'.'+theimtype+'.'+thestokes+themodtext 

786 

787 if ' ' in rval: 

788 raise Exception("ERROR: generated name contains spaces: \""+rval+"\"") 

789 if '/' in rval: 

790 raise Exception("ERROR: generated name contains slash: \""+rval+"\"") 

791 if '*' in rval: 

792 raise Exception("ERROR: generated name contains star: \""+rval+"\"") 

793 if '?' in rval: 

794 raise Exception("ERROR: generated name contains question mark: \""+rval+"\"") 

795 

796 return rval
