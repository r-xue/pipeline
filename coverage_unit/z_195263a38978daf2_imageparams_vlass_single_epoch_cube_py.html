41 def meanfreq_spwgroup(self, spw_selection): 

42 """Calculate the mean frequency of a spw group (specified by a selection string, e.g. '2,3,4').""" 

43 vis = self.vislist[0] 

44 ms = self.observing_run.get_ms(vis) 

45 spwid_list = [int(spw) for spw in spw_selection.split(',')] 

46 

47 spwfreq_list = [] 

48 for spwid in spwid_list: 

49 real_spwid = self.observing_run.virtual2real_spw_id(spwid, ms) 

50 spw = ms.get_spectral_window(real_spwid) 

51 spwfreq_list.append(float(spw.mean_frequency.to_units(measures.FrequencyUnits.GIGAHERTZ))) 

52 meanfreq_value = np.mean(spwfreq_list) 

53 

54 return str(meanfreq_value)+'GHz'

56 def _flagpct_spwgroup(self, results_list: Union[list, None] = None, spw_selection=None): 

57 """Get the flag percentage of a spw group (specified by a selection string, e.g. '2,3,4'). 

58 

59 Note: Quick check using cached results from hifv_flagtargetsdata() (PIPE-1401). 

60 For more comprehensive validation, use ImageParamsHeuristics.has_data() (see PIPE-557). 

61 Deprecated since PIPE-2641 - scheduled for removal. 

62 """ 

63 flagpct = None 

64 

65 # Catch exception as the success of result parsing is not guaranteed. 

66 try: 

67 if results_list and type(results_list) is list: 

68 for result in results_list: 

69 result_meta = result.read() 

70 if hasattr(result_meta, 'pipeline_casa_task') and result_meta.pipeline_casa_task.startswith( 

71 'hifv_flagtargetsdata' 

72 ): 

73 flagtargets_summary = [r.summaries for r in result_meta][0][-1] 

74 

75 n_flagged = n_total = 0.0 

76 for spwid in [spw.strip() for spw in spw_selection.split(',')]: 

77 n_flagged += flagtargets_summary['spw'][str(spwid)]['flagged'] 

78 n_total += flagtargets_summary['spw'][str(spwid)]['total'] 

79 flagpct = n_flagged / n_total 

80 except Exception as e: 

81 pass 

82 

83 return flagpct

85 def _plane_rejection(self, imlist_entry, vlass_plane_reject_ms): 

86 """Decide whether to reject a spw group based on the number of high-flagging-percentage fields.""" 

87 vis_name = self.vislist[-1] 

88 job = casa_tasks.flagdata(vis=vis_name, mode='summary', fieldcnt=True) 

89 flag_stats = job.execute() 

90 

91 # PIPE-1800: for plane-rejection, we restrict the flagging stats evaluation within the 

92 # 1deg^2 box based on the cutout layout. 

93 mosaic_side_arcsec = 3600 # 1 degree 

94 dist = mosaic_side_arcsec / 2.0 

95 dist_arcsec = str(dist) + 'arcsec' 

96 

97 fid_list = self.select_fields( 

98 offsets=dist_arcsec, 

99 intent='TARGET', 

100 phasecenter=imlist_entry['phasecenter'], 

101 name='0*,"0*,1*,"1*,2*,"2*,T*,"T*', 

102 ) 

103 # for the existing VLASS workflow, only one MS is used, though this might change in the future. 

104 fid_list = fid_list[0] 

105 

106 msobj = self.observing_run.get_ms(vis_name) 

107 field_objs = msobj.get_fields(field_id=fid_list) 

108 n_spwgroup = len(imlist_entry['spw']) 

109 n_flagged_field_spwgroup = np.zeros((len(field_objs), n_spwgroup)) 

110 n_total_field_spwgroup = np.zeros((len(field_objs), n_spwgroup)) 

111 scan_list = np.zeros(len(field_objs)) 

112 fname_list = [] 

113 

114 for field_idx, field_obj in enumerate(field_objs): 

115 fname_list.append(field_obj.name) 

116 for spwgroup_idx, spwgroup_sel in enumerate(imlist_entry['spw']): 

117 n_flagged = n_total = 0.0 

118 spw_list = spwgroup_sel.split(',') 

119 for spw_str in spw_list: 

120 n_flagged += flag_stats[field_obj.name]['spw'][spw_str]['flagged'] 

121 n_total += flag_stats[field_obj.name]['spw'][spw_str]['total'] 

122 n_flagged_field_spwgroup[field_idx, spwgroup_idx] = n_flagged 

123 n_total_field_spwgroup[field_idx, spwgroup_idx] = n_total 

124 scan_list[field_idx] = list(flag_stats[field_obj.name]['scan'].keys())[0] 

125 

126 nfield_above_flagpct = np.sum( 

127 n_flagged_field_spwgroup / n_total_field_spwgroup > vlass_plane_reject_ms['flagpct_thresh'], axis=0 

128 ) 

129 spwgroup_reject = [False] * n_spwgroup 

130 for idx, nfield in enumerate(nfield_above_flagpct): 

131 is_spwgroup_excluded = set(imlist_entry['spw'][idx].split(',')) & set( 

132 vlass_plane_reject_ms['exclude_spw'].split(',') 

133 ) 

134 if ( 

135 vlass_plane_reject_ms['apply'] 

136 and nfield >= vlass_plane_reject_ms['nfield_thresh'] 

137 and not is_spwgroup_excluded 

138 ): 

139 spwgroup_reject[idx] = True 

140 

141 vlass_flag_stats = {} 

142 vlass_flag_stats['spwgroup_list'] = imlist_entry['spw'] 

143 vlass_flag_stats['scan_list'] = scan_list 

144 vlass_flag_stats['fname_list'] = fname_list 

145 vlass_flag_stats['flagpct_field_spwgroup'] = n_flagged_field_spwgroup / n_total_field_spwgroup 

146 vlass_flag_stats['flagpct_spwgroup'] = np.sum(n_flagged_field_spwgroup, axis=0) / np.sum( 

147 n_total_field_spwgroup, axis=0 

148 ) 

149 vlass_flag_stats['nfield_above_flagpct'] = nfield_above_flagpct 

150 vlass_flag_stats['flagpct_thresh'] = vlass_plane_reject_ms['flagpct_thresh'] 

151 vlass_flag_stats['nfield_thresh'] = vlass_plane_reject_ms['nfield_thresh'] 

152 vlass_flag_stats['spwgroup_reject'] = spwgroup_reject 

153 

154 return vlass_flag_stats

156 def get_subtargets(self, cleantarget: CleanTarget, inputs: StandardInputs) -> list[CleanTarget] | None: 

157 """Derive sub-/child-targets from the original clean target for VLASS-SE-CUBE imaging. 

158 

159 Processes the initial full-bandwidth continuum CleanTarget specification to generate per-"plane" CleanTarget 

160 list for the VLASS "Coarse Cube" imaging. The function derives relevant child-targets based on VLASS plane 

161 rejection settings and also attaches flagging statistics for downstream analysis. 

162 

163 For the "coarse cube" workflow, we perform the following operations using the base CleanTarget specs: 

164 - Loop over individual spw groups 

165 - Generate corresponding per-spw(plane)-targets using modified copies of the original CleanTarget object 

166 as a template 

167 - Filter clean targets list after the VLASS-SE-CUBE plane rejection criteria is applied 

168 

169 Note: The initial 'spw' from the base CleanTarget object template, i.e., imlist_entry['spw'], is expected to 

170 be a list here (different from norm). For VLASS-SE-CUBE, we also attach additional attributes (`misc` field) 

171 to CleanTarget objects which can be rendered by Mako templates later. 

172 

173 Args: 

174 cleantarget: The original clean target object to process 

175 inputs: TaskInput object containing VLASS plane rejection settings 

176 

177 Returns: 

178 List ofsub-targets 

179 """ 

180 vlass_plane_reject_keys_allowed = {'apply', 'exclude_spw', 'flagpct_thresh', 'nfield_thresh'} 

181 vlass_plane_reject_ms = inputs.vlass_plane_reject_ms 

182 

183 # Validate input keys and warn about unexpected ones 

184 for key in vlass_plane_reject_ms: 

185 if key not in vlass_plane_reject_keys_allowed: 

186 LOG.warning( 

187 "The key %r in the 'vlass_plane_reject_ms' task input dictionary is not expected and will be " 

188 'ignored.', 

189 key, 

190 ) 

191 

192 vlass_flag_stats = self._plane_rejection(cleantarget, vlass_plane_reject_ms) 

193 cleantarget['misc_vlass'] = (cleantarget['misc_vlass'] or {}) | vlass_flag_stats 

194 

195 spwgroup_reject_list = [] 

196 imlist_entries = [] 

197 

198 for idx, spw in enumerate(cleantarget['spw']): 

199 imlist_entry_per_spwgroup = copy.deepcopy(cleantarget) 

200 imlist_entry_per_spwgroup['spw'] = spw 

201 imlist_entry_per_spwgroup['imagename'] = ( 

202 cleantarget['imagename'] + '.spw' + spw.replace('~', '-').replace(',', '_') 

203 ) 

204 imlist_entry_per_spwgroup['reffreq'] = self.meanfreq_spwgroup(spw) 

205 

206 # PIPE-1800: flagpct per spw group within the 1deg^2 box 

207 imlist_entry_per_spwgroup['misc_vlass']['flagpct'] = vlass_flag_stats['flagpct_spwgroup'][idx] 

208 

209 # PIPE-1800/PIPE-2641: flagpct_threshold is the flag percent rejection threshold over selected fields. 

210 # We hardcode the value to 1.0 which means we reject any spw that is completely flagged. 

211 # Note that this differs from vlass_plane_reject_ms['flagpct_thresh'] which is a per-field flagging 

212 # threshold to define "bad" fields. 

213 flagpct_threshold = 1.0 

214 if imlist_entry_per_spwgroup['misc_vlass']['flagpct'] >= flagpct_threshold: 

215 LOG.warning( 

216 'VLASS Data for spw=%r is %.2f%% flagged, and we will skip it as an imaging target.', 

217 spw, 

218 imlist_entry_per_spwgroup['misc_vlass']['flagpct'] * 100, 

219 ) 

220 continue 

221 

222 if vlass_flag_stats['spwgroup_reject'][idx]: 

223 spwgroup_reject_list.append(spw) 

224 continue 

225 

226 imlist_entries.append(imlist_entry_per_spwgroup) 

227 

228 if spwgroup_reject_list: 

229 LOG.warning( 

230 'VLASS Data for spw=%r meets the plane rejection criteria: nfield>=%i with flagpct>=%.2f%%.', 

231 ','.join(spwgroup_reject_list), 

232 vlass_plane_reject_ms['nfield_thresh'], 

233 vlass_plane_reject_ms['flagpct_thresh'] * 100, 

234 ) 

235 

236 return imlist_entries

274 return mask_list
