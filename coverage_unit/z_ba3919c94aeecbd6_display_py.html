86 def _plot(self) -> Generator[logger.Plot, None, None]: 

87 """ 

88 Create a plot with: 

89 - Primary x-axis: Centre Frequency (GHz) 

90 - Secondary x-axis: SPW IDs 

91  

92 Yields: 

93 Plot instance 

94 

95 Plot style depends on the number of measurement sets (MS): 

96 - If MS count is below a threshold, a scatter plot is used (each MS’s data from ms_dict). 

97 - Otherwise, a boxplot is drawn (using all_factors) with manually overlaid outlier points. 

98 """ 

99 MS_THRESHOLD = 5 # Use scatter plot if MS count is less than this threshold 

100 ALPHA = 1.0 # Weight for blending uniform spacing with normalized frequency differences 

101 LEGEND_LIMIT = 3 # If number of MS > LEGEND_LIMIT, shorten labels and add extra right-space. 

102 

103 # helper: shorten label if needed 

104 # e.g. process_label("uid___A002_X1234_X56.ms") will return "X56" 

105 def process_label(labels_list, label) -> str: 

106 return label.split('_')[-1].split('.')[0] if len(labels_list) > LEGEND_LIMIT else label 

107 

108 # Extract and sort SPWs by centre frequency 

109 spw_freq_pairs = [] 

110 for spw_id in list(self.valid_factors.keys()): 

111 spw_obj = self.valid_factors[spw_id]["spw_obj"] 

112 f = float(spw_obj.centre_frequency.to_units(FrequencyUnits.GIGAHERTZ)) 

113 spw_freq_pairs.append((spw_id, round(f, 1))) 

114 spw_freq_pairs.sort(key=lambda pair: pair[1]) 

115 if spw_freq_pairs: 

116 spw_ids, frequencies = zip(*spw_freq_pairs) 

117 spw_ids = list(spw_ids) 

118 frequencies = list(frequencies) 

119 else: 

120 spw_ids, frequencies = [], [] 

121 positions = self.__compute_x_positions(frequencies, ALPHA) 

122 

123 xlabel_bottom = 'Frequency (GHz)' 

124 ylabel = 'Jy/K factor' 

125 

126 fig = Figure(figsize=(8, 6)) 

127 canvas = FigureCanvas(fig) 

128 ax = fig.add_subplot(111) 

129 ax.set_xlabel(xlabel_bottom, fontsize=11) 

130 ax.set_ylabel(ylabel, fontsize=11) 

131 ax.set_title('Jy/K Factors across Frequencies', fontsize=11, fontweight='bold') 

132 

133 ms_count = len(self.ms_labels) 

134 ms_styles = { 

135 ms: (marker, color) 

136 for ms, marker, color in zip(self.ms_labels, itertools.cycle('osDv^<>'), 

137 itertools.cycle(cm.get_cmap('tab10').colors)) 

138 } 

139 

140 # CASE 1: Scatter Plot for MS count < MS_THRESHOLD 

141 if ms_count < MS_THRESHOLD: 

142 for ms in self.ms_labels: 

143 marker, color = ms_styles[ms] 

144 for i, spw_id in enumerate(spw_ids): 

145 ms_data = self.valid_factors[spw_id]["ms_dict"].get(ms, []) 

146 if ms_data: 

147 y_vals = [d[0] for d in ms_data] 

148 x_vals = [positions[i]] * len(y_vals) 

149 ax.scatter(x_vals, y_vals, marker=marker, color=color, alpha=1.0, 

150 label=process_label(self.ms_labels, ms) if i == 0 else "") 

151 if ms_count <= 20: 

152 # Expand x-axis to add space for legend if needed. 

153 lims = ax.get_xlim() 

154 extra = (max(positions) - min(positions)) / (len(spw_ids)/2) if spw_ids else 0 

155 if ms_count > LEGEND_LIMIT: 

156 ax.set_xlim((lims[0], lims[1] + extra)) 

157 ax.legend(title='MS', loc='best') 

158 

159 # CASE 2: Boxplot for MS count >= MS_THRESHOLD 

160 else: 

161 # Build boxplot data from all_factors for each SPW. 

162 box_data = [self.valid_factors[spw_id]["all_factors"] for spw_id in spw_ids] 

163 bp = ax.boxplot(box_data, positions=positions, patch_artist=True, showfliers=False) 

164 for box in bp['boxes']: 

165 box.set_facecolor('orange') 

166 box.set_alpha(0.7) 

167 

168 # Scatter outliers manually 

169 handles = {} 

170 for i, spw_id in enumerate(spw_ids): 

171 outliers = self.valid_factors[spw_id]["outliers"] 

172 for ms, factor in outliers: 

173 marker, color = ms_styles[ms] 

174 scatter = ax.scatter(positions[i], factor, marker=marker, color=color, 

175 edgecolor="black", zorder=3, label=ms) 

176 if ms not in handles: 

177 handles[ms] = scatter 

178 

179 if handles and len(handles) <= 20: 

180 # Expand x-axis if needed. 

181 if (len(handles) > LEGEND_LIMIT): 

182 lims = ax.get_xlim() 

183 extra = (max(positions) - min(positions)) / (len(spw_ids)/2) if spw_ids else 0 

184 keys = [process_label(handles.keys(), key) for key in handles.keys()] 

185 ax.set_xlim((lims[0], lims[1] + extra)) 

186 else: 

187 keys = handles.keys() 

188 

189 ax.legend(list(handles.values()), keys, 

190 title="MS with outliers", loc="best") 

191 

192 # Adjust y-limits if needed. 

193 y_min, y_max = ax.get_ylim() 

194 if (y_max - y_min) < 0.5: 

195 ax.set_ylim(y_min - 0.25, y_max + 0.25) 

196 

197 ax_top = ax.twiny() 

198 ax_top.set_xlim(ax.get_xlim()) 

199 ax_top.set_xticks(positions) 

200 ax_top.set_xticklabels(spw_ids) 

201 ax_top.set_xlabel('SPW ID', fontsize=11) 

202 ax.set_xticks(positions) 

203 ax.set_xticklabels(frequencies) # for the main x-axis, display the original frequency values. 

204 ax.grid(True) 

205 

206 plotfile = os.path.join(self.stage_dir, 'k2jy_factors_across_frequencies.png') 

207 canvas.print_figure(plotfile, format='png', dpi=DPISummary) 

208 plot = self._create_plot(plotfile, xlabel_bottom, ylabel) 

209 yield plot

212 def __compute_x_positions(frequencies: List[float], alpha: float) -> np.ndarray: 

213 """ 

214 The x-positions for SPWs plotting are computed as a blend between uniform spacing (by index) and 

215 normalized frequency differences: 

216 x = i + alpha * ((f - f_min) / (f_max - f_min)) 

217 ensuring that SPWs very close in frequency are evenly spaced while distant ones remain separated. 

218 

219 Args: 

220 frequencies: List of centre frequencies (in GHz) for the SPWs. 

221 alpha: Weight for the normalized frequency term. 

222 

223 Returns: 

224 Numpy array of x-positions. 

225 """ 

226 N = len(frequencies) 

227 if N == 0: 

228 return np.array([]) 

229 f_min, f_max = min(frequencies), max(frequencies) 

230 if f_max - f_min > 0: 

231 return np.array([i + alpha * ((f - f_min) / (f_max - f_min)) for i, f in enumerate(frequencies)]) 

232 return np.arange(N, dtype=float)
