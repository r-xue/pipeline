103 # PIPE-2710: export flat noise images 

104 # The flat noise images are the same as the 'imagename` associated with each target, but without the 

105 # '.pbcor' extension. 

106 # note: results.targetimages[0] is the same as self.inputs.context.sciimlist.get_imlist() 

107 for target in results.targetimages[0]: 

108 if not target['imagename'].endswith('.pbcor') or '.cube.' in target['imagename']: 

109 # skip images that are not pbcor images or are cubes. 

110 continue 

111 # process the flat noise images associated with this target pbcor image 

112 if target['multiterm']: 

113 # if multiterm, add the extension to the image name 

114 ext_list = [f'.tt{nt}' for nt in range(target['multiterm'])] 

115 else: 

116 # if not multiterm, add the empty extension 

117 ext_list = [''] 

118 

119 for nt in ext_list: 

120 flatnoiseimage_basename = target['imagename'][:-6] 

121 flatnoiseimage = flatnoiseimage_basename + nt 

122 flatnoisefits = fitsname(self.inputs.products_dir, flatnoiseimage) 

123 if os.path.exists(flatnoiseimage) and flatnoisefits not in target['fitsfiles']: 

124 self._shorten_spwlist(flatnoiseimage) 

125 task = casa_tasks.exportfits( 

126 imagename=flatnoiseimage, 

127 fitsimage=flatnoisefits, 

128 velocity=False, 

129 optical=False, 

130 bitpix=-32, 

131 minpix=0, 

132 maxpix=-1, 

133 overwrite=True, 

134 dropstokes=False, 

135 stokeslast=True, 

136 ) 

137 self._executor.execute(task) 

138 

139 target['auxfitsfiles'].append(flatnoisefits) 

140 

141 oussid = self.inputs.context.get_oussid() # returns string of 'unknown' for VLA 

142 

143 # Make the imaging vislist and the sessions lists. 

144 # Force this regardless of the value of imaging_only_products 

145 _, _, _, vislist = super()._make_lists( 

146 self.inputs.context, self.inputs.session, self.inputs.vis, imaging_only_mses=None) 

147 

148 # Export the auxiliary file products into a single tar file 

149 # These are optional for reprocessing but informative to the user 

150 # The calibrator source fluxes file 

151 # The antenna positions file 

152 # The continuum regions file 

153 # The target flagging file 

154 recipe_name = self.get_recipename(self.inputs.context) 

155 if not recipe_name: 

156 prefix = oussid 

157 else: 

158 prefix = oussid + '.' + recipe_name 

159 auxfproducts = self._do_if_auxiliary_products(prefix, self.inputs.output_dir, self.inputs.products_dir, vislist, 

160 self.inputs.imaging_products_only) 

161 

162 # Export the AQUA report 

163 pipe_aqua_reportfile = self._export_aqua_report(context=self.inputs.context, 

164 oussid=prefix, 

165 products_dir=self.inputs.products_dir, 

166 report_generator=vlaifaqua.VLAAquaXmlGenerator(), 

167 weblog_filename=results.weblog) 

168 

169 # Update the manifest 

170 if auxfproducts is not None or pipe_aqua_reportfile is not None: 

171 manifest_file = os.path.join(self.inputs.products_dir, results.manifest) 

172 recipe_name = self.inputs.context.project_structure.recipe_name 

173 self._add_to_manifest(manifest_file, auxfproducts, False, [], pipe_aqua_reportfile, oussid, recipe_name) 

174 

175 # Set specline_spws and smoothed_spw info that was used in the calibration run 

176 self._export_specline_smoothing(results, vislist) 

177 

178 return results

180 def _shorten_spwlist(self, image): 

181 # PIPE-325: abbreviate 'spw' and/or 'virtspw' for FITS header when spw string is "too long" 

182 # TODO: elevate this function to h exportdata after the PL2021 release so that it can be used 

183 # here as well as in h_exportdata. Too close to a release candidate at the moment to disrupt 

184 # ALMA validation. 

185 with casa_tools.ImageReader(image) as img: 

186 info = img.miscinfo() 

187 if 'spw' in info: 

188 if len(info['spw']) >= 68: 

189 spw_sorted = sorted([int(x) for x in info['spw'].split(',')]) 

190 info['spw'] = '{},...,{}'.format(spw_sorted[0], spw_sorted[-1]) 

191 img.setmiscinfo(info) 

192 if 'virtspw' in info: 

193 if len(info['virtspw']) >= 68: 

194 spw_sorted = sorted([int(x) for x in info['virtspw'].split(',')]) 

195 info['virtspw'] = '{},...,{}'.format(spw_sorted[0], spw_sorted[-1]) 

196 img.setmiscinfo(info)

199 """ 

200 Save the CASA restore script. 

201 """ 

202 

203 # Generate the file list 

204 

205 # Get the output file name 

206 ps = context.project_structure 

207 script_file = os.path.join(context.report_dir, script_name)

209 project_structure=ps,

216 tmpvislist = []

221 for vis in vislist:

223 if filename.endswith('.ms'): 

224 filename, _ = os.path.splitext(filename) 

225 tmpvislist.append(filename)

244 result_meta = result.read() 

245 if hasattr(result_meta, 'pipeline_casa_task'): 

246 if result_meta.pipeline_casa_task.startswith('hif_selfcal'): 

247 selfcal = True 

248 if result_meta.pipeline_casa_task.startswith('hif_uvcontsub'): 

249 uvcontsub = True 

250 

251 if vlassmode: 

252 task_string += "\n hifv_fixpointing()" 

253 

254 task_string += "\n hifv_statwt()" 

255 task_string += "\n hifv_mstransform()" 

256 if uvcontsub: 

257 task_string += "\n hif_uvcontsub()"

260

262try: 

263%s 

264finally: 

265 h_save() 

266''' % task_string 

267 

268 with open(script_file, 'w') as casa_restore_file: 

269 casa_restore_file.write(template) 

270 

271 LOG.info('Copying casa restore script %s to %s', script_file, out_script_file) 

272 shutil.copy(script_file, out_script_file)

276 def _export_final_flagversion(self, context, vis, flag_version_name, products_dir): 

277 """ 

278 PIPE-1553: include additional flag versions in tarfile 

279 """ 

280 

281 # Define the name of the output tarfile 

282 visname = os.path.basename(vis) 

283 tarfilename = visname + '.flagversions.tgz' 

284 if os.path.exists(tarfilename): 

285 os.remove(tarfilename) 

286 LOG.info('Storing final flags for %s in %s', visname, tarfilename) 

287 

288 # Create the tar file 

289 tar = tarfile.open(os.path.join(products_dir, tarfilename), "w:gz") 

290 

291 # Define the versions list file to be saved 

292 flag_version_list = os.path.join(visname + '.flagversions', 'FLAG_VERSION_LIST') 

293 tar_info = tarfile.TarInfo(flag_version_list) 

294 LOG.info('Saving flag version list') 

295 

296 # retrieve all flagversions saved 

297 task = casa_tasks.flagmanager(vis=visname, mode='list') 

298 flag_dict = self._executor.execute(task) 

299 # remove MS key entry if it exists; MS key does not conform with other entries 

300 # more information about flagmanager return dictionary here: 

301 # https://casadocs.readthedocs.io/en/stable/api/tt/casatasks.flagging.flagmanager.html#mode 

302 flag_dict.pop('MS', None) 

303 flag_keys = [y['name'] for y in flag_dict.values()] 

304 

305 export_final_flags_dict = dict() 

306 if flag_version_name in flag_keys: 

307 flag_comment = [y for y in flag_dict.values() if y['name'] == flag_version_name][0]['comment'] 

308 export_final_flags_dict[flag_version_name] = flag_comment 

309 

310 # rename flagversions to make them more deterministic 

311 if 'Applycal_Final' not in flag_keys: 

312 applycal_flags = sorted([y for y in flag_dict.values() if 'applycal' in y['name']], 

313 key=lambda x: x['name']) 

314 if applycal_flags: 

315 export_final_flags_dict['Applycal_Final'] = applycal_flags[-1]['comment'] 

316 task = casa_tasks.flagmanager(vis=vis, mode='rename', oldname=applycal_flags[-1]['name'], 

317 versionname='Applycal_Final', comment=applycal_flags[-1]['comment']) 

318 self._executor.execute(task) 

319 else: 

320 applycal_comment = [y for y in flag_dict.values() if y['name'] == 'Applycal_Final'][0]['comment'] 

321 export_final_flags_dict['Applycal_Final'] = applycal_comment 

322 

323 if 'hifv_checkflag_target-vla' not in flag_keys: 

324 target_RFI_key = [y for y in flag_dict.values() if 'hifv_checkflag_target-vla' in y['name']] 

325 if target_RFI_key: 

326 export_final_flags_dict['hifv_checkflag_target-vla'] = target_RFI_key[-1]['comment'] 

327 task = casa_tasks.flagmanager(vis=vis, mode='rename', oldname=target_RFI_key[-1]['name'], 

328 versionname='hifv_checkflag_target-vla', comment=target_RFI_key[-1]['comment']) 

329 self._executor.execute(task) 

330 else: 

331 hifv_comment = [y for y in flag_dict.values() if y['name'] == 'hifv_checkflag_target-vla'][0]['comment'] 

332 export_final_flags_dict['hifv_checkflag_target-vla'] = hifv_comment 

333 

334 if 'statwt_1' in flag_keys: 

335 statwt_comment = [y for y in flag_dict.values() if y['name'] == 'statwt_1'][0]['comment'] 

336 export_final_flags_dict['statwt_1'] = statwt_comment 

337 

338 # recreate tar file 

339 line = "" 

340 for flag_version, flag_version_comment in export_final_flags_dict.items(): 

341 # Define the directory to be saved, and where to store in tar archive. 

342 flagsname = os.path.join(vis + '.flagversions', 'flags.' + flag_version) 

343 flagsarcname = os.path.join(visname + '.flagversions', 'flags.' + flag_version) 

344 LOG.info('Saving flag version %s', flag_version) 

345 tar.add(flagsname, arcname=flagsarcname) 

346 line += "{} : {}\n".format(flag_version, flag_version_comment) 

347 

348 line = line.encode(sys.stdout.encoding) 

349 tar_info.size = len(line) 

350 tar.addfile(tar_info, io.BytesIO(line)) 

351 tar.close() 

352 

353 return tarfilename

355 def _export_specline_smoothing(self, results, vislist): 

356 """ 

357 Export the specline_spws and smoothed_spws information to the manifest 

358 """ 

359 # hifv_importdata only allows one set of specline_spws to be specified for all MSes, so pick the first MS 

360 mses = self.inputs.context.observing_run.get_measurement_sets()[0] 

361 spws = mses.get_spectral_windows(science_windows_only=True) 

362 

363 hanning_smoothed = None 

364 with casa_tools.TableReader(mses.name + '/SPECTRAL_WINDOW') as table: 

365 if 'OFFLINE_HANNING_SMOOTH' in table.colnames(): 

366 hanning_smoothed = table.getcol('OFFLINE_HANNING_SMOOTH') 

367 LOG.debug('Found smoothed spw information in SPECTRAL_WINDOW table') 

368 else: 

369 LOG.info("Could not find hanning smoothing information in SPECTRAL WINDOW table.") 

370 

371 smoothed_spws = [] 

372 if hanning_smoothed is not None: 

373 for spw in spws: 

374 real_spwid = self.inputs.context.observing_run.virtual2real_spw_id(spw.id, mses) 

375 if hanning_smoothed[real_spwid]: 

376 smoothed_spws.append(spw.id) 

377 

378 smoothed_spws_str = '' 

379 if len(smoothed_spws) > 0: 

380 smoothed_spws_str = utils.find_ranges(smoothed_spws) 

381 

382 specline_spws = '' 

383 specline_spws_list = [str(spw.id) for spw in spws if spw.specline_window] 

384 if len(specline_spws_list) > 0: 

385 specline_spws = utils.find_ranges(specline_spws_list) 

386 

387 # Add the specline_spws and smoothed_spws information to the mainfest 

388 manifest_inputs = {} 

389 manifest_inputs['specline_spws'] = specline_spws 

390 

391 # Only include hanning smoothing information if available 

392 if hanning_smoothed is not None: 

393 manifest_inputs['smoothed_spws'] = smoothed_spws_str 

394 

395 pipemanifest = manifest.PipelineManifest('') 

396 manifest_file = os.path.join(self.inputs.products_dir, results.manifest) 

397 pipemanifest.import_xml(manifest_file) 

398 

399 for asdm in pipemanifest.get_ous().findall(f".//asdm[@name=\'{vislist[0]}\']"): 

400 newinputs = {key: str(value) for (key, value) in manifest_inputs.items()} # stringify the values 

401 eltree.SubElement(asdm, "restoredata", newinputs) 

402 pipemanifest.write(manifest_file)
