115 def primary_beam_size(self, spwid, intent): 

116 

117 '''Calculate primary beam size in arcsec.''' 

118 

119 cqa = casa_tools.quanta 

120 

121 # get the diameter of the smallest antenna used among all vis sets 

122 antenna_ids = self.antenna_ids(intent) 

123 diameters = [] 

124 for vis in self.vislist: 

125 ms = self.observing_run.get_ms(name=vis) 

126 antennas = ms.antennas 

127 for antenna in antennas: 

128 if antenna.id in antenna_ids[os.path.basename(vis)]: 

129 diameters.append(antenna.diameter) 

130 smallest_diameter = np.min(np.array(diameters)) 

131 

132 msname = self.get_ref_msname(spwid) 

133 real_spwid = self.observing_run.virtual2real_spw_id(spwid, self.observing_run.get_ms(msname)) 

134 ms = self.observing_run.get_ms(name=msname) 

135 spw = ms.get_spectral_window(real_spwid) 

136 

137 ref_frequency = float( 

138 spw.ref_frequency.to_units(measures.FrequencyUnits.HERTZ)) 

139 

140 # use the smallest antenna diameter and the reference frequency 

141 # to estimate the primary beam radius - 

142 # radius of first null in arcsec = 1.22*lambda/D 

143 primary_beam_size = \ 

144 1.22 \ 

145 * cqa.getvalue(cqa.convert(cqa.constants('c'), 'm/s')) \ 

146 / ref_frequency \ 

147 / smallest_diameter \ 

148 * (180.0 * 3600.0 / math.pi) 

149 

150 return primary_beam_size

722 def phasecenter(self, fields, centreonly=True, vislist=None, shift_to_nearest_field=False, 

723 primary_beam=None, intent='TARGET'): 

724 

725 cme = casa_tools.measures 

726 cqa = casa_tools.quanta 

727 

728 # Return None for each return value if no fields are input 

729 if len(fields) == 1 and fields[0] == '': 

730 if centreonly: 

731 return None, None 

732 else: 

733 return None, None, None, None 

734 

735 if vislist is None: 

736 vislist = self.vislist 

737 

738 mdirections = [] 

739 sdirections = [] 

740 field_names = [] 

741 for ivis, vis in enumerate(vislist): 

742 

743 # Get the visibilities 

744 ms = self.observing_run.get_ms(name=vis) 

745 visfields = fields[ivis] 

746 if visfields == '': 

747 continue 

748 visfields = visfields.split(',') 

749 visfields = list(map(int, visfields)) 

750 

751 # Get the phase directions and convert to ICRS 

752 for field in visfields: 

753 # get field centres as measures 

754 fieldobj = ms.get_fields(field_id=field)[0] 

755 ref = cme.getref(fieldobj.mdirection) 

756 if ref == 'ICRS' or ref == 'J2000' or ref == 'B1950': 

757 phase_dir = cme.measure(fieldobj.mdirection, 'ICRS') 

758 else: 

759 phase_dir = fieldobj.mdirection 

760 mdirections.append(phase_dir) 

761 # Store source direction for ephemeris objects 

762 # for later correction of mosaic center position. 

763 if self.is_eph_obj(fieldobj.name): 

764 ref = cme.getref(fieldobj.source.direction) 

765 if ref == 'ICRS' or ref == 'J2000' or ref == 'B1950': 

766 phase_dir = cme.measure(fieldobj.source.direction, 'ICRS') 

767 else: 

768 phase_dir = fieldobj.source.direction 

769 sdirections.append(phase_dir) 

770 field_names.append(fieldobj.name) 

771 

772 # Correct field directions for ephemeris source motion 

773 if sdirections != []: 

774 sdirection_m0_rad = [] 

775 sdirection_m1_rad = [] 

776 for sdirection in sdirections: 

777 sdirection_m0_rad.append(cqa.getvalue(cqa.convert(sdirection['m0'], 'rad'))) 

778 sdirection_m1_rad.append(cqa.getvalue(cqa.convert(sdirection['m1'], 'rad'))) 

779 avg_sdirection_m0_rad = np.average(sdirection_m0_rad) 

780 avg_sdirection_m1_rad = np.average(sdirection_m1_rad) 

781 for i in range(len(mdirections)): 

782 mdirections[i]['m0'] = cqa.sub(mdirections[i]['m0'], cqa.sub(sdirections[i]['m0'], cqa.quantity(avg_sdirection_m0_rad, 'rad'))) 

783 mdirections[i]['m1'] = cqa.sub(mdirections[i]['m1'], cqa.sub(sdirections[i]['m1'], cqa.quantity(avg_sdirection_m1_rad, 'rad'))) 

784 

785 # sanity check - for single field images the field centres from 

786 # different measurement sets should be coincident and can 

787 # collapse the list 

788 # Set the maximum separation to 200 microarcsec and test via 

789 # a tolerance rather than an equality. 

790 max_separation = cqa.quantity('200uarcsec') 

791 

792 is_mos_or_het = self._is_mosaic(fields) or len(self.antenna_diameters()) > 1 

793 if not is_mos_or_het: 

794 max_separation_uarcsec = cqa.getvalue(cqa.convert(max_separation, "uarcsec"))[0] # in micro arcsec 

795 # PIPE-1504/PIPE-2859: only issue this message at the WARNING level if it's executed by hifa_imageprecheck 

796 if current_task_name.get() == 'hifa_imageprecheck': 

797 log_level = logging.WARNING 

798 else: 

799 log_level = logging.INFO 

800 for mdirection in mdirections: 

801 separation = cme.separation(mdirection, mdirections[0]) 

802 if cqa.gt(separation, max_separation): 

803 separation_arcsec = cqa.getvalue(cqa.convert(separation, "arcsec"))[0] 

804 LOG.log( 

805 log_level, 

806 "The separation between %s field centers across EBs is %f arcseconds (larger than the limit of %.1f microarcseconds). " 

807 "This is only normal for an ephemeris source or a source with a large proper motion or parallax." % 

808 (field_names[0], 

809 separation_arcsec, max_separation_uarcsec)) 

810 mdirections = [mdirections[0]] 

811 

812 # it should be easy to calculate some 'average' direction 

813 # from the contributing fields but it doesn't seem to be 

814 # at the moment - no conversion between direction measures, 

815 # no calculation of a direction from a direction and an 

816 # offset. Consequently, what follows is a bit crude. 

817 

818 # First, find the offset of all field from field 0. 

819 xsep = [] 

820 ysep = [] 

821 for mdirection in mdirections: 

822 pa = cme.posangle(mdirections[0], mdirection) 

823 sep = cme.separation(mdirections[0], mdirection) 

824 xs = cqa.mul(sep, cqa.sin(pa)) 

825 ys = cqa.mul(sep, cqa.cos(pa)) 

826 xs = cqa.convert(xs, 'arcsec') 

827 ys = cqa.convert(ys, 'arcsec') 

828 xsep.append(cqa.getvalue(xs)) 

829 ysep.append(cqa.getvalue(ys)) 

830 

831 xsep = np.array(xsep) 

832 ysep = np.array(ysep) 

833 xspread = xsep.max() - xsep.min() 

834 yspread = ysep.max() - ysep.min() 

835 xcen = xsep.min() + xspread / 2.0 

836 ycen = ysep.min() + yspread / 2.0 

837 

838 # get direction of image centre crudely by adding offset 

839 # of centre to ref values of first field. 

840 ref = cme.getref(mdirections[0]) 

841 md = cme.getvalue(mdirections[0]) 

842 m0 = cqa.quantity(md['m0']) 

843 m1 = cqa.quantity(md['m1']) 

844 

845 m0 = cqa.add(m0, cqa.div('%sarcsec' % xcen, cqa.cos(m1))) 

846 m1 = cqa.add(m1, '%sarcsec' % ycen) 

847 

848 # convert to strings (CASA 4.0 returns as list for some reason 

849 # hence 0 index) 

850 if ref == 'ICRS' or ref == 'J2000' or ref == 'B1950': 

851 m0 = cqa.time(m0, prec=10)[0] 

852 else: 

853 m0 = cqa.angle(m0, prec=9)[0] 

854 m1 = cqa.angle(m1, prec=9)[0] 

855 

856 center = cme.direction(ref, m0, m1) 

857 phase_center = '%s %s %s' % (ref, m0, m1) 

858 psf_phase_center = phase_center 

859 

860 # If the image center is outside of the mosaic pointings, calculate a PSF phase center 

861 # pointing to the nearest field. Both the actual and the PSF phase centers are returned. 

862 if shift_to_nearest_field: 

863 nearest_field_to_center = self.center_field_ids(vislist, field_names[0], intent, phase_center)[0] 

864 ms = self.observing_run.get_ms(name=vislist[0]) 

865 nearest = ms.get_fields(field_id=nearest_field_to_center)[0].mdirection 

866 if primary_beam: 

867 pb_dist = 0.408 

868 if cqa.getvalue(cqa.convert(cme.separation(center, nearest), 'arcsec'))[0] > pb_dist * primary_beam: 

869 LOG.info('The nearest pointing is > {pb_dist}pb away from image center. ' 

870 'Shifting the PSF phase center to the ' 

871 'nearest field (id = {nf})'.format(pb_dist=pb_dist, nf=nearest_field_to_center)) 

872 LOG.info('Old phasecenter: {}'.format(phase_center)) 

873 # convert to strings (CASA 4.0 returns as list for some reason hence 0 index) 

874 m0 = cme.getvalue(nearest)['m0'] 

875 m1 = cme.getvalue(nearest)['m1'] 

876 if ref == 'ICRS' or ref == 'J2000' or ref == 'B1950': 

877 m0 = cqa.time(m0, prec=10)[0] 

878 else: 

879 m0 = cqa.angle(m0, prec=9)[0] 

880 m1 = cqa.angle(m1, prec=9)[0] 

881 psf_phase_center = '%s %s %s' % (ref, m0, m1) 

882 LOG.info('New PSF phasecenter: {}'.format(phase_center)) 

883 else: 

884 LOG.warning('No primary beam supplied. Will not attempt to shift PSF phasecenter to ' 

885 'nearest field w/o a primary beam distance check.') 

886 

887 if centreonly: 

888 return phase_center, psf_phase_center 

889 else: 

890 return phase_center, psf_phase_center, xspread, yspread

932 def select_fields(self, intent='TARGET', name=None, phasecenter=None, offsets=None): 

933 """Select fields id based on intent, field name, or separation constrains from the phasecenter 

934 

935 This method selects fields based on the following search rules: 

936 intent: restrict the search by the field intent 

937 name: restrict the search id by the field name. 

938 phasecenter, offsets: restrict the search area in an on-sky box centered around phasecenter 

939 

940 Args: 

941 intent (str, optional): intent string. Defaults to 'TARGET'. 

942 name (str, optional): name string, which could be a wildcard like '1*,2*,0*' Defaults to None. 

943 phasecenter (str, optional): center of the search box. Defaults to None. 

944 offsets (optional): sky offsets search limits along the longitude/ latitude direction in the reference frame.  

945 Defaults to None. 

946 

947 Returns: 

948 list: a list; each element is the selected field id list per measurement sets. 

949 """ 

950 

951 fields_list = [] 

952 

953 for vis in self.vislist: 

954 ms = self.observing_run.get_ms(name=vis) 

955 fields = list(ms.fields) 

956 select = np.full(len(fields), True) 

957 

958 if isinstance(intent, str) and intent not in ('', '*'): 

959 intent_set = intent.split(',') 

960 select = select & np.array([not field.intents.isdisjoint(intent_set) for field in fields]) 

961 

962 if isinstance(name, str) and name not in ('', '*'): 

963 name_pats = name.split(',') 

964 select = select & np.array([any([fnmatch.fnmatch(field.name, name_pat) for name_pat in name_pats]) for field in fields]) 

965 

966 if isinstance(phasecenter, str): 

967 

968 coord_phasecenter = phasecenter_to_skycoord(phasecenter) 

969 frame = refcode_to_skyframe(fields[0].frame) 

970 coord_phasecenter = coord_phasecenter.transform_to(frame) 

971 

972 coords = SkyCoord( 

973 [field.longitude['value'] for field in fields], 

974 [field.latitude['value'] for field in fields], 

975 frame=frame, 

976 unit=(fields[0].longitude['unit'], fields[0].latitude['unit'])) 

977 dra, ddec = coord_phasecenter.spherical_offsets_to(coords) 

978 if not isinstance(offsets, (list, tuple)): 

979 offsets_limit = (offsets, offsets) 

980 else: 

981 offsets_limit = offsets 

982 LOG.info('Searching for fields within the maximum sky offsets of %s from %s .', offsets_limit, phasecenter) 

983 select = select & (np.abs(dra) < u.Quantity(offsets_limit[0])) & (np.abs(ddec) < u.Quantity(offsets_limit[1])) 

984 

985 LOG.info('Found %s of %s fields out meeting the selection rule(s) in %s', select.sum(), select.size, os.path.basename(vis)) 

986 fields_list.append([field.id for idx, field in enumerate(fields) if select[idx]]) 

987 

988 return fields_list

1171 def imsize_from_cutout( 

1172 self, cutout_imsize: list[int], cell: str, primary_beam: float, sfpblimit: float | None = None 

1173 ) -> list[int]: 

1174 """Calculate optimal imaging size based on requested cutout size. 

1175 

1176 Determines the optimal image size for CASA imaging by adding a buffer around the requested 

1177 cutout size. The buffer is calculated as 1.5 times the primary beam radius (0.75 beams on 

1178 each side), ensuring adequate coverage for imaging. Final image dimensions are adjusted to 

1179 composite numbers for optimal FFT performance using CASA's synthesis utilities. 

1180 

1181 Args: 

1182 cutout_imsize: Two-element list [nxpix, nypix] defining the requested cutout/final 

1183 image size in pixels. 

1184 cell: Cell size of imaging in string format with units (e.g. '0.1arcsec'). 

1185 primary_beam: Primary beam size in arcseconds (full width). 

1186 sfpblimit: Single field primary beam response limit. Reserved for future use. 

1187 

1188 Returns: 

1189 Two-element list [nxpix, nypix] for the calculated image size in pixels, optimized 

1190 for FFT performance. 

1191 

1192 Example: 

1193 >>> heuristics.imsize_from_cutout([512, 512], '0.1arcsec', 60.0) 

1194 [1440, 1440] 

1195 """ 

1196 qa_tool = casa_tools.quanta 

1197 cell_arcsec = qa_tool.convert(cell, 'arcsec')['value'] 

1198 

1199 buffer_size = math.ceil(primary_beam * 1.5 / cell_arcsec * 0.5) 

1200 

1201 csu = casa_tools.synthesisutils 

1202 imsize = [csu.getOptimumSize(size + 2 * buffer_size) for size in cutout_imsize] 

1203 csu.done() 

1204 

1205 return imsize

1207 def imsize(self, fields, cell, primary_beam, sfpblimit=None, max_pixels=None, 

1208 centreonly=False, vislist=None, spwspec=None, intent: str = '', joint_intents: str = '', specmode=None): 

1209 """ 

1210 Image size heuristics for single fields and mosaics. The pixel count along x and y image dimensions 

1211 is determined by the cell size, primary beam size and the spread of phase centers in case of mosaics. 

1212 

1213 :param fields: list of comma separated strings of field IDs per MS. 

1214 :param cell: pixel (cell) size in arcsec. 

1215 :param primary_beam: primary beam width in arcsec. 

1216 :param sfpblimit: single field primary beam response. If provided then imsize is chosen such that the image 

1217 edge is at normalised primary beam level equals to sfpblimit. 

1218 :param max_pixels: maximum allowed pixel count, integer. The same limit is applied along both image axes. 

1219 :param centreonly: if True, then ignore the spread of field centers. 

1220 :param vislist: list of visibility path string to be used for imaging. If not set then use all visibilities 

1221 in the context. 

1222 :param spwspec: ID list of spectral windows used to create image product. List or string containing comma 

1223 separated spw IDs list. Not used in the base method. 

1224 :param intent: field/source intent 

1225 :param joint_intents: stage intents 

1226 :return: two element list of pixel count along x and y image axes. 

1227 """ 

1228 

1229 # For the time being PIPE-1829 asks for a fixed imsize 

1230 # for polarization calibrators. The subsequent image 

1231 # analysis depends on the exact numbers. The IQUV imaging 

1232 # and analysis is only performed if the stage asks for 

1233 # polarization intent only. 

1234 if intent == 'POLARIZATION' and joint_intents == 'POLARIZATION': 

1235 return [256, 256] 

1236 

1237 if vislist is None: 

1238 vislist = self.vislist 

1239 

1240 # get spread of beams 

1241 if centreonly: 

1242 xspread = yspread = 0.0 

1243 else: 

1244 _, _, xspread, yspread = self.phasecenter(fields, centreonly=centreonly, vislist=vislist) 

1245 

1246 cqa = casa_tools.quanta 

1247 

1248 cellx = cell[0] 

1249 if len(cell) > 1: 

1250 celly = cell[1] 

1251 else: 

1252 celly = cell[0] 

1253 

1254 if (cellx == 'invalid') or (celly == 'invalid'): 

1255 return [0, 0] 

1256 

1257 # get cell and beam sizes in arcsec 

1258 cellx_v = cqa.getvalue(cqa.convert(cellx, 'arcsec'))[0] 

1259 celly_v = cqa.getvalue(cqa.convert(celly, 'arcsec'))[0] 

1260 beam_radius_v = primary_beam 

1261 

1262 # set size of image to spread of field centres plus a 

1263 # border of 0.75 (0.825) * beam radius (radius is to 

1264 # first null) wide 

1265 nfields = int(np.median([len(field_ids.split(',')) for field_ids in fields])) 

1266 is_mos_or_het = self._is_mosaic(fields) or len(self.antenna_diameters()) > 1 

1267 

1268 if is_mos_or_het and nfields <= 3: 

1269 # PIPE-209 asks for a slightly larger size for small (2-3 field) mosaics. 

1270 nxpix = int((1.65 * beam_radius_v + xspread) / cellx_v) 

1271 nypix = int((1.65 * beam_radius_v + yspread) / celly_v) 

1272 else: 

1273 nxpix = int((1.5 * beam_radius_v + xspread) / cellx_v) 

1274 nypix = int((1.5 * beam_radius_v + yspread) / celly_v) 

1275 

1276 if (not is_mos_or_het) and (sfpblimit is not None): 

1277 beam_fwhp = 1.12 / 1.22 * beam_radius_v 

1278 nxpix = int(utils.round_half_up(1.1 * beam_fwhp * math.sqrt(-math.log(sfpblimit) / math.log(2.)) / cellx_v)) 

1279 nypix = int(utils.round_half_up(1.1 * beam_fwhp * math.sqrt(-math.log(sfpblimit) / math.log(2.)) / celly_v)) 

1280 

1281 if max_pixels is not None: 

1282 nxpix = min(nxpix, max_pixels) 

1283 nypix = min(nypix, max_pixels) 

1284 

1285 # set nxpix, nypix to next highest 'composite number' 

1286 csu = casa_tools.synthesisutils 

1287 nxpix = csu.getOptimumSize(nxpix) 

1288 nypix = csu.getOptimumSize(nypix) 

1289 csu.done() 

1290 

1291 return [nxpix, nypix]

1710 def freq_intersection(self, vis, field, intent, spw, frame='LSRK'): 

1711 """ 

1712 Calculate LSRK frequency intersection of a list of MSs for a 

1713 given field and spw. Exclude flagged channels. 

1714 """ 

1715 

1716 cqa = casa_tools.quanta 

1717 csu = casa_tools.synthesisutils 

1718 

1719 per_eb_flagged_freq_ranges = [] 

1720 per_eb_full_freq_ranges = [] 

1721 channel_widths = [] 

1722 

1723 for msname in vis: 

1724 ms = self.observing_run.get_ms(name=msname) 

1725 real_spw = str(self.observing_run.virtual2real_spw_id(spw, ms)) 

1726 

1727 # Loop over mosaic fields and determine the channel flags per field. 

1728 # Skip channels that have only partial pointing coverage. 

1729 field_ids = self.field(intent=intent, field=field, vislist=[msname])[0].split(',') 

1730 per_field_flagged_freq_ranges = [] 

1731 per_field_full_freq_ranges = [] 

1732 for field_id in field_ids: 

1733 

1734 # For multi-tuning EBs, a field may be observed in just a subset of 

1735 # science spws. Filter out spws not applicable to this field. 

1736 spw_do = ms.get_spectral_window(real_spw) 

1737 field_dos = ms.get_fields(field, intent=intent) 

1738 # field(s) not in MS 

1739 if not field_dos: 

1740 continue 

1741 # spw not observed for the field(s) 

1742 if not [f for f in field_dos if spw_do in f.valid_spws]: 

1743 continue 

1744 

1745 # Get the channel flags (uses virtual spw ID !) 

1746 channel_flags = self.get_channel_flags(msname, field_id, spw) 

1747 

1748 # Get indices of unflagged channels 

1749 nfi = np.where(channel_flags == False)[0] 

1750 

1751 # Use unflagged edge channels to determine LSRK frequency range 

1752 if nfi.shape != (0,): 

1753 # Use the edges. Another heuristic will skip one extra channel later in the final frequency range. 

1754 if frame in ('REST', 'SOURCE'): 

1755 result = csu.advisechansel(msname=msname, fieldid=int(field_id), spwselection='%s:%d~%d' % (real_spw, nfi[0], nfi[-1]), getfreqrange=True, freqframe='SOURCE', ephemtable='TRACKFIELD') 

1756 else: 

1757 result = csu.advisechansel(msname=msname, fieldid=int(field_id), spwselection='%s:%d~%d' % (real_spw, nfi[0], nfi[-1]), getfreqrange=True, freqframe=frame) 

1758 csu.done() 

1759 

1760 f0_flagged = float(cqa.getvalue(cqa.convert(result['freqstart'], 'Hz'))[0]) 

1761 f1_flagged = float(cqa.getvalue(cqa.convert(result['freqend'], 'Hz'))[0]) 

1762 

1763 per_field_flagged_freq_ranges.append((f0_flagged, f1_flagged)) 

1764 # The frequency range from advisechansel is from channel edge 

1765 # to channel edge. To get the width, one needs to divide by the 

1766 # number of channels in the selection. 

1767 channel_widths.append((f1_flagged - f0_flagged) / (nfi[-1] - nfi[0] + 1)) 

1768 

1769 # Also get the full ranges to trim the final LSRK range for 

1770 # odd tunings near the LO range edges (PIPE-526). 

1771 if frame in ('REST', 'SOURCE'): 

1772 result = csu.advisechansel(msname=msname, fieldid=int(field_id), spwselection='%s' % (real_spw), getfreqrange=True, freqframe='SOURCE', ephemtable='TRACKFIELD') 

1773 else: 

1774 result = csu.advisechansel(msname=msname, fieldid=int(field_id), spwselection='%s' % (real_spw), getfreqrange=True, freqframe=frame) 

1775 csu.done() 

1776 

1777 f0_full = float(cqa.getvalue(cqa.convert(result['freqstart'], 'Hz'))[0]) 

1778 f1_full = float(cqa.getvalue(cqa.convert(result['freqend'], 'Hz'))[0]) 

1779 

1780 per_field_full_freq_ranges.append((f0_full, f1_full)) 

1781 

1782 # Calculate weighted intersection with threshold of 1.0 to avoid 

1783 # edge channels that have drifted too much in LSRK or REST during 

1784 # the EB. 

1785 if per_field_flagged_freq_ranges != []: 

1786 per_eb_flagged_intersect_range = utils.intersect_ranges_by_weight(per_field_flagged_freq_ranges, max(channel_widths), 1.0) 

1787 per_eb_full_intersect_range = utils.intersect_ranges_by_weight(per_field_full_freq_ranges, max(channel_widths), 1.0) 

1788 if per_eb_flagged_intersect_range != () and per_eb_full_intersect_range != (): 

1789 per_eb_flagged_freq_ranges.append(per_eb_flagged_intersect_range) 

1790 per_eb_full_freq_ranges.append(per_eb_full_intersect_range) 

1791 

1792 if per_eb_flagged_freq_ranges == []: 

1793 return -1, -1, 0 

1794 

1795 # Calculate weighted intersection with threshold of 0.6 to avoid 

1796 # partially flagged spws of individual EBs restricting the frequency 

1797 # axis (PIPE-180). 

1798 intersect_range_flagged = utils.intersect_ranges_by_weight(per_eb_flagged_freq_ranges, max(channel_widths), 0.6) 

1799 # Calculate the weighted intersection of all ranges with threshold 

1800 # of 1.0 to get the maximum possible range with contributions from 

1801 # all EBs (PIPE-526). 

1802 intersect_range_full = utils.intersect_ranges_by_weight(per_eb_full_freq_ranges, max(channel_widths), 1.0) 

1803 if intersect_range_flagged != () and intersect_range_full != (): 

1804 intersect_range_final = utils.intersect_ranges_by_weight([intersect_range_flagged, intersect_range_full], max(channel_widths), 1.0) 

1805 if0, if1 = intersect_range_final 

1806 return if0, if1, max(channel_widths) 

1807 else: 

1808 return -1, -1, 0

1814 def get_bw_corr_factor(self, ms_do, spw, nchan): 

1815 """ 

1816 Calculate effective bandwidth correction factor. 

1817 """ 

1818 

1819 real_spwid = self.observing_run.virtual2real_spw_id(spw, ms_do) 

1820 spw_do = ms_do.get_spectral_window(real_spwid) 

1821 spwchan = spw_do.num_channels 

1822 physicalBW_of_1chan = float(spw_do.channels[0].getWidth().convert_to(measures.FrequencyUnits.HERTZ).value) 

1823 effectiveBW_of_1chan = float(spw_do.channels[0].effective_bw.convert_to(measures.FrequencyUnits.HERTZ).value) 

1824 

1825 BW_ratio = effectiveBW_of_1chan / physicalBW_of_1chan 

1826 

1827 if (BW_ratio <= 1.0): 

1828 N_smooth = 0 

1829 elif (utils.equal_to_n_digits(BW_ratio, 2.667, 4)): 

1830 N_smooth = 1 

1831 elif (utils.equal_to_n_digits(BW_ratio, 1.600, 4)): 

1832 N_smooth = 2 

1833 elif (utils.equal_to_n_digits(BW_ratio, 1.231, 4)): 

1834 N_smooth = 4 

1835 elif (utils.equal_to_n_digits(BW_ratio, 1.104, 4)): 

1836 N_smooth = 8 

1837 elif (utils.equal_to_n_digits(BW_ratio, 1.049, 4)): 

1838 N_smooth = 16 

1839 else: 

1840 LOG.warning('Could not evaluate channel bandwidths ratio. Physical: %s Effective: %s Ratio: %s' % (physicalBW_of_1chan, effectiveBW_of_1chan, BW_ratio)) 

1841 N_smooth = 0 

1842 

1843 if N_smooth > 0 and nchan > 1: 

1844 optimisticBW = nchan * float(effectiveBW_of_1chan) 

1845 approximateEffectiveBW = (nchan + 1.12 * (spwchan - nchan) / spwchan / N_smooth) * float(physicalBW_of_1chan) 

1846 SCF = (optimisticBW / approximateEffectiveBW) ** 0.5 

1847 else: 

1848 approximateEffectiveBW = nchan * float(physicalBW_of_1chan) 

1849 SCF = 1.0 

1850 

1851 return SCF, physicalBW_of_1chan, effectiveBW_of_1chan, approximateEffectiveBW

2439 def check_psf(self, psf_name, field, spw): 

2440 """Check for problematic PSF based on per-plane beam results from .psf image. 

2441 

2442 This method primarily catches issues with CASA's internal FitGaussianPSF algorithm and was 

2443 initially developed for ALMA cube imaging cases as a trigger for calling .find_good_commonbeam(). 

2444  

2445 Note that CASA's internal FitGaussianPSF algorithm (CAS-13022) performs channel-wise 

2446 interpolation/extrapolation, so header values may not exactly reflect the actual per-channel 

2447 beam size in edge cases. For example, a blank channel might still report a beam fit value 

2448 derived from interpolation. 

2449  

2450 Reference: 

2451 https://open-bitbucket.nrao.edu/projects/CASA/repos/casa6/browse/casatools/src/code/synthesis/TransformMachines/StokesImageUtil.cc#481 

2452 """ 

2453 cqa = casa_tools.quanta 

2454 bad_psf_fit = False 

2455 

2456 LOG.info('checking psf: %s from field=%s / spw=%s', psf_name, field, spw) 

2457 

2458 with casa_tools.ImageReader(psf_name) as image: 

2459 try: 

2460 beams = image.restoringbeam()['beams'] 

2461 bmaj = np.array([cqa.getvalue(cqa.convert(b['*0']['major'], 'arcsec')) for b in beams.values()]) 

2462 

2463 # Filter empty psf planes 

2464 bmaj = bmaj[np.where(bmaj > 1e-6)] 

2465 bmaj_median = np.median(bmaj) 

2466 

2467 # theshold (in relative scaling) to detect outliers and trigger the .find_good_common heuristic method 

2468 outlier_threshold = 2.0 

2469 

2470 cond1 = np.logical_and(0.0 < bmaj, bmaj < bmaj_median / outlier_threshold) 

2471 cond2 = bmaj > outlier_threshold * bmaj_median 

2472 LOG.info( 

2473 'Per-plane beams bmajor - min/max/medium - %s/%s/%s', 

2474 np.min(bmaj), 

2475 np.max(bmaj), 

2476 np.median(bmaj), 

2477 ) 

2478 if np.logical_or(cond1, cond2).any(): 

2479 bad_psf_fit = True 

2480 LOG.warning( 

2481 'The PSF fit has one or more outlier channels for field %s SPW %s, please check the ' 

2482 'results for this cube carefully, there are likely data issues.', 

2483 field, 

2484 spw, 

2485 ) 

2486 else: 

2487 LOG.info('No outlier per-plane beams with bmaj < 0.5*median(bmaj) or bmaj > 2*median(bmaj)') 

2488 except Exception as ex: 

2489 traceback_msg = traceback.format_exc() 

2490 LOG.debug(traceback_msg) 

2491 LOG.info(ex) 

2492 

2493 return bad_psf_fit

2496 def find_good_commonbeam(psf_filename: str): 

2497 """Find and replace outlier beams to calculate a good common beam. 

2498 

2499 Method from Urvashi Rao. 

2500 

2501 Returns new common beam and array of channel numbers with invalid beams. 

2502 Leaves old beams in the PSF as is. 

2503 """ 

2504 cqa = casa_tools.quanta 

2505 

2506 with casa_tools.ImageReader(psf_filename) as image: 

2507 allbeams = image.restoringbeam() 

2508 commonbeam = image.commonbeam() 

2509 # note that the beam measure dictionaries return by .restoringbeam() and .commonbeam() have slightly 

2510 # different keys: 

2511 # from .restoringbeam: major/minor/positionangle 

2512 # fro .commonbeam: major/minior/pa 

2513 # https://casadocs.readthedocs.io/en/latest/api/tt/casatools.image.html#casatools.image.image.commonbeam 

2514 # https://casadocs.readthedocs.io/en/latest/api/tt/casatools.image.html#casatools.image.image.restoringbeam 

2515 nchan = allbeams['nChannels'] 

2516 areas = np.zeros(nchan, 'float') 

2517 axratio = np.zeros(nchan, 'float') 

2518 weight = np.zeros(nchan, 'bool') 

2519 

2520 for ii in range(0, nchan): 

2521 axmajor = cqa.convert(cqa.quantity(allbeams['beams']['*' + str(ii)]['*0']['major']), 'arcsec')['value'] 

2522 axminor = cqa.convert(cqa.quantity(allbeams['beams']['*' + str(ii)]['*0']['minor']), 'arcsec')['value'] 

2523 areas[ii] = axmajor * axminor 

2524 axratio[ii] = axmajor / axminor 

2525 weight[ii] = np.isfinite(axratio[ii]) 

2526 

2527 # Detect outliers based on axis ratio 

2528 # The iterative loop is to get robust autoflagging 

2529 # Add a linear fit instead of just a 'mean' to account for slopes (useful for flagging on beam_area) 

2530 local_axratio = axratio.copy() 

2531 for steps in range(0, 2): ### Heuristic : how many iterations here ? 

2532 local_axratio[~weight] = np.nan 

2533 mean_axrat = np.nanmean(local_axratio) 

2534 std_axrat = np.nanstd(local_axratio) 

2535 ## Flag all points deviating from the mean by 3 times stdev 

2536 weight[np.fabs(local_axratio - mean_axrat) > 3 * std_axrat] = False 

2537 

2538 ## Find the first channel with a valid beam 

2539 validbeamchans = np.where(weight) 

2540 chanid = 0 

2541 if len(validbeamchans) > 0: 

2542 chanid = validbeamchans[0][0] 

2543 else: 

2544 LOG.error('No valid beams in {!s}'.format(psf_filename)) 

2545 return None, np.arange(nchan) 

2546 

2547 # Fill all flagged channels with the largest/first valid beam 

2548 dummybeam = allbeams['beams']['*' + str(chanid)]['*0'] 

2549 for ii in range(0, nchan): 

2550 if not weight[ii]: 

2551 image.setrestoringbeam( 

2552 major=dummybeam['major'], minor=dummybeam['minor'], pa=dummybeam['positionangle'], channel=ii 

2553 ) 

2554 

2555 # Need to close and reopen for commonbeam() to see the new chan beams ! 

2556 with casa_tools.ImageReader(psf_filename) as image: 

2557 ## Recalculate common beam 

2558 newcommonbeam = image.commonbeam() 

2559 

2560 # Reinstate the old beam to get back to the original iter0 product 

2561 with casa_tools.ImageReader(psf_filename) as image: 

2562 for ii in range(0, nchan): 

2563 if not weight[ii]: 

2564 beam = allbeams['beams']['*' + str(ii)]['*0'] 

2565 image.setrestoringbeam( 

2566 major=beam['major'], minor=beam['minor'], pa=beam['positionangle'], channel=ii 

2567 ) 

2568 

2569 LOG.info( 

2570 'Commonbeam of %s, from ia.commonbeam(): %s', 

2571 os.path.basename(psf_filename), 

2572 ImageParamsHeuristics._commonbeam_to_string(commonbeam), 

2573 ) 

2574 LOG.info( 

2575 'Commonbeam of %s, from find_good_commonbeam(): %s', 

2576 os.path.basename(psf_filename), 

2577 ImageParamsHeuristics._commonbeam_to_string(newcommonbeam), 

2578 ) 

2579 

2580 bad_psf_channels = np.where(np.logical_not(weight))[0] 

2581 

2582 return newcommonbeam, bad_psf_channels
