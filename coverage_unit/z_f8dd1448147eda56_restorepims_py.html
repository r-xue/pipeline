120 def _check_resources(self): 

121 

122 # Create the resource request list 

123 # - self.imagename at this point is from SEIP_parameter.list of the SEIP products. 

124 # - inputs.vis should be identical to the SEIP input with the same flag state. 

125 # Therefore, we verify these names against the file list inside reimaging_resources.tgz 

126 

127 restore_resources = {} 

128 reimaging_resources_tgz = self.inputs.reimaging_resources 

129 flagd_pat_key_desc = (self.inputs.vis+'.flagversions*', 'flag_dir', 'flag directory') 

130 flagv_pat_key_desc = (self.inputs.vis+f'.flagversions/flags.{self.flagversion}', 'flag_version', 'flag version') 

131 sctab_pat_key_desc = (self.inputs.vis+'.*.phase-self-cal.tbl*', 'selfcal_table', 'selfcal table') 

132 model_pat_key_desc = ('s*_0.'+self.imagename+'.I.iter1.model*', 'model_images', 'model image(s)') 

133 tier1_pat_key_desc = ('s*_0.'+self.imagename+'.QLcatmask-tier1.mask*', 'tier1_mask', 'tier1 mask') 

134 tier2_pat_key_desc = ('s*_0.'+self.imagename+'.combined-tier2.mask*', 'tier2_mask', 'tier2 mask') 

135 pkd_list = [flagd_pat_key_desc, flagv_pat_key_desc, sctab_pat_key_desc, 

136 model_pat_key_desc, tier1_pat_key_desc, tier2_pat_key_desc] 

137 

138 # check the reimaging resources tarball before trying unpacking. 

139 

140 is_resources_available = os.path.isfile(reimaging_resources_tgz) and tarfile.is_tarfile(reimaging_resources_tgz) 

141 restore_resources['reimaging_resources'] = (reimaging_resources_tgz, is_resources_available) 

142 

143 if is_resources_available: 

144 LOG.info(f"Found the reimaging resources tarball: {reimaging_resources_tgz}") 

145 else: 

146 LOG.error( 

147 f"The reimaging resources tarball {reimaging_resources_tgz} doesn't exist or is not a tarfile.") 

148 return restore_resources, is_resources_available 

149 

150 # untar any files required for the restorepims operation: 

151 

152 with tarfile.open(reimaging_resources_tgz, 'r:gz') as tar: 

153 

154 members = [] 

155 for member in tar.getmembers(): 

156 is_resource = any([fnmatch(member.name, pkd[0]) for pkd in pkd_list]) 

157 if is_resource: 

158 members.append(member) 

159 # if the request flagversion already exists, do not override the flagversions directory to preserve 

160 # any recent flagversions, e.g. the hifv_restorepims backup. 

161 if os.path.exists(flagv_pat_key_desc[0]): 

162 LOG.warning( 

163 f'{flagv_pat_key_desc[0]} exists and the .flagversions directory will be re-used and not extracted.') 

164 members = [member for member in members if not fnmatch(member.name, flagd_pat_key_desc[0])] 

165 for member in members: 

166 LOG.info(f'extracting: {member.name}') 

167 

168 tar.extractall(path='.', members=members) 

169 

170 # check against the resource requirement list 

171 

172 for pkd in pkd_list: 

173 paths = utils.glob_ordered(pkd[0]) 

174 if paths: 

175 restore_resources[pkd[1]] = (paths, True) # (file_list, True) 

176 LOG.info(f"Found the request {pkd[2]}: {', '.join(paths)}") 

177 else: 

178 restore_resources[pkd[1]] = (pkd[0], False) # (file_pattern, False) 

179 is_resources_available = False 

180 LOG.error(f"Cannot find the request {pkd[2]} using the name pattern: {pkd[0]}") 

181 

182 return restore_resources, is_resources_available

213 def _do_restoremodel(self, model_images): 

214 

215 ms = self.inputs.context.observing_run.get_ms(self.inputs.vis) 

216 

217 with casa_tools.TableReader(ms.name) as table: 

218 is_modelcolumn_present = 'MODEL_DATA' in table.colnames() 

219 with casa_tools.TableReader(ms.name+'/SOURCE') as table: 

220 is_virtualmodel_present = 'SOURCE_MODEL' in table.colnames() 

221 

222 # remove any (unlikely) virtual/otf model in the MS/SOURCE subtable to prevent it overriding the effect of the modelcolumn. 

223 # see: https://casadocs.readthedocs.io/en/latest/api/tt/casatasks.imaging.tclean.html?highlight=savemodel#savemodel 

224 if is_virtualmodel_present: 

225 job = casa_tasks.delmod(vis=self.inputs.vis, otf=True, scr=False) 

226 self._executor.execute(job) 

227 

228 # re-initialze FLAGs/WEIGHTs if needed to ensure identical predicting (vs. SE-CONT) from model image created with normtype='flatnoise'. 

229 if is_modelcolumn_present: 

230 LOG.warning(f'MODEL_DATA column found in {ms.basename} and will be overwritten.') 

231 self._reinitialize_pims() 

232 else: 

233 LOG.info('Writing model data to {}'.format(ms.basename)) 

234 

235 mp_nthreads = casa_tools.casalog.ompGetNumThreads() 

236 LOG.info(f'Predicting the MODEL column with openmp_nthreads : {mp_nthreads}') 

237 

238 model_image = model_images[0] 

239 last_idx = model_image.rfind('.model') 

240 restore_imagename = model_image[:last_idx] 

241 restore_startmodel = model_images 

242 

243 inputs = self.inputs 

244 for target in inputs.context.clean_list_pending: # just one target for VLASS-SE-CONT* 

245 target['niter'] = 0 # not essential but ensure this indicated this is for a selfcal restoration. 

246 target['mask'] = [] # empty the mask list to stop tclean beyond iter=0 (just for the model restoration) 

247 target['heuristics'].restore_imagename = restore_imagename # this has been sorted: xx.tt0, xx.tt1. 

248 # the tclean imagename used in the production selfcal-imaging flow 

249 target['heuristics'].restore_startmodel = restore_startmodel 

250 # note: both restore_imagename/_startmodel here can be used to recover the modelcolumn via tclean() 

251 # we add both but later use 'restore_imagname' due to the csys mismatch issue (CAS-13338) during the parallel=True->False switch in 

252 # stage5 when the vlass-se-cont workflow is done in mpicasa. 

253 # tclean/startmodel will always regrid when csys mismatch 

254 # tclean/imagename will just reset to use the cys on the disk, leaving the model untouch. 

255 # the later is what we prefer to reproduce the sequence in the mpicasa+awp situation 

256 

257 makeimages_inputs = MakeImages.Inputs(inputs.context, vis=inputs.vis, hm_masking='manual') 

258 makeimages_task = MakeImages(makeimages_inputs) 

259 self._executor.execute(makeimages_task, True) 

260 

261 return

289 def _do_applycal(self, gaintable): 

290 """Rerun applycal using the selfcal table from the SE reimaging resources. 

291 

292 Also see the SEIP setting in hifv_selfcal() 

293 """ 

294 

295 m = self.inputs.context.observing_run.get_ms(self.inputs.vis) 

296 spwsobjlist = m.get_spectral_windows(science_windows_only=True) 

297 spws = [int(spw.id) for spw in spwsobjlist] 

298 numspws = len(m.get_spectral_windows(science_windows_only=False)) 

299 lowestscispwid = min(spws) # PIPE-101, PIPE-1042: spwmap parameter in applycal must be a list of integers 

300 # VLASS mode 

301 applycal_task_args = {'vis': self.inputs.vis, 

302 'gaintable': gaintable, 

303 'interp': ['nearestPD'], 

304 'spwmap': [numspws * [lowestscispwid]], 

305 'parang': False, 

306 'applymode': 'calonly'} 

307 

308 applycal_task_args['calwt'] = False 

309 applycal_task_args['interp'] = ['nearest'] 

310 

311 job = casa_tasks.applycal(**applycal_task_args) 

312 

313 return self._executor.execute(job)
