13def get_spwmaps(context: Context, results: ResultsList, include_empty: bool = True) -> list[SpwMapInfo]:

18 

19 Args: 

20 context: The pipeline context. 

21 results: List of task results. 

22 include_empty: Include empty SpwMapInfo entries for MS if result

24 

25 Returns: 

26 List of SpwMapInfo instances. 

27 """ 

28 spwmaps = []

30 for result in results: 

31 ms = context.observing_run.get_ms(result.vis) 

32 

33 if result.spwmaps:

35 science_spw_ids = [spw.id for spw in ms.get_spectral_windows(science_windows_only=True)]

38 # Get ID of field and scans. 

39 fieldid = ms.get_fields(name=[field])[0].id

41 

42 # Append info on spwmap to list. 

43 spwmaps.append(SpwMapInfo(ms.basename, intent, field, fieldid, spwmapping.combine, spwmapping.spwmap, 

44 scanids, science_spw_ids, spwmapping.solint, spwmapping.gaintype)) 

45 elif include_empty:
