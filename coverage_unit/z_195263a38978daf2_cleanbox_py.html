428 """Compute statistics for each Stokes plane of a 4D CASA image cube.

432 [nx, ny, nstokes, nfreq]. 

433 

434 It is used as a workaround for CAS-14660 when deriving per-Stokes cube statistics. 

435 

436 Args: 

437 image: A CASA image object with a .shape() method and .statistics() method. 

438 stokescontrol: If 'f', only the first Stokes plane (index 0) is processed; otherwise all planes are processed. 

439 region: Optional region record to constrain the statistics to a specific area. 

440 **kwargs: Passed directly to the image's .statistics() method (e.g., verbose, stretch). 

441 

442 Returns: 

443 A dictionary mapping statistic names (e.g., 'max', 'mean', 'rms') to NumPy arrays. 

444 Each array contains values for the processed Stokes planes. 

445 

446 Raises: 

447 AttributeError: If the image object lacks a .shape() method. 

448 ValueError: If the image is not 4-dimensional. 

449 

450 Example: 

451 >>> stats = image_statistics_per_stokes(my_image, stokescontrol='a', verbose=True) 

452 >>> print(stats['mean']) # Mean values for all Stokes planes 

453 [0.001, 0.002, 0.001, 0.003] 

454 """ 

455 # Validate input image 

456 try: 

457 img_shape = image.shape() 

458 except AttributeError as e: 

459 raise AttributeError("The 'image' object must provide a .shape() method.") from e 

460 if len(img_shape) != 4: 

461 raise ValueError(f'Expected 4D image, got {len(img_shape)}D with shape {img_shape}.') 

462 

463 # Avoid CAS-14660 bug by discarding explicit axis specification 

464 kwargs.pop('axes', None) 

465 

466 # Initialize statistics containers 

467 stats_items = [ 

468 'max', 

469 'mean', 

470 'medabsdevmed', 

471 'median', 

472 'min', 

473 'npts', 

474 'q1', 

475 'q3', 

476 'rms', 

477 'sigma', 

478 'sum', 

479 'sumsq', 

480 ] 

481 stats_per_stokes: dict[str, list[float]] = {key: [] for key in stats_items} 

482 

483 # Handle region specification 

484 if isinstance(region, dict) and {'blc', 'trc'} <= region.keys() and region.get('name') == 'LCSlicer': 

485 # rg.box uses zero-based indices; region records return one-based indices, so convert back 

486 blc = region['blc'].copy() - 1 

487 trc = region['trc'].copy() - 1 

488 if stokescontrol == 'f': 

489 trc[2] = blc[2] 

490 else: 

491 if region: 

492 LOG.warning('Unsupported region format %r; ignoring.', region) 

493 blc = [0, 0, 0, 0] 

494 stokes_last = 0 if stokescontrol == 'f' else img_shape[2] - 1 

495 trc = [img_shape[0] - 1, img_shape[1] - 1, stokes_last, img_shape[3] - 1] 

496 

497 rg = casa_tools.regionmanager 

498 for stokes_idx in range(int(blc[2]), int(trc[2]) + 1): 

499 blc_stokes, trc_stokes = blc.copy(), trc.copy() 

500 blc_stokes[2] = trc_stokes[2] = stokes_idx 

501 region_stokes = rg.box(blc=blc_stokes, trc=trc_stokes) 

502 

503 stats_current = image.statistics(region=region_stokes, **kwargs) 

504 for key in stats_per_stokes: 

505 if key in stats_current and stats_current[key].size: 

506 stats_per_stokes[key].append(stats_current[key][0]) 

507 

508 return {key: np.array(values) for key, values in stats_per_stokes.items()}
