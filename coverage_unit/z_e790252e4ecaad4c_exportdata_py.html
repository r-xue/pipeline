111 def prepare(self) -> exportdata.ExportDataResults: 

112 """Prepare and execute an export data job appropriate to the task inputs. 

113 

114 Returns: 

115 object of exportdata.ExportDataResults 

116 """ 

117 results = super(NROExportData, self).prepare() 

118 

119 # manifest file 

120 manifest_file = os.path.join(self.inputs.context.products_dir, 

121 results.manifest) 

122 

123 # export NRO data reduction template 

124 template_script = \ 

125 self._export_reduction_template(self.inputs.products_dir) 

126 

127 if template_script is not None: 

128 self._update_manifest(manifest_file, script=template_script) 

129 

130 # export NRO scaling file template 

131 template_file = \ 

132 self._export_nroscalefile_template(self.inputs.products_dir) 

133 

134 if template_file is not None: 

135 self._update_manifest(manifest_file, scalefile=template_file) 

136 

137 return results

139 def _export_reduction_template(self, products_dir: str) -> str: 

140 """Export reduction template script. 

141 

142 Exports rebase_and_image.py. It is a Python script to perform baseline 

143 subtraction and imaging by CASA tasks. 

144 

145 Args: 

146 products_dir : product directory path 

147 

148 Returns: 

149 script name 

150 """ 

151 script_name = 'rebase_and_image.py' 

152 config_name = 'rebase_and_image_config.py' 

153 script_path = os.path.join(products_dir, script_name) 

154 config_path = os.path.join(products_dir, config_name) 

155 

156 status = nrotemplategenerator.generate_script(self.inputs.context, 

157 script_path, 

158 config_path) 

159 return script_name if status is True else None

177 def _update_manifest(self, manifest_file: str, script: str=None, 

178 scalefile: str=None): 

179 """Add Nobeyama specific products to manifest file. 

180 

181 Args: 

182 manifest_file : manifest file path 

183 script : the name of template reduction script 

184 scalefile : the name of NRO scale file 

185 """ 

186 pipemanifest = manifest.NROPipelineManifest('') 

187 pipemanifest.import_xml(manifest_file) 

188 ouss = pipemanifest.get_ous() 

189 

190 if script: 

191 pipemanifest.add_reduction_script(ouss, script) 

192 pipemanifest.write(manifest_file) 

193 

194 if scalefile: 

195 pipemanifest.add_scalefile(ouss, scalefile) 

196 pipemanifest.write(manifest_file)

198 def _export_casa_restore_script(self, context: Context, script_name: str, 

199 products_dir: str, oussid: str, 

200 vislist: List[str], 

201 session_list: List[str]) -> str:

226 def _get_hm_rasterscan_value(self, context: Context) -> str: 

227 """Retrieve hm_rasterscan value from Results object. 

228 

229 This method checks if either NROImportDataResults or 

230 NRORestoreDataResults object is registered to Pipeline context. 

231 If exists, hm_rasterscan value is retrieved from the results 

232 object. If no NROImportDataResults nor NRORestoreDataResults 

233 object is registered, 'time' will be returned. 

234 

235 Args: 

236 context: Pipeline context 

237 

238 Returns: 

239 The hm_rasterscan value. 

240 """ 

241 results_filter = filter( 

242 lambda x: isinstance(x, basetask.ResultsList) and isinstance(x[0], (NROImportDataResults, NRORestoreDataResults)), 

243 map(lambda x: x.read(), context.results) 

244 ) 

245 importdata_results = next(results_filter, None) 

246 if importdata_results: 

247 hm_rasterscan = importdata_results.inputs.get('hm_rasterscan', 'time') 

248 else: 

249 hm_rasterscan = 'time' 

250 

251 return hm_rasterscan
