142 Merge results with context.

147 super(NRORestoreDataResults, self).merge_with_context(context)

149 # set amplitude scaling factor to ms domain objects 

150 if isinstance(self.applycal_results, ResultsList):

156 def _merge_ampcal(self, context: Context, applycal_results: ApplycalResults): 

157 """ 

158 Merge results of applycal with context. 

159 

160 Args: 

161 context: Context object 

162 applycal_results: results of applycal 

163 """ 

164 for calapp in applycal_results.applied: 

165 msobj = context.observing_run.get_ms(name=os.path.basename(calapp.vis)) 

166 if not hasattr(msobj, 'k2jy_factor'): 

167 for _calfrom in calapp.calfrom: 

168 if _calfrom.caltype == 'amp' or _calfrom.caltype == 'gaincal': 

169 LOG.debug('Adding k2jy factor to {0}'.format(msobj.basename)) 

170 # k2jy gaincal table 

171 k2jytable = _calfrom.gaintable 

172 k2jy_factor = {} 

173 with casa_tools.TableReader(k2jytable) as tb: 

174 spws = tb.getcol('SPECTRAL_WINDOW_ID') 

175 antennas = tb.getcol('ANTENNA1') 

176 params = tb.getcol('CPARAM').real 

177 nrow = tb.nrows() 

178 for irow in range(nrow): 

179 spwid = spws[irow] 

180 antenna = antennas[irow] 

181 param = params[:, 0, irow] 

182 npol = param.shape[0] 

183 antname = msobj.get_antenna(antenna)[0].name 

184 dd = msobj.get_data_description(spw=int(spwid)) 

185 if dd is None: 

186 continue 

187 for ipol in range(npol): 

188 polname = dd.get_polarization_label(ipol) 

189 k2jy_factor[(spwid, antname, polname)] = 1.0 / (param[ipol] * param[ipol]) 

190 msobj.k2jy_factor = k2jy_factor 

191 LOG.debug('msobj.k2jy_factor = {0}'.format(getattr(msobj, 'k2jy_factor', 'N/A')))

206 results = super(NRORestoreData, self).prepare()

216 def _do_importasdm(self, sessionlist: List[str], vislist: List[str]): 

217 """ 

218 Execute importasdm. 

219 

220 Args: 

221 sessionlist: a list of sessions 

222 vislist: a list of vis 

223 """ 

224 inputs = self.inputs 

225 # NROImportDataInputs operate in the scope of a single measurement set. 

226 # To operate in the scope of multiple MSes we must use an 

227 # InputsContainer. 

228 

229 LOG.debug('_do_importasdm inputs = {0}'.format(inputs)) 

230 

231 container = vdp.InputsContainer(importdata.NROImportData, inputs.context, vis=vislist, 

232 output_dir=None, hm_rasterscan=inputs.hm_rasterscan) 

233 importdata_task = importdata.NROImportData(container) 

234 return self._executor.execute(importdata_task, merge=True)

236 def _do_applycal(self): 

237 """Execute applycal.""" 

238 inputs = self.inputs 

239 LOG.debug('_do_applycal inputs = {0}'.format(inputs)) 

240 

241 # Before applycal, sensitively (amplitude) correction using k2jycal task and 

242 # a scalefile (=reffile) given by Observatory. This is the special operation for NRO data. 

243 # If no scalefile exists in the working directory, skip this process. 

244 if os.path.exists(inputs.reffile): 

245 container = vdp.InputsContainer(ampcal.SDAmpCal, inputs.context, reffile=inputs.reffile) 

246 else: 

247 LOG.info('No scale factor file exists. Skip scaling.') 

248 container = vdp.InputsContainer(ampcal.SDAmpCal, inputs.context) 

249 LOG.debug('ampcal container = {0}'.format(container)) 

250 ampcal_task = ampcal.SDAmpCal(container) 

251 self.ampcal_results = self._executor.execute(ampcal_task, merge=True) 

252 

253 # SDApplyCalInputs operates in the scope of a single measurement set. 

254 # To operate in the scope of multiple MSes we must use an 

255 # InputsContainer. 

256 container = vdp.InputsContainer(applycal.SerialSDApplycal, inputs.context) 

257 applycal_task = applycal.SerialSDApplycal(container) 

258 LOG.debug('_do_applycal container = {0}'.format(container)) 

259 return self._executor.execute(applycal_task, merge=True)
