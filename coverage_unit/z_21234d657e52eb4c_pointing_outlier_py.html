14class PointingOutlierHeuristics(api.Heuristic): 

15 # Variables below are the threshold factors for the heuristics. 

16 # Which factor to use depends on the situation of the scan, 

17 # especially, whether the last scan can be interrupted or not, 

18 # and the number of raster rows per map. 

19 

20 # (1) Interrupted scan is not allowed by the system. 

21 # The factor is 2.0 in this case. 

22 FACTOR_NO_INTERRUPT = 2 

23 

24 # (2) Raster scan can be interrupted in the middle, and 

25 # number of raster rows per map is always greater than 2. 

26 # The factor is sqrt(13 * pi / 8) in this case. 

27 FACTOR_INTERRUPT = 2.26 

28 

29 # (3) Raster scan can be interrupted in the middle, and 

30 # number of raster rows per map can be 2. 

31 # The factor is 4 * sqrt(5) / 3 in this case. 

32 FACTOR_INTERRUPT_LESS_RASTER_ROWS = 2.982 

33 

34 def calculate(self, 

35 dir_frame: str, 

36 x: np.ndarray, 

37 y: np.ndarray, 

38 iterate: bool = False) -> PointingOutlierHeuristicsResult: 

39 """Perform pointing outlier heuristics. 

40 

41 Algorithm is as follows: 

42 

43 1. take median of x and y coordinates -> med_x, med_y 

44 2. compute distance from (med_x, med_y) for each data -> dist 

45 3. take median of distance array, dist -> med_dist 

46 4. mask data if dist > threshold * med_dist 

47 (mask is False if dist > threshold * med_dist) 

48 please see below for threshold factor 

49 5. if iterate is True, do the following: 

50 5-1. re-compute med_x and med_y using a subset of 

51 the data where mask is True 

52 5-2. re-compute distance from updated (med_x, med_y) 

53 to update dist 

54 5-3. take median of updated dist 

55 5-4. update mask using updated med_dist 

56 

57 Regarding the step 4, three threshold factors are available. 

58 They are defined as class attributes. Which factor should use 

59 depends on the observing strategy. Current observing strategy 

60 in ALMA corresponds to the first case, so the current 

61 implementation uses FACTOR_INTERRUPT. 

62 

63 - If interrupted scan can happen, and there is a guarantee 

64 that number of raster rows > 2, then threshold will be 

65 FACTOR_INTERRUPT. 

66 - if interrupted scan can happen, and number of raster 

67 rows can be 2, then threshold will be 

68 FACTOR_INTERRUPT_LESS_RASTER_ROWS. 

69 - if interrupted scan should not happen, threshold will be 

70 FACTOR_NO_INTERRUPT. 

71 

72 Args: 

73 dir_frame: direction reference frame 

74 x: List of RA values in degrees. For ephemeris source, 

75 RA values must be shifted in advance (use SHIFT_RA column 

76 in datatable). 

77 y: List of DEC values in degrees. For ephemeris source, 

78 DEC values must be shifted in advance (use SHIFT_DEC column 

79 in datatable). 

80 iterate: whether or not perform iteration. Default is False. 

81 

82 Returns: 

83 PointingOutlierHeuristicsResult object, which includes: 

84 

85 - cx: x coordinate of nominal center position 

86 - cy: y coordinate of nominal center position 

87 - med_dist: median distance from the nominal center 

88 - threshold: factor to derive the threshold from 

89 median distance 

90 - mask: validity mask (False means flagged by this heuristic) 

91 - dist: computed distance array with respect to 

92 the nominal center 

93 """ 

94 med_x = np.median(x) 

95 med_y = np.median(y) 

96 distance = coordinate_utils.angular_distances(dir_frame, x, y, med_x, med_y) 

97 median_distance = np.median(distance) 

98 # raster scan can be interrupted, and number of raster rows > 2 

99 threshold = self.FACTOR_INTERRUPT 

100 if iterate: 100 ↛ 101line 100 didn't jump to line 101 because the condition on line 100 was never true

101 unflagged = distance <= threshold * median_distance 

102 med_x = np.median(x[unflagged]) 

103 med_y = np.median(y[unflagged]) 

104 distance = np.sqrt(np.square(x - med_x) + np.square(y - med_y)) 

105 median_distance = np.median(distance) 

106 mask = distance <= threshold * median_distance 

107 

108 return PointingOutlierHeuristicsResult( 

109 med_x, 

110 med_y, 

111 median_distance, 

112 threshold, 

113 mask, 

114 distance
